67      private BorderPane root_;
68      private SplitPane mainSplitPane;
69      private MenuBar menuBar;
70      private BorderPane loadWait;
71      private boolean preloadExecuted = false;

74      @Inject
75      private IterableProvider<IsaacViewWithMenusI> moduleViews_;
76      @Inject
77      private IterableProvider<DockedViewI> dockedViews_;
78      @Inject

81      //Just a hashed view of all of the menus (including nested menus)
82      private final Hashtable<String, Menu> allMenus_ = new Hashtable<>();
83  
84      public AppController() {
85  
86          AppContext.getServiceLocator().inject(this);
87          ToolTipDefaultsFixer.setTooltipTimers(100, 20000, 200);
88          
89          root_ = new BorderPane();
90          mainSplitPane = new SplitPane();
91          mainSplitPane.setDividerPositions(0.60);
92          mainSplitPane.setOrientation(Orientation.HORIZONTAL);
93          mainSplitPane.getStyleClass().add("hashedBackground");
94          
95          root_.setCenter(mainSplitPane);
96          root_.setMaxHeight(Double.MAX_VALUE);
97          root_.setMaxWidth(Double.MAX_VALUE);
98          
99          loadWait = new BorderPane();
100         loadWait.setCenter(LightWeightDialogs.buildLoadingDialog());
101         mainSplitPane.getItems().add(loadWait);
102         mainSplitPane.setMaxWidth(Double.MAX_VALUE);
103         mainSplitPane.setMaxHeight(Double.MAX_VALUE);
104         
105         
106         menuBar = new MenuBar();
107         for (ApplicationMenus menu : ApplicationMenus.values())
108         {
109             Menu m = new Menu(menu.getMenuName());
110             m.setId(menu.getMenuId());
111             m.setDisable(true);
112             m.setMnemonicParsing(false);
113             menuBar.getMenus().add(m);
114             //index these for ease in adding module menus
115             allMenus_.put(m.getId(), m);
116         }
117         
118         root_.setTop(menuBar);
119 
120         //Sort them...
121         TreeSet<MenuItemI> menusToAdd = new TreeSet<>();
122         for (IsaacViewWithMenusI view : moduleViews_)
123         {
124             for (MenuItemI menuItem : view.getMenuBarMenus())
125             {
126                 menusToAdd.add(menuItem);
127             }
128         }
129 
130         for (final MenuItemI menuItemsToCreate : menusToAdd)
131         {
132             Menu parentMenu = allMenus_.get(menuItemsToCreate.getParentMenuId());
133             if (parentMenu == null)
134             {
135                 LOG.error("Cannot add module menu '" + menuItemsToCreate.getMenuId() + "' because the specified parent menu doesn't exist");
136             }
137             else
138             {
139                 MenuItem menuItem = new MenuItem();
140                 menuItem.setId(menuItemsToCreate.getMenuId());
141                 menuItem.setText(menuItemsToCreate.getMenuName());
142                 menuItem.setMnemonicParsing(menuItemsToCreate.enableMnemonicParsing());
143                 menuItem.setOnAction(new EventHandler<ActionEvent>()
144                 {
145                     @Override
146                     public void handle(ActionEvent arg0)
147                     {
148                         menuItemsToCreate.handleMenuSelection(root_.getScene().getWindow());
149                     }
150                 });
151                 if (menuItemsToCreate.getImage() != null)
152                 {
153                     menuItem.setGraphic(new ImageView(menuItemsToCreate.getImage()));
154                 }
155                 if (menuItemsToCreate.getDisableBinding() != null)
156                 {
157                     menuItem.disableProperty().bind(menuItemsToCreate.getDisableBinding());
158                 }
159                 parentMenu.getItems().add(menuItem);
160                 //TODO fix this sorting API stuff... supposed to be sorted by the menu order in the menu API - but was never finished... see other TODO below.
161                 parentMenu.getItems().sort(new Comparator<MenuItem>() {
162                   @Override
163                   public int compare(MenuItem o1, MenuItem o2) {
164                     return o1.getText().compareTo(o2.getText());
165                   }
166                   
167                 });
168             }
169         }
170 
171         for (final DockedViewI dv : dockedViews_)
172         {
173             try
174             {
175                 Menu parentMenu = allMenus_.get(dv.getMenuBarMenuToShowView().getParentMenuId());
176                 if (parentMenu == null)
177                 {
178                     LOG.error("Cannot add module menu '" + dv.getMenuBarMenuToShowView().getMenuId() + "' because the specified parent menu doesn't exist");
179                 }
180                 else
181                 {
182                     final BorderPane bp = buildPanelForView(dv);
183                     //TODO this isn't honoring sort order... need to sort all of the menus from the DockedViewI at once....
184                     final CheckMenuItem mi = new CheckMenuItem();
185                     mi.setText(dv.getMenuBarMenuToShowView().getMenuName());
186                     mi.setId(dv.getMenuBarMenuToShowView().getMenuId());
187                     mi.setMnemonicParsing(dv.getMenuBarMenuToShowView().enableMnemonicParsing());
188                     mi.selectedProperty().addListener(new InvalidationListener()
189                     {
190                         @Override
191                         public void invalidated(Observable observable)
192                         {
193                             //This is a convenience call... not expected to actually show the view.
194                             dv.getMenuBarMenuToShowView().handleMenuSelection(root_.getScene().getWindow());
195                             if (mi.isSelected() && !mainSplitPane.getItems().contains(bp))
196                             {
197                                 mainSplitPane.getItems().add(bp);
198                             }
199                             else if (!mi.isSelected() && mainSplitPane.getItems().contains(bp))
200                             {
201                                 hidePanelView(bp);
202                             }
203                         }
204                     });
205                     if (dv.getMenuBarMenuToShowView().getImage() != null)
206                     {
207                         mi.setGraphic(new ImageView(dv.getMenuBarMenuToShowView().getImage()));
208                     }
209                     mi.selectedProperty().bindBidirectional(bp.visibleProperty());
210                     parentMenu.getItems().add(mi);
211                 }
212             }
213             catch (Exception e)
214             {
215                 LOG.error("Unexpected error configuring DockedViewI " + (dv == null ? "?" : dv.getViewTitle()), e);
216             }
217         }
218     }

225     protected void finishInit() {
226         // Make sure in application thread.
227         FxUtils.checkFxUserThread();
228         
229         //Kick off other preloads
230         for (ServicesToPreloadI service : preloadRequested_)
231         {
232             LOG.debug("Preloading {}", service);
233             service.loadRequested();
234         }
235         preloadExecuted = true;
236         
237         loadWait.getChildren().clear();
238         
239         AtomicLong loginFailCount = new AtomicLong(0);
240         loadWait.setCenter(LightWeightDialogs.buildLoginDialog(new Consumer<Boolean>()
241         {
242             @Override
243             public void accept(Boolean t)
244             {
245                 if (!t)
246                 {
247                     if (loginFailCount.incrementAndGet() > 3)
248                     {
249                         ((App)ExtendedAppContext.getMainApplicationWindow()).shutdown();
250                     }
251                 }
252                 else
253                 {
254                     mainSplitPane.getItems().remove(loadWait);
255                     loadWait = null;
256                     // Enable the menus.
257                     for (Menu menu : menuBar.getMenus())
258                     {
259                         menu.setDisable(false);
260                     }
261                 }
262                 
263             }
264         }));
265         BorderPane.setAlignment(loadWait.getCenter(), Pos.CENTER);
266     }

268     private BorderPane buildPanelForView(DockedViewI dockedView)
269     {
270         final BorderPane bp = new BorderPane();
271         bp.setVisible(false);
272         AnchorPane ap = new AnchorPane();
273         ap.getStyleClass().add("headerBackground");
274 
275         Label l = new Label(dockedView.getViewTitle());
276         l.getStyleClass().add("titleLabel");
277         AnchorPane.setLeftAnchor(l, 5.0);
278         AnchorPane.setTopAnchor(l, 2.0);
279         ap.getChildren().add(l);
280 
281         Button b = new Button();
282         b.setMnemonicParsing(false);
283         b.setStyle("-fx-cursor:hand");
284         b.getStyleClass().add("tab-close-button");
285         b.setOnAction(new EventHandler<ActionEvent>()
286         {
287             @Override
288             public void handle(ActionEvent arg0)
289             {
290                 hidePanelView(bp);
291             }
292         });
293         AnchorPane.setTopAnchor(b, 0.0);
294         AnchorPane.setRightAnchor(b, 5.0);
295         ap.getChildren().add(b);
296 
297         bp.setTop(ap);
298         Label placeholder = new Label("");
299         bp.setCenter(placeholder);
300         //Delay this, so we don't force the init of all of these panels up front
301         bp.visibleProperty().addListener((change) -> 
302         {
303             if (bp.isVisible() && bp.getCenter() == placeholder)
304             {
305                 bp.setCenter(dockedView.getView());
306             }
307         });
308         return bp;
309     }

311     private void hidePanelView(BorderPane bp)
312     {
313         bp.setVisible(false);
314         mainSplitPane.getItems().remove(bp);
315     }

317     public void ensureDockedViewIsVisible(DockedViewI view)
318     {
319         Menu parentMenu = allMenus_.get(view.getMenuBarMenuToShowView().getParentMenuId());
320         String id = view.getMenuBarMenuToShowView().getMenuId();
321         for (MenuItem menu : parentMenu.getItems())
322         {
323             if (menu.getId().equals(id))
324             {
325                 CheckMenuItem cmi = (CheckMenuItem)menu;
326                 cmi.selectedProperty().set(true);
327                 break;
328             }
329         }
330     }
331     
332     protected void shutdown()
333     {
334         AppContext.getService(RuntimeGlobals.class).shutdown();
335         if (preloadExecuted)
336         {
337             //notify anything that was preloaded
338             for (ServicesToPreloadI service : preloadRequested_)
339             {
340                 LOG.debug("Shutdown notify {}", service);
341                 service.shutdown();
342             }
343         }
344     }
