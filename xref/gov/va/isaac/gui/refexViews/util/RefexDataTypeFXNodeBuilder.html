81  	private static Logger logger = LoggerFactory.getLogger(RefexDataTypeFXNodeBuilder.class);

83  	/**
84  	 * @param dt - The datatype of the node to be built
85  	 * @param defaultValue - The default value for the node (null allowed)
86  	 * @param currentValue - The current value for the node to prepopulate GUI (null allowed)
87  	 * @param valueIsRequired - If this value must be set, pass in a SimpleStringProperty - it will be updated with a suitable message when the node is blank.
88  	 * If this value is not required, pass null. 
89  	 * @param defaultValueTooltip - If this field has a default value - as a convenience, this pass in a SimpleStringProperty here, and it will update the value
90  	 * as appropriate with a user friendly string explaining the default value.  (null allowed)
91  	 * @param polymorphicSelection - If the specified datatype of this node is polymorphic - the user has to choose a type for this instance.  This is for the users
92  	 * choice.  Null allowed, if the datatype is not polymorphic.
93  	 * @param allValid - Bindings will be added to this UpdateableBooleanBinding depending on the node construction.  The objects that were bound will be returned in the 
94  	 * result object.
95  	 * @param validatorType - if not null, data entered into the field will be validated against the validator.
96  	 * @param validatorData - if not null (and validatorType is not null) data entered into the field will be validated against the validator.
97  	 * 
98  	 */
99  	public static RefexDataTypeNodeDetails buildNodeForType(RefexDynamicDataType dt, RefexDynamicDataBI defaultValue, RefexDynamicDataBI currentValue, 
100 			SimpleStringProperty valueIsRequired, SimpleStringProperty defaultValueTooltip, ReadOnlyObjectProperty<RefexDynamicDataType> polymorphicSelection, 
101 			UpdateableBooleanBinding allValid, ObjectProperty<RefexDynamicValidatorType> validatorType, ObjectProperty<RefexDynamicDataBI> validatorData)
102 	{
103 		return buildNodeForType(dt, defaultValue, currentValue, valueIsRequired, defaultValueTooltip, polymorphicSelection, allValid, validatorType, validatorData, true);
104 	}

106 	/**
107 	 * when valueIsRequired is null, it is understood to be optional.  If it is not null, need to tie it in to the listeners on the field - setting
108 	 * an appropriate message if the field is empty.
109 	 */
110 	private static RefexDataTypeNodeDetails buildNodeForType(RefexDynamicDataType dt, RefexDynamicDataBI defaultValue, RefexDynamicDataBI currentValue, 
111 			SimpleStringProperty valueIsRequired, SimpleStringProperty defaultValueAndValidatorTooltip, ReadOnlyObjectProperty<RefexDynamicDataType> polymorphicSelection, 
112 			UpdateableBooleanBinding allValid, ObjectProperty<RefexDynamicValidatorType> validatorType, ObjectProperty<RefexDynamicDataBI> validatorData,
113 			boolean isFirstLevel)
114 	{
115 		if (validatorType != null && validatorData == null)
116 		{
117 			throw new RuntimeException("If a validator type is supplied, you need a validator data reference");
118 		}
119 		
120 		RefexDataTypeNodeDetails returnValue = new RefexDataTypeNodeDetails();;
121 		if (RefexDynamicDataType.BOOLEAN == dt)
122 		{
123 			ChoiceBox<String> cb = new ChoiceBox<>();
124 			cb.getItems().add("No Value");
125 			cb.getItems().add("True");
126 			cb.getItems().add("False");
127 			
128 			if (valueIsRequired != null)
129 			{
130 				cb.getSelectionModel().selectedIndexProperty().addListener((change) ->
131 				{
132 					if (cb.getSelectionModel().getSelectedIndex() == 0)
133 					{
134 						valueIsRequired.set("You must select True or False");
135 					}
136 					else
137 					{
138 						valueIsRequired.set("");
139 					}
140 				});
141 			}
142 			
143 			if (currentValue == null)
144 			{
145 				if (defaultValue != null)
146 				{
147 					if (((RefexDynamicBooleanBI)defaultValue).getDataBoolean())
148 					{
149 						cb.getSelectionModel().select(1);
150 					}
151 					else
152 					{
153 						cb.getSelectionModel().select(2);
154 					}
155 				}
156 				else
157 				{
158 					cb.getSelectionModel().select(0);
159 				}
160 			}
161 			else
162 			{
163 				if (((RefexDynamicBooleanBI)currentValue).getDataBoolean())
164 				{
165 					cb.getSelectionModel().select(1);
166 				}
167 				{
168 					cb.getSelectionModel().select(2);
169 				}
170 			}
171 			if (defaultValue != null && defaultValueAndValidatorTooltip != null)
172 			{
173 				defaultValueAndValidatorTooltip.set("The default value for this field is '" + defaultValue.getDataObject().toString() + "'");
174 			}
175 			if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
176 			{
177 				throw new RuntimeException("It doesn't makse sense to assign a validator to a boolean");
178 			}
179 			returnValue.dataField = cb;
180 			returnValue.nodeForDisplay = cb;
181 		}
182 		else if (RefexDynamicDataType.BYTEARRAY == dt)
183 		{
184 			HBox hbox = new HBox();
185 			hbox.setMaxWidth(Double.MAX_VALUE);
186 			Label choosenFile = new Label("- no data attached -");
187 			if (valueIsRequired != null)
188 			{
189 				valueIsRequired.set("You must select a file to attach");
190 			}
191 			choosenFile.setAlignment(Pos.CENTER_LEFT);
192 			choosenFile.setMaxWidth(Double.MAX_VALUE);
193 			choosenFile.setMaxHeight(Double.MAX_VALUE);
194 			Tooltip tt = new Tooltip("Select a file to attach to the sememe");
195 			Tooltip.install(choosenFile, tt);
196 			Button fileChooser = new Button("Choose File...");
197 			final ByteArrayDataHolder dataHolder = new ByteArrayDataHolder();
198 			
199 			if (currentValue != null)
200 			{
201 				dataHolder.data = ((RefexDynamicByteArray)currentValue).getData();
202 				choosenFile.setText("Currently has " + dataHolder.data.length + " bytes attached");
203 				if (valueIsRequired != null)
204 				{
205 					valueIsRequired.set("");
206 				}
207 			}
208 			else if (defaultValue != null)
209 			{
210 				dataHolder.data = ((RefexDynamicByteArray)defaultValue).getData();
211 				choosenFile.setText("Will attach the default value of " + dataHolder.data.length + " bytes");
212 				if (valueIsRequired != null)
213 				{
214 					valueIsRequired.set("");
215 				}
216 			}
217 			
218 			fileChooser.setOnAction((event) -> 
219 			{
220 				FileChooser fc = new FileChooser();
221 				fc.setTitle("Select a file to attach to the sememe");
222 				File selectedFile = fc.showOpenDialog(fileChooser.getScene().getWindow());
223 				if (selectedFile != null && selectedFile.isFile())
224 				{
225 					
226 					try
227 					{
228 						dataHolder.data = Files.readAllBytes(selectedFile.toPath());
229 						Platform.runLater(() -> 
230 						{
231 							choosenFile.setText(selectedFile.getName());
232 							tt.setText(selectedFile.getAbsolutePath());
233 							if (valueIsRequired != null)
234 							{
235 								valueIsRequired.set("");
236 							}
237 						});
238 					}
239 					catch (Exception e)
240 					{
241 						AppContext.getCommonDialogs().showErrorDialog("Error reading the selected file", e);
242 					}
243 				}
244 				else
245 				{
246 					Platform.runLater(() -> 
247 					{
248 						if (defaultValue != null)
249 						{
250 							dataHolder.data = ((RefexDynamicByteArray)defaultValue).getData();
251 							choosenFile.setText("Will attach the default value of " + dataHolder.data.length + " bytes");
252 							if (valueIsRequired != null)
253 							{
254 								valueIsRequired.set("");
255 							}
256 						}
257 						else
258 						{
259 							choosenFile.setText("- no data attached -");
260 							tt.setText("Select a file to attach that file to the sememe");
261 							if (valueIsRequired != null)
262 							{
263 								valueIsRequired.set("You must select a file to attach");
264 							}
265 						}
266 					});
267 				}
268 			});
269 			
270 			returnValue.dataField = dataHolder;
271 			
272 			hbox.getChildren().add(choosenFile);
273 			hbox.getChildren().add(fileChooser);
274 			HBox.setHgrow(choosenFile, Priority.ALWAYS);
275 			HBox.setHgrow(fileChooser, Priority.NEVER);
276 			returnValue.nodeForDisplay = hbox;
277 			if (defaultValue != null && defaultValueAndValidatorTooltip != null)
278 			{
279 				defaultValueAndValidatorTooltip.set("If no file is selected, the default value of " + ((RefexDynamicByteArray)defaultValue).getData().length +  " bytes will be used");
280 			}
281 			if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
282 			{
283 				throw new RuntimeException("There are currently no supported cases for a validator on a byte array");
284 			}
285 		}
286 		else if (RefexDynamicDataType.DOUBLE == dt || RefexDynamicDataType.FLOAT == dt || RefexDynamicDataType.INTEGER == dt || RefexDynamicDataType.LONG == dt
287 				|| RefexDynamicDataType.STRING == dt || RefexDynamicDataType.UUID == dt)
288 		{
289 			TextField tf = new TextField();
290 			returnValue.dataField = tf;
291 
292 			if (defaultValue != null)
293 			{
294 				tf.setPromptText(defaultValue.getDataObject().toString());
295 			}
296 			SimpleStringProperty valueInvalidReason = new SimpleStringProperty("");
297 			returnValue.boundToAllValid.add(valueInvalidReason);
298 			allValid.addBinding(valueInvalidReason);
299 			Node n = ErrorMarkerUtils.setupErrorMarker(tf, valueInvalidReason);
300 
301 			tf.textProperty().addListener(new ChangeListener<String>()
302 			{
303 				@Override
304 				public void changed(ObservableValue<? extends String> observable, String oldValue, String newValue)
305 				{
306 					if (newValue.length() == 0)
307 					{
308 						valueInvalidReason.setValue("");
309 						if (valueIsRequired != null && defaultValue == null)
310 						{
311 							valueIsRequired.set("You must specify a value for this field");
312 						}
313 					}
314 					else if (RefexDynamicDataType.DOUBLE == dt)
315 					{
316 						try
317 						{
318 							if (valueIsRequired != null)
319 							{
320 								valueIsRequired.set("");
321 							}
322 							RefexDynamicDouble data = new RefexDynamicDouble(Double.parseDouble(tf.getText()));
323 							if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
324 							{
325 								valueInvalidReason.set(validatorType.get().passesValidatorStringReturn(data, validatorData.get(), WBUtility.getViewCoordinate()));
326 							}
327 							else
328 							{
329 								valueInvalidReason.set("");
330 							}
331 						}
332 						catch (Exception e)
333 						{
334 							valueInvalidReason.set("The value (if present) must be a double");
335 						}
336 
337 					}
338 					else if (RefexDynamicDataType.FLOAT == dt)
339 					{
340 						try
341 						{
342 							if (valueIsRequired != null)
343 							{
344 								valueIsRequired.set("");
345 							}
346 							RefexDynamicFloat data = new RefexDynamicFloat(Float.parseFloat(tf.getText()));
347 							if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
348 							{
349 								valueInvalidReason.set(validatorType.get().passesValidatorStringReturn(data, validatorData.get(), WBUtility.getViewCoordinate()));
350 							}
351 							else
352 							{
353 								valueInvalidReason.set("");
354 							}
355 						}
356 						catch (Exception e)
357 						{
358 							valueInvalidReason.set("The value (if present) must be a float");
359 						}
360 					}
361 					else if (RefexDynamicDataType.INTEGER == dt)
362 					{
363 						try
364 						{
365 							if (valueIsRequired != null)
366 							{
367 								valueIsRequired.set("");
368 							}
369 							RefexDynamicInteger data = new RefexDynamicInteger(Integer.parseInt(tf.getText()));
370 							if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
371 							{
372 								valueInvalidReason.set(validatorType.get().passesValidatorStringReturn(data, validatorData.get(), WBUtility.getViewCoordinate()));
373 							}
374 							else
375 							{
376 								valueInvalidReason.set("");
377 							}
378 						}
379 						catch (Exception e)
380 						{
381 							valueInvalidReason.set("The value (if present) must be an integer");
382 						}
383 					}
384 					else if (RefexDynamicDataType.LONG == dt)
385 					{
386 						try
387 						{
388 							if (valueIsRequired != null)
389 							{
390 								valueIsRequired.set("");
391 							}
392 							RefexDynamicLong data = new RefexDynamicLong(Long.parseLong(tf.getText()));
393 							if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
394 							{
395 								valueInvalidReason.set(validatorType.get().passesValidatorStringReturn(data, validatorData.get(), WBUtility.getViewCoordinate()));
396 							}
397 							else
398 							{
399 								valueInvalidReason.set("");
400 							}
401 						}
402 						catch (Exception e)
403 						{
404 							valueInvalidReason.set("The value (if present) must be a long");
405 						}
406 					}
407 					else if (RefexDynamicDataType.STRING == dt)
408 					{
409 						if (valueIsRequired != null)
410 						{
411 							valueIsRequired.set("");
412 						}
413 						if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
414 						{
415 							try
416 							{
417 								valueInvalidReason.set(validatorType.get().passesValidatorStringReturn(new RefexDynamicString(tf.getText()), validatorData.get(), 
418 										WBUtility.getViewCoordinate()));
419 							}
420 							catch (PropertyVetoException e)
421 							{
422 								valueInvalidReason.set(e.getMessage());
423 							}
424 						}
425 						else
426 						{
427 							valueInvalidReason.set("");
428 						}
429 					}
430 					else if (RefexDynamicDataType.UUID == dt)
431 					{
432 						if (valueIsRequired != null)
433 						{
434 							valueIsRequired.set("");
435 						}
436 						if (Utility.isUUID(tf.getText()))
437 						{
438 							if (validatorType != null && validatorType.get() != null && validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
439 							{
440 								try
441 								{
442 									valueInvalidReason.set(validatorType.get().passesValidatorStringReturn(new RefexDynamicUUID(UUID.fromString(tf.getText())), 
443 										validatorData.get(), WBUtility.getViewCoordinate()));
444 								}
445 								catch (PropertyVetoException e)
446 								{
447 									valueInvalidReason.set(e.getMessage());
448 								}
449 							}
450 							else
451 							{
452 								valueInvalidReason.set("");
453 							}
454 						}
455 						else
456 						{
457 							valueInvalidReason.set("The value (if present) must be a properly formatted UUID");
458 						}
459 					}
460 				}
461 			});
462 			
463 			if (validatorType != null)
464 			{
465 				validatorType.addListener((change) -> 
466 				{
467 					//hack way to re-run the validator logic above...
468 					tf.setText(tf.getText() + " ");
469 					tf.setText(tf.getText().substring(0, tf.getText().length() - 1));
470 				});
471 				
472 				validatorData.addListener((change) ->
473 				{
474 					//hack way to re-run the validator logic above...
475 					tf.setText(tf.getText() + " ");
476 					tf.setText(tf.getText().substring(0, tf.getText().length() - 1));
477 				});
478 			}
479 			
480 			if (currentValue != null)
481 			{
482 				tf.setText(currentValue.getDataObject().toString());
483 			}
484 			if (currentValue == null && valueIsRequired != null && defaultValue == null)
485 			{
486 				valueIsRequired.set("You must specify a value for this field");
487 			}
488 			returnValue.nodeForDisplay = n;
489 			setupInfoTooltip(defaultValue, defaultValueAndValidatorTooltip, validatorType, validatorData);
490 		}
491 		else if (RefexDynamicDataType.NID == dt)
492 		{
493 			ConceptNode cn = new ConceptNode(null, false);
494 			returnValue.dataField = cn;
495 			returnValue.boundToAllValid.add(cn.isValid().getReasonWhyInvalid());
496 			allValid.addBinding(cn.isValid().getReasonWhyInvalid());
497 			
498 			if (currentValue != null)
499 			{
500 				//TODO this doesn't work, if the nid isn't a concept nid.  We need a NidNode, rather than a ConceptNode
501 				cn.set(WBUtility.getConceptVersion(((RefexDynamicNidBI)currentValue).getDataNid()));
502 			}
503 			
504 			if (valueIsRequired != null && defaultValue == null)
505 			{
506 				if (currentValue == null)
507 				{
508 					valueIsRequired.set("You must specify a value for this field");
509 				}
510 			}
511 			
512 			cn.getConceptProperty().addListener((change) ->
513 			{
514 				if (cn.getConceptProperty().getValue() == null && valueIsRequired != null && defaultValue == null)
515 				{
516 					valueIsRequired.set("You must specify a value for this field");
517 				}
518 				else
519 				{
520 					if (cn.isValid().get() && valueIsRequired != null)
521 					{
522 						valueIsRequired.setValue("");
523 					}
524 					if (validatorType != null && validatorType.get() != null && cn.isValid().get() && cn.getConceptProperty().get() != null &&
525 							validatorType.get() != RefexDynamicValidatorType.UNKNOWN)
526 					{
527 						try
528 						{
529 							String isInvalid = validatorType.get().passesValidatorStringReturn(new RefexDynamicNid(cn.getConceptProperty().get().getNid()), 
530 									validatorData.get(), WBUtility.getViewCoordinate());
531 							if (isInvalid.length() > 0)
532 							{
533 								cn.isValid().setInvalid(isInvalid);
534 							}
535 						}
536 						catch (PropertyVetoException e)
537 						{
538 							cn.isValid().setInvalid(e.getMessage());
539 						}
540 					}
541 				}
542 			});
543 			
544 			
545 			if (validatorType != null)
546 			{
547 				validatorType.addListener((change) -> 
548 				{
549 					//hack way to re-run the validator logic above...
550 					cn.revalidate();
551 				});
552 				
553 				validatorData.addListener((change) ->
554 				{
555 					//hack way to re-run the validator logic above...
556 					cn.revalidate();
557 				});
558 			}
559 
560 			returnValue.nodeForDisplay = cn.getNode();
561 			setupInfoTooltip(defaultValue, defaultValueAndValidatorTooltip, validatorType, validatorData);
562 		}
563 		else if (RefexDynamicDataType.POLYMORPHIC == dt)
564 		{
565 			//a slick little bit of recursion... but a bit tricky to keep the validators aligned properly...
566 			HBox hBox = new HBox();
567 			hBox.setMaxWidth(Double.MAX_VALUE);
568 			NestedPolymorphicData nestedData = new NestedPolymorphicData();
569 			
570 			nestedData.nestedNode = buildNodeForType(polymorphicSelection.get(), null, currentValue, valueIsRequired, defaultValueAndValidatorTooltip, null, allValid, 
571 					validatorType, validatorData, false);
572 			
573 			hBox.getChildren().add(nestedData.nestedNode.nodeForDisplay);
574 			HBox.setHgrow(hBox.getChildren().get(0), Priority.ALWAYS);
575 			
576 			nestedData.dataType = polymorphicSelection.get();
577 			returnValue.dataField = nestedData;
578 			if (nestedData.nestedNode.boundToAllValid.size() == 1 )
579 			{
580 				returnValue.boundToAllValid.add(nestedData.nestedNode.boundToAllValid.get(0));
581 				allValid.invalidate(); //the new binding will already be made to allValid_ during the recursion - but it may have computed without ths full list...
582 			}
583 			
584 			polymorphicSelection.addListener((change) ->
585 			{
586 				hBox.getChildren().remove(0);
587 				if (nestedData.nestedNode.boundToAllValid.size() == 1)
588 				{
589 					returnValue.boundToAllValid.remove(nestedData.nestedNode.boundToAllValid.get(0));
590 					allValid.removeBinding(nestedData.nestedNode.boundToAllValid.get(0));
591 				}
592 				nestedData.nestedNode = buildNodeForType(polymorphicSelection.get(), null, currentValue, valueIsRequired, defaultValueAndValidatorTooltip, null, allValid, 
593 						validatorType, validatorData, false);
594 				hBox.getChildren().add(nestedData.nestedNode.nodeForDisplay);
595 				HBox.setHgrow(hBox.getChildren().get(0), Priority.ALWAYS);
596 				nestedData.dataType = polymorphicSelection.get();
597 				if (nestedData.nestedNode.boundToAllValid.size() == 1)
598 				{
599 					returnValue.boundToAllValid.add(nestedData.nestedNode.boundToAllValid.get(0));
600 					allValid.invalidate(); //the new binding will already be made to allValid_ during the recursion - but it may have computed without ths full list...
601 				}
602 			});
603 			returnValue.nodeForDisplay = hBox;
604 		}
605 		else
606 		{
607 			throw new RuntimeException("Unexpected datatype " + dt);
608 		}
609 		if (isFirstLevel && valueIsRequired != null)
610 		{
611 			returnValue.boundToAllValid.add(valueIsRequired);
612 			allValid.addBinding(valueIsRequired);
613 		}
614 		return returnValue;

617 	public static RefexDynamicDataBI getDataForType(Object data, RefexDynamicColumnInfo ci) throws PropertyVetoException
618 	{
619 		//TODO the way this is currently set up - if there is a default value, and the value is not required - it is impossible to leave the row blank.
620 		//not sure if that is a necessary use case, or not.
621 		if (RefexDynamicDataType.BOOLEAN == ci.getColumnDataType())
622 		{
623 			@SuppressWarnings("unchecked")
624 			ChoiceBox<String> cb = (ChoiceBox<String>) data;
625 
626 			Boolean value = null;
627 			if (cb.getSelectionModel().getSelectedItem().equals("True"))
628 			{
629 				value = true;
630 			}
631 			else if (cb.getSelectionModel().getSelectedItem().equals("False"))
632 			{
633 				value = false;
634 			}
635 			else if (ci.getDefaultColumnValue() != null)
636 			{
637 				value =  ((RefexDynamicBooleanBI) ci.getDefaultColumnValue()).getDataBoolean();
638 			}
639 			return (value == null ? null : new RefexDynamicBoolean(value));
640 		}
641 		else if (RefexDynamicDataType.BYTEARRAY == ci.getColumnDataType())
642 		{
643 			if (data == null && ci.getDefaultColumnValue() == null)
644 			{
645 				return null;
646 			}
647 			ByteArrayDataHolder holder = (ByteArrayDataHolder) data;
648 			if (holder == null || holder.data == null)
649 			{
650 				return (RefexDynamicByteArray)ci.getDefaultColumnValue();
651 			}
652 			return new RefexDynamicByteArray(holder.data);
653 		}
654 		else if (RefexDynamicDataType.DOUBLE == ci.getColumnDataType() || RefexDynamicDataType.FLOAT == ci.getColumnDataType()
655 				|| RefexDynamicDataType.INTEGER == ci.getColumnDataType() || RefexDynamicDataType.LONG == ci.getColumnDataType()
656 				|| RefexDynamicDataType.STRING == ci.getColumnDataType() || RefexDynamicDataType.UUID == ci.getColumnDataType())
657 		{
658 			TextField tf = (TextField) data;
659 			String text = tf.getText();
660 			if (text.length() == 0 && ci.getDefaultColumnValue() == null)
661 			{
662 				return null;
663 			}
664 			if (RefexDynamicDataType.DOUBLE == ci.getColumnDataType())
665 			{
666 				return (text.length() > 0 ? new RefexDynamicDouble(Double.parseDouble(text)) : (RefexDynamicDoubleBI)ci.getDefaultColumnValue());
667 			}
668 			else if (RefexDynamicDataType.FLOAT == ci.getColumnDataType())
669 			{
670 				return (text.length() > 0 ? new RefexDynamicFloat(Float.parseFloat(text)) : (RefexDynamicFloatBI)ci.getDefaultColumnValue());
671 			}
672 			else if (RefexDynamicDataType.INTEGER == ci.getColumnDataType())
673 			{
674 				return (text.length() > 0 ? new RefexDynamicInteger(Integer.parseInt(text)) : (RefexDynamicIntegerBI)ci.getDefaultColumnValue());
675 			}
676 			else if (RefexDynamicDataType.LONG == ci.getColumnDataType())
677 			{
678 				return (text.length() > 0 ? new RefexDynamicLong(Long.parseLong(text)) : (RefexDynamicLongBI)ci.getDefaultColumnValue());
679 			}
680 			else if (RefexDynamicDataType.STRING == ci.getColumnDataType())
681 			{
682 				return (text.length() > 0 ? new RefexDynamicString(text) : (RefexDynamicStringBI)ci.getDefaultColumnValue());
683 			}
684 			else if (RefexDynamicDataType.UUID == ci.getColumnDataType())
685 			{
686 				return (text.length() > 0 ? new RefexDynamicUUID(UUID.fromString(text)) : (RefexDynamicUUIDBI)ci.getDefaultColumnValue());
687 			}
688 			else
689 			{
690 				throw new RuntimeException("oops");
691 			}
692 		}
693 		else if (RefexDynamicDataType.NID == ci.getColumnDataType())
694 		{
695 			ConceptNode cn = (ConceptNode)data;
696 			if (cn.getConcept() == null)
697 			{
698 				return (RefexDynamicNid)ci.getDefaultColumnValue();
699 			}
700 			return new RefexDynamicNid(cn.getConcept().getNid());
701 		}
702 		else if (RefexDynamicDataType.POLYMORPHIC == ci.getColumnDataType())
703 		{
704 			NestedPolymorphicData nestedData = (NestedPolymorphicData)data;
705 			// only need the data type field... (default value isn't allowed for polymorphic)
706 			RefexDynamicColumnInfo nestedCI = new RefexDynamicColumnInfo();
707 			nestedCI.setColumnDataType(nestedData.dataType);
708 			return getDataForType(nestedData.nestedNode.dataField, nestedCI);
709 		}
710 		else
711 		{
712 			throw new RuntimeException("Unexpected datatype " + ci.getColumnDataType());
713 		}
714 	}
