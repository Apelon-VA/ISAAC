61  	public static RefexValidatorTypeNodeDetails buildNodeForType(RefexDynamicValidatorType dt, RefexDynamicDataBI currentValue, 
62  			ObjectProperty<RefexDynamicDataType> refexDataType, UpdateableBooleanBinding allValid)
63  	{
64  		RefexValidatorTypeNodeDetails returnValue = new RefexValidatorTypeNodeDetails();
65  		if (RefexDynamicValidatorType.GREATER_THAN == dt || RefexDynamicValidatorType.GREATER_THAN_OR_EQUAL == dt || RefexDynamicValidatorType.INTERVAL == dt
66  				|| RefexDynamicValidatorType.LESS_THAN == dt || RefexDynamicValidatorType.LESS_THAN_OR_EQUAL == dt || RefexDynamicValidatorType.REGEXP == dt)
67  		{
68  			TextField tf = new TextField();
69  
70  			SimpleStringProperty valueInvalidReason = new SimpleStringProperty("");
71  			returnValue.boundToAllValid.add(valueInvalidReason);
72  			allValid.addBinding(valueInvalidReason);
73  			Node n = ErrorMarkerUtils.setupErrorMarker(tf, valueInvalidReason);
74  
75  			tf.textProperty().addListener(new ChangeListener<String>()
76  			{
77  				@Override
78  				public void changed(ObservableValue<? extends String> observable, String oldValue, String newValue)
79  				{
80  					if (newValue.length() == 0)
81  					{
82  						valueInvalidReason.setValue("You must specify a value for this field");
83  					}
84  					else if (RefexDynamicValidatorType.GREATER_THAN == dt || RefexDynamicValidatorType.GREATER_THAN_OR_EQUAL == dt
85  							|| RefexDynamicValidatorType.LESS_THAN == dt || RefexDynamicValidatorType.LESS_THAN_OR_EQUAL == dt)
86  					{
87  						try
88  						{
89  							returnValue.validatorData.set(NumberUtilities.wrapIntoRefexHolder(NumberUtilities.parseNumber(tf.getText().trim())));
90  							valueInvalidReason.set("");
91  						}
92  						catch (Exception e)
93  						{
94  							valueInvalidReason.set(e.getMessage());
95  							returnValue.validatorData.set(null);
96  						}
97  					}
98  					else if (RefexDynamicValidatorType.INTERVAL == dt)
99  					{
100 						try
101 						{
102 							//Don't actually care if it passes the validator - just want to know if the validator can parse it.
103 							RefexDynamicString s = new RefexDynamicString(tf.getText());
104 							RefexDynamicValidatorType.INTERVAL.passesValidator(new RefexDynamicInteger(0), s, null);
105 							returnValue.validatorData.set(s);
106 							valueInvalidReason.set("");
107 						}
108 						catch (Exception e)
109 						{
110 							valueInvalidReason.set(e.getMessage());
111 							returnValue.validatorData.set(null);
112 						}
113 					}
114 					else if (RefexDynamicValidatorType.REGEXP == dt)
115 					{
116 						try
117 						{
118 							//Don't actually care if it passes the validator - just want to know if the validator can parse it.
119 							RefexDynamicString s = new RefexDynamicString(tf.getText());
120 							RefexDynamicValidatorType.REGEXP.passesValidator(new RefexDynamicString("a"), s, null);
121 							returnValue.validatorData.set(s);
122 							valueInvalidReason.set("");
123 						}
124 						catch (Exception e)
125 						{
126 							valueInvalidReason.set(e.getMessage());
127 							returnValue.validatorData.set(null);
128 						}
129 					}
130 					else
131 					{
132 						throw new RuntimeException("oops");
133 					}
134 				}
135 			});
136 
137 			if (currentValue != null)
138 			{
139 				tf.setText(currentValue.getDataObject().toString());
140 			}
141 			if (currentValue == null)
142 			{
143 				valueInvalidReason.setValue("You must specify a value for this field");
144 			}
145 			returnValue.nodeForDisplay = n;
146 		}
147 		else if (RefexDynamicValidatorType.IS_CHILD_OF == dt || RefexDynamicValidatorType.IS_KIND_OF == dt)
148 		{
149 			ConceptNode cn = new ConceptNode(null, true);
150 			returnValue.boundToAllValid.add(cn.isValid().getReasonWhyInvalid());
151 			allValid.addBinding(cn.isValid().getReasonWhyInvalid());
152 
153 			if (currentValue != null)
154 			{
155 				cn.set(WBUtility.getConceptVersion(((RefexDynamicNidBI) currentValue).getDataNid()));
156 			}
157 
158 			returnValue.nodeForDisplay = cn.getNode();
159 			
160 			cn.getConceptProperty().addListener((change) ->
161 			{
162 				if (cn.getConceptProperty().get() == null)
163 				{
164 					returnValue.validatorData.set(null);
165 				}
166 				else
167 				{
168 					try
169 					{
170 						returnValue.validatorData.set(new RefexDynamicNid(cn.getConceptProperty().getValue().getNid()));
171 					}
172 					catch (Exception e)
173 					{
174 						returnValue.validatorData.set(null);
175 						cn.isValid().setInvalid("Unexpected error parsing concept");
176 					}
177 				}
178 			});
179 		}
180 		else if (RefexDynamicValidatorType.EXTERNAL == dt)
181 		{
182 			ChoiceBox<RefexDroolsValidatorImplInfo> cb = new ChoiceBox<>();
183 			cb.setMaxWidth(Double.MAX_VALUE);
184 			cb.setConverter(new StringConverter<RefexDroolsValidatorImplInfo>()
185 			{
186 				@Override
187 				public String toString(RefexDroolsValidatorImplInfo object)
188 				{
189 					return object.getDisplayName();
190 				}
191 
192 				@Override
193 				public RefexDroolsValidatorImplInfo fromString(String string)
194 				{
195 					//not needed
196 					return null;
197 				}
198 			});
199 			
200 			SimpleStringProperty valueInvalidReason = new SimpleStringProperty("");
201 			returnValue.boundToAllValid.add(valueInvalidReason);
202 			allValid.addBinding(valueInvalidReason);
203 			
204 			Node n = ErrorMarkerUtils.setupErrorMarker(cb, valueInvalidReason);
205 			
206 			ArrayList<String> longDescriptions = new ArrayList<>();
207 			
208 			try
209 			{
210 				DroolsExecutorsManager dem = AppContext.getService(DroolsExecutorsManager.class);
211 				
212 				for (RefexDroolsValidatorImplInfo rdvii : RefexDroolsValidatorImplInfo.values())
213 				{
214 					cb.getItems().add(rdvii);
215 					StringBuilder temp = new StringBuilder();
216 					DroolsExecutor de = dem.getDroolsExecutor(rdvii.getDroolsPackageName());
217 					if (de == null)
218 					{
219 						logger.error("Unexpected - couldn't locate a DroolsExecutor for " + rdvii.getDroolsPackageName());
220 					}
221 					else
222 					{
223 						for (String ruleNames : de.getAllRuleNames())
224 						{
225 							temp.append(ruleNames + "\r");
226 						}
227 					}
228 					if (temp.length() > 1)
229 					{
230 						temp.setLength(temp.length() - 1);
231 					}
232 					longDescriptions.add(temp.toString());
233 				}
234 				
235 				Tooltip tt = new Tooltip("");
236 				tt.setWrapText(true);
237 				tt.setMaxWidth(400);
238 				cb.setTooltip(tt);
239 				
240 				refexDataType.addListener((change) ->
241 				{
242 					boolean matched = false;
243 					for (RefexDynamicDataType rddt : cb.getValue().getApplicableDataTypes())
244 					{
245 						if (rddt == refexDataType.get())
246 						{
247 							matched = true;
248 							break;
249 						}
250 					}
251 					if (!matched)
252 					{
253 						valueInvalidReason.set("The selected validator is not appliable to the selected data type");
254 					}
255 					else
256 					{
257 						valueInvalidReason.set("");
258 					}
259 				});
260 				
261 				cb.valueProperty().addListener((change) ->
262 				{
263 					try
264 					{
265 						boolean matched = false;
266 						for (RefexDynamicDataType rddt : cb.getValue().getApplicableDataTypes())
267 						{
268 							if (rddt == refexDataType.get())
269 							{
270 								matched = true;
271 								break;
272 							}
273 						}
274 						if (!matched)
275 						{
276 							valueInvalidReason.set("The selected validator is not appliable to the selected data type");
277 						}
278 						else
279 						{
280 							valueInvalidReason.set("");
281 						}
282 						returnValue.validatorData.set(RefexDroolsValidator.createValidatorDefinitionData(cb.getValue()));
283 						tt.setText(longDescriptions.get(cb.getSelectionModel().getSelectedIndex()));
284 					}
285 					catch (Exception e)
286 					{
287 						logger.error("Unexpected!", e);
288 					}
289 				});
290 				
291 				RefexDroolsValidatorImplInfo rdvii = RefexDroolsValidator.readFromData(currentValue);
292 				
293 				if (rdvii != null)
294 				{
295 					cb.getSelectionModel().select(rdvii);
296 				}
297 				else
298 				{
299 					cb.getSelectionModel().select(0);
300 				}
301 				returnValue.nodeForDisplay = n;
302 			}
303 			catch (Exception e)
304 			{
305 				throw new RuntimeException("Unexpected", e);
306 			}
307 		}
308 		else if (RefexDynamicValidatorType.COMPONENT_TYPE == dt)
309 		{
310 			ChoiceBox<ComponentType> cb = new ChoiceBox<>();
311 			cb.setMaxWidth(Double.MAX_VALUE);
312 			
313 			for (ComponentType ct : ComponentType.values())
314 			{
315 				if (ct != ComponentType.UNKNOWN && ct != ComponentType.CONCEPT_ATTRIBUTES)
316 				{
317 					cb.getItems().add(ct);
318 				}
319 			}
320 			cb.setConverter(new StringConverter<ComponentType>()
321 			{
322 				@Override
323 				public String toString(ComponentType object)
324 				{
325 					return "Must be a " + object.toString();
326 				}
327 
328 				@Override
329 				public ComponentType fromString(String string)
330 				{
331 					//not needed
332 					return ComponentType.UNKNOWN;
333 				}
334 			});
335 			
336 			cb.valueProperty().addListener((change) ->
337 			{
338 				try
339 				{
340 					returnValue.validatorData.set(new RefexDynamicString(cb.getValue().name()));
341 				}
342 				catch (Exception e)
343 				{
344 					logger.error("Unexpected!", e);
345 				}
346 			});
347 			returnValue.nodeForDisplay = cb;
348 			cb.getSelectionModel().select(0);
349 		}
350 		else
351 		{
352 			throw new RuntimeException("Unexpected datatype " + dt);
353 		}
354 		return returnValue;
355 	}
