41  /**
42   * {@link DynamicRefexDataColumnListCell}
43   *
44   * Display code for a data column of a Dynamic Refex, when shown within a list view (one cell per refex column)
45   *
46   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
47   */
48  public class DynamicRefexDataColumnListCell extends ListCell<RefexDynamicColumnInfo>
49  {
50  	/**
51  	 * @see javafx.scene.control.Cell#updateItem(java.lang.Object, boolean)
52  	 */
53  	@Override
54  	protected void updateItem(RefexDynamicColumnInfo item, boolean empty)
55  	{
56  		super.updateItem(item, empty);
57  		if (item != null)
58  		{
59  			setText("");
60  
61  			GridPane gp = new GridPane();
62  			gp.setHgap(0.0);
63  			gp.setVgap(0.0);
64  			gp.setPadding(new Insets(5, 5, 5, 5));
65  			gp.setMinWidth(250);
66  
67  			ColumnConstraints constraint1 = new ColumnConstraints();
68  			constraint1.setFillWidth(false);
69  			constraint1.setHgrow(Priority.NEVER);
70  			constraint1.setMinWidth(160);
71  			constraint1.setMaxWidth(160);
72  			gp.getColumnConstraints().add(constraint1);
73  
74  			ColumnConstraints constraint2 = new ColumnConstraints();
75  			constraint2.setFillWidth(true);
76  			constraint2.setHgrow(Priority.ALWAYS);
77  			gp.getColumnConstraints().add(constraint2);
78  
79  			int row = 0;
80  
81  			gp.add(wrapAndStyle(makeBoldLabel("Attribute Name"), row), 0, row);
82  			Label name = new Label(item.getColumnName());
83  			name.setWrapText(true);
84  			name.maxWidthProperty().bind(this.widthProperty().subtract(210));
85  			gp.add(wrapAndStyle(name, row), 1, row++);
86  
87  			gp.add(wrapAndStyle(makeBoldLabel("Attribute Description"), row), 0, row);
88  			Label description = new Label(item.getColumnDescription());
89  			description.setWrapText(true);
90  			description.maxWidthProperty().bind(this.widthProperty().subtract(210));
91  
92  			gp.add(wrapAndStyle(description, row), 1, row++);
93  
94  			gp.add(wrapAndStyle(makeBoldLabel("Attribute Order"), row), 0, row);
95  			gp.add(wrapAndStyle(new Label(item.getColumnOrder() + 1 + ""), row), 1, row++);
96  
97  			gp.add(wrapAndStyle(makeBoldLabel("Data Type"), row), 0, row);
98  			gp.add(wrapAndStyle(new Label(item.getColumnDataType().getDisplayName()), row), 1, row++);
99  
100 			gp.add(wrapAndStyle(makeBoldLabel("Attribute Required"), row), 0, row);
101 			gp.add(wrapAndStyle(new Label(item.isColumnRequired() + ""), row), 1, row++);
102 
103 			gp.add(wrapAndStyle(makeBoldLabel("Default Value"), row), 0, row);
104 			String temp = "";
105 			if (item.getDefaultColumnValue() != null)
106 			{
107 				if (item.getColumnDataType() == RefexDynamicDataType.BYTEARRAY)
108 				{
109 					temp = "Byte array of size " + ((RefexDynamicByteArray) item.getDefaultColumnValue()).getDataByteArray().length;
110 				}
111 				else if (item.getColumnDataType() == RefexDynamicDataType.NID)
112 				{
113 					temp = WBUtility.getDescriptionIfConceptExists(((RefexDynamicNid)item.getDefaultColumnValue()).getDataNid());
114 					if (temp == null)
115 					{
116 						temp = "NID: " + item.getDefaultColumnValue().getDataObject().toString();
117 					}
118 				}
119 				else if (item.getColumnDataType() == RefexDynamicDataType.UUID)
120 				{
121 					temp = WBUtility.getDescriptionIfConceptExists(((RefexDynamicUUID)item.getDefaultColumnValue()).getDataUUID());
122 					if (temp == null)
123 					{
124 						temp = "UUID: " + item.getDefaultColumnValue().getDataObject().toString();
125 					}
126 				}
127 				else
128 				{
129 					temp = item.getDefaultColumnValue().getDataObject().toString();
130 				}
131 			}
132 			Label defaultValue = new Label(temp);
133 			defaultValue.setWrapText(true);
134 			defaultValue.maxWidthProperty().bind(this.widthProperty().subtract(210));
135 			gp.add(wrapAndStyle(defaultValue, row), 1, row++);
136 
137 			gp.add(wrapAndStyle(makeBoldLabel("Validator"), row), 0, row);
138 			gp.add(wrapAndStyle(new Label(item.getValidator() == null ? "" : 
139 				(item.getValidator() == RefexDynamicValidatorType.EXTERNAL ? "Drools" : item.getValidator().getDisplayName())),
140 				row), 1, row++);
141 
142 			if (item.getValidator() != null)
143 			{
144 				gp.add(wrapAndStyle(makeBoldLabel("Validator Data"), row), 0, row);
145 				String validiatorData = "";
146 				if (item.getValidatorData().getRefexDataType() == RefexDynamicDataType.BYTEARRAY)
147 				{
148 					validiatorData = "Byte array of size " + ((RefexDynamicByteArray) item.getValidatorData()).getDataByteArray().length;
149 				}
150 				else if (item.getValidatorData().getRefexDataType() == RefexDynamicDataType.NID)
151 				{
152 					validiatorData = WBUtility.getDescriptionIfConceptExists(((RefexDynamicNid)item.getValidatorData()).getDataNid());
153 					if (validiatorData == null)
154 					{
155 						validiatorData = "NID: " + item.getValidatorData().getDataObject().toString();
156 					}
157 				}
158 				else if (item.getValidatorData().getRefexDataType() == RefexDynamicDataType.UUID)
159 				{
160 					validiatorData = WBUtility.getDescriptionIfConceptExists(((RefexDynamicUUID)item.getValidatorData()).getDataUUID());
161 					if (validiatorData == null)
162 					{
163 						validiatorData = "UUID: " + item.getValidatorData().getDataObject().toString();
164 					}
165 				}
166 				else if (item.getValidator() == RefexDynamicValidatorType.EXTERNAL)
167 				{
168 					RefexDroolsValidatorImplInfo rdvi = RefexDroolsValidator.readFromData(item.getValidatorData());
169 					if (rdvi == null)
170 					{
171 						//this should be impossible....
172 						validiatorData = "!ERROR!";
173 					}
174 					else
175 					{
176 						validiatorData = rdvi.getDisplayName();
177 					}
178 				}
179 						
180 				else
181 				{
182 					validiatorData = item.getValidatorData().getDataObject().toString();
183 				}
184 				Label valData = new Label(validiatorData);
185 				valData.setWrapText(true);
186 				valData.maxWidthProperty().bind(this.widthProperty().subtract(210));
187 				gp.add(wrapAndStyle(valData, row), 1, row++);
188 			}
189 
190 			setGraphic(gp);
191 			this.setStyle("-fx-border-width:  0 0 2 0; -fx-border-color: grey; ");
192 
193 		}
194 		else
195 		{
196 			setText("");
197 			setGraphic(null);
198 			this.setStyle("");
199 		}
200 	}

202 	private Node wrapAndStyle(Region node, int rowNumber)
203 	{
204 		Pane p = new Pane(node);
205 		node.setPadding(new Insets(5.0));
206 		GridPane.setFillWidth(p, true);
207 		p.minHeightProperty().bind(node.heightProperty());
208 		//Hack - wrapped labels don't seem to fire their height property changes at the right time - leaving the surrounding Pane node too small.
209 		//this seems to help...
210 		Platform.runLater(() -> p.autosize());
211 		p.getStyleClass().add(((rowNumber % 2 == 0) ? "evenGridRow" : "oddGridRow"));
212 		return p;
213 	}
