52  public class SummaryController implements PanelControllersI {
53  	@FXML private Label actualRefexName;
54  	@FXML private Label actualRefexDescription;
55  	@FXML private Label actualParentConcept;
56  	@FXML private Label actualRefexType;
57  	@FXML private Label actualComponentTypeRestriction;
58  	@FXML private BorderPane summaryPane;
59  	@FXML private ListView<RefexDynamicColumnInfo> detailsListView;
60  	@FXML private Button cancelButton;
61  	@FXML private Button startOverButton;
62  	@FXML private Button commitButton;
63  	@FXML private Button backButton;
64  
65  	ScreensController processController_;
66  	Region sceneParent_;
67  
68  	private final Logger logger = LoggerFactory.getLogger(SummaryController.class);
69  
70  	@Override
71  	public void initialize() {
72  		assert actualRefexDescription != null : "fx:id=\"actualRefexDescription\" was not injected: check your FXML file 'summary.fxml'.";
73  		assert actualRefexName != null : "fx:id=\"actualRefexName\" was not injected: check your FXML file 'summary.fxml'.";
74  		assert cancelButton != null : "fx:id=\"cancelButton\" was not injected: check your FXML file 'summary.fxml'.";
75  		assert startOverButton != null : "fx:id=\"startOverButton\" was not injected: check your FXML file 'summary.fxml'.";
76  		assert commitButton != null : "fx:id=\"commitButton\" was not injected: check your FXML file 'summary.fxml'.";
77  		assert backButton != null : "fx:id=\"commitButton\" was not injected: check your FXML file 'summary.fxml'.";
78  		assert actualRefexType != null : "fx:id=\"actualRefexType\" was not injected: check your FXML file 'summary.fxml'.";
79  		assert summaryPane != null : "fx:id=\"summaryPane\" was not injected: check your FXML file 'summary.fxml'.";
80  		assert actualParentConcept != null : "fx:id=\"actualParentConcept\" was not injected: check your FXML file 'summary.fxml'.";
81  		assert detailsListView != null : "fx:id=\"detailsListView\" was not injected: check your FXML file 'summary.fxml'.";
82  
83  		cancelButton.setOnAction(e -> ((Stage)summaryPane.getScene().getWindow()).close());
84  	
85  		commitButton.setOnAction(e -> 
86  		{
87  			try
88  			{
89  				AppContext.getRuntimeGlobals().disableAllCommitListeners();
90  				storeValues();
91  			} catch (Exception e1) {
92  				logger.error("Coudn't Disable WF Init & Commit New Dynamic Sememe Concept", e1);
93  			}
94  			finally
95  			{
96  			    AppContext.getRuntimeGlobals().enableAllCommitListeners();
97  			}
98  
99  			((Stage)summaryPane.getScene().getWindow()).close();
100 		});
101 
102 		startOverButton.setOnAction(e -> processController_.showFirstScreen());
103 		backButton.setOnAction(e -> processController_.showPreviousScreen());
104 		
105 		detailsListView.setCellFactory(new Callback<ListView<RefexDynamicColumnInfo>, ListCell<RefexDynamicColumnInfo>>()
106 		{
107 			@Override
108 			public ListCell<RefexDynamicColumnInfo> call(ListView<RefexDynamicColumnInfo> param)
109 			{
110 				return new DynamicRefexDataColumnListCell();
111 			}
112 		});
113 	}
114 
115 	private void setupRefexContent(RefexData refexData) {
116 		actualRefexName.setText(refexData.getRefexName());
117 		actualRefexDescription.setText(refexData.getRefexDescription());
118 		actualParentConcept.setText(OTFUtility.getDescription(refexData.getParentConcept()));
119 		actualComponentTypeRestriction.setText(refexData.getComponentRestrictionType() == ComponentType.UNKNOWN ? "No Restriction" 
120 				: refexData.getComponentRestrictionType().toString());
121 		if (refexData.isAnnotatedStyle()) {
122 			actualRefexType.setText("Annotated");
123 		} else {
124 			actualRefexType.setText("Refset");
125 		}
126 		
127 		detailsListView.getItems().clear();
128 		detailsListView.getItems().addAll(refexData.getColumnInfo());
129 		detailsListView.scrollTo(0);
130 	}
131 
132 	public void storeValues() {
133 		try {
134 			RefexData refexData = processController_.getWizardData();
135 			RefexDynamicUsageDescriptionBuilder.createNewRefexDynamicUsageDescriptionConcept(refexData.getRefexName(),
136 					refexData.getRefexName(), refexData.getRefexDescription(), refexData.getColumnInfo().toArray(new RefexDynamicColumnInfo[0]), 
137 					refexData.getParentConcept().getPrimordialUuid(), 
138 					refexData.isAnnotatedStyle(),
139 					refexData.getComponentRestrictionType());
140 		} catch (IOException | ContradictionException | InvalidCAB | PropertyVetoException e) {
141 			logger.error("Unable to create and/or commit refset concept and metadata", e);
142 			AppContext.getCommonDialogs().showErrorDialog("Error Creating Sememe", "Unexpected error creating the Sememe", e.getMessage(), summaryPane.getScene().getWindow());
143 		}
144 	}
145 	
146 	public void updateValues(RefexData refexData)
147 	{
148 		setupRefexContent(refexData);
149 	}

154 	@Override
155 	public void finishInit(ScreensController screenController, Region parent)

157 		processController_ = screenController;

159 	}

168 	}
