448 		{

454 		}
455 		else
456 		{
457 			try
458 			{
459 				assemblageInfo_ = RefexDynamicUsageDescriptionBuilder.readRefexDynamicUsageDescriptionConcept(createRefexFocus_.getAssemblyNid());
460 				gp_.add(unselectableComponentLabel_, 1, 1);
461 				unselectableComponentLabel_.setText(WBUtility.getDescription(createRefexFocus_.getAssemblyNid()));
462 				if (assemblageInfo_.getReferencedComponentTypeRestriction() != null 
463 						&& ComponentType.CONCEPT != assemblageInfo_.getReferencedComponentTypeRestriction() 
464 						&& ComponentType.CONCEPT_ATTRIBUTES != assemblageInfo_.getReferencedComponentTypeRestriction())
465 				{
466 					conceptNodeIsConceptType_ = false;
467 					gp_.add(selectableComponentNode_, 1, 0);
468 					selectableComponent_.setPromptText("UUID or NID of a " + assemblageInfo_.getReferencedComponentTypeRestriction().toString());
469 					selectableComponent_.setTooltip(new Tooltip("UUID or NID of a " + assemblageInfo_.getReferencedComponentTypeRestriction().toString()));
470 					selectableComponentNodeValid_.invalidate();
471 				}
472 				else
473 				{
474 					conceptNodeIsConceptType_ = true;
475 					gp_.add(selectableConcept_.getNode(), 1, 0);
476 				}
477 				refexDropDownOptions.clear();
478 				refexDropDownOptions.addAll(AppContext.getService(CommonlyUsedConcepts.class).getObservableConcepts());
479 				assemblageIsValid_.set(true);
480 				buildDataFields(true, null);
481 			}
482 			catch (Exception e)
483 			{
484 				logger_.error("Unexpected", e);
485 				AppContext.getCommonDialogs().showErrorDialog("Unexpected Error reading Assembly concept", e);
486 			}
487 		}

490 	private void buildDataFields(boolean assemblageValid, RefexDynamicDataBI[] currentValues)
491 	{
492 		if (assemblageValid)
493 		{
494 			for (ReadOnlyStringProperty ssp : currentDataFieldWarnings_)
495 			{
496 				allValid_.removeBinding(ssp);
497 			}
498 			currentDataFieldWarnings_.clear();
499 			for (RefexDataTypeNodeDetails nd : currentDataFields_)
500 			{
501 				nd.cleanupListener();
502 			}
503 			currentDataFields_.clear();
504 
505 			GridPane gp = new GridPane();
506 			gp.setHgap(10.0);
507 			gp.setVgap(10.0);
508 			gp.setStyle("-fx-padding: 5;");
509 			int row = 0;
510 			boolean extraInfoColumnIsRequired = false;
511 			for (RefexDynamicColumnInfo ci : assemblageInfo_.getColumnInfo())
512 			{
513 				SimpleStringProperty valueIsRequired = (ci.isColumnRequired() ? new SimpleStringProperty("") : null);
514 				SimpleStringProperty defaultValueTooltip = ((ci.getDefaultColumnValue() == null && ci.getValidator() == null) ? null : new SimpleStringProperty());
515 				ComboBox<RefexDynamicDataType> polymorphicType = null;
516 				
517 				Label l = new Label(ci.getColumnName());
518 				l.getStyleClass().add("boldLabel");
519 				l.setMinWidth(FxUtils.calculateNecessaryWidthOfBoldLabel(l));
520 				Tooltip.install(l, new Tooltip(ci.getColumnDescription()));
521 				int col = 0;
522 				gp.add(l, col++, row);
523 				
524 				if (ci.getColumnDataType() == RefexDynamicDataType.POLYMORPHIC)
525 				{
526 					polymorphicType = new ComboBox<>();
527 					polymorphicType.setEditable(false);
528 					polymorphicType.setConverter(new StringConverter<RefexDynamicDataType>()
529 					{
530 						
531 						@Override
532 						public String toString(RefexDynamicDataType object)
533 						{
534 							return object.getDisplayName();
535 						}
536 						
537 						@Override
538 						public RefexDynamicDataType fromString(String string)
539 						{
540 							throw new RuntimeException("unecessary");
541 						}
542 					});
543 					for (RefexDynamicDataType type : RefexDynamicDataType.values())
544 					{
545 						if (type == RefexDynamicDataType.POLYMORPHIC || type == RefexDynamicDataType.UNKNOWN)
546 						{
