54  public class ComponentDataCell extends TreeTableCell<RefexDynamicGUI, RefexDynamicGUI>
55  {
56  	private static Logger logger_ = LoggerFactory.getLogger(ComponentDataCell.class);
57  	
58  	private boolean isAssemblage_ = false;
59  	private ToIntFunction<RefexDynamicVersionBI<?>> nidFetcher_;
60  	
61  	protected ComponentDataCell(boolean isAssemblage, ToIntFunction<RefexDynamicVersionBI<?>> nidFetcher)
62  	{
63  		nidFetcher_ = nidFetcher;
64  		isAssemblage_ = isAssemblage;
65  	}
66  	
67  	protected ComponentDataCell(ToIntFunction<RefexDynamicVersionBI<?>> nidFetcher)
68  	{
69  		nidFetcher_ = nidFetcher;
70  		isAssemblage_ = false;
71  	}

73  	/**
74  	 * @see javafx.scene.control.Cell#updateItem(java.lang.Object, boolean)
75  	 */
76  	@Override
77  	protected void updateItem(RefexDynamicGUI item, boolean empty)
78  	{
79  		super.updateItem(item, empty);
80  		
81  		if (empty || item == null)
82  		{
83  			setText("");
84  			setGraphic(null);
85  		}
86  		else if (item != null)
87  		{
88  			conceptLookup(item);
89  		}
90  	}

92  	private void conceptLookup(RefexDynamicGUI item)
93  	{
94  		ProgressBar pb = new ProgressBar();
95  		pb.setMaxWidth(Double.MAX_VALUE);
96  		setGraphic(pb);
97  		
98  		setText(null);
99  		Task<Void> t = new Task<Void>()
100 		{
101 			String text;
102 			boolean setStyle = false;
103 			ContextMenu cm = new ContextMenu();
104 			int nid = nidFetcher_.applyAsInt(item.getRefex());
105 			
106 			@Override
107 			protected Void call() throws Exception
108 			{
109 				try
110 				{
111 					ConceptVersionBI c = WBUtility.getConceptVersion(nid);
112 					if (c == null) 
113 					{
114 						//This may be a different component - like a description, or another refex... need to handle.
115 						ComponentVersionBI cv = ExtendedAppContext.getDataStore().getComponentVersion(WBUtility.getViewCoordinate(), nid);
116 						
117 						if (cv instanceof DescriptionVersionBI<?>)
118 						{
119 							DescriptionVersionBI<?> dv = (DescriptionVersionBI<?>) cv;
120 							text = "Description: " + dv.getText();
121 							
122 							CommonMenuBuilderI menuBuilder = CommonMenus.CommonMenuBuilder.newInstance();
123 							menuBuilder.setMenuItemsToExclude(CommonMenuItem.COPY_SCTID);
124 							
125 							CommonMenus.addCommonMenus(cm, menuBuilder, new CommonMenusNIdProvider()
126 							{
127 								
128 								@Override
129 								public Collection<Integer> getNIds()
130 								{
131 									return Arrays.asList(new Integer[] {dv.getNid()});
132 								}
133 							});
134 						}
135 						else if (cv instanceof RelationshipVersionBI<?>)
136 						{
137 							RelationshipVersionBI<?> rv = (RelationshipVersionBI<?>) cv;
138 							text = "Relationship: " + WBUtility.getDescription(rv.getOriginNid()) + "->" 
139 									+ WBUtility.getDescription(rv.getTypeNid()) + "->"
140 									+ WBUtility.getDescription(rv.getDestinationNid());
141 							
142 							CommonMenuBuilderI menuBuilder = CommonMenus.CommonMenuBuilder.newInstance();
143 							menuBuilder.setMenuItemsToExclude(CommonMenuItem.COPY_SCTID);
144 							
145 							CommonMenus.addCommonMenus(cm, menuBuilder, new CommonMenusNIdProvider()
146 							{
147 								
148 								@Override
149 								public Collection<Integer> getNIds()
150 								{
151 									return Arrays.asList(new Integer[] {rv.getNid()});
152 								}
153 							});
154 						}
155 						else if (cv instanceof RefexDynamicVersionBI<?>)
156 						{
157 							RefexDynamicVersionBI<?> rdv = (RefexDynamicVersionBI<?>) cv;
158 							text = "Nested Refex Dynamic: using assemblage " + WBUtility.getDescription(rdv.getAssemblageNid());
159 							
160 							CommonMenuBuilderI menuBuilder = CommonMenus.CommonMenuBuilder.newInstance();
161 							menuBuilder.setMenuItemsToExclude(CommonMenuItem.COPY_SCTID);
162 							
163 							CommonMenus.addCommonMenus(cm, menuBuilder, new CommonMenusNIdProvider()
164 							{
165 								
166 								@Override
167 								public Collection<Integer> getNIds()
168 								{
169 									return Arrays.asList(new Integer[] {rdv.getNid()});
170 								}
171 							});
172 						}
173 						else if (cv instanceof RefexVersionBI<?>)
174 						{
175 							RefexVersionBI<?> rv = (RefexVersionBI<?>) cv;
176 							text = "Nested Refex: using assemblage " + WBUtility.getDescription(rv.getAssemblageNid());
177 							
178 							CommonMenuBuilderI menuBuilder = CommonMenus.CommonMenuBuilder.newInstance();
179 							menuBuilder.setMenuItemsToExclude(CommonMenuItem.COPY_SCTID);
180 							
181 							CommonMenus.addCommonMenus(cm, menuBuilder, new CommonMenusNIdProvider()
182 							{
183 								
184 								@Override
185 								public Collection<Integer> getNIds()
186 								{
187 									return Arrays.asList(new Integer[] {rv.getNid()});
188 								}
189 							});
190 						}
191 						else
192 						{
193 							logger_.warn("The component type " + cv + " is not handled yet!");
194 							//Not sure what else there may be?  media - doesn't seem to be used... anything else?
195 							text = cv.toUserString();
196 						}
197 					}
198 					else
199 					{
200 						if (isAssemblage_)
201 						{
202 							MenuItem mi = new MenuItem("View Refex Assemblage Usage");
203 							mi.setOnAction((action) ->
204 							{
205 								SimpleDisplayConcept sdc = new SimpleDisplayConcept(c);
206 								DynamicReferencedItemsView driv = new DynamicReferencedItemsView(sdc);
207 								driv.showView(null);
208 							});
209 							mi.setGraphic(Images.SEARCH.createImageView());
210 							cm.getItems().add(mi);
211 							
212 							mi = new MenuItem("Configure Refex Indexing");
213 							mi.setOnAction((action) ->
214 							{
215 								new ConfigureDynamicRefexIndexingView(c).showView(null);
216 							});
217 							mi.setGraphic(Images.CONFIGURE.createImageView());
218 							cm.getItems().add(mi);
219 						}
220 						
221 						
222 						CommonMenus.addCommonMenus(cm, new CommonMenusNIdProvider()
223 						{
224 							
225 							@Override
226 							public Collection<Integer> getNIds()
227 							{
228 								return Arrays.asList(new Integer[] {item.getRefex().getNid()});
229 							}
230 						});
231 						text = WBUtility.getDescription(c);
232 						setStyle = true;
233 					}
234 				}
235 				catch (Exception e)
236 				{
237 					logger_.error("Unexpected error", e);
238 					text= "-ERROR-";
239 				}
240 				return null;
241 			}
242 
243 			/**
244 			 * @see javafx.concurrent.Task#succeeded()
245 			 */
246 			@Override
247 			protected void succeeded()
248 			{
249 				//default text is a label, which doesn't wrap properly.
250 				setText(null);
251 				Text textHolder = new Text(text);
252 				textHolder.wrappingWidthProperty().bind(widthProperty().subtract(10));
253 				if (cm.getItems().size() > 0)
254 				{
255 					setContextMenu(cm);
256 				}
257 				setGraphic(textHolder);
258 				if (setStyle)
259 				{
260 					if (item.isCurrent())
261 					{
262 						getTreeTableRow().getStyleClass().removeAll("historical");
263 					}
264 					else
265 					{
266 						if (!getTreeTableRow().getStyleClass().contains("historical"))
267 						{
268 							getTreeTableRow().getStyleClass().add("historical");
269 						}
270 					}
271 				}
272 			}
273 		};
274 		Utility.execute(t);
275 	}
