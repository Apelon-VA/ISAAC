167 	private final ListChangeListener<Object> filterListener_ = new ListChangeListener<Object>() {
168 		@Override
169 		public void onChanged(
170 				javafx.collections.ListChangeListener.Change<? extends Object> c) {
171 			while (c.next()) {
172 				if (c.wasPermutated()) {
173 					// irrelevant
174 				} else if (c.wasUpdated()) {
175 					// irrelevant
176 				} else {
177 					refresh();
178 					break;
179 				}
180 			}
181 		}
182 	};
183 	private void addFilterCacheListeners() {
184 		removeFilterCacheListeners();
185 		filterCache_.addListener(filterCacheListener_);
186 		for (HeaderNode.Filter<?> filter : filterCache_.values()) {
187 			filter.getFilterValues().addListener(filterListener_);
188 		}
189 	}
190 	private void removeFilterCacheListeners() {
191 		filterCache_.removeListener(filterCacheListener_);
192 		for (HeaderNode.Filter<?> filter : filterCache_.values()) {
193 			filter.getFilterValues().removeListener(filterListener_);
194 		}
195 	}
