124 	/**
125 	 * For cases when it was built from an existing refex only
126 	 * @param attachedDataColumn - optional - ignored (can be null) except applicable to {@link DynamicRefexColumnType#ATTACHED_DATA}
127 	 */
128 	public int compareTo(DynamicRefexColumnType columnTypeToCompare, Integer attachedDataColumn, RefexDynamicGUI other)
129 	{
130 		try
131 		{
132 			switch (columnTypeToCompare)
133 			{
134 				case STATUS_CONDENSED:
135 				{
136 					//sort by uncommitted first, then current / historical, then active / inactive
137 					if (this.getRefex().getTime() == Long.MAX_VALUE)
138 					{
139 						return -1;
140 					}
141 					else if (other.getRefex().getTime() == Long.MAX_VALUE)
142 					{
143 						return 1;
144 					}
145 					
146 					if (this.isCurrent() && !other.isCurrent())
147 					{
148 						return -1;
149 					}
150 					else if (!this.isCurrent() && other.isCurrent())
151 					{
152 						return 1;
153 					}
154 					
155 					if (this.getRefex().isActive() && !other.getRefex().isActive())
156 					{
157 						return -1;
158 					}
159 					else if (!this.getRefex().isActive() && other.getRefex().isActive())
160 					{
161 						return 1;
162 					}
163 					return 0;
164 				}
165 				case TIME:
166 				{
167 					if (this.getRefex().getTime() < other.getRefex().getTime())
168 					{
169 						return -1;
170 					}
171 					else if (this.getRefex().getTime() > other.getRefex().getTime())
172 					{
173 						return -1;
174 					}
175 					else
176 					{
177 						return 0;
178 					}
179 				}
180 				case COMPONENT: case ASSEMBLAGE: case STATUS_STRING: case AUTHOR: case MODULE: case PATH:
181 				{
182 					String myString = this.getDisplayStrings(columnTypeToCompare, null).getKey();
183 					String otherString = other.getDisplayStrings(columnTypeToCompare, null).getKey();
184 					return AlphanumComparator.compare(myString, otherString, true);
185 				}
186 				case ATTACHED_DATA:
187 				{
188 					if (attachedDataColumn == null)
189 					{
190 						throw new RuntimeException("API misuse");
191 					}
192 					RefexDynamicDataBI myData = this.refex_.getData().length > attachedDataColumn ? this.refex_.getData()[attachedDataColumn] : null;
193 					RefexDynamicDataBI otherData = other.refex_.getData().length > attachedDataColumn ? other.refex_.getData()[attachedDataColumn] : null;
194 					
195 					if (myData == null && otherData != null)
196 					{
197 						return -1;
198 					}
199 					else if (myData != null && otherData == null)
200 					{
201 						return 1;
202 					}
203 					else if (myData == null && otherData == null)
204 					{
205 						return 0;
206 					}
207 					else if (myData instanceof RefexDynamicFloatBI && otherData instanceof RefexDynamicFloatBI)
208 					{
209 						return NumberUtilities.compare(((RefexDynamicFloatBI) myData).getDataFloat(), ((RefexDynamicFloatBI) otherData).getDataFloat());
210 					}
211 					else if (myData instanceof RefexDynamicDoubleBI && otherData instanceof RefexDynamicDoubleBI) 
212 					{
213 						return NumberUtilities.compare(((RefexDynamicDoubleBI) myData).getDataDouble(), ((RefexDynamicDoubleBI) otherData).getDataDouble());
214 					}
215 					else if (myData instanceof RefexDynamicIntegerBI && otherData instanceof RefexDynamicIntegerBI) 
216 					{
217 						return NumberUtilities.compare(((RefexDynamicIntegerBI) myData).getDataInteger(), ((RefexDynamicIntegerBI) otherData).getDataInteger());
218 					}
219 					else if (myData instanceof RefexDynamicLongBI && otherData instanceof RefexDynamicLongBI)
220 					{
221 						return NumberUtilities.compare(((RefexDynamicLongBI) myData).getDataLong(), ((RefexDynamicLongBI) otherData).getDataLong());
222 					}
223 					else
224 					{
225 						String myString = this.getDisplayStrings(columnTypeToCompare, attachedDataColumn).getKey();
226 						String otherString = other.getDisplayStrings(columnTypeToCompare, attachedDataColumn).getKey();
227 						return AlphanumComparator.compare(myString, otherString, true);
228 					}
229 				}
230 
231 				default:
232 					throw new RuntimeException("Missing implementation: " + columnTypeToCompare);
233 			}
234 		}
235 		catch (IOException e)
236 		{
237 			//TODO (artf231428) how to handle?
238 			throw new RuntimeException("Sort error", e);
239 		}
240 	}

243 	 * Returns the string for display, and the tooltip, if applicable.  Either / or may be null.
244 	 * Key is for the display, value is for the tooltip.
245 	 * @param attachedDataColumn should be null for most types - applicable to {@link DynamicRefexColumnType#ATTACHED_DATA}
246 	 */
247 	public AbstractMap.SimpleImmutableEntry<String, String> getDisplayStrings(DynamicRefexColumnType desiredColumn, Integer attachedDataColumn)
248 	{
249 		String cacheKey = desiredColumn.name() + attachedDataColumn;  //null is ok on the attachedDataColumn...
250 		
251 		AbstractMap.SimpleImmutableEntry<String, String> returnValue = stringCache_.get(cacheKey);
252 		if (returnValue != null)
253 		{
254 			return returnValue;
255 		}
256 		
257 		
258 		switch (desiredColumn)
259 		{
260 			case STATUS_CONDENSED:
261 			{
262 				//Just easier to leave the impl in StatusCell for this one.  We don't need filters on this column either.
263 				throw new RuntimeException("No text for this field");
264 			}
265 			case COMPONENT: case ASSEMBLAGE: case AUTHOR: case PATH: case MODULE:
266 			{
267 				String text = getComponentText(getNidFetcher(desiredColumn, attachedDataColumn));
268 				returnValue = new AbstractMap.SimpleImmutableEntry<String, String>(text, text);
269 				break;
270 			}
271 			case STATUS_STRING:
272 			{
273 				returnValue = new AbstractMap.SimpleImmutableEntry<String, String>(refex_.getStatus().toString(), null);
274 				break;
275 			}
276 			case TIME:
277 			{
278 				returnValue = new AbstractMap.SimpleImmutableEntry<String, String>((refex_.getTime() == Long.MAX_VALUE ? "-Uncommitted-" : 
279 					new Date(refex_.getTime()).toString()), null);
280 				break;
281 			}
282 			case ATTACHED_DATA:
283 			{
284 				if (attachedDataColumn == null)
285 				{
286 					throw new RuntimeException("API misuse");
287 				}
288 				RefexDynamicDataBI data = this.refex_.getData().length > attachedDataColumn ? this.refex_.getData()[attachedDataColumn] : null;
289 				if (data != null)
290 				{
291 					if (data instanceof RefexDynamicByteArrayBI)
292 					{
293 						returnValue = new AbstractMap.SimpleImmutableEntry<String, String>("[Binary]", null);
294 					}
295 					else if (data instanceof RefexDynamicNidBI)
296 					{
297 						String desc = getComponentText(((RefexDynamicNidBI)data).getDataNid());
298 						returnValue = new AbstractMap.SimpleImmutableEntry<String, String>(desc, data.getDataObject().toString());
299 					}
300 					else if (data instanceof RefexDynamicUUIDBI)
301 					{
302 						String desc;
303 						if (ExtendedAppContext.getDataStore().hasUuid(((RefexDynamicUUIDBI)data).getDataUUID()))
304 						{
305 							desc = getComponentText(ExtendedAppContext.getDataStore().getNidForUuids(((RefexDynamicUUIDBI)data).getDataUUID()));
306 						}
307 						else
308 						{
309 							desc = ((RefexDynamicUUIDBI)data).getDataUUID() + "";
310 						}
311 						returnValue = new AbstractMap.SimpleImmutableEntry<String, String>(desc, data.getDataObject().toString());
312 					}
313 					else if (data instanceof RefexDynamicArrayBI<?>)
314 					{
315 						RefexDynamicArrayBI<?> instanceData = (RefexDynamicArrayBI<?>)data;
316 						switch (instanceData.getArrayDataType())
317 						{
318 							case ARRAY:
319 								//Could recurse... but I can't imagine a use case at the moment.
320 								returnValue = new AbstractMap.SimpleImmutableEntry<String, String>("[" + instanceData.getDataArray().length + " nested arrays]",
321 										"An array of nested arrays");
322 								break;
323 							case STRING: case BOOLEAN: case DOUBLE: case FLOAT: case INTEGER: case LONG: case NID: case UUID:
324 							{
325 								//NID and UUID could be turned into strings... but, unusual use case... leave like this for now.
326 								StringBuilder sb = new StringBuilder();
327 								sb.append("[");
328 								for (RefexDynamicDataBI d : instanceData.getDataArray())
329 								{
330 									sb.append(d.getDataObject().toString());
331 									sb.append(", ");
332 								}
333 								if (sb.length() > 1)
334 								{
335 									sb.setLength(sb.length() - 2);
336 								}
337 								sb.append("]");
338 								returnValue = new AbstractMap.SimpleImmutableEntry<String, String>(sb.toString(), "Array of " + instanceData.getDataArray().length + " items: " + sb.toString());
339 								break;
340 							}
341 							
342 							case BYTEARRAY:
343 								returnValue = new AbstractMap.SimpleImmutableEntry<String, String>("[" + instanceData.getDataArray().length + " Binary items]",
344 										"An array of binary objects");
345 								break;
346 							case UNKNOWN: case POLYMORPHIC:
347 							{
348 								//shouldn't happen - but just do the toString
349 								returnValue = new AbstractMap.SimpleImmutableEntry<String, String>("[" + instanceData.getDataArray().length + " items]",
350 										"An array of unknown data elements");
351 								break;
352 							}
353 							default:
354 								logger_.error("Unhandled case: {}, {}", instanceData, Arrays.toString(instanceData.getDataArray()));
355 								returnValue = new AbstractMap.SimpleImmutableEntry<String, String>("-ERROR-", "Internal error computing value");
356 								break;
357 						}
358 					}
359 					else
360 					{
361 						returnValue = new AbstractMap.SimpleImmutableEntry<String, String>(data.getDataObject().toString(), data.getDataObject().toString());
362 					}
363 				}
364 				else
365 				{
366 					returnValue = new AbstractMap.SimpleImmutableEntry<String, String>("", null);
367 				}
368 				break;
369 			}
370 
371 			default:
372 				throw new RuntimeException("Missing implementation: " + desiredColumn);
373 		}
374 		
375 		stringCache_.put(cacheKey, returnValue);
376 		return returnValue;
377 		
378 	}

385 	private String getComponentText(int nid)
386 	{
387 		String text;
388 		
389 		try
390 		{
391 			ConceptVersionBI c = WBUtility.getConceptVersion(nid);
392 			if (c == null) 
393 			{
394 				//This may be a different component - like a description, or another refex... need to handle.
395 				ComponentVersionBI cv = ExtendedAppContext.getDataStore().getComponentVersion(WBUtility.getViewCoordinate(), nid);
396 				if (cv == null)
397 				{
398 					text = "[NID] " + nid;
399 				}
400 				
401 				else if (cv instanceof DescriptionVersionBI<?>)
402 				{
403 					DescriptionVersionBI<?> dv = (DescriptionVersionBI<?>) cv;
404 					text = "Description: " + dv.getText();
405 				}
406 				else if (cv instanceof RelationshipVersionBI<?>)
407 				{
408 					RelationshipVersionBI<?> rv = (RelationshipVersionBI<?>) cv;
409 					text = "Relationship: " + WBUtility.getDescription(rv.getOriginNid()) + "->" 
410 							+ WBUtility.getDescription(rv.getTypeNid()) + "->"
411 							+ WBUtility.getDescription(rv.getDestinationNid());
412 				}
413 				else if (cv instanceof RefexDynamicVersionBI<?>)
414 				{
415 					RefexDynamicVersionBI<?> rdv = (RefexDynamicVersionBI<?>) cv;
416 					text = "Nested Sememe Dynamic: using assemblage " + WBUtility.getDescription(rdv.getAssemblageNid());
417 				}
418 				else if (cv instanceof RefexVersionBI<?>)
419 				{
420 					RefexVersionBI<?> rv = (RefexVersionBI<?>) cv;
421 					text = "Nested Sememe: using assemblage " + WBUtility.getDescription(rv.getAssemblageNid());
422 				}
423 				else if (cv instanceof MediaVersionBI<?>)
424 				{
425 					MediaVersionBI<?> mv = (MediaVersionBI<?>) cv;
426 					text = "Media of format " + mv.getFormat();
427 				}
428 				else
429 				{
430 					logger_.warn("The component type " + cv + " is not handled yet!");
431 					//Not sure what else there may be?
432 					text = cv.toUserString();
433 				}
434 			}
435 			else
436 			{
437 				text = WBUtility.getDescription(c);
438 			}
439 		}
440 		catch (Exception e)
441 		{
442 			logger_.error("Unexpected error", e);
443 			text = "-ERROR-";
444 		}
445 		return text;
446 	}

453 	public ToIntFunction<RefexDynamicVersionBI<?>> getNidFetcher(DynamicRefexColumnType desiredColumn, Integer attachedDataColumn)
454 	{
455 		switch (desiredColumn)

464 				{
465 					@Override

469 					}
470 				};
471 			}
472 			case ASSEMBLAGE:
473 			{
474 				return new ToIntFunction<RefexDynamicVersionBI<?>>()
475 				{
476 					@Override
477 					public int applyAsInt(RefexDynamicVersionBI<?> value)
478 					{
479 						return refex_.getAssemblageNid();
480 					}
481 				};
482 			}
483 			case AUTHOR:
484 			{
485 				return new ToIntFunction<RefexDynamicVersionBI<?>>()
486 				{
487 					@Override
488 					public int applyAsInt(RefexDynamicVersionBI<?> value)
489 					{

491 					}
492 				};
493 			}
494 			case MODULE:
495 			{
496 				return new ToIntFunction<RefexDynamicVersionBI<?>>()
497 				{
498 					@Override
499 					public int applyAsInt(RefexDynamicVersionBI<?> value)
500 					{
501 						return refex_.getModuleNid();
502 					}
503 				};
504 			}
505 			case PATH:
506 			{
507 				return new ToIntFunction<RefexDynamicVersionBI<?>>()
508 				{
509 					@Override
510 					public int applyAsInt(RefexDynamicVersionBI<?> value)
511 					{
512 						return refex_.getPathNid();
513 					}
514 				};
515 			}
516 			
517 			case ATTACHED_DATA:
518 			{
519 				if (attachedDataColumn == null)
520 				{
521 					throw new RuntimeException("API misuse");
522 				}
523 				return new ToIntFunction<RefexDynamicVersionBI<?>>()
524 				{
525 					@Override
526 					public int applyAsInt(RefexDynamicVersionBI<?> value)
527 					{
528 						RefexDynamicDataBI data = refex_.getData().length > attachedDataColumn ? refex_.getData()[attachedDataColumn] : null;
529 						if (data != null)
530 						{
531 							if (data instanceof RefexDynamicNidBI)
532 							{
533 								return ((RefexDynamicNidBI)data).getDataNid();
534 							}
535 							else if (data instanceof RefexDynamicUUIDBI)
536 							{
537 								if (ExtendedAppContext.getDataStore().hasUuid(((RefexDynamicUUIDBI)data).getDataUUID()))
538 								{
539 									return ExtendedAppContext.getDataStore().getNidForUuids(((RefexDynamicUUIDBI)data).getDataUUID());
540 								}
541 							}
542 						}
543 						return 0;
544 					}
545 				};
546 				
547 			}
548 
549 			default:
550 				throw new RuntimeException("Missing implementation: " + desiredColumn);
551 		}
