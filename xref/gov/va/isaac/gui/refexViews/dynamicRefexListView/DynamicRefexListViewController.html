265 	private void rebuildList(boolean fullRebuild)
266 	{
267 		if (disableRead)
268 		{
269 			log.debug("Skip rebuild");
270 			return;
271 		}
272 
273 		synchronized (readStatusLock)
274 		{
275 			if (readStatusTracker != PendingRead.IDLE)
276 			{
277 				//already a read in progress.  Add this request to the list, return.
278 				if (fullRebuild && readStatusTracker != PendingRead.DO_FULL_READ)
279 				{
280 					readStatusTracker = PendingRead.DO_FULL_READ;
281 				}
282 				else if (readStatusTracker != PendingRead.DO_FULL_READ && readStatusTracker != PendingRead.DO_FILTER_READ)
283 				{
284 					readStatusTracker = PendingRead.DO_FILTER_READ;
285 				}
286 				log.debug("Queued rebuild " + readStatusTracker);
287 				return;
288 			}
289 			else
290 			{
291 				readStatusTracker = (fullRebuild ? PendingRead.FULL_READ_IN_PROGRESS : PendingRead.FILTER_UPDATE_PROGRESS);
292 			}
293 		}
294 
295 		statusLabel.setText("Reading Sememes");
296 		readingRefexProgress.setVisible(true);
297 		SimpleDisplayConcept selectedBefore = refexList.getSelectionModel().getSelectedItem();
298 		refexList.getSelectionModel().clearSelection();
299 		refexList.getItems().clear();
300 
301 		Task<Void> t = new Task<Void>()
302 		{
303 			ArrayList<SimpleDisplayConcept> filteredList;
304 
305 			@Override
306 			protected Void call() throws Exception
307 			{
308 				log.debug("Rebuild request running: " + readStatusTracker);
309 				if (fullRebuild)
310 				{
311 					allRefexDefinitions = null;
312 				}
313 
314 				if (allRefexDefinitions == null)
315 				{
316 					allRefexDefinitions = new HashSet<>();
317 					
318 					LuceneDynamicRefexIndexer indexer = AppContext.getService(LuceneDynamicRefexIndexer.class);
319 					List<SearchResult> refexes = indexer.queryAssemblageUsage(RefexDynamic.REFEX_DYNAMIC_DEFINITION_DESCRIPTION.getNid(), 1000, Long.MAX_VALUE);
320 					for (SearchResult sr : refexes)
321 					{
322 						RefexDynamicChronicleBI<?> rc = (RefexDynamicChronicleBI<?>) ExtendedAppContext.getDataStore().getComponent(sr.getNid());
323 						if (rc == null)
324 						{
325 							log.info("Out of date index?  Search result for refexes contained a NID that can't be resolved: {}" + sr.getNid());
326 							continue;
327 						}
328 						//These are nested refex references - it returns a description component - concept we want is the parent of that.
329 						allRefexDefinitions.add(new SimpleDisplayConcept(
330 								ExtendedAppContext.getDataStore().getComponent(rc.getReferencedComponentNid()).getEnclosingConcept(),
331 								null));
332 					}
333 				}
334 				
335 				//This code for adding the concept from the concept filter panel can be removed, if we fix the above code to actually
336 				//find all dynamic refexes in the system.
337 				boolean conceptFromOutsideTheList = true;
338 				SimpleDisplayConcept enteredConcept = null;
339 				if (conceptNode.getConcept() != null && conceptNode.isValid().get())
340 				{
341 					enteredConcept = new SimpleDisplayConcept(conceptNode.getConcept());
342 				}
343 
344 				filteredList = new ArrayList<>();
345 				for (SimpleDisplayConcept sdc : allRefexDefinitions)
346 				{
347 					if (enteredConcept != null && sdc.getNid() == enteredConcept.getNid())
348 					{
349 						conceptFromOutsideTheList = false;
350 					}
351 					if (passesFilters(sdc))
352 					{
353 						filteredList.add(sdc);
354 					}
355 				}
356 				
357 				if (enteredConcept != null && conceptFromOutsideTheList)
358 				{
359 					filteredList.add(enteredConcept);
360 				}
361 				
362 				Collections.sort(filteredList);
363 				
364 				return null;
365 			}
366 
367 			/**
368 			 * @see javafx.concurrent.Task#succeeded()
369 			 */
370 			@Override
371 			protected void succeeded()
372 			{
373 				finished();
374 			}
375 
376 			/**
377 			 * @see javafx.concurrent.Task#failed()
378 			 */
379 			@Override
380 			protected void failed()
381 			{
382 				log.error("Unexpected error building Sememe List", this.getException());
383 				AppContext.getCommonDialogs().showErrorDialog("Error reading Dynamic sememes", this.getException());
384 				finished();
385 			}
386 
387 			private void finished()
388 			{
389 				log.debug("Sememe Definition refresh complete");
390 				refexList.getItems().addAll(filteredList);
391 				if (selectedBefore != null && refexList.getItems().contains(selectedBefore))
392 				{
393 					refexList.getSelectionModel().select(selectedBefore);
394 				}
395 				showRefexDetails(refexList.getSelectionModel().getSelectedItem());
396 				statusLabel.setText("Showing " + filteredList.size() + " of " + allRefexDefinitions.size() + " Sememes");
397 				readingRefexProgress.setVisible(false);
398 				synchronized (readStatusLock)
399 				{
400 					if (readStatusTracker == PendingRead.DO_FILTER_READ || readStatusTracker == PendingRead.DO_FULL_READ)
401 					{
402 						boolean rebuild = readStatusTracker == PendingRead.DO_FULL_READ ? true : false;
403 						readStatusTracker = PendingRead.IDLE;
404 						//Another request came in while we were running.  Run again.
405 						rebuildList(rebuild);
406 					}
407 					else
408 					{
409 						readStatusTracker = PendingRead.IDLE;
410 					}
411 				}
412 			}
413 		};
414 
415 		Utility.execute(t);
416 	}

418 	private boolean passesFilters(SimpleDisplayConcept sdc) throws IOException
419 	{
420 		if (!conceptNode.isValid().get())
421 		{
422 			return false;
423 		}
424 		else if (conceptNode.getConcept() != null && conceptNode.getConcept().getConceptNid() != sdc.getNid())
425 		{
426 			return false;
427 		}
428 		else if (descriptionMatchesFilter.getText().length() > 0)
429 		{
430 			if (!sdc.getDescription().toLowerCase().contains(descriptionMatchesFilter.getText().toLowerCase()))
431 			{
432 				return false;
433 			}
434 		}
435 		else if (refexStyleFilter.getSelectionModel().getSelectedIndex() != 0)
436 		{
437 			ConceptVersionBI c = WBUtility.getConceptVersion(sdc.getNid());
438 			if ((c.isAnnotationStyleRefex() && refexStyleFilter.getSelectionModel().getSelectedIndex() == 2)
439 					|| (!c.isAnnotationStyleRefex() && refexStyleFilter.getSelectionModel().getSelectedIndex() == 1))
440 			{
441 				return false;
442 			}
443 		}
444 		return true;
445 	}

447 	private void showRefexDetails(SimpleDisplayConcept sdn)
448 	{
449 		if (sdn != null && sdn.getNid() == currentlyRenderedRefexNid)
450 		{
451 			return;
452 		}
453 		else
454 		{
455 			currentlyRenderedRefexNid = (sdn == null ? 0 : sdn.getNid());
456 		}
457 		selectedRefexNameLabel.setText("");
458 		selectedRefexDescriptionLabel.setText("");
459 		refexStyleLabel.setText("");
460 		referencedComponentTypeLabel.setText("");
461 		extensionFields.getItems().clear();
462 		
463 		if (sdn == null)
464 		{
465 			viewUsage.setDisable(true);
466 			return;
467 		}
468 		else
469 		{
470 			viewUsage.setDisable(false);
471 		}
472 		selectedRefexProgressIndicator.setVisible(true);
473 		selectedRefexNameLabel.setText(sdn.getDescription());
474 
475 		Task<Void> t = new Task<Void>()
476 		{
477 			ArrayList<RefexDynamicColumnInfo> tempColumnInfo = new ArrayList<>();
478 			
479 			@Override
480 			protected Void call() throws Exception
481 			{
482 				RefexDynamicUsageDescription rdud = RefexDynamicUsageDescriptionBuilder.readRefexDynamicUsageDescriptionConcept(sdn.getNid());
483 				//fill in the header stuff
484 				Platform.runLater(() -> 
485 				{
486 					selectedRefexNameLabel.setText(rdud.getRefexName());
487 					selectedRefexDescriptionLabel.setText(rdud.getRefexUsageDescription());
488 					refexStyleLabel.setText(rdud.isAnnotationStyle() ? "Annotation" : "Embedded");
489 					referencedComponentTypeLabel.setText(rdud.getReferencedComponentTypeRestriction() == null ? "No restriction" : 
490 						"Must be " + rdud.getReferencedComponentTypeRestriction().toString());
491 				});
492 				
493 				//now fill in the data column details...
494 				
495 				for (RefexDynamicColumnInfo rdci : rdud.getColumnInfo())
496 				{
497 					//force the read on the column info - this may have to be read from the DB.
498 					rdci.getColumnName();
499 					tempColumnInfo.add(rdci);
500 				}
501 				
502 				return null;
503 			}
504 
505 			/**
506 			 * @see javafx.concurrent.Task#succeeded()
507 			 */
508 			@Override
509 			protected void succeeded()
510 			{
511 				extensionFields.getItems().addAll(tempColumnInfo);
512 				extensionFields.scrollTo(0);
513 				finished();
514 			}
515 
516 			/**
517 			 * @see javafx.concurrent.Task#failed()
518 			 */
519 			@Override
520 			protected void failed()
521 			{
522 				log.error("Unexpected error building selected sememe", this.getException());
523 				AppContext.getCommonDialogs().showErrorDialog("Error reading Dynamic Sememe", this.getException());
524 				finished();
525 			}
526 
527 			private void finished()
528 			{
529 				selectedRefexProgressIndicator.setVisible(false);
530 			}
531 		};

533 		Utility.execute(t);
534 	}
