38  /**
39   * A GUI for handling imports.
40   *
41   * @author ocarlsen
42   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
43   */
44  public class ImportView extends GridPane {
45  
46      private static final Logger LOG = LoggerFactory.getLogger(ImportView.class);
47  
48      private final Label modelTypeLabel = new Label();
49      private final Label fileNameLabel = new Label();
50      private final Label resultLabel = new Label();
51  
52      public ImportView() {
53          super();
54  
55          // GUI placeholders.
56          add(new Label("Information Model: "), 0, 0);
57          add(modelTypeLabel, 1, 0);
58          add(new Label("File Name: "), 0, 1);
59          add(fileNameLabel, 1, 1);
60          add(new Label("Result: "), 0, 2);
61          add(resultLabel, 1, 2);
62  
63          // Set minimum dimensions.
64          setMinHeight(200);
65          setMinWidth(600);
66      }
67  
68      public void doImport(InformationModelType modelType, final String fileName) {
69          Preconditions.checkNotNull(modelType);
70          Preconditions.checkNotNull(fileName);
71  
72          // Make sure in application thread.
73          FxUtils.checkFxUserThread();
74  
75          // Update UI.
76          modelTypeLabel.setText(modelType.getDisplayName());
77          fileNameLabel.setText(fileName);
78  
79          if (modelType == InformationModelType.CEM) {
80              importCEM(modelType, fileName);
81          } else {
82              throw new UnsupportedOperationException(modelType.getDisplayName() +
83                      " import not yet supported in ISAAC.");
84          }
85      }
86  
87      private void importCEM(InformationModelType modelType, final String fileName) {
88  
89          // Do work in background.
90          Task<ConceptChronicleBI> task = new Task<ConceptChronicleBI>() {
91  
92              @Override
93              protected ConceptChronicleBI call() throws Exception {
94  
95                  // Do work.
96                  CEMImporter importer = new CEMImporter();
97                  return importer.importModel(new File(fileName));
98              }
99  
100             @Override
101             protected void succeeded() {
102                 ConceptChronicleBI result = this.getValue();
103 
104                 // Update UI.
105                 resultLabel.setText("Successfully imported concept: " + result.toUserString());
106            }
107 
108             @Override
109             protected void failed() {
110 
111                 // Update UI.
112                 resultLabel.setText("Failed to import model from file: " + fileName);
113 
114                 // Show dialog.
115                 Throwable ex = getException();
116                 String title = ex.getClass().getName();
117                 String msg = String.format("Unexpected error importing from file \"%s\"", fileName);
118                 LOG.error(msg, ex);
119                 AppContext.getCommonDialogs().showErrorDialog(title, msg, ex.getMessage());
120             }
121         };
122 
123         // Bind cursor to task state.
124         ObjectBinding<Cursor> cursorBinding = Bindings.when(task.runningProperty()).then(Cursor.WAIT).otherwise(Cursor.DEFAULT);
125         this.getScene().cursorProperty().bind(cursorBinding);
126 
127         Thread t = new Thread(task, "Importer_" + modelType);
128         t.setDaemon(true);
129         t.start();
130     }
131 }
