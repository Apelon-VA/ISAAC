19  package gov.va.isaac.gui.dialog;
20  
21  import java.io.IOException;
22  import java.net.URL;
23  
24  import javafx.fxml.FXMLLoader;
25  import javafx.scene.Parent;
26  import javafx.scene.Scene;
27  import javafx.stage.Modality;
28  import javafx.stage.Stage;
29  import javafx.stage.StageStyle;
30  import javafx.stage.Window;
31  
32  /**
33   * A {@link Stage} which can be used to show an export settings dialog.
34   *
35   * @author ocarlsen
36   */
37  public class ExportFileSettingsDialog extends Stage {
38  
39    /** The controller. */
40    private final ExportFileSettingsDialogController controller;
41  
42    /**
43     * Instantiates a {@link ExportFileSettingsDialog} from the specified
44     * parameters.
45     *
46     * @param parent the parent
47     * @throws IOException Signals that an I/O exception has occurred.
48     */
49    public ExportFileSettingsDialog(Window parent) throws IOException {
50      super();
51      setTitle("Export Settings");
52      setResizable(true);
53  
54      initOwner(parent);
55      initModality(Modality.WINDOW_MODAL);
56      initStyle(StageStyle.DECORATED);
57  
58      // Load from FXML.
59      URL resource = this.getClass().getResource("ExportFileSettingsDialog.fxml");
60      FXMLLoader loader = new FXMLLoader(resource);
61      Parent root = (Parent) loader.load();
62      Scene scene = new Scene(root);
63      setScene(scene);
64  
65      controller = loader.getController();
66      controller.setVariables(this, parent);
67    }
68  }
