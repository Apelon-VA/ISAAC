120 			private volatile AtomicBoolean refreshQueued = new AtomicBoolean(false);
121 			{
122 				setComputeOnInvalidate(true);
123 				addBinding(AppContext.getService(UserProfileBindings.class).getViewCoordinatePath(),

132 			protected boolean computeValue()
133 			{
134 				synchronized (refreshQueued)
135 				{
136 					if (refreshQueued.get())
137 					{
138 						LOG.info("Skip relationship refresh() due to pending refresh");
139 						return false;
140 					}
141 					else
142 					{
143 						refreshQueued.set(true);
144 
145 						LOG.debug("Rel refresh() due to change of an observed user property");
146 						Utility.schedule(() ->
147 						{
148 							Platform.runLater(() ->
149 							{
150 								try
151 								{
152 									synchronized (refreshQueued)
153 									{
154 										refreshQueued.set(false);
155 									}
156 									refresh(null);
157 									if (sourceColumn != null 
158 											&& AppContext.getService(UserProfileBindings.class).getDisplayRelDirection().get() != RelationshipDirection.SOURCE)
159 									{
160 										//this defaults to off, turn it on, if they switch to a view mode that includes targets

162 									}
163 								}
164 								catch (Exception e)
165 								{
166 									LOG.error("Unexpected error running refresh", e);
167 								}
168 							});
169 						}, 10, TimeUnit.MILLISECONDS);
170 					}
171 				}
172 				return false;
173 			}

422 	private void sizeAndPosition(Node node, StackPane sp, Pos position)
423 	{
424 		if (node instanceof ImageView)
425 		{
426 			((ImageView)node).setFitHeight(12);
427 			((ImageView)node).setFitWidth(12);
428 		}
429 		
430 		if (position == Pos.TOP_LEFT)
431 		{
432 			StackPane.setMargin(node, new Insets(0, 0, 0, 0));
433 		}
434 		else if (position == Pos.TOP_RIGHT)
435 		{
436 			StackPane.setMargin(node, new Insets(0, 0, 0, 13));
437 		}
438 		else if (position == Pos.BOTTOM_LEFT)
439 		{
440 			StackPane.setMargin(node, new Insets(13, 0, 0, 0));
441 		}
442 		else if (position == Pos.BOTTOM_RIGHT)
443 		{
444 			StackPane.setMargin(node, new Insets(13, 0, 0, 13));
445 		}
446 		else
447 		{
448 			throw new RuntimeException("Unsupported Position!");
449 		}
450 		sp.getChildren().add(node);
451 		StackPane.setAlignment(node, Pos.TOP_LEFT);
452 	}

454 	private Node backgroundLookup(final TableCell<RelationshipVersion, RelationshipVersion> cell, RelationshipColumnType type, RelationshipVersion ref)
455 	{
456 		Utility.execute(() ->
457 		{
458 			String value = ref.getDisplayStrings(type).getKey();
459 			Platform.runLater(() ->
460 			{
461 				if (cell.isEmpty() || cell.getItem() == null)
462 				{
463 					//We are updating a cell that should no longer be populated - stop!
464 					return;
465 				}
466 				ContextMenu cm = cell.getContextMenu();
467 				if (cm == null)
468 				{
469 					cm = new ContextMenu();
470 					cell.setContextMenu(cm);
471 				}
472 				
473 				Text text = new Text(value);
474 				cell.setGraphic(text);
475 				text.wrappingWidthProperty().bind(cell.getTableColumn().widthProperty());
476 
477 				CommonMenus.addCommonMenus(cm, new CommonMenusNIdProvider()
478 				{
479 					@Override
480 					public Collection<Integer> getNIds()
481 					{
482 						int nid = ref.getNidFetcher(type).applyAsInt(ref.getRelationshipVersion());
483 
484 						ArrayList<Integer> nids = new ArrayList<>();
485 						if (nid != 0)
486 						{
487 							nids.add(nid);
488 						}
489 						return nids;
490 					}
491 				});
492 				
493 				// Menu item to copy cell text.
494 				//for some reason, we get called multiple times - only add the Copy Value method if it doesn't exit
495 				boolean found = false;
496 				for (MenuItem mi : cm.getItems())
497 				{
498 					if (mi.getText().equals("Copy Value"))
499 					{
500 						found = true;
501 						break;
502 					}
503 				}
504 				
505 				if (!found)
506 				{
507 					// Menu item to copy cell text.
508 					MenuItem mi = new MenuItem("Copy Value");
509 					mi.setOnAction(new EventHandler<ActionEvent>()
510 					{
511 						@Override
512 						public void handle(ActionEvent arg0)
513 						{
514 							CustomClipboard.set(text.getText());
515 						}
516 					});
517 					mi.setGraphic(Images.COPY.createImageView());
518 					
519 					cm.getItems().add(mi);
520 				}
521 				
522 				AppContext.getService(DragRegistry.class).setupDragOnly(text, new SingleConceptIdProvider()
523 				{
524 					@Override
525 					public String getConceptId()
526 					{
527 						return ref.getNidFetcher(type).applyAsInt(ref.getRelationshipVersion()) +"";
528 					}
529 				});
530 			});
531 		});
532 		ProgressBar pb = new ProgressBar();
533 		pb.prefWidthProperty().bind(cell.widthProperty());
534 		return pb;
535 	}

537 	private void addTableColumns(RelationshipColumnType[] columns, boolean isStampColumnSet, 
538 			Callback<TableColumn<RelationshipVersion, RelationshipVersion>, TableCell<RelationshipVersion, RelationshipVersion>> cellFactory)
539 	{
540 		TableColumn<RelationshipVersion, RelationshipVersion> nestingParent = null;
541 		if (isStampColumnSet)
542 		{
543 			nestingParent = new TableColumn<>("Stamp Fields");
544 			relationshipsTable.getColumns().add(nestingParent);
545 			stampColumns.add(nestingParent);
546 		}
547 		for (RelationshipColumnType col : columns)
548 		{
549 			TableColumn<RelationshipVersion, RelationshipVersion> tc = new TableColumn<RelationshipVersion, RelationshipVersion>(col.toString());
550 			tc.setUserData(col);
551 			tc.setCellValueFactory(new Callback<TableColumn.CellDataFeatures<RelationshipVersion, RelationshipVersion>, ObservableValue<RelationshipVersion>>()
552 			{
553 				@Override
554 				public ObservableValue<RelationshipVersion> call(CellDataFeatures<RelationshipVersion, RelationshipVersion> param)
555 				{
556 					return new SimpleObjectProperty<RelationshipVersion>(param.getValue());
557 				}
558 			});
559 			tc.setCellFactory(cellFactory);
560 
561 			//off by default
562 			if (col == RelationshipColumnType.UUID || col == RelationshipColumnType.SOURCE)
563 			{
564 				tc.setVisible(false);
565 			}
566 			
567 			if (col == RelationshipColumnType.SOURCE)
568 			{
569 				sourceColumn = tc;
570 			}
571 
572 			if (nestingParent == null)
573 			{
574 				relationshipsTable.getColumns().add(tc);
575 			}
576 			else
577 			{
578 				nestingParent.getColumns().add(tc);
579 				stampColumns.add(tc);
580 			}
581 		}

591 	private void refresh(ConceptVersionBI concept)
592 	{
593 		if (concept == null && conceptUUID_ == null)
594 		{
595 			LOG.info("refesh called while concept null");
596 			return;
597 		}
598 		
599 		Utility.execute(() ->
600 		{
601 			ArrayList<RelationshipVersionBI<?>> allRelationships = new ArrayList<>();
602 			try
603 			{
604 				ConceptVersionBI localConcept = (concept == null ? OTFUtility.getConceptVersion(conceptUUID_) : concept);
605 				
606 				//target is the only option where we would exclude source
607 				if (AppContext.getService(UserProfileBindings.class).getDisplayRelDirection().get() != RelationshipDirection.TARGET)
608 				{
609 					for (RelationshipChronicleBI chronicle : localConcept.getRelationshipsOutgoing())
610 					{
611 						for (RelationshipVersionBI<?> rv : chronicle.getVersions())
612 						{
613 							allRelationships.add(rv);
614 						}
615 					}
616 				}
617 				
618 				//source is the only option where we would exclude target
619 				if (AppContext.getService(UserProfileBindings.class).getDisplayRelDirection().get() != RelationshipDirection.SOURCE)
620 				{
621 					for (RelationshipChronicleBI chronicle : localConcept.getRelationshipsIncoming())
622 					{
623 						for (RelationshipVersionBI<?> rv : chronicle.getVersions())
624 						{
625 							allRelationships.add(rv);
626 						}
627 					}
628 				}
629 				
630 				//Sort the newest to the top per UUID
631 				Collections.sort(allRelationships, new Comparator<RelationshipVersionBI<?>>()
632 				{
633 					@Override
634 					public int compare(RelationshipVersionBI<?> o1, RelationshipVersionBI<?> o2)
635 					{
636 						if (o1.getPrimordialUuid().equals(o2.getPrimordialUuid()))
637 						{
638 							return o2.getStamp() - o1.getStamp();
639 						}
640 						else
641 						{
642 							return o1.getPrimordialUuid().compareTo(o2.getPrimordialUuid());
643 						}
644 					}
645 				});
646 			}
647 			catch (Exception e)
648 			{
649 				AppContext.getCommonDialogs().showErrorDialog("Error reading relationships", e);
650 				LOG.error("Unexpected error reading relationships", e);
651 			}
652 			
653 			Platform.runLater(() ->
654 			{
655 				int count = 0;
656 				try
657 				{
658 					UUID lastSeenRefex = null;
659 					
660 					relationshipsTable.getItems().clear();
661 					for (RelationshipVersionBI<?> r : allRelationships)
662 					{
663 						if (!showHistory_.get() && r.getPrimordialUuid().equals(lastSeenRefex))
664 						{
665 							//Only want the newest one per UUID if history isn't requested
666 							continue;
667 						}
668 						if (showActiveOnly_.get() == false || r.isActive())
669 						{
670 							//first one we see with a new UUID is current, others are historical
671 							RelationshipVersion newRelationshipVersion = new RelationshipVersion(r, !r.getPrimordialUuid().equals(lastSeenRefex));
672 							count++;
673 							relationshipsTable.getItems().add(newRelationshipVersion);
674 						}
675 						lastSeenRefex = r.getPrimordialUuid();
676 					}
677 				}
678 				catch (Exception e)
679 				{
680 					AppContext.getCommonDialogs().showErrorDialog("Error reading relationships", e);
681 					LOG.error("Unexpected error reading relationships", e);
682 				}
683 				summaryText.set(count + " relationships");
684 			});
685 		});
686 	}
687 	
688 	public Node getNode()
689 	{
690 		return relationshipsTable;
691 	}

694 	{
695 		return summaryText.getReadOnlyProperty();
