98     * Instantiates an empty {@link ExportView}.

100   public ExportView() {

104     this.setHgap(10);

106     this.setPadding(new javafx.geometry.Insets(10, 10, 10, 10));
107     GridPaneBuilder builder = new GridPaneBuilder(this);
108     builder.addRow("Export Type: ", exportTypeLabel);
109     builder.addRow("Path Name: ", pathNameLabel);
110     builder.addRow("Folder Name: ", folderNameLabel);
111     builder.addRow("Progress: ", progressBar);
112     progressBar.setMinWidth(500);
113     builder.addRow("Status: ", statusLabel);
114     builder.addRow("Result: ", resultLabel);
115     @SuppressWarnings("deprecation")
116     javafx.scene.layout.HBox hbox = javafx.scene.layout.HBoxBuilder.create()
117     .alignment(javafx.geometry.Pos.TOP_CENTER).children(cancelButton).build();
118 
119     builder.addRow(hbox);
120     cancelButton.setOnAction(new EventHandler<ActionEvent>() {
121       @Override
122       public void handle(ActionEvent event) {
123         doCancel();
124       }
125     });
126     setConstraints();
127 
128     // Set minimum dimensions.
129     setMinHeight(100);
130     setMinWidth(600);
131   }

133   /**
134    * Perform an export according the specifed parameters.
135    *
136    * @param exportType the export type
137    * @param pathNid the path nid
138    * @param folderName the file name
139    * @param zipChecked the flag indicating whether to compress output
140    */
141   public void doExport(ExportType exportType, int pathNid,
142     final String folderName, boolean zipChecked) {
143     Preconditions.checkNotNull(exportType);
144     Preconditions.checkNotNull(folderName);
145 
146     // Make sure in application thread.
147     FxUtils.checkFxUserThread();
148 
149     // Update UI.
150     exportTypeLabel.setText(exportType.getDisplayName());
151     pathNameLabel.setText(OTFUtility.getConPrefTerm(pathNid));
152     folderNameLabel.setText(folderName);
153 
154     File folder = new File(folderName);
155     // Inject into an ExportFileHandler.
156     exportFileHandler =
157         new ExportFileHandler(pathNid, exportType, folder, zipChecked);
158 
159     // Do work in background.
160     task = new ExporterTask(exportFileHandler);
161 
162     // Bind cursor to task state.
163     ObjectBinding<Cursor> cursorBinding =
164         Bindings.when(task.runningProperty()).then(Cursor.WAIT)
165             .otherwise(Cursor.DEFAULT);
166     this.getScene().cursorProperty().bind(cursorBinding);
167 
168     taskThread = new Thread(task, "Exporter_" + exportType);
169     taskThread.setDaemon(true);
170     taskThread.start();
171   }
172 
173   /**
174    * Do cancel.
175    */
176   public void doCancel() {
177     LOG.info("Cancel import.");
178     if (requestCancel) {
179       // complete the process
180       ((Stage) getScene().getWindow()).close();
181       return;
182     }
183 
184     // Set requestCancel to true and cancel task
185     requestCancel = true;
186     exportFileHandler.doCancel();
187     cancelButton.setText("Close");
188   }
189 
190   /**
191    * Sets the FX constraints.
192    */
193   private void setConstraints() {
194 
195     // Column 1 has empty constraints.
196     this.getColumnConstraints().add(new ColumnConstraints());
197 
198     // Column 2 should grow to fill space.
199     ColumnConstraints column2 = new ColumnConstraints();
200     column2.setHgrow(Priority.ALWAYS);
201     this.getColumnConstraints().add(column2);
202 
203     // Rows 1-4 have empty constraints.
204     this.getRowConstraints().add(new RowConstraints());
205     this.getRowConstraints().add(new RowConstraints());
206     this.getRowConstraints().add(new RowConstraints());
207     this.getRowConstraints().add(new RowConstraints());
208 
209     // Row 5 should
210     RowConstraints row5 = new RowConstraints();
211     row5.setVgrow(Priority.ALWAYS);
212     this.getRowConstraints().add(row5);
213   }

215   /**
216    * Inner class to handle the export task.
217    */
218   class ExporterTask extends Task<Boolean> {
219 
220     /** The export handler. */
221     ExporterTask(ExportFileHandler exportFileHandler) {
222       // do nothing
223     }
224 
225     /*
226      * (non-Javadoc)
227      * 
228      * @see javafx.concurrent.Task#call()
229      */
230     @Override
231     protected Boolean call() throws Exception {
232 
233       ProgressListener listener = new ProgressListener() {
234         @Override
235         public void updateProgress(ProgressEvent pe) {
236           Platform.runLater(() -> {
237             progressBar.setProgress(((double) pe.getProgress()) / 100);
238             statusLabel.setText(pe.getNote());
239           });
240         }
241       };
242       Platform.runLater(() -> {
243         statusLabel.setText("Starting...");
244         progressBar.setProgress(0);
245       });
246       exportFileHandler.doExport(listener);
247       Platform.runLater(() -> {
248         statusLabel.setText("Finished");
249         progressBar.setProgress(1);
250       });
251       return true;
252     }
253 
254     /*
255      * (non-Javadoc)
256      * 
257      * @see javafx.concurrent.Task#succeeded()
258      */
259     @Override
260     protected void succeeded() {
261       // Update UI.
262       progressBar.setProgress(100);
263       cancelButton.setText("Close");
264       if (requestCancel) {
265         resultLabel.setText("Successfully cancelled export");
266       } else {
267         statusLabel.setText("");
268         resultLabel.setText("Successfully exported data");
269       }
270       requestCancel = true;
271     }
272 
273     /*
274      * (non-Javadoc)
275      * 
276      * @see javafx.concurrent.Task#failed()
277      */
278     @Override
279     protected void failed() {
280       Throwable ex = getException();
281       progressBar.setProgress(100);
282       cancelButton.setText("Close");
283       requestCancel = true;
284       // leave last comment on failure, e.g. do not set status label here
285       resultLabel.setText("Failed to export data");
286 
287       String title = ex.getClass().getName();
288       String msg = String.format("Unexpected error exporting to file");
289       LOG.error(msg, ex);
290       AppContext.getCommonDialogs()
291           .showErrorDialog(title, msg, ex.getMessage());
292     }
