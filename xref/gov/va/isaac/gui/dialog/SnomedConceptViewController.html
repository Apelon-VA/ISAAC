92  	@FXML private ToggleButton activeOnlyToggle;

94  	@FXML private HBox sourceRelTitleHBox;
95  
96  	private Button showInTreeButton;
97  	private ProgressIndicator treeViewProgress;
98  	private final BooleanProperty treeViewSearchRunning = new SimpleBooleanProperty(false);

100 	private SctTreeViewIsaacView sctTree;

102 	private UUID conceptUuid;
103 	private int conceptNid = 0;

105 	public Region getRootNode()
106 	{
107 		return anchorPane;

110 	public void setConcept(ConceptChronicleDdo concept) {
111 		conceptUuid = concept.getPrimordialUuid();
112 		splitPane.setDividerPositions(0.7);
113 
114 		// Update text of labels.
115 		ConceptAttributesChronicleDdo attributeChronicle = concept.getConceptAttributes();
116 		final ConceptAttributesVersionDdo conceptAttributes = attributeChronicle.getVersions().get(attributeChronicle.getVersions().size() - 1);
117 		conceptDefinedLabel.setText(conceptAttributes.isDefined() + "");
118 		conceptStatusLabel.setText(conceptAttributes.getStatus().name());
119 		fsnLabel.setText(OTFUtility.getDescription(concept));
120 		CopyableLabel.addCopyMenu(fsnLabel);
121 		
122 		MenuItem copyFull = new MenuItem("Copy Full Concept");
123 		copyFull.setGraphic(Images.COPY.createImageView());
124 
125 		copyFull.setOnAction(e -> CustomClipboard.set(OTFUtility.getConceptVersion(concept.getPrimordialUuid()).toLongString()));
126 
127 		fsnLabel.getContextMenu().getItems().add(copyFull);
128 		
129 		AppContext.getService(DragRegistry.class).setupDragOnly(fsnLabel, new SingleConceptIdProvider()
130 		{
131 			@Override
132 			public String getConceptId()
133 			{
134 				return uuidLabel.getText();
135 			}
136 		});
137 		uuidLabel.setText(concept.getPrimordialUuid().toString());
138 		AppContext.getService(DragRegistry.class).setupDragOnly(uuidLabel, new SingleConceptIdProvider()
139 		{
140 			@Override
141 			public String getConceptId()
142 			{
143 				return uuidLabel.getText();
144 			}
145 		});
146 		CopyableLabel.addCopyMenu(uuidLabel);
147 		
148 		stampToggle.setText("");
149 		stampToggle.setGraphic(Images.STAMP.createImageView());
150 		stampToggle.setTooltip(new Tooltip("Show Stamp columns when pressed, Hides Stamp columns when not pressed"));
151 		stampToggle.setSelected(true);
152 
153 		activeOnlyToggle.setText("");
154 		activeOnlyToggle.setGraphic(Images.FILTER_16.createImageView());
155 		activeOnlyToggle.setTooltip(new Tooltip("Filter to only show active items when pressed, Show all items when not pressed"));
156 		activeOnlyToggle.setSelected(true);
157 		
158 		historyToggle.setText("");
159 		historyToggle.setGraphic(Images.HISTORY.createImageView());
160 		historyToggle.setTooltip(new Tooltip("Shows full history when pressed, Only shows current items when not pressed"));
161 		
162 		descriptionTypeButton.setText("");
163 		ImageView displayFsn = Images.DISPLAY_FSN.createImageView();
164 		Tooltip.install(displayFsn, new Tooltip("Displaying the Fully Specified Name - click to display the Preferred Term"));
165 		displayFsn.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getDisplayFSN());
166 		ImageView displayPreferred = Images.DISPLAY_PREFERRED.createImageView();
167 		displayPreferred.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getDisplayFSN().not());
168 		Tooltip.install(displayPreferred, new Tooltip("Displaying the Preferred Term - click to display the Fully Specified Name"));
169 		descriptionTypeButton.setGraphic(new StackPane(displayFsn, displayPreferred));
170 		descriptionTypeButton.prefHeightProperty().bind(historyToggle.heightProperty());
171 		descriptionTypeButton.prefWidthProperty().bind(historyToggle.widthProperty());
172 		descriptionTypeButton.setOnAction(new EventHandler<ActionEvent>()
173 		{
174 			@Override
175 			public void handle(ActionEvent event)
176 			{
177 				try
178 				{
179 					UserProfile up = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
180 					up.setDisplayFSN(AppContext.getService(UserProfileBindings.class).getDisplayFSN().not().get());
181 					ExtendedAppContext.getService(UserProfileManager.class).saveChanges(up);
182 				}
183 				catch (Exception e)
184 				{
185 					LOG.error("Unexpected error storing pref change", e);
186 				}
187 			}
188 		});
189 		
190 		
191 		ConceptVersionBI conceptVersionBI = OTFUtility.getConceptVersion(concept.getPrimordialUuid());
192 		AppContext.getService(CommonlyUsedConcepts.class).addConcept(new SimpleDisplayConcept(conceptVersionBI));
193 
194 		// Add context menu items for additional identifiers.
195 		for (final IdentifierDdo id : attributeChronicle.getAdditionalIds()) {
196 
197 			MenuItem mi = new MenuItem("View Concept");
198 			mi.setOnAction(new EventHandler<ActionEvent>() {
199 				@Override
200 				public void handle(ActionEvent ignore) {
201 					AppContext.getCommonDialogs().showConceptDialog(id.getAuthorityRef().getUuid());
202 				}
203 			});
204 
205 			CopyableLabel l = new CopyableLabel(id.getAuthorityRef().getText());
206 			l.getContextMenu().getItems().add(mi);
207 			l.getStyleClass().add("boldLabel");
208 
209 			HBox hbox = new HBox();
210 			hbox.getChildren().add(l);
211 			hbox.getChildren().add(new CopyableLabel(id.getDenotation().toString()));
212 			hbox.setSpacing(5.0);
213 
214 			idVBox.getChildren().add(hbox);
215 		}
216 		
217 		try
218 		{
219 			DescriptionTableView dtv = new DescriptionTableView(stampToggle.selectedProperty(), historyToggle.selectedProperty(), activeOnlyToggle.selectedProperty());
220 			dtv.setConcept(conceptVersionBI);
221 			descriptionsTableHolder.getChildren().add(dtv.getNode());
222 			VBox.setVgrow(dtv.getNode(), Priority.ALWAYS);
223 		}
224 		catch (Exception e)
225 		{
226 			LOG.error("Error configuring description view", e);
227 			descriptionsTableHolder.getChildren().add(new Label("Unexpected error configuring descriptions view"));
228 		}
229 		
230 		//rel table section
231 		try
232 		{
233 			Button taxonomyViewMode = new Button();
234 			taxonomyViewMode.setPadding(new Insets(2.0));
235 			ImageView taxonomyInferred = Images.TAXONOMY_INFERRED.createImageView();
236 			taxonomyInferred.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getStatedInferredPolicy().isEqualTo(StatedInferredOptions.INFERRED));
237 			Tooltip.install(taxonomyInferred, new Tooltip("Displaying the Inferred view- click to display the Inferred then Stated view"));
238 			ImageView taxonomyStated = Images.TAXONOMY_STATED.createImageView();
239 			taxonomyStated.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getStatedInferredPolicy().isEqualTo(StatedInferredOptions.STATED));
240 			Tooltip.install(taxonomyStated, new Tooltip("Displaying the Stated view- click to display the Inferred view"));
241 			ImageView taxonomyInferredThenStated = Images.TAXONOMY_INFERRED_THEN_STATED.createImageView();
242 			taxonomyInferredThenStated.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getStatedInferredPolicy().isEqualTo(StatedInferredOptions.INFERRED_THEN_STATED));
243 			Tooltip.install(taxonomyInferredThenStated, new Tooltip("Displaying the Inferred then Stated view- click to display the Stated view"));
244 			taxonomyViewMode.setGraphic(new StackPane(taxonomyInferred, taxonomyStated, taxonomyInferredThenStated));
245 			taxonomyViewMode.setOnAction(new EventHandler<ActionEvent>()
246 			{
247 				@Override
248 				public void handle(ActionEvent event)
249 				{
250 					try
251 					{
252 						UserProfile up = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
253 						StatedInferredOptions sip = null;
254 						if (AppContext.getService(UserProfileBindings.class).getStatedInferredPolicy().get() == StatedInferredOptions.STATED)
255 						{
256 							sip = StatedInferredOptions.INFERRED;
257 						}
258 						else if (AppContext.getService(UserProfileBindings.class).getStatedInferredPolicy().get() == StatedInferredOptions.INFERRED)
259 						{
260 							sip = StatedInferredOptions.INFERRED_THEN_STATED;
261 						}
262 						else if (AppContext.getService(UserProfileBindings.class).getStatedInferredPolicy().get() == StatedInferredOptions.INFERRED_THEN_STATED)
263 						{
264 							sip = StatedInferredOptions.STATED;
265 						}
266 						else
267 						{
268 							LOG.error("Unexpected error!");
269 							return;
270 						}
271 						up.setStatedInferredPolicy(sip);
272 						ExtendedAppContext.getService(UserProfileManager.class).saveChanges(up);
273 					}
274 					catch (Exception e)
275 					{
276 						LOG.error("Unexpected error storing pref change", e);
277 					}
278 				}
279 			});
280 			HBox.setMargin(taxonomyViewMode, new Insets(0, 0, 0, 5.0));
281 			sourceRelTitleHBox.getChildren().add(taxonomyViewMode);
282 			
283 			Button relationshipViewMode = new Button();
284 			relationshipViewMode.setPadding(new Insets(2.0));
285 			ImageView taxonomySource = Images.TAXONOMY_SOURCE.createImageView();
286 			taxonomySource.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getDisplayRelDirection().isEqualTo(RelationshipDirection.SOURCE));
287 			Tooltip.install(taxonomySource, new Tooltip("Displaying the Source Relationships only, click to display Target"));
288 			ImageView taxonomyTarget = Images.TAXONOMY_TARGET.createImageView();
289 			taxonomyTarget.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getDisplayRelDirection().isEqualTo(RelationshipDirection.TARGET));
290 			Tooltip.install(taxonomyTarget, new Tooltip("Displaying the Target Relationships only, click to display Source and Target"));
291 			ImageView taxonomySourceAndTarget = Images.TAXONOMY_SOURCE_AND_TARGET.createImageView();
292 			taxonomySourceAndTarget.visibleProperty().bind(AppContext.getService(UserProfileBindings.class).getDisplayRelDirection().isEqualTo(RelationshipDirection.SOURCE_AND_TARGET));
293 			Tooltip.install(taxonomySourceAndTarget, new Tooltip("Displaying the Source and Target Relationships, click to display Source only"));
294 			relationshipViewMode.setGraphic(new StackPane(taxonomySource, taxonomyTarget, taxonomySourceAndTarget));
295 			relationshipViewMode.setOnAction(new EventHandler<ActionEvent>()
296 			{
297 				@Override
298 				public void handle(ActionEvent event)
299 				{
300 					try
301 					{
302 						UserProfile up = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
303 						if (up.getDisplayRelDirection() == RelationshipDirection.SOURCE)
304 						{
305 							up.setDisplayRelDirection(RelationshipDirection.TARGET);
306 						}
307 						else if (up.getDisplayRelDirection() == RelationshipDirection.TARGET)
308 						{
309 							up.setDisplayRelDirection(RelationshipDirection.SOURCE_AND_TARGET);
310 						}
311 						else if (up.getDisplayRelDirection() == RelationshipDirection.SOURCE_AND_TARGET)
312 						{
313 							up.setDisplayRelDirection(RelationshipDirection.SOURCE);
314 						}
315 						else 
316 						{
317 							LOG.error("Unhandeled case!");
318 						}
319 						ExtendedAppContext.getService(UserProfileManager.class).saveChanges(up);
320 					}
321 					catch (Exception e)
322 					{
323 						LOG.error("Unexpected error storing pref change", e);
324 					}
325 				}
326 			});
327 			HBox.setMargin(relationshipViewMode, new Insets(0, 0, 0, 5.0));
328 			sourceRelTitleHBox.getChildren().add(relationshipViewMode);
329 
330 			RelationshipTableView rtv = new RelationshipTableView(stampToggle.selectedProperty(), historyToggle.selectedProperty(), activeOnlyToggle.selectedProperty());
331 			rtv.setConcept(conceptVersionBI);
332 			relationshipsTableHolder.getChildren().add(rtv.getNode());
333 			VBox.setVgrow(rtv.getNode(), Priority.ALWAYS);
334 			
335 			Label summary = new Label();
336 			HBox.setMargin(summary, new Insets(0, 0, 0, 5.0));
337 			sourceRelTitleHBox.getChildren().add(summary);
338 			summary.textProperty().bind(rtv.getSummaryText());
339 			
340 		}
341 		catch (Exception e)
342 		{
343 			LOG.error("Error configuring relationship view", e);
344 			descriptionsTableHolder.getChildren().add(new Label("Unexpected error configuring descriptions view"));
345 		}
346 		
347 		RefexViewI v = AppContext.getService(RefexViewI.class, "DynamicRefexView");
348 		v.setComponent(conceptVersionBI.getNid(), stampToggle.selectedProperty(), activeOnlyToggle.selectedProperty(), historyToggle.selectedProperty(), false);
349 		v.getView().setMinHeight(100.0);
350 		VBox.setVgrow(v.getView(), Priority.ALWAYS);
351 		annotationsRegion.getChildren().add(v.getView());
352 		
353 		//TODO this is a hack - I want these off by default, but there is a bug in javafx that messes up the table
354 		Platform.runLater(new Runnable()
355 		{
356 			@Override
357 			public void run()
358 			{
359 				stampToggle.setSelected(false);
360 			}
361 		});
362 
363 		// Load the inner tree view.
364 		try {
365 			sctTree = AppContext.getService(SctTreeViewIsaacView.class); 
366 			sctTree.init();
367 			
368 			showInTreeButton = new Button();
369 			showInTreeButton.setPadding(new Insets(2.0));
370 			showInTreeButton.setGraphic(Images.TAXONOMY_SEARCH_RESULT_ANCESTOR.createImageView());
371 			showInTreeButton.setTooltip(new Tooltip("Find Concept"));
372 			showInTreeButton.visibleProperty().bind(treeViewSearchRunning.not());
373 			showInTreeButton.setOnAction(new EventHandler<ActionEvent>() {
374 				@Override
375 				public void handle(ActionEvent event) {
376 					treeViewSearchRunning.set(true);
377 					sctTree.locateConcept(conceptAttributes.getConcept().getPrimordialUuid(), treeViewSearchRunning);
378 				}
379 			});
380 			
381 			treeViewProgress = new ProgressIndicator(-1);
382 			treeViewProgress.setMaxSize(16, 16);
383 			treeViewProgress.visibleProperty().bind(treeViewSearchRunning);
384 
385 			StackPane sp = new StackPane(showInTreeButton, treeViewProgress);
386 			sctTree.addToToolBar(sp);
387 			
388 			Region r = sctTree.getView();
389 			splitRight.getChildren().add(r);
390 			VBox.setVgrow(r, Priority.ALWAYS);
391 			treeViewSearchRunning.set(true);
392 			sctTree.locateConcept(concept.getPrimordialUuid(), treeViewSearchRunning);
393 		} catch (Exception ex) {
394 			LOG.error("Error creating tree view", ex);
395 			splitRight.getChildren().add(new Label("Unexpected error building tree"));
396 		}
397 	}
398 
399 	public StringProperty getTitle() {
400 		return fsnLabel.textProperty();
401 	}
402 
403 	public UUID getConceptUuid() {
404 		return conceptUuid;
405 	}
406 
407 	public int getConceptNid() {
408 		if (conceptNid == 0) {
409 			conceptNid = OTFUtility.getConceptVersion(conceptUuid).getNid();
410 		}
411 		
412 		return conceptNid;
413 	}
