113     public void setConcept(ConceptChronicleDdo concept) {
114         conceptUuid = concept.getPrimordialUuid();
115 
116         // Update text of labels.
117         ConceptAttributesChronicleDdo attributeChronicle = concept.getConceptAttributes();
118         final ConceptAttributesVersionDdo conceptAttributes = attributeChronicle.getVersions().get(attributeChronicle.getVersions().size() - 1);
119         conceptDefinedLabel.setText(conceptAttributes.isDefined() + "");
120         conceptStatusLabel.setText(conceptAttributes.getStatus().name());
121         fsnLabel.setText(WBUtility.getDescription(concept));
122         CopyableLabel.addCopyMenu(fsnLabel);
123

125         copyFull.setGraphic(Images.COPY.createImageView());
126 
127         copyFull.setOnAction(e -> CustomClipboard.set(WBUtility.getConceptVersion(concept.getPrimordialUuid()).toLongString()));

142             @Override
143             public String getConceptId()

146             }
147         });
148         CopyableLabel.addCopyMenu(uuidLabel);
149         
150         stampToggle.setText("");
151         stampToggle.setGraphic(Images.STAMP.createImageView());
152         stampToggle.setTooltip(new Tooltip("Show/Hide Stamp Columns"));
153         stampToggle.setSelected(true);
154

156         activeOnlyToggle.setGraphic(Images.FILTER_16.createImageView());
157         activeOnlyToggle.setTooltip(new Tooltip("Show Active Only / Show All"));

160         historyToggle.setText("");

162         historyToggle.setTooltip(new Tooltip("Show Current Only / Show Full History"));
163         //TODO (artf231887) make the other view tables aware of the show/hide stamp call

168         Callback<TableColumn.CellDataFeatures<StringWithRefList, StringWithRef>, ObservableValue<StringWithRef>> cellValueFactory =

268         // Populate relationship table data model.
269         for (RelationshipChronicleDdo chronicle : concept.getOriginRelationships()) {
270             RelationshipVersionDdo relationship = chronicle.getVersions().get(chronicle.getVersions().size() - 1);
271             StringWithRef typeRef = new StringWithRef(relationship.getTypeReference().getText(), relationship.getTypeReference().getUuid());
272             StringWithRef destRef = new StringWithRef(relationship.getDestinationReference().getText(), relationship.getDestinationReference().getUuid());
273             StringWithRefList row = new StringWithRefList(typeRef, destRef);
274             relationshipsTable.getItems().add(row);
275         }
276 
277         setupTable(new String[] { "Type", "Destination" }, relationshipsTable,
278                 cellValueFactory, cellFactory);
279         
280         RefexViewI v = AppContext.getService(RefexViewI.class, "DynamicRefexView");
281         v.setComponent(conceptVersionBI.getNid(), stampToggle.selectedProperty(), activeOnlyToggle.selectedProperty(), historyToggle.selectedProperty());
282         v.getView().setMinHeight(100.0);
283         VBox.setVgrow(v.getView(), Priority.ALWAYS);
284         annotationsRegion.getChildren().add(v.getView());
285 
286         // Load the inner tree view.
287         try {
288             sctTree = AppContext.getService(SctTreeViewIsaacView.class); 
289             sctTree.init();
290             
291             showInTreeButton = new Button();
292             showInTreeButton.setPadding(new Insets(2.0));
293             showInTreeButton.setGraphic(Images.TAXONOMY_SEARCH_RESULT_ANCESTOR.createImageView());

306             treeViewProgress.visibleProperty().bind(treeViewSearchRunning);

308             StackPane sp = new StackPane(showInTreeButton, treeViewProgress);
309             sctTree.addToToolBar(sp);
310             
311             Region r = sctTree.getView();
312             splitRight.getChildren().add(r);
313             VBox.setVgrow(r, Priority.ALWAYS);
314             treeViewSearchRunning.set(true);
315             sctTree.locateConcept(concept.getPrimordialUuid(), treeViewSearchRunning);
316         } catch (Exception ex) {
317             LOG.error("Error creating tree view", ex);
318             splitRight.getChildren().add(new Label("Unexpected error building tree"));
319         }

338     private void setupTable(String[] columns, TableView<StringWithRefList> tableView,
339             Callback<TableColumn.CellDataFeatures<StringWithRefList, StringWithRef>, ObservableValue<StringWithRef>> cellValueFactory,
340             Callback<TableColumn<StringWithRefList, StringWithRef>, TableCell<StringWithRefList, StringWithRef>> cellFactory) {
341 
342         // Configure table columns.
343         for (int i = 0; i < columns.length; i++) {
344             TableColumn<StringWithRefList, StringWithRef> tc =
345                     new TableColumn<StringWithRefList, StringWithRef>(columns[i]);
346             tc.setId(i + "");
347             tc.setCellValueFactory(cellValueFactory);
348             tc.setCellFactory(cellFactory);
349 
350             // Bind preferred column width to function of column count.
351             float colWidth = 1.0f / columns.length;
352             tc.prefWidthProperty().bind(tableView.widthProperty().multiply(colWidth).subtract(5.0));
353 
354             tableView.getColumns().add(tc);
355         }
356 
357         tableView.setPrefHeight(tableView.getMinHeight() + (20.0 * tableView.getItems().size()));
358         tableView.setPlaceholder(new Label());
359     }

364     private static final class StringWithRefList {
365 
366         private final ArrayList<StringWithRef> items = new ArrayList<StringWithRef>();
367 
368         private StringWithRefList(StringWithRef... items) {
369             for (StringWithRef item : items) {
370                 this.items.add(item);
371             }
372         }
373 
374         public StringWithRef get(int index) {
375             return items.get(index);
376         }
377     }

382     private static final class StringWithRef {
383

386
