135   /**
136    * Do import for the specified type and file name.
137    *
138    * @param modelType the model type
139    * @param fileName the file name
140    * @throws TransformerConfigurationException
141    */
142   public void doImport(InformationModelType modelType, final String fileName)
143     throws TransformerConfigurationException {
144     Preconditions.checkNotNull(modelType);
145     Preconditions.checkNotNull(fileName);
146 
147     // Make sure in application thread.
148     FxUtils.checkFxUserThread();
149 
150     // Update UI.
151     modelTypeLabel.setText(modelType.getDisplayName());
152     fileNameLabel.setText(fileName);
153 
154     // Instantiate appropriate importer class.
155     ImportHandler importHandler = null;
156     switch (modelType) {
157       case CEM:
158         importHandler = new CEMImporter();
159         break;
160       case HeD:
161         importHandler = new HeDImporter();
162         break;
163       case FHIM:
164         importHandler = new FHIMImporter();
165         break;
166       default:
167         throw new UnsupportedOperationException(modelType.getDisplayName()
168             + " import not yet supported in ISAAC.");
169     }
170 
171     // Do work in background.
172     task = new ImporterTask(fileName, importHandler);
173 
174     // Bind cursor to task state.
175     ObjectBinding<Cursor> cursorBinding =
176         Bindings.when(task.runningProperty()).then(Cursor.WAIT)
177             .otherwise(Cursor.DEFAULT);
178     this.getScene().cursorProperty().bind(cursorBinding);
179 
180     Thread t = new Thread(task, "Importer_" + modelType);
181     t.setDaemon(true);
182     t.start();
183   }
184 
185   /**
186    * Do cancel.
187    */
188   public void doCancel() {
189     LOG.info("Cancel import.");
190     if (requestCancel) {
191       // complete the process
192       ((Stage) getScene().getWindow()).close();
193       return;
194     }
195 
196     // Set requestCancel to true and cancel task
197     requestCancel = true;
198     if (task.isRunning()) {
199       task.cancel(true);
200     }
201     cancelButton.setText("Close");
202     requestCancel = true;
203     resultLabel.setText("Successfully cancelled import.");
204   }

221     this.getRowConstraints().add(new RowConstraints());
222     this.getRowConstraints().add(new RowConstraints());
223     this.getRowConstraints().add(new RowConstraints());
224 
225     // Row 5 should
226     RowConstraints row5 = new RowConstraints();
227     row5.setVgrow(Priority.ALWAYS);
228     this.getRowConstraints().add(row5);
229   }

231   /**
232    * Concrete {@link Task} for executing the import.
233    *
234    * @author ocarlsen
235    * @author bcarlsenca
236    */
237   class ImporterTask extends Task<InformationModel> {
238 
239     /** The file name. */
240     private final String fileName;
241 
242     /** The import handler. */
243     private final ImportHandler importHandler;
244 
245     /**
246      * Instantiates a {@link ImporterTask} from the specified parameters.
247      *
248      * @param fileName the file name
249      * @param importHandler the import handler
250      */
251     ImporterTask(String fileName, ImportHandler importHandler) {
252       this.fileName = fileName;
253       this.importHandler = importHandler;
254     }
255 
256     /*
257      * (non-Javadoc)
258      * 
259      * @see javafx.concurrent.Task#call()
260      */
261     @Override
262     protected InformationModel call() throws Exception {
263       InformationModel returnValue = null;
264       // Do work - loop if .zip file case
265       boolean errorFound = false;
266       SimpleDateFormat df = new SimpleDateFormat("YYYYmmDD");
267       File f =
268           new File(System.getenv("user.home"), 
269               "importError." + df.format(new Date()) + ".txt");
270       if (f.exists()) {
271         f.delete();
272       }
273       if (this.fileName.endsWith(".zip")) {
274         FileWriter out = new FileWriter(f);
275         ZipFile zipFile = new ZipFile(new File(this.fileName));
276         int progress = 0;
277         int maxProgress = Collections.list(zipFile.entries()).size();
278         for (final ZipEntry entry : Collections.list(zipFile.entries())) {
279           if (requestCancel) {
280             break;
281           }
282           Platform.runLater(() -> {
283             statusLabel.setText("Processing " + entry.getName());
284           });
285           final int progressFinal = progress;
286           Platform.runLater(() -> {
287             progressBar.setProgress((progressFinal * 1.0) / maxProgress);
288           });
289           // Process each .zip or .uml file
290           if (entry.getName().endsWith(".xml")
291               || entry.getName().endsWith(".uml")) {
292             InputStream stream = zipFile.getInputStream(entry);
293             try {
294               returnValue = importHandler.importModel(stream);
295             } catch (Exception e) {
296               // Add case to errors file
297               if (!errorFound) {
298                 out.write("Problems importing:" + System.lineSeparator());
299               }
300               errorFound = true;
301               out.write(entry.getName() + " - " + e.getMessage()
302                   + System.lineSeparator());
303             }
304             stream.close();
305           } else {
306             Platform.runLater(() -> {
307               statusLabel.setText("Skipping" + entry.getName() + ", "
308                   + " wrong file type");
309             });
310           }
311           progress++;
312         }
313         zipFile.close();
314         out.flush();
315         out.close();
316       } else {
317         Platform.runLater(() -> {
318           statusLabel.setText("Processing " + fileName);
319         });
320         Platform.runLater(() -> {
321           progressBar.setProgress(.3);
322         });
323         returnValue = importHandler.importModel(new File(fileName));
324       }
325       if (errorFound) {
326         Platform.runLater(() -> {
327           statusLabel.setText("Error loading models, see: "
328               + f.getAbsolutePath());
329         });
330       } else {
331         Platform.runLater(() -> {
332           statusLabel.setText("done");
333         });
334       }
335       Platform.runLater(() -> {
336         progressBar.setProgress(1);
337       });
338 
339       // return value
340       return returnValue;
341     }
342 
343     /*
344      * (non-Javadoc)
345      * 
346      * @see javafx.concurrent.Task#succeeded()
347      */
348     @Override
349     protected void succeeded() {
350       // Update UI.
351       progressBar.setProgress(1);
352       cancelButton.setText("Close");
353       if (requestCancel) {
354         resultLabel.setText("Successfully cancelled imported.");
355       } else {
356         resultLabel.setText("Successfully imported model(s).");
357       }
358       requestCancel = true;
359     }
360 
361     /*
362      * (non-Javadoc)
363      * 
364      * @see javafx.concurrent.Task#failed()
365      */
366     @Override
367     protected void failed() {
368       Throwable ex = getException();
369 
370       // Update UI.
371       progressBar.setProgress(1);
372       cancelButton.setText("Close");
373       // leave last status value
374       resultLabel.setText("Failed to import model.");
375 
376       // Show dialog.
377       String title = ex.getClass().getName();
378       String msg =
379           String
380               .format("Unexpected error importing from file \"%s\"", fileName);
381       LOG.error(msg, ex);
382       AppContext.getCommonDialogs()
383           .showErrorDialog(title, msg, ex.getMessage());
384     }
385   }
