58  	EventHandler<Event> getCompTooltipEnterHandler(ComponentVersionBI comp, ComponentType type) {
59  		return new EventHandler<Event> () {
60  			@Override
61  			public void handle(Event event) {
62  				Label l = (Label)event.getSource();
63  				if (controlKeyPressed) {
64  					boolean errorFound = true;
65  					StringBuffer tpText = new StringBuffer();
66  	
67  					try {
68  						Collection<? extends RefexVersionBI<?>> annots = comp.getAnnotationsActive(WBUtility.getViewCoordinate());
69  						
70  						for (RefexVersionBI<?> annot : annots) {
71  							if (annot.getAssemblageNid() != ConceptViewerHelper.getSnomedAssemblageNid()) {
72  								try {
73  									tpText.append(getRefsetTooltip(annot));
74  										tpText.append("\n\n");
75  								} catch (Exception e) {
76  									LOG.error("Unable to access annotations", e);
77  									tpText = new StringBuffer("Unable to access annotations");
78  									errorFound = true;
79  								}
80  							}
81  						}
82  					} catch (IOException e) {
83  						e.printStackTrace();
84  					}
85  	
86  					if (!errorFound && tpText.toString().trim().isEmpty()) {
87  						l.getTooltip().setText("There are no refsets for this component");
88  					} else {
89  						l.getTooltip().setText(tpText.toString().trim());
90  					}
91  				} else {
92  					setDefaultTooltip(l, comp, type);
93  				}
94  			}
95  		};
96  	}

108 	void setDefaultTooltip(Label node, ComponentVersionBI comp, ComponentType type) {
109 		int compNid = comp.getNid();
110 		long lastCommitTime = comp.getTime();
111 		
112 		if (tooltipCache.hasLatestTooltip(compNid, lastCommitTime)) {
113 			node.setTooltip(tooltipCache.getTooltip(compNid));
114 		} else {
115 			final Tooltip tp = createComponentTooltip(type, comp);
116 			node.setTooltip(tp);
117 			tooltipCache.updateCache(compNid, lastCommitTime, tp);
118 		}
119 	}

122 	private String getRefsetTooltip(RefexVersionBI<?> annot) throws IOException, ContradictionException {
123 		int annotNid = annot.getNid();
124 		long lastCommitTime = annot.getTime();
125 		
126 		if (tooltipCache.hasLatestTooltip(annotNid, lastCommitTime)) {
127 			return tooltipCache.getRefsetTooltip(annotNid);
128 		} else {
129 			final String refsetTooltipString = createRefsetTooltip(annot);
130 			tooltipCache.updateRefsetCache(annotNid, lastCommitTime, refsetTooltipString);
131 
132 			return refsetTooltipString;
133 		}
134 	}

136 	private String createRefsetTooltip(RefexVersionBI<?> annot) throws IOException, ContradictionException {
137 		String refset = WBUtility.getConPrefTerm(annot.getAssemblageNid());
138 		StringBuffer strBuf = new StringBuffer();
139 
140 		strBuf.append(refset + " is a ");
141 		
142 		if (annot.getRefexType() == RefexType.MEMBER) {
143 			strBuf.append("Member Annotation ");
144 		} else {
145 			if (annot.getRefexType() == RefexType.CID) {
146 				strBuf.append("Component Annotation");
147 			} else if (annot.getRefexType() == RefexType.STR) {
148 				strBuf.append("String Annotation");
149 			} else if (annot.getRefexType() == RefexType.LONG) {
150 				strBuf.append("Long Annotation");
151 			}
152 
153 			strBuf.append(" with extension content:\n");
154 
155 			if (annot.getRefexType() == RefexType.CID) {
156 				int nidExt = ((RefexNidVersionBI<?>)annot).getNid1();
157 				String compExt = WBUtility.getConceptVersion(nidExt).getPreferredDescription().getText();
158 				strBuf.append("Component #1: " + compExt);
159 			} else if (annot.getRefexType() == RefexType.STR) {
160 				String strExt = ((RefexStringVersionBI<?>)annot).getString1();
161 				strBuf.append("String #1: " + strExt);
162 			} else if (annot.getRefexType() == RefexType.LONG) {
163 				String longExt = Long.toString(((RefexLongVersionBI<?>)annot).getLong1());
164 				strBuf.append("String #1: " + longExt);
165 			}
166 		}
167 		
168 		strBuf.append(getStampTooltip(annot));
169 		return strBuf.toString();
170 	}

172 	private String createDescTooltipText(DescriptionVersionBI<?> desc) {
173 		String lang = desc.getLang();
174 		String text = desc.getText();
175 		String type = WBUtility.getConPrefTerm(desc.getTypeNid());
176 		String caseSig = desc.isInitialCaseSignificant() ? "Is Case Significant" : "Not Case Significant";
177 
178 		return "Term: " + text + "\nType: " + type + "  Case Significant: " + caseSig + "  Language: " + lang + getStampTooltip(desc);

185 	private String createRelTooltipText(RelationshipVersionBI<?> rel) {
186 
187 		
188 		String refinCharType = "";
189 		try {
190 			refinCharType = RelationshipType.getRelationshipType(rel.getRefinabilityNid(), rel.getCharacteristicNid()).toString();
191 		} catch (NullPointerException npe) {
192 			LOG.error("RelationshipType.getRelationshipType() doesn't handle AdditionalRelationship.  Tracker created");
193 			refinCharType = "AdditionalRelationshipType not handled properly";
194 		} catch (Exception e) {
195 			LOG.error("Unknown error in identifying relationship type");
196 		}
197 		String group = Integer.toString(rel.getGroup());
198 		String type = WBUtility.getConPrefTerm(rel.getTypeNid());
199 		String target = WBUtility.getConPrefTerm(rel.getDestinationNid());
200 		String statInf = rel.isInferred() ? "False" : "True";
201 
202 		return "Destination: " + target + "  Type: " + type + "\nStated: " + statInf + "  Relationship Type: " + refinCharType + "  Role Group: " + group + getStampTooltip(rel);
203 	}

205 	private Tooltip createComponentTooltip(ComponentType type, ComponentVersionBI comp) {
206 		final Tooltip tp = new Tooltip();
207 		
208 		String txt;
209 		if (type == ComponentType.CONCEPT) {
210 			txt = createConTooltipText((ConceptAttributeVersionBI<?>)comp);
211 		} else if (type == ComponentType.DESCRIPTION) {
212 			txt = createDescTooltipText((DescriptionVersionBI<?>)comp);
213 		} else {
214 			txt = createRelTooltipText((RelationshipVersionBI<?>)comp);
215 		}
216 
217 		tp.setText(txt);
218 
219 		return tp;
220 	}
