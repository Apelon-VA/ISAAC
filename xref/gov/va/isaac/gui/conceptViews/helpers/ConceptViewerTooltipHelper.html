49  public class ConceptViewerTooltipHelper {
50  	private static final Logger LOG = LoggerFactory.getLogger(ConceptViewerTooltipHelper.class);
51  	private static boolean controlKeyPressed = false;
52  	private static boolean shiftKeyPressed = false;
53  	private EnhancedConceptViewerTooltipCache tooltipCache = new EnhancedConceptViewerTooltipCache();
54  
55  	EventHandler<Event> getCompTooltipEnterHandler(ComponentVersionBI comp, ComponentType type) {
56  		return new EventHandler<Event> () {
57  			@Override
58  			public void handle(Event event) {
59  				Label l = (Label)event.getSource();
60  				if (shiftKeyPressed) {
61  					boolean errorFound = true;
62  					StringBuffer tpText = new StringBuffer();
63  	
64  					try {
65  						Collection<? extends RefexVersionBI<?>> annots = comp.getAnnotationsActive(OTFUtility.getViewCoordinate());
66  						
67  						for (RefexVersionBI<?> annot : annots) {
68  							if (annot.getAssemblageNid() != ConceptViewerHelper.getSnomedAssemblageNid()) {
69  								try {
70  									tpText.append(getRefsetTooltip(annot));
71  										tpText.append("\n\n");
72  								} catch (Exception e) {
73  									LOG.error("Unable to access annotations", e);
74  									tpText = new StringBuffer("Unable to access annotations");
75  									errorFound = true;
76  								}
77  							}
78  						}
79  					} catch (IOException e) {
80  						e.printStackTrace();
81  					}
82  	
83  					if (!errorFound && tpText.toString().trim().isEmpty()) {
84  						l.getTooltip().setText("There are no refsets for this component");
85  					} else {
86  						l.getTooltip().setText(tpText.toString().trim());
87  					}
88  				} else if (controlKeyPressed){
89  					setDefaultTooltip(l, comp, type);
90  				} else {
91  					l.setTooltip(null);
92  				}
93  			}
94  		};
95  	}

97  	EventHandler getCompTooltipExitHandler(ComponentVersionBI comp, ComponentType type) {
98  		return new EventHandler<Event>() {
99  			@Override
100 			public void handle(Event event) {
101 				Label l = (Label)event.getSource();
102 				setDefaultTooltip(l, comp, type);
103 			}
104 		};
105 	}

107 	void setDefaultTooltip(Label node, ComponentVersionBI comp, ComponentType type) {
108 		int compNid = comp.getNid();
109 		long lastCommitTime = comp.getTime();
110 		
111 		if (tooltipCache.hasLatestTooltip(compNid, lastCommitTime)) {
112 			node.setTooltip(tooltipCache.getTooltip(compNid));
113 		} else {
114 			final Tooltip tp = createComponentTooltip(type, comp);
115 			node.setTooltip(tp);
116 			tooltipCache.updateCache(compNid, lastCommitTime, tp);
117 		}
118 	}

121 	private String getRefsetTooltip(RefexVersionBI<?> annot) throws IOException, ContradictionException {
122 		int annotNid = annot.getNid();
123 		long lastCommitTime = annot.getTime();
124 		
125 		if (tooltipCache.hasLatestTooltip(annotNid, lastCommitTime)) {
126 			return tooltipCache.getRefsetTooltip(annotNid);
127 		} else {
128 			final String refsetTooltipString = createRefsetTooltip(annot);
129 			tooltipCache.updateRefsetCache(annotNid, lastCommitTime, refsetTooltipString);
130 
131 			return refsetTooltipString;
132 		}
133 	}

135 	private String createRefsetTooltip(RefexVersionBI<?> annot) throws IOException, ContradictionException {
136 		String refset = OTFUtility.getConPrefTerm(annot.getAssemblageNid());
137 		StringBuffer strBuf = new StringBuffer();
138 
139 		strBuf.append(refset + " is a ");
140 		
141 		if (annot.getRefexType() == RefexType.MEMBER) {
142 			strBuf.append("Member Annotation ");
143 		} else {
144 			if (annot.getRefexType() == RefexType.CID) {
145 				strBuf.append("Component Annotation");
146 			} else if (annot.getRefexType() == RefexType.STR) {
147 				strBuf.append("String Annotation");
148 			} else if (annot.getRefexType() == RefexType.LONG) {
149 				strBuf.append("Long Annotation");
150 			}
151 
152 			strBuf.append(" with extension content:\n");
153 
154 			if (annot.getRefexType() == RefexType.CID) {
155 				int nidExt = ((RefexNidVersionBI<?>)annot).getNid1();
156 				String compExt = OTFUtility.getConceptVersion(nidExt).getPreferredDescription().getText();
157 				strBuf.append("Component #1: " + compExt);
158 			} else if (annot.getRefexType() == RefexType.STR) {
159 				String strExt = ((RefexStringVersionBI<?>)annot).getString1();
160 				strBuf.append("String #1: " + strExt);
161 			} else if (annot.getRefexType() == RefexType.LONG) {
162 				String longExt = Long.toString(((RefexLongVersionBI<?>)annot).getLong1());
163 				strBuf.append("String #1: " + longExt);
164 			}
165 		}
166 		
167 		strBuf.append(getStampTooltip(annot));
168 		return strBuf.toString();
169 	}

171 	private String createDescTooltipText(DescriptionVersionBI<?> desc) {
172 		String lang = desc.getLang();
173 		String text = desc.getText();
174 		String type = OTFUtility.getConPrefTerm(desc.getTypeNid());
175 		String caseSig = desc.isInitialCaseSignificant() ? "Is Case Significant" : "Not Case Significant";
176 
177 		return "Term: " + text + "\nType: " + type + "  Case Significant: " + caseSig + "  Language: " + lang + getStampTooltip(desc);
178 	}

180 	private String createConTooltipText(ConceptAttributeVersionBI<?> attr) {
181 		return "SctId: " + ConceptViewerHelper.getSctId(attr)+ " " + ConceptViewerHelper.getPrimDef(attr) + getStampTooltip(attr);
182 	}

184 	private String createRelTooltipText(RelationshipVersionBI<?> rel) {
185 
186 		
187 		String refinCharType = "";
188 		try {
189 			refinCharType = RelationshipType.getRelationshipType(rel.getRefinabilityNid(), rel.getCharacteristicNid()).toString();
190 		} catch (NullPointerException npe) {
191 			LOG.error("RelationshipType.getRelationshipType() doesn't handle Additional OR RF1 Relationship Types.  Tracker created");
192 			refinCharType = "Additional and RF1 relationship type not handled properly";
193 		} catch (Exception e) {
194 			LOG.error("Unknown error in identifying relationship type");
195 		}
196 		String group = Integer.toString(rel.getGroup());
197 		String type = OTFUtility.getConPrefTerm(rel.getTypeNid());
198 		String target = OTFUtility.getConPrefTerm(rel.getDestinationNid());
199 		String statInf = rel.isInferred() ? "False" : "True";
200 
201 		return "Destination: " + target + "  Type: " + type + "\nStated: " + statInf + "  Relationship Type: " + refinCharType + "  Role Group: " + group + getStampTooltip(rel);
202 	}

204 	private Tooltip createComponentTooltip(ComponentType type, ComponentVersionBI comp) {
205 		final Tooltip tp = new Tooltip();
206 		
207 		String txt;
208 		if (type == ComponentType.CONCEPT) {
209 			txt = createConTooltipText((ConceptAttributeVersionBI<?>)comp);
210 		} else if (type == ComponentType.DESCRIPTION) {
211 			txt = createDescTooltipText((DescriptionVersionBI<?>)comp);
212 		} else {
213 			txt = createRelTooltipText((RelationshipVersionBI<?>)comp);
214 		}
215 
216 		tp.setText(txt);
217 
218 		return tp;
219 	}

221 	// Control Handling Functionality
222 	public EventHandler<KeyEvent> getCtrlKeyReleasedEventHandler(){
223 		return new EventHandler<KeyEvent>() {
224 			@Override
225 			public void handle(KeyEvent event) {
226 				if (event.getCode() == KeyCode.CONTROL)
227 				{
228 					controlKeyPressed = false;
229 				} 
230 
231 				if (event.getCode() == KeyCode.SHIFT)
232 				{
233 					shiftKeyPressed = false;
234 				}
235 			}
236 		};
237 	}
238 
239 	public EventHandler<KeyEvent> getCtrlKeyPressEventHandler(){
240 		return new EventHandler<KeyEvent>() {
241 			@Override
242 			public void handle(KeyEvent event) {
243 				if (event.getCode() == KeyCode.CONTROL)
244 				{
245 					controlKeyPressed = true;
246 				} 
247 
248 				if (event.getCode() == KeyCode.SHIFT){
249 					shiftKeyPressed = true;
250 				}
251 			}
252 		};
253 	}

255 	private String getStampTooltip(ComponentVersionBI comp) {
256 		String status = OTFUtility.getStatusString(comp);
257 		String time =  OTFUtility.getTimeString(comp);
258 		String author = OTFUtility.getAuthorString(comp); 
259 		String module = OTFUtility.getModuleString(comp);
260 		String path = OTFUtility.getPathString(comp);
261 
262 		return "\nStatus: " + status + "  Time: " + time + "  Author: " + author + "  Module: " + module + "  Path: " + path;
263 	}
