141 	private void executeTermBuilder(ConceptViewMode mode) {
142 		// Descriptions
143 		try {
144 			// Sort Descriptions filtering out FSN and special storage for PT
145 			Map<Integer, Set<DescriptionVersionBI<?>>> sortedDescs = new HashMap<>();
146 			DescriptionVersionBI<?> ptDesc = null;
147 			DescriptionVersionBI<?> fsnDesc = null;
148 			
149 			for (DescriptionVersionBI<?> desc : con.getDescriptionsActive()) {
150 				if (desc.getNid() == con.getFullySpecifiedDescription().getNid()) {
151 					fsnDesc = desc;
152 				} else if (desc.getNid() == con.getPreferredDescription().getNid()) {
153 					ptDesc = desc;
154 				} else {
155 					if (!sortedDescs.containsKey(desc.getTypeNid())) {
156 						Set<DescriptionVersionBI<?>> descs = new HashSet<>();
157 						sortedDescs.put(desc.getTypeNid(), descs);
158 					}
159 
160 					sortedDescs.get(desc.getTypeNid()).add(desc);
161 				}
162 			}
163 		   	
164 			// Create GridPane
165 			tr.createGridPane();
166 
167 			if (mode != ConceptViewMode.SIMPLE_VIEW) {
168 				// Add PT Row to GridPane
169 				tr.addTermRow(fsnDesc, false);
170 			}
171 			
172 			// Add PT Row to GridPane
173 			tr.addTermRow(ptDesc, true);
174 
175 			// Add other terms to GridPane
176 			for (Integer descType: sortedDescs.keySet()) {
177 				for (DescriptionVersionBI<?> desc: sortedDescs.get(descType)) {
178 					if (desc.getNid() != con.getPreferredDescription().getNid()) {
179 						tr.addTermRow(desc, false);
180 					}
181 				}
182 			}
183 			
184 			// Add GridPane to VBox
185 			termVBox.getChildren().add(tr.getGridPane());
186 		} catch (Exception e) {
187 			LOG.error("Cannot access descriptions for concept: " + con.getPrimordialUuid());
188 		}		
189 	}

196 	private void executeRelBuilder(ConceptViewMode mode) {
197 
198 		try {
199 			rr.createGridPane();
200 			executeRelBuilderWithSpecifiedRels(con.getRelationshipsOutgoingActive());
201 			relVBox.getChildren().add(rr.getGridPane());
202 
203 			if (mode == ConceptViewMode.DETAIL_VIEW) {
204 				rr.resetCounter();
205 				executeRelBuilderWithSpecifiedRels(con.getRelationshipsIncomingActive());
206 				GridPane gp = rr.getDestinationGridPane();
207 
208 				if (!gp.getChildren().isEmpty()) {
209 					destVBox.getChildren().add(gp);
210 					destVBox.setVisible(true);
211 					destScrollPane.setVisible(true);
212 				}
213 			} else {
214 				destVBox.setVisible(false);
215 				destScrollPane.setVisible(false);
216 			}
217 		} catch (IOException | ContradictionException e) {
218 			LOG.error("Cannot access relationships for concept: " + con.getPrimordialUuid());
219 		}		
220 	}

223 	private void executeRelBuilderWithSpecifiedRels(Collection<? extends RelationshipVersionBI<?>> rels) throws ValidationException, IOException {
224 		// Capture for sorting (storing is-a in different collection
225 		Map<Integer, Set<RelationshipVersionBI<?>>> sortedRels = new HashMap<>();
226 		Set<RelationshipVersionBI<?>> isaRels = new HashSet<>();
227 		sortRels(sortedRels, isaRels, rels);
228 
229 		// Display IS-As
230 		addRels(isaRels);
231 
232 		for (Integer relType: sortedRels.keySet()) {
233 			// Display non-IS-As
234 			addRels(sortedRels.get(relType));
235 		}
236 	}

258 	private void sortRels(Map<Integer, Set<RelationshipVersionBI<?>>> sortedRels, Set<RelationshipVersionBI<?>> isaRels,
259 			Collection<? extends RelationshipVersionBI<?>> relsToSort) throws ValidationException, IOException {
260 		for (RelationshipVersionBI<?> rel : relsToSort) {
261 			if (rel.getTypeNid() == Snomed.IS_A.getNid()) {
262 				isaRels.add(rel);
263 			} else {
264 				if (!sortedRels.containsKey(rel.getTypeNid())) {
265 					Set<RelationshipVersionBI<?>> rels = new HashSet<>();
266 					sortedRels.put(rel.getTypeNid(), rels);
267 				}
268 
269 				sortedRels.get(rel.getTypeNid()).add(rel);
270 			}
271 		}
