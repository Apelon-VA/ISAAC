15  public class EnhancedConceptDynamicRefexPopup {
16  	static class Prompt extends Stage {
17  		public Prompt( String title, Stage owner, Scene scene) {
18  		    setTitle( title );
19  		    initStyle( StageStyle.UTILITY );
20  		    initModality( Modality.APPLICATION_MODAL );
21  		    initOwner( owner );
22  		    setResizable( true );
23  		    setScene( scene );
24  		}
25  		public void showDialog() {
26  		    sizeToScene();
27  		    centerOnScreen();
28  		    showAndWait();
29  		}
30  	}
31  
32  	public static void showDynamicRefexForConcept( Stage owner, String title, int conNid ) {
33          VBox vb = new VBox(10);
34          vb.setAlignment(Pos.CENTER);
35          Scene scene = new Scene( vb );
36  	    final Prompt prompt = new Prompt( title, owner, scene);
37  		
38          RefexViewI v = AppContext.getService(RefexViewI.class, "DynamicRefexView");
39          v.setComponent(conNid, null, null, null, true);
40          v.getView().setMinHeight(200.0);
41          v.getView().setMinWidth(200.0);
42  
43          VBox.setVgrow(v.getView(), Priority.ALWAYS);
44          VBox.setVgrow(v.getView(), Priority.ALWAYS);
45  
46  	    Button closeButton = new Button( "Close" );
47  	    closeButton.setPadding(new Insets(10));
48  	    closeButton.setOnAction((e) -> {
49  	            prompt.close();
50  	    } );
51  	    
52  	    vb.getChildren().addAll(v.getView(), closeButton);
53  	    prompt.showDialog();
54  	}
55  	
56  }
