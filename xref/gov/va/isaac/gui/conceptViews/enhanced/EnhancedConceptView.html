60  @Service @Named(value=SharedServiceNames.MODERN_STYLE)
61  @PerLookup
62  public class EnhancedConceptView implements PopupConceptViewI {
63  	private final Logger LOG = LoggerFactory.getLogger(this.getClass());
64  	
65  	private Stage s;
66  	private EnhancedConceptViewController controller = null;
67  
68  	private ObservableList<Integer> conceptHistoryStack = FXCollections.observableArrayList();
69  	
70  	private static int currentConNid;
71  	private static UUID currentConUuid;
72  	private static ConceptChronicleDdo currentCon;
73  	
74  	private EnhancedConceptView() throws IOException {
75  		//This is for HK2 to construct...
76  		super();
77  
78  		// Load from FXML.
79  		URL resource = this.getClass().getResource("EnhancedView.fxml");
80  		FXMLLoader loader = new FXMLLoader(resource);
81  		Region root = loader.load();
82  		root.getStylesheets().add(EnhancedConceptView.class.getResource("EnhancedView.css").toString());
83  		controller = loader.getController();
84  		controller.setConceptView(this);
85  	}
86  
87  	private void setConcept(ConceptChronicleDdo concept) {
88  		// Make sure in application thread.
89  		FxUtils.checkFxUserThread();
90  		controller.setConcept(concept.getPrimordialUuid(), controller.getViewMode(), conceptHistoryStack);
91  	}

93  	@Override
94  	public void showView(Window parent) {
95  		s = new Stage();
96  		s.initOwner(parent);
97  		s.initModality(Modality.NONE);
98  		s.initStyle(StageStyle.DECORATED);
99  
100 		s.setScene(new Scene(getView()));
101 		s.sizeToScene(); 
102 		
103 		s.getScene().getStylesheets().add(EnhancedConceptView.class.getResource("/isaac-shared-styles.css").toString());
104 		s.getIcons().add(Images.CONCEPT_VIEW.getImage());
105 
106 		// Title will change after concept is set.
107 		s.setTitle(controller.getTitle());
108 		s.show();
109 		
110 
111 		//doesn't come to the front unless you do this (on linux, at least)
112 		Platform.runLater(() -> 
113 		{
114 			s.setHeight(s.getHeight() + 30);  //coming up slightly to small on linux for some reason
115 			s.toFront();
116 		});
117 	}

119 	@Override
120 	public void setConcept(UUID conceptUUID) {
121 		// TODO this needs to be rewritten so that the dialog displays immediately
122 		//but with a progress indicator while we wait for the concept to be found..
123 		
124 		Task<ConceptChronicleDdo> task = new Task<ConceptChronicleDdo>()
125 		{
126 
127 			@Override
128 			protected ConceptChronicleDdo call() throws Exception
129 			{
130 				LOG.info("Loading concept with UUID " + conceptUUID);
131 				ConceptChronicleDdo concept = ExtendedAppContext.getDataStore().getFxConcept(conceptUUID, OTFUtility.getViewCoordinate(),
132 						VersionPolicy.ACTIVE_VERSIONS, RefexPolicy.REFEX_MEMBERS, RelationshipPolicy.ORIGINATING_AND_DESTINATION_TAXONOMY_RELATIONSHIPS);
133 				 LOG.info("Finished loading concept with UUID " + conceptUUID);
134 
135 				return concept;
136 			}
137 
138 			@Override
139 			protected void succeeded()
140 			{
141 				try
142 				{
143 					currentCon = this.getValue();
144 					setConcept(currentCon);
145 				}
146 				catch (Exception e)
147 				{
148 					String title = "Unexpected error loading concept with UUID " + conceptUUID;
149 					String msg = e.getClass().getName();
150 					LOG.error(title, e);
151 					AppContext.getCommonDialogs().showErrorDialog(title, msg, e.getMessage());
152 				}
153 			}
154 
155 			@Override
156 			protected void failed()
157 			{
158 				Throwable ex = getException();
159 				String title = "Unexpected error loading concept with UUID " + conceptUUID;
160 				String msg = ex.getClass().getName();
161 				LOG.error(title, ex);
162 				AppContext.getCommonDialogs().showErrorDialog(title, msg, ex.getMessage());
163 			}
164 		};
165 
166 		Utility.execute(task);	}
167 
168 	@Override
169 	public void setConcept(int conceptNid) {
170 		//TODO fix threading issues on this too...
171 		try
172 		{
173 			ConceptChronicleBI concept = ExtendedAppContext.getDataStore().getConcept(conceptNid);
174 			if (concept != null)
175 			{
176 				setConcept(concept.getPrimordialUuid());
177 			}
178 		}
179 		catch (IOException e)
180 		{
181 			String title = "Unexpected error loading concept with nid " + conceptNid;
182 			String msg = e.getClass().getName();
183 			LOG.error(title, e);
184 			AppContext.getCommonDialogs().showErrorDialog(title, msg, e.getMessage());
185 		}
186 	}
