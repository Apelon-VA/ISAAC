150 	public void finishInit(ComponentVersionBI comp, PopupConceptViewI callingView)
151 	{
152 		origComp = comp;
153 		conceptNid = comp.getConceptNid();
154 		callingView_ = callingView;
155 		
156 		finishInit();
157 		setupValidations();
158 		saveButton.disableProperty().bind(allValid_.not());
159 	}
160 	
161 	protected void doSave()
162 	{
163 		if (passesQA()) {
164 			addNewVersion();
165 	
166 			if (callingView_ != null)
167 			{
168 				ExtendedAppContext.getDataStore().waitTillWritesFinished();
169 				callingView_.setConcept(conceptNid);
170 			}
171 			close();
172 		}
173 	}
174 	
175 	protected void createTitleLabel(String title) {
176 		Label ltitle = new Label(title + ":");
177 		ltitle.getStyleClass().add("boldLabel");
178 		maxLengthTitleLabel = (FxUtils.calculateNecessaryWidthOfBoldLabel(ltitle) > FxUtils.calculateNecessaryWidthOfBoldLabel(maxLengthTitleLabel)) ? ltitle : maxLengthTitleLabel;
179 		gp_.add(ltitle, 0, row);		
180 	}
181 
182 	protected void createOriginalLabel(String origValue) {
183 		Label lOrig = new Label(origValue);
184 		maxLengthOriginalLabel = (FxUtils.calculateNecessaryWidthOfBoldLabel(lOrig) > FxUtils.calculateNecessaryWidthOfBoldLabel(maxLengthOriginalLabel)) ? lOrig : maxLengthOriginalLabel;
185 		gp_.add(lOrig, 1, row);		
186 		
187 		row++;
188 	}
189 
190 	protected void setupGridPaneConstraints() {
191 		ColumnConstraints cc = new ColumnConstraints();
192 		cc.setHgrow(Priority.NEVER);
193 		cc.setMinWidth(FxUtils.calculateNecessaryWidthOfBoldLabel(maxLengthTitleLabel));
194 		cc.setPrefWidth(FxUtils.calculateNecessaryWidthOfBoldLabel(maxLengthTitleLabel));
195 		gp_.getColumnConstraints().add(cc);
196 
197 		cc = new ColumnConstraints();
198 		cc.setHgrow(Priority.NEVER);
199 		cc.setMinWidth(FxUtils.calculateNecessaryWidthOfLabel(maxLengthOriginalLabel));
200 		cc.setPrefWidth(FxUtils.calculateNecessaryWidthOfLabel(maxLengthOriginalLabel));
201 		gp_.getColumnConstraints().add(cc);
202 
203 		cc = new ColumnConstraints();
204 		cc.setHgrow(Priority.ALWAYS);
205 		gp_.getColumnConstraints().add(cc);
206 	}
207 
208 	protected void setupGridPane(VBox topItems) {
209 		title = new Label(popupTitle);
210 		title.getStyleClass().add("titleLabel");
211 		title.setAlignment(Pos.CENTER);
212 		title.prefWidthProperty().bind(topItems.widthProperty());
213 		topItems.getChildren().add(title);
214 		VBox.setMargin(title, new Insets(10, 10, 10, 10));
215 
216 		gp_.setHgap(10.0);
217 		gp_.setVgap(10.0);
218 		VBox.setMargin(gp_, new Insets(5, 5, 5, 5));
219 		topItems.getChildren().add(gp_);
220 	}

223 	{
