21  import gov.va.isaac.gui.util.Images;
22  import gov.va.isaac.interfaces.gui.views.commonFunctionality.XMLViewI;

24  import gov.va.isaac.util.XMLUtils;
25  import java.util.function.Supplier;
26  import javafx.application.Platform;
27  import javafx.event.EventHandler;
28  import javafx.scene.Cursor;
29  import javafx.scene.Scene;
30  import javafx.scene.control.ProgressIndicator;
31  import javafx.scene.layout.BorderPane;
32  import javafx.scene.layout.Region;
33  import javafx.scene.web.WebView;
34  import javafx.stage.Modality;
35  import javafx.stage.Stage;
36  import javafx.stage.StageStyle;
37  import javafx.stage.Window;
38  import javafx.stage.WindowEvent;
39  import org.glassfish.hk2.api.PerLookup;
40  import org.jvnet.hk2.annotations.Service;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  
44  /**
45   * 
46   * {@link XMLDisplayWindow}
47   *
48   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
49   */
50  @Service
51  @PerLookup
52  public class XMLDisplayWindow extends Stage implements XMLViewI
53  {
54  	static Logger logger = LoggerFactory.getLogger(XMLDisplayWindow.class);
55  	private WebView wv_;
56  	ProgressIndicator pi_;
57  
58  	private XMLDisplayWindow()
59  	{
60  		// created by HK2
61  		super(StageStyle.DECORATED);
62  		initModality(Modality.NONE);
63  	}
64  	
65  	
66  
67  	/**
68  	 * @see gov.va.isaac.interfaces.gui.views.PopupViewI#showView(javafx.stage.Window)
69  	 */
70  	@Override
71  	public void showView(Window parent)
72  	{
73  		initOwner(parent);
74  		show();
75  	}
76  
77  	/**
78  	 * @see gov.va.isaac.interfaces.gui.views.commonFunctionality.XMLViewI#showView(Window, String, Supplier, int, int)
79  	 */
80  	@Override
81  	public void setParameters(String title, Supplier<String> xmlContent, int width, int height)
82  	{
83  		setWidth(width);
84  		setHeight(height);
85  		setTitle(title);
86  		getIcons().add(Images.XML_VIEW_16.getImage());
87  		getIcons().add(Images.XML_VIEW_32.getImage());
88  
89  		BorderPane bp = new BorderPane();
90  		bp.setCursor(Cursor.WAIT);
91  
92  		pi_ = new ProgressIndicator(ProgressIndicator.INDETERMINATE_PROGRESS);
93  		pi_.setMaxHeight(Region.USE_PREF_SIZE);
94  		pi_.setMaxWidth(Region.USE_PREF_SIZE);
95  		bp.setCenter(pi_);
96  
97  		wv_ = new WebView();
98  
99  		Scene scene = new Scene(bp);
100 		setScene(scene);
101 
102 		addEventHandler(WindowEvent.WINDOW_CLOSE_REQUEST, new EventHandler<WindowEvent>()
103 		{
104 			@Override
105 			public void handle(WindowEvent event)
106 			{
107 				wv_ = null;
108 			}
109 		});
110 		
111 		Utility.execute(() -> 
112 		{
113 			final StringBuilder formattedContent = new StringBuilder();
114 			
115 			try
116 			{
117 				formattedContent.append(XMLUtils.toHTML(xmlContent.get()));
118 			}
119 			catch (Exception e)
120 			{
121 				logger.error("There was an error formatting the lego as XML", e);
122 				formattedContent.append("There was an error formatting the XML for display");
123 			}
124 			
125 			Platform.runLater(() ->
126 			{
127 				WebView wv = wv_;
128 				if (wv != null)
129 				{
130 					wv_.getEngine().loadContent(formattedContent.toString());
131 					getScene().getRoot().setCursor(Cursor.DEFAULT);
132 					((BorderPane) getScene().getRoot()).setCenter(wv_);
133 					((BorderPane) getScene().getRoot()).getChildren().remove(pi_);
134 					pi_ = null;
135 				}
136 			});
137 		});
138 	}
139 }
