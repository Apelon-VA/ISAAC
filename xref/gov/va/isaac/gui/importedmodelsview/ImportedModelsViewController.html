295   /**
296    * Display imported models.
297    *
298    * @param modelType the model type
299    */
300   private void displayImportedModels(final InformationModelType modelType) {
301 
302     // Get this in scope.
303     final String modelTypeName =
304         (modelType != null ? modelType.getDisplayName() : ALL);
305 
306     // Do work in background.
307     Task<List<InformationModel>> task = new Task<List<InformationModel>>() {
308 
309       @Override
310       protected List<InformationModel> call() throws Exception {
311 
312         // Do work.
313         FetchHandler fetchHandler = new FetchHandler();
314         List<InformationModel> list = fetchHandler.fetchModels(modelType);
315         Collections.sort(list);
316         return list;
317       }
318 
319       @Override
320       protected void succeeded() {
321 
322         // Update UI.
323         List<InformationModel> informationModels = this.getValue();
324         updateUI(informationModels);
325       }
326 
327       @Override
328       protected void failed() {
329 
330         // Show dialog.
331         Throwable ex = getException();
332         String msg =
333             String.format("Unexpected error fetching models of type \"%s\"",
334                 modelTypeName);
335         LOG.error(msg, ex);
336         AppContext.getCommonDialogs().showErrorDialog(msg, ex);
337       }
338     };
339 
340     // Bind cursor to task state.
341     ObjectBinding<Cursor> cursorBinding =
342         Bindings.when(task.runningProperty()).then(Cursor.WAIT)
343             .otherwise(Cursor.DEFAULT);
344     parent.getScene().cursorProperty().bind(cursorBinding);
345 
346     // Bind progress indicator to task state.
347     lookupProgress.visibleProperty().bind(task.runningProperty());
348 
349     Thread t = new Thread(task, "InformationModelFetcher_" + modelTypeName);
350     t.setDaemon(true);
351     t.start();
352   }

369   /**
370    * Returns the model type.
371    *
372    * @param comboBoxItem the combo box item
373    * @return the model type
374    */
375   private InformationModelType getModelType(String comboBoxItem) {
376     for (InformationModelType modelType : InformationModelType.values()) {
377       String displayName = modelType.getDisplayName();
378       if (displayName.equals(comboBoxItem)) {
379         return modelType;
380       }
381     }
382 
383     // Must have been "All".
384     return null;
385   }

387   /**
388    * Gather combo box items.
389    *
390    * @return the observable list
391    */
392   private ObservableList<String> gatherComboBoxItems() {
393     ObservableList<String> items = FXCollections.observableArrayList();
394 
395     for (InformationModelType modelType : InformationModelType.values()) {
396       String displayName = modelType.getDisplayName();
397       items.add(displayName);
398     }

401 
402     return items;
403   }
