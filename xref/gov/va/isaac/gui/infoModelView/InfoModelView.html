100 	@Override
101 	public void setConcept(UUID conceptID)
102 	{
103 		if (this.conceptUUID != null)
104 		{
105 			throw new RuntimeException("Currently only supports being set once");
106 		}
107 		this.conceptUUID = conceptID;
108 		
109 		HBox h = new HBox();
110 		h.setPadding(new Insets(5, 5, 5, 5));
111 		h.getStyleClass().add("headerBackground");
112 		
113 		Label title = new Label("Clinical Element Model");
114 		h.getChildren().add(title);
115 		title.getStyleClass().add("boldLabel");
116 		HBox.setMargin(title, new Insets(4, 0, 0, 0));
117 		
118 		Label l = new Label(WBUtility.getDescription(conceptUUID));
119 		HBox.setMargin(l, new Insets(4, 0, 0, 10));
120 		h.getChildren().add(l);
121 		
122 		final ListView<String> list = new ListView<String>();
123 		
124 		configure.setOnAction(new EventHandler<ActionEvent>()
125 		{
126 			@Override
127 			public void handle(ActionEvent event)
128 			{
129 				if (popup.isShowing())
130 				{
131 					popup.hide();
132 				}
133 				else
134 				{
135 					popup.show(stage, stage.getScene().getWindow().getX() + 220, stage.getScene().getWindow().getY() + 60);
136 				}
137 				
138 			}
139 		});
140 		HBox.setMargin(configure, new Insets(0, 10, 0, 10));
141 		Region spacer = new Region();
142 		HBox.setHgrow(spacer, Priority.ALWAYS);
143 		h.getChildren().add(spacer);
144 		
145 		h.getChildren().add(activeOnly);
146 		HBox.setMargin(activeOnly, new Insets(4, 0, 0, 5));
147 		activeOnly.setOnAction(new EventHandler<ActionEvent>()
148 		{
149 			//TODO this is pretty hackish... ideally, the refset view would be monitoring a property we pass in, 
150 			//or maybe even a global property, and it would update itself.  But this should work for now...
151 			@Override
152 			public void handle(ActionEvent event)
153 			{
154 				refsetArea.getChildren().clear();
155 				refsetsOnDisplay.clear();
156 				for (String s : list.getSelectionModel().getSelectedItems())
157 				{
158 					addRefsetView(s);
159 				}
160 			}
161 		});
162 		activeOnly.setSelected(true);
163 		h.getChildren().add(configure);
164 	
165 		BorderPane bp = new BorderPane();
166 		bp.getStyleClass().add("itemBorder");
167 		bp.setTop(new Label("Select Desired Model Elements"));
168 		BorderPane.setMargin(bp.getTop(), new Insets(5, 5, 5, 5));
169 		
170 		list.getSelectionModel().setSelectionMode(SelectionMode.MULTIPLE);
171 
172 		for (ConceptSpec cs : CEMMetadataBinding.getAllRefsets())
173 		{
174 			list.getItems().add(cs.getDescription());
175 		}
176 		
177 		list.setPrefHeight(400);
178 		list.setPrefWidth(500);
179 		bp.setCenter(list);
180 		bp.setStyle("-fx-background-color: gainsboro");
181 		
182 		popup.getContent().add(bp);
183 		Button ok = new Button("Ok");
184 		ok.setOnAction(new EventHandler<ActionEvent>()
185 		{
186 			@Override
187 			public void handle(ActionEvent event)
188 			{
189 				popup.hide();
190 			}
191 		});
192 		BorderPane.setMargin(ok, new Insets(5, 5, 5, 5));
193 		BorderPane.setAlignment(ok, Pos.CENTER);
194 		bp.setBottom(ok);
195 		
196 		
197 		root.getChildren().add(h);
198 		ScrollPane sp = new ScrollPane();
199 		sp.setPrefWidth(Double.MAX_VALUE);
200 		refsetArea.setFillWidth(true);
201 		sp.setContent(refsetArea);
202 		VBox.setVgrow(sp, Priority.ALWAYS);
203 		root.getChildren().add(sp);
204 
205 		for (ConceptSpec cs : CEMMetadataBinding.getAllRefsets())
206 		{
207 			if (cs == CEMMetadataBinding.CEM_DATA_REFSET || cs == CEMMetadataBinding.CEM_TYPE_REFSET)
208 			{
209 				add(cs);
210 				list.getSelectionModel().select(cs.getDescription());
211 			}
212 		}
213 		
214 		list.getSelectionModel().getSelectedItems().addListener(new ListChangeListener<String>()
215 		{
216 			@Override
217 			public void onChanged(Change<? extends String> c)
218 			{
219 				//asking c for the remove and added sublists doesn't seem to work properly.
220 				//manually compute what needs to change.  Sigh.
221 				//TODO file a bug on JavaFX.  The old code here should have worked...
222 				HashSet<String> all = new HashSet<>();
223 				for (ConceptSpec cs : CEMMetadataBinding.getAllRefsets())
224 				{
225 					all.add(cs.getDescription());
226 				}
227 				
228 				//Add the ones that we currently aren't showing
229 				for (String s : list.getSelectionModel().getSelectedItems())
230 				{
231 					all.remove(s);
232 					if (!refsetsOnDisplay.containsKey(s))
233 					{
234 						addRefsetView(s);
235 					}
236 				}
237 				
238 				//Anything that remains in the all list should _not_ be on display
239 				for (String s : all)
240 				{
241 					removeRefsetView(s);
242 				}
243 			}
244 		});
245 	}
246 	
247 	private void addRefsetView(String name)
248 	{
249 		for (ConceptSpec cs : CEMMetadataBinding.getAllRefsets())
250 		{
251 			if (cs.getDescription().equals(name))
252 			{
253 				add(cs);
254 				break;
255 			}
256 		}
257 	}
258 	
259 	private void removeRefsetView(String name)
260 	{
261 		refsetArea.getChildren().remove(refsetsOnDisplay.remove(name));
262 	}
263 	
264 	private Region getRefsetView(ConceptSpec refset)
265 	{
266 		RefsetViewI rv = AppContext.getService(RefsetViewI.class);
267 		
268 		rv.setViewActiveOnly(activeOnly.isSelected());
269 		rv.setRefsetAndComponent(refset.getUuids()[0], conceptUUID);
270 		Region r =  rv.getView();
271 		r.getStyleClass().add("itemBorder");
272 		r.setPrefHeight(300);
273 		r.prefWidthProperty().bind(root.widthProperty().subtract(20));
274 		return r;
275 	}
276 	
277 	protected void add(ConceptSpec refset)
278 	{
279 		if (refsetsOnDisplay.keySet().contains(refset.getDescription()))
280 		{
281 			logger.error("Who called add?");
282 			return;
283 		}
284 		
285 		Region r = getRefsetView(refset);
286 		DragResizer.makeResizable(r);
287 		refsetsOnDisplay.put(refset.getDescription(), r);
288 		refsetArea.getChildren().add(r);
289 	}
