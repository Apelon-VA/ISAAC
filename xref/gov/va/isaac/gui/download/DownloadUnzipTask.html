19  package gov.va.isaac.gui.download;
20  
21  import java.io.File;
22  import java.io.FileOutputStream;
23  import java.io.InputStream;
24  import java.net.HttpURLConnection;
25  import java.net.URL;
26  import java.util.Base64;
27  import javafx.concurrent.Task;
28  import net.lingala.zip4j.core.ZipFile;
29  import net.lingala.zip4j.progress.ProgressMonitor;
30  import org.slf4j.Logger;
31  import org.slf4j.LoggerFactory;
32  
33  /**
34   * {@link DownloadUnzipTask}
35   *
36   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
37   */
38  public class DownloadUnzipTask extends Task<Integer>
39  {
40  	private static Logger log = LoggerFactory.getLogger(DownloadUnzipTask.class);
41  	String username_, password_;
42  	URL url_;
43  	private boolean cancel = false;
44  
45  	public DownloadUnzipTask(String username, String password, URL url)
46  	{
47  		this.username_ = username;
48  		this.password_ = password;
49  		this.url_ = url;
50  	}
51  
52  	/**
53  	 * @see javafx.concurrent.Task#call()
54  	 */
55  	@Override
56  	protected Integer call() throws Exception
57  	{
58  		log.debug("Beginning download from " + url_);
59  		HttpURLConnection httpCon = (HttpURLConnection) url_.openConnection();
60  		if (username_.length() > 0 || password_.length() > 0)
61  		{
62  			String encoded = Base64.getEncoder().encodeToString((username_ + ":" + password_).getBytes());
63  			httpCon.setRequestProperty("Authorization", "Basic " + encoded);
64  		}
65  		httpCon.setDoInput(true);
66  		httpCon.setRequestMethod("GET");
67  		httpCon.setConnectTimeout(30 * 1000);
68  		httpCon.setReadTimeout(60 * 60 * 1000);
69  		long fileLength = httpCon.getContentLengthLong();
70  		InputStream in = httpCon.getInputStream();
71  		
72  		File file = File.createTempFile("DBDownload", "");
73  
74  		byte[] buf = new byte[1048576];
75  		FileOutputStream fos= new FileOutputStream(file);
76  		int read = 0;
77  		long totalRead = 0;
78  		while (!cancel && (read = in.read(buf, 0, buf.length)) > 0)
79  		{
80  			totalRead += read;
81  			//update every 1 MB
82  			updateProgress(totalRead, fileLength);
83  			float percentDone = (float)((int)(((float)totalRead / (float)fileLength) * 1000)) / 10f;
84  			updateTitle("Downloading - " + percentDone + " % - out of " + fileLength + " bytes");
85  			fos.write(buf, 0, read);
86  		}
87  		fos.flush();
88  		fos.close();
89  		in.close();
90  		
91  		if (cancel)
92  		{
93  			log.debug("Download cancelled");
94  			throw new Exception("Cancelled!");
95  		}
96  		log.debug("Download complete - start unzip");
97  		
98  		updateTitle("Unzipping");
99  		try
100 		{
101 			ZipFile zipFile = new ZipFile(file);
102 			zipFile.setRunInThread(true);
103 			zipFile.extractAll(new File("").getAbsolutePath());
104 			while (zipFile.getProgressMonitor().getState() == ProgressMonitor.STATE_BUSY)
105 			{
106 				if (cancel)
107 				{
108 					zipFile.getProgressMonitor().cancelAllTasks();
109 					break;
110 				}
111 				updateProgress(zipFile.getProgressMonitor().getPercentDone(), 100);
112 				try
113 				{
114 					Thread.sleep(100);
115 				}
116 				catch (InterruptedException e)
117 				{
118 					// noop
119 				}
120 			}
121 			log.debug("Unzip complete");
122 		}
123 		catch (Exception e)
124 		{
125 			log.error("error unzipping", e);
126 			throw new Exception("The downloaded file doesn't appear to be a zip file");
127 		}
128 		finally
129 		{
130 			file.delete();
131 		}
132 		
133 		return 0;
134 	}
135 
136 	/**
137 	 * @see javafx.concurrent.Task#cancel(boolean)
138 	 */
139 	@Override
140 	public boolean cancel(boolean mayInterruptIfRunning)
141 	{
142 		super.cancel(mayInterruptIfRunning);
143 		cancel = true;
144 		return true;
145 	}
146 
147 }
