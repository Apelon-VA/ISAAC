45  /**
46   * 
47   * {@link DownloaderController}
48   *
49   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
50   */
51  public class DownloaderController
52  {
53  	private static Logger log = LoggerFactory.getLogger(DownloaderController.class);
54  
55  	@FXML private ResourceBundle resources;
56  	@FXML private URL location;
57  	@FXML private Button cancel;
58  	@FXML private Button download;
59  	@FXML private PasswordField password;
60  	@FXML private Label statusLabel;
61  	@FXML private ProgressBar progressBar;
62  	@FXML private ComboBox<String> url;
63  	@FXML private TextField username;
64  	@FXML private GridPane paramGridPane;
65  	@FXML private Hyperlink link;
66  	
67  	Consumer<Boolean> callOnCompletion_;
68  	Stage stage_;
69  	ValidBooleanBinding urlValid_;
70  	DownloadUnzipTask dut;
71  
72  	@FXML
73  	void initialize()
74  	{
75  		assert cancel != null : "fx:id=\"cancel\" was not injected: check your FXML file 'Downloader.fxml'.";
76  		assert download != null : "fx:id=\"download\" was not injected: check your FXML file 'Downloader.fxml'.";
77  		assert password != null : "fx:id=\"password\" was not injected: check your FXML file 'Downloader.fxml'.";
78  		assert statusLabel != null : "fx:id=\"statusLabel\" was not injected: check your FXML file 'Downloader.fxml'.";
79  		assert progressBar != null : "fx:id=\"progressBar\" was not injected: check your FXML file 'Downloader.fxml'.";
80  		assert url != null : "fx:id=\"url\" was not injected: check your FXML file 'Downloader.fxml'.";
81  		assert username != null : "fx:id=\"username\" was not injected: check your FXML file 'Downloader.fxml'.";
82  		
83  		cancel.setOnAction((event) -> 
84  		{
85  			if (dut != null)
86  			{
87  				dut.cancel();
88  			}
89  			else
90  			{
91  				stage_.hide();
92  				callOnCompletion_.accept(false);
93  			}
94  		});
95  		
96  		download.setOnAction((event) -> 
97  		{
98  			download.disableProperty().unbind();
99  			download.setDisable(true);
100 			url.setDisable(true);
101 			username.setDisable(true);
102 			password.setDisable(true);
103 			progressBar.setProgress(-1);
104 			statusLabel.setText("Downloading...");
105 			download();
106 		});
107 		
108 		urlValid_ = new ValidBooleanBinding()
109 		{
110 			{
111 				bind(url.getEditor().textProperty());
112 				setComputeOnInvalidate(true);
113 			}
114 			@Override
115 			protected boolean computeValue()
116 			{
117 				if (url.getEditor().getText().length() == 0)
118 				{
119 					setInvalidReason("The URL is required");
120 					return false;
121 				}
122 				else if (!url.getEditor().getText().trim().toLowerCase().matches("https{0,1}://.*"))
123 				{
124 					setInvalidReason("The URL must start with http:// or https://");
125 					return false;
126 				}
127 				
128 				try 
129 				{
130 					new URL(url.getEditor().getText());
131 					clearInvalidReason();
132 					return true;
133 				}
134 				catch (Exception e)
135 				{
136 					setInvalidReason("The entered text is not a valid URL.");
137 					return false;
138 				}
139 			}
140 		};
141 		
142 		download.disableProperty().bind(urlValid_.not());
143 		
144 		StackPane sp = new StackPane();
145 		ErrorMarkerUtils.swapGridPaneComponents(url, sp, paramGridPane);
146 		ErrorMarkerUtils.setupErrorMarker(url, sp, urlValid_.getReasonWhyInvalid());
147 		
148 		url.getItems().add("https://va.maestrodev.com/archiva/repository/data-files/gov/va/isaac/db/solor-snomed/2014.11.12/solor-snomed-2014.11.12-active-only.bdb.zip");
149 		url.getItems().add("https://va.maestrodev.com/archiva/repository/data-files/gov/va/isaac/db/solor-snomed-loinc/2014.11.12/solor-snomed-loinc-2014.11.12-active-only.bdb.zip");
150 		url.getItems().add("https://va.maestrodev.com/archiva/repository/data-files/gov/va/isaac/db/solor-all/2014.11.12/solor-all-2014.11.12-active-only.bdb.zip");
151 		
152 		link.setOnAction((event) ->
153 		{
154 			AppContext.getMainApplicationWindow().browseURL(
155 					"https://csfe.aceworkspace.net/sf/frs/do/viewRelease/projects.veterans_administration_project/frs.isaac.isaac_databases");
156 		});
157 	}
158 	
159 	protected void setStage(Stage stage)
160 	{
161 		stage_ = stage;
162 	}
163 	
164 	protected void callOnCompletion(Consumer<Boolean> notifyOfResult)
165 	{
166 		callOnCompletion_ = notifyOfResult;
167 	}
168 	
169 	private void download()
170 	{
171 		try
172 		{
173 			dut = new DownloadUnzipTask(username.getText(), password.getText(), new URL(url.getEditor().getText()));
174 			dut.setOnSucceeded((event) -> 
175 			{
176 				taskFinished(dut);
177 			});
178 			
179 			dut.setOnFailed((event) -> 
180 			{
181 				taskFinished(dut);
182 			});
183 			dut.setOnCancelled((event) -> 
184 			{
185 				taskFinished(dut);
186 			});
187 			progressBar.progressProperty().bind(dut.progressProperty());
188 			statusLabel.textProperty().bind(dut.titleProperty());
189 			Utility.execute(dut);
190 		}
191 		catch (Exception e)
192 		{
193 			log.error("Unexpected error", e);
194 			AppContext.getCommonDialogs().showErrorDialog("Error during download", e.getClass().getName(), e.getMessage(), stage_);
195 		}
196 	}
197 	
198 	private void taskFinished(Task<?> task)
199 	{
200 		statusLabel.textProperty().unbind();
201 		progressBar.progressProperty().unbind();
202 		
203 		if (task.isCancelled() || task.getException() != null)
204 		{
205 			Platform.runLater(() ->
206 			{
207 				AppContext.getCommonDialogs().showErrorDialog("Error during download", "Error during download", 
208 						(task.isCancelled() ? "The download was cancelled" : task.getException().getMessage()));
209 				download.setDisable(false);
210 				download.disableProperty().bind(urlValid_.not());
211 				url.setDisable(false);
212 				username.setDisable(false);
213 				password.setDisable(false);
214 				progressBar.setProgress(0);
215 				statusLabel.setText("Please Enter Download Information");
216 			});
217 		}
218 		else
219 		{
220 			stage_.hide();
221 			callOnCompletion_.accept(true);
222 		}
223 		dut = null;
224 	}
225 }
