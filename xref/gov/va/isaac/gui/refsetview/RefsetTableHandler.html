111 	private void setColSizes(TableView<RefsetInstance> refsetRows)
112 	{
113 		//Horrible hack to move the stamp column to the end
114 		//TODO remove this hack after code gets refactored to combine these various init methods
115 		for (int i = 0; i < refsetRows.getColumns().size(); i++)
116 		{
117 			if (refsetRows.getColumns().get(i).getText().equals("STAMP"))
118 			{
119 				refsetRows.getColumns().add(refsetRows.getColumns().remove(i));
120 				break;
121 			}
122 		}
123 		
124 		//Horrible hack to set a reasonable default size on the columns.
125 		//Min width to the with of the header column.  Preferred width - divide space out evenly.
126 		Font f = new Font("System Bold", 13.0);
127 		float prefColWidthPercentage = 1.0f / (float) refsetRows.getColumns().size();
128 		for (TableColumn<RefsetInstance, ?> col : refsetRows.getColumns())
129 		{
130 			for (TableColumn<RefsetInstance, ?> nestedCol : col.getColumns())
131 			{
132 				nestedCol.setMinWidth(Toolkit.getToolkit().getFontLoader().computeStringWidth(nestedCol.getText(), f) + 20);
133 				nestedCol.prefWidthProperty().bind(refsetRows.widthProperty().subtract(5.0).multiply(prefColWidthPercentage).divide(col.getColumns().size()));
134 			}
135 			col.setMinWidth(Toolkit.getToolkit().getFontLoader().computeStringWidth(col.getText(), f) + 20);
136 			col.prefWidthProperty().bind(refsetRows.widthProperty().subtract(5.0).multiply(prefColWidthPercentage));
137 		}
138 	}

276 	private TableColumn<RefsetInstance, String> createCidColumn(String columnTitle, String associatedValue, final int columnNumber) {
277 		TableColumn<RefsetInstance, String> col = new TableColumn<>(columnTitle);	
278 		col.setCellValueFactory(new PropertyValueFactory<RefsetInstance,String>(associatedValue));
279 		col.setCellFactory(RefsetTableEditingCell.create());
280 
281 		col.setOnEditCommit(new EventHandler<CellEditEvent<RefsetInstance, String>>() {
282 			@Override
283 			public void handle(CellEditEvent<RefsetInstance, String> t) {
284 				try
285 				{
286 					if (!Utility.isUUID(t.getNewValue()))
287 					{
288 						AppContext.getCommonDialogs().showErrorDialog("Invalid UUID", "Value requested is not a valid UUID", t.getNewValue());
289 						return;
290 					}
291 	
292 					ConceptVersionBI comp = OTFUtility.lookupIdentifier(t.getNewValue());
293 					if (comp == null) {
294 						AppContext.getCommonDialogs().showErrorDialog("UUID Not Found", "Could not find the UUID in the database", t.getNewValue());
295 					} else {
296 						RefexCAB bp = null;
297 						
298 						if (refsetType == RefexType.CID ||
299 							refsetType == RefexType.CID_STR ||
300 							refsetType == RefexType.UNKNOWN)
301 						{
302 							if (columnNumber == 1) {
303 								NidExtRefsetInstance instance = (NidExtRefsetInstance) t.getTableView().getItems().get(t.getTablePosition().getRow());
304 								if (OTFUtility.getRefsetMember(instance.getMemberNid()) == null) {
305 									RefexCAB newMember = new RefexCAB(RefexType.CID, instance.getRefCompConNid(), refsetNid, IdDirective.GENERATE_RANDOM, RefexDirective.EXCLUDE);
306 
307 									newMember.put(ComponentProperty.COMPONENT_EXTENSION_1_ID, comp.getPrimordialUuid());
308 									
309 									RefexChronicleBI<?> newMemChron = OTFUtility.getBuilder().construct(newMember);
310 									instance.setMemberNid(newMemChron.getNid());
311 									
312 									
313 									ConceptVersionBI refCompCon;
314 									if (!isAnnotation) {
315 										refCompCon = OTFUtility.getConceptVersion(instance.getRefCompConNid());
316 									} else {
317 										refCompCon = OTFUtility.getConceptVersion(refsetNid);
318 									}
319 									refCompCon.addAnnotation(newMemChron);
320 									
321 									OTFUtility.addUncommitted(instance.getRefCompConNid());
322 									rvc_.reloadData();
323 									return;
324 								} else {
325 									instance.setCidExtFsn(comp.getFullySpecifiedDescription().getText());
326 									instance.setCidExtUuid(comp.getPrimordialUuid());
327 	
328 									bp = createBlueprint(instance.getMemberNid());
329 									bp.put(ComponentProperty.COMPONENT_EXTENSION_1_ID, comp.getPrimordialUuid());
330 								}
331 							}
332 						}
333 
334 						addUncommitted(bp, isAnnotation);
335 						rvc_.reloadData();
336 					}
337 				}
338 				catch (Exception e)
339 				{
340 					logger.error("Unexpected error handling edit", e);
341 				}
342 			}
343 		});
344 
345 		return col;
346 	}
