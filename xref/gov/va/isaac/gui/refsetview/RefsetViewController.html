213 	private void handleComplexRefset(RefexVersionBI<?> member, RefexVersionBI<?> previousMember, ConceptVersionBI refCompCon) {
214 		if (!rth_.isSetupFinished() && member.getRefexType() != RefexType.MEMBER) {
215 			rth_.finishTableSetup(member, isAnnotation, refsetRows, refCompCon, member.getAssemblageNid());
216 			refsetType = member.getRefexType();
217 		}
218 		
219 		// Have needed member, add to data
220 		CEMCompositRefestInstance instance = (CEMCompositRefestInstance)RefsetInstanceAccessor.getInstance(refCompCon, member, previousMember, RefexType.UNKNOWN);
221 		
222 		data.add(instance);
223 	}

225 	private void processMembers(RefexVersionBI<?> member, RefexVersionBI<?> previousMember, ConceptVersionBI refCompCon) throws IOException, ContradictionException {
226 		if (member.getAssemblageNid() == refsetNid_) {
227 			// Setup if Necessary
228 			//TODO (artf231876) The current code only works if every member has the same RefexType (and there is _no_ guarantee about that in the APIs.  
229 			//The entire column display of the tables needs to be reworked, as the columns that are displayed needs to be dynamically detected 
230 			//from the data in the table, so it can take into account multiple refex types.
231 			if (!rth_.isSetupFinished() && member.getRefexType() != RefexType.MEMBER) {
232 				rth_.finishTableSetup(member, isAnnotation, refsetRows, refCompCon, member.getAssemblageNid());
233 				refsetType = member.getRefexType();
234 			}
235 
236 			// Have needed member, add to data
237 			RefsetInstance instance = RefsetInstanceAccessor.getInstance(refCompCon, member, previousMember, refsetType);
238 			data.add(instance);
239 		}
240 
241 		// Search for member's annotations
242 		Collection<? extends RefexChronicleBI<?>> refAnnots = (activeOnly_ ? member.getAnnotationsActive(OTFUtility.getViewCoordinate()) : member.getAnnotations());
243 		for (RefexChronicleBI<?> annot : refAnnots) {
244 			List<RefexVersionBI<?>> annotVersions = new ArrayList<>();
245 			if (activeOnly_)
246 			{
247 				RefexVersionBI<?> version = (RefexVersionBI<?>) annot.getVersion(OTFUtility.getViewCoordinate());
248 				if (version.isActive()) {
249 					annotVersions.add(version);
250 				}
251 			}
252 			else {
253 				annotVersions.addAll((Collection<? extends RefexVersionBI<?>>) annot.getVersions());
254 			}
255 			
256 			RefexVersionBI<?> prevAnnotVersion = null;
257 			for (RefexVersionBI<?> annotVersion : annotVersions)
258 			{
259 				processMembers(annotVersion, prevAnnotVersion, refCompCon);
260 				
261 				prevAnnotVersion = annotVersion;
262 			}
263 		}
264 	}
