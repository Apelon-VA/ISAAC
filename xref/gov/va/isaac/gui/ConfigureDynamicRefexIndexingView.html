52  /**
53   * {@link ConfigureDynamicRefexIndexingView}
54   *
55   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
56   */
57  public class ConfigureDynamicRefexIndexingView implements PopupViewI
58  {
59  	ConceptVersionBI assemblageConcept_;
60  	private BorderPane root_;
61  	private CheckBox indexMember_;
62  	private List<CheckBox> indexColumns_;
63  	private Logger logger_ = LoggerFactory.getLogger(this.getClass());
64  
65  	public ConfigureDynamicRefexIndexingView(ConceptVersionBI assemblageConcept)
66  	{
67  		try
68  		{
69  			assemblageConcept_ = assemblageConcept;
70  			root_ = new BorderPane();
71  			root_.setPrefWidth(450);
72  			
73  			VBox titleBox = new VBox();
74  			
75  			Label title = new Label("Dynamic Sememe Index Configuration");
76  			title.getStyleClass().add("titleLabel");
77  			title.setAlignment(Pos.CENTER);
78  			title.setMaxWidth(Double.MAX_VALUE);
79  			titleBox.getChildren().add(title);
80  			
81  			title = new Label(OTFUtility.getDescription(assemblageConcept));
82  			title.setAlignment(Pos.CENTER);
83  			title.setMaxWidth(Double.MAX_VALUE);
84  			titleBox.getChildren().add(title);
85  			
86  			titleBox.getStyleClass().add("headerBackground");
87  			titleBox.setPadding(new Insets(5, 5, 5, 5));
88  			root_.setTop(titleBox);
89  			
90  			indexColumns_ = new ArrayList<>();
91  			
92  			VBox vbox = new VBox();
93  			vbox.setPadding(new Insets(10));
94  			vbox.setSpacing(5.0);
95  			
96  			RefexDynamicUsageDescription rdud = RefexDynamicUsageDescription.read(assemblageConcept.getNid());
97  			Integer[] currentIndexConfig = LuceneDynamicRefexIndexerConfiguration.readIndexInfo(assemblageConcept.getNid());
98  			
99  			if (rdud.isAnnotationStyle())
100 			{
101 				indexMember_ = new CheckBox("Index sememe members");
102 				indexMember_.setTooltip(new Tooltip("Allows queries to find all components that have an instance of this sememe.\n"
103 						+ "Only applicable to Annotation-style sememes"));
104 				if (currentIndexConfig != null)
105 				{
106 					indexMember_.setSelected(true);
107 				}
108 				vbox.getChildren().add(indexMember_);
109 			}
110 			else if (rdud.getColumnInfo().length == 0)
111 			{
112 				vbox.getChildren().add(new Label("Member-style Sememes without columns are not indexable"));
113 			}
114 			
115 			for (RefexDynamicColumnInfo col : rdud.getColumnInfo())
116 			{
117 				CheckBox cb = new CheckBox("Index Attribute: " + col.getColumnName());
118 				cb.setTooltip(new Tooltip("Attribute Type: " + col.getColumnDataType().getDisplayName() + "\nAttribute Description: " + col.getColumnDescription()));
119 				
120 				
121 				if (!LuceneDynamicRefexIndexerConfiguration.isColumnTypeIndexable(col.getColumnDataType()))
122 				{
123 					//disabled check boxes don't show tooltips...
124 					StackPane silly = new StackPane();
125 					silly.getChildren().add(cb);
126 					cb.setDisable(true);
127 					cb.setMaxWidth(Double.MAX_VALUE);
128 					Tooltip.install(silly, new Tooltip("The datatype for this attribute isn't indexable"));
129 					vbox.getChildren().add(silly);
130 				}
131 				else
132 				{
133 					vbox.getChildren().add(cb);
134 				}
135 				indexColumns_.add(cb);
136 			}
137 			
138 			if (currentIndexConfig != null)
139 			{
140 				for (int i : currentIndexConfig)
141 				{
142 					indexColumns_.get(i).setSelected(true);
143 				}
144 			}
145 			
146 			ScrollPane sp = new ScrollPane();
147 			sp.setContent(vbox);
148 			root_.setCenter(sp);
149 			
150 			HBox buttons = new HBox();
151 			buttons.setMaxWidth(Double.MAX_VALUE);
152 			buttons.setAlignment(Pos.CENTER);
153 			buttons.setPadding(new Insets(5));
154 			buttons.setSpacing(30);
155 
156 			Button cancel = new Button("Cancel");
157 			cancel.setOnAction((action) ->
158 			{
159 				cancel.getScene().getWindow().hide();
160 			});
161 			buttons.getChildren().add(cancel);
162 
163 			Button ok = new Button("Ok");
164 			buttons.getChildren().add(ok);
165 			ok.minWidthProperty().bind(cancel.widthProperty());
166 			ok.setOnAction((action) ->
167 			{
168 				try
169 				{
170 					ArrayList<Integer> colsChecked = new ArrayList<>();
171 					for (int i = 0; i < indexColumns_.size(); i++)
172 					{
173 						if (indexColumns_.get(i).isSelected())
174 						{
175 							colsChecked.add(i);
176 						}
177 					}
178 					Task<Void> t = new Task<Void>()
179 					{
180 						@Override
181 						protected Void call() throws Exception
182 						{
183 							if (colsChecked.size() > 0 || (indexMember_ != null && indexMember_.isSelected()))
184 							{
185 								Integer[] toIndex = colsChecked.toArray(new Integer[colsChecked.size()]);
186 								if (Arrays.deepEquals(toIndex, LuceneDynamicRefexIndexerConfiguration.readIndexInfo(assemblageConcept_.getNid())))
187 								{
188 									logger_.info("Skipping reindex - no change detected.");
189 								}
190 								else
191 								{
192 									AppContext.getRuntimeGlobals().disableAllCommitListeners();
193 									try
194 									{
195 										LuceneDynamicRefexIndexerConfiguration.configureColumnsToIndex(assemblageConcept_.getNid(), toIndex);
196 									}
197 									finally
198 									{
199 										AppContext.getRuntimeGlobals().enableAllCommitListeners();
200 									}
201 								}
202 							}
203 							else
204 							{
205 								AppContext.getRuntimeGlobals().disableAllCommitListeners();
206 								try
207 								{
208 									LuceneDynamicRefexIndexerConfiguration.disableIndex(assemblageConcept_.getNid());
209 								}
210 								finally
211 								{
212 									AppContext.getRuntimeGlobals().enableAllCommitListeners();
213 								}
214 							}
215 							return null;
216 						}
217 
218 						/**
219 						 * @see javafx.concurrent.Task#failed()
220 						 */
221 						@Override
222 						protected void failed()
223 						{
224 							AppContext.getCommonDialogs().showErrorDialog("Unexpected Error updating index configuration", getException());
225 						}
226 						
227 					};
228 					
229 					Utility.submit(t);
230 					AppContext.getCommonDialogs().showInformationDialog("Reindex running", "An index operation has begun in the background.\nIt may take some time to complete.", 
231 							root_.getScene().getWindow());
232 				}
233 				catch (Exception e)
234 				{
235 					logger_.error("Unexpected Error updating index configuration", e);
236 					AppContext.getCommonDialogs().showErrorDialog("Unexpected Error updating index configuration", e);
237 				}
238 				ok.getScene().getWindow().hide();
239 			});
240 			
241 			root_.setBottom(buttons);
242 			
243 		}
244 		catch (Exception e)
245 		{
246 			logger_.error("Unexpected Error reading index configuration", e);
247 			AppContext.getCommonDialogs().showErrorDialog("Unexpected Error reading index configuration", e);
248 		}
249 	}
250

252 	/**
253 	 * @see gov.va.isaac.interfaces.gui.views.PopupViewI#showView(javafx.stage.Window)
254 	 */
255 	@Override
256 	public void showView(Window parent)
257 	{
258 		Stage stage = new Stage(StageStyle.DECORATED);
259 		stage.initModality(Modality.NONE);
260 		stage.initOwner(parent);
261 		Scene scene = new Scene(root_);
262 		stage.setScene(scene);
263 		stage.setTitle("Configure Dynamic Sememe Indexing");
264 		stage.getScene().getStylesheets().add(ConfigureDynamicRefexIndexingView.class.getResource("/isaac-shared-styles.css").toString());
265 		stage.sizeToScene();
266 		stage.show();
267 	}
