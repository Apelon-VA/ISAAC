66      @Override
67      public void start(Stage primaryStage) throws Exception {
68          primaryStage_ = primaryStage;
69          ServiceLocatorUtilities.addOneConstant(AppContext.getServiceLocator(), this);
70          //Set up the CommonDialogs class (which needs a references to primaryStage_ and gets it via injection)
71          commonDialog_ = AppContext.getServiceLocator().getService(CommonDialogs.class);
72  
73          URL resource = this.getClass().getResource("App.fxml");
74          FXMLLoader loader = new FXMLLoader(resource);
75          Parent root = (Parent) loader.load();
76          this.controller = loader.getController();
77  
78          primaryStage.getIcons().add(new Image("/icons/16x16/application-block.png"));
79          primaryStage.setTitle("ISAAC App");
80          primaryStage.setScene(new Scene(root));
81  
82          // Set minimum dimensions.
83          primaryStage.setMinHeight(400);
84          primaryStage.setMinWidth(400);
85  
86          // Handle window close event.
87          primaryStage.setOnHiding(new EventHandler<WindowEvent>() {
88  
89              @Override
90              public void handle(WindowEvent event) {
91                  shutdown();
92              }
93          });
94  
95          primaryStage.show();
96  
97          // Reduce size to fit in user's screen.
98          // (Need to do after stage is shown, because otherwise
99          // the primary stage width & height are NaN.)
100         Screen screen = Screen.getPrimary();
101         double screenW = screen.getVisualBounds().getWidth();
102         double screenH = screen.getVisualBounds().getHeight();
103         if (primaryStage.getWidth() > screenW) {
104             LOG.debug("Resizing width to " + screenW);
105             primaryStage.setWidth(screenW);
106         }
107         if (primaryStage.getHeight() > screenH) {
108             LOG.debug("Resizing height to " + screenH);
109             primaryStage.setHeight(screenH);
110         }
111 
112         // Kick off a thread to open the DB connection.
113         loadDataStore();
114     }

116     private void loadDataStore() {
117 
118         // Do work in background.
119         Task<BdbTerminologyStore> task = new Task<BdbTerminologyStore>() {
120 
121             @Override
122             protected BdbTerminologyStore call() throws Exception {
123                 LOG.info("Opening Workbench database");
124                 BdbTerminologyStore dataStore = AppContext.getServiceLocator().getService(BdbTerminologyStore.class);
125                 LOG.info("Finished opening Workbench database");
126 
127                 // Check if user shut down early.
128                 if (shutdown) {
129                     dataStore.shutdown();
130                     return null;
131                 }
132 
133                 return dataStore;
134             }
135 
136             @Override
137             protected void succeeded() {
138                 controller.finishInit();
139             }
140 
141             @Override
142             protected void failed() {
143                 Throwable ex = getException();
144 
145                 // Display helpful dialog to users.
146                 if (ex instanceof FileNotFoundException) {
147                     String title = "No Snomed Database";
148                     String message = "The Snomed Database was not found.";
149                     LOG.error(message, ex);
150                     String details = "Please download the file\n\n"
151                             + "https://mgr.servers.aceworkspace.net/apps/va-archiva/repository/all/org/ihtsdo/otf/tcc-test-data/3.0/tcc-test-data-3.0.zip"
152                             + "\n\nand unzip it into\n\n"
153                             + System.getProperty("user.dir")
154                             + "\n\nand then restart the editor.";
155                     commonDialog_.showErrorDialog(title, message, details);
156 
157                     // Close app since no DB to load.
158                     // (The #shutdown method will be also invoked by
159                     // the handler we hooked up with Stage#setOnHiding.)
160                     primaryStage_.hide();
161 
162                 } else {
163                     String title = "Unexpected error connecting to workbench database";
164                     String msg = ex.getClass().getName();
165                     String details = ex.getMessage();
166                     LOG.error(title, ex);
167                     commonDialog_.showErrorDialog(title, msg, details);
168                 }
169             }
170         };
171 
172         Thread t = new Thread(task, "SCT_DB_Open");
173         t.setDaemon(true);
174         t.start();
175     }

177     private static void configDataStorePaths(File bdbFolder) throws IOException {
178 
179         // Default value if null.
180         if (bdbFolder == null) {
181             //do nothing... let the store use its own default
182             //TODO OTF fix: note - the BDB defaults are not in sync with the Lucene defaults...  lucene will init in the wrong place and fail....
183         }
184         else
185         {
186             
187             if (!bdbFolder.exists())
188             {
189                 throw new FileNotFoundException("Couldn't find specified bdb data store '" + bdbFolder.getAbsolutePath() + "'");
190             }
191             if (!bdbFolder.isDirectory())
192             {
193                 throw new IOException("The specified bdb data store: '" + bdbFolder.getAbsolutePath() + "' is not a folder");
194             }
195             
196             if (System.getProperty(BdbTerminologyStore.BDB_LOCATION_PROPERTY) == null)
197             {
198                 System.setProperty(BdbTerminologyStore.BDB_LOCATION_PROPERTY, bdbFolder.getCanonicalPath());
199             }
200             else
201             {
202                 LOG.warn("The application specified '" + bdbFolder.getCanonicalPath() + "' but the system property " + BdbTerminologyStore.BDB_LOCATION_PROPERTY 
203                         + "is set to " + System.getProperty(BdbTerminologyStore.BDB_LOCATION_PROPERTY + " this will override the application path"));
204             }
205             if (System.getProperty(LuceneIndexer.LUCENE_ROOT_LOCATION_PROPERTY) == null)
206             {
207                 System.setProperty(LuceneIndexer.LUCENE_ROOT_LOCATION_PROPERTY, bdbFolder.getCanonicalPath());
208             }
209             else
210             {
211                 LOG.warn("The application specified '" + bdbFolder.getCanonicalPath() + "' but the system property " + LuceneIndexer.LUCENE_ROOT_LOCATION_PROPERTY 
212                         + "is set to " + System.getProperty(LuceneIndexer.LUCENE_ROOT_LOCATION_PROPERTY) + " this will override the application path");
213             }
214             
215         }
216     }

218     private void shutdown() {
219         LOG.info("Shutting down");
220         shutdown = true;
221         try {
222             ExtendedAppContext.getDataStore().shutdown();
223             for (ShutdownBroadcastListenerI s : shutdownListeners_)
224             {
225                 if (s != null)
226                 {
227                     s.shutdown();
228                 }
229             }
230         } catch (Exception ex) {
231             String message = "Trouble shutting down";
232             LOG.warn(message, ex);
233             commonDialog_.showErrorDialog("Oops!", message, ex.getMessage());
234         }
235 
236         LOG.info("Finished shutting down");
237     }
