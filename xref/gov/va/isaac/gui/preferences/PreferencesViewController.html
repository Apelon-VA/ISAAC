71  public class PreferencesViewController {
72  	private Logger logger = LoggerFactory.getLogger(PreferencesViewController.class);
73  
74  	@Inject
75  	private IterableProvider<PreferencesPluginViewI> plugins_;
76  
77  	private @FXML TabPane tabPane_;
78  
79  	private @FXML Button okButton_;
80  	private @FXML Button cancelButton_;
81  	
82  	private PreferencesView stage_;
83  	
84  	private ValidBooleanBinding allValid_ = null;
85  	
86  	public PreferencesViewController() {
87  		AppContext.getServiceLocator().inject(this);
88  	}
89  
90  	@FXML
91  	void initialize()
92  	{
93  		assert tabPane_ != null : "fx:id=\"tabPane\" was not injected: check your FXML file 'PreferencesView.fxml'.";
94  		assert okButton_ != null : "fx:id=\"okButton\" was not injected: check your FXML file 'PreferencesView.fxml'.";
95  		assert cancelButton_ != null : "fx:id=\"cancelButton_\" was not injected: check your FXML file 'PreferencesView.fxml'.";
96  
97  		tabPane_.setTabClosingPolicy(TabClosingPolicy.UNAVAILABLE);
98  		tabPane_.setMaxWidth(Double.MAX_VALUE);
99  
100 		okButton_.setOnAction((e) -> saveAndExitIfSuccessful());
101 
102 		cancelButton_.setOnAction((e) -> stage_.close());
103 	}
104 
105 	void setStage(PreferencesView stage) {
106 		this.stage_ = stage;
107 	}
108 	
109 	public void aboutToShow()
110 	{
111 		// Using allValid_ to prevent rerunning content of aboutToShow()
112 		if (allValid_ == null) {
113 			// These listeners are for debug and testing only. They may be removed at any time.
114 			UserProfileManager  userProfileManager = AppContext.getService(UserProfileManager.class);
115 			for (UserProfileProperty property : UserProfileProperty.values()) {
116 				userProfileManager.addUserProfilePropertyChangeListener(property, new PropertyChangeListener() {
117 					@Override
118 					public void propertyChange(PropertyChangeEvent evt) {
119 						logger.debug("{} property changed from {} to {}", evt.getPropertyName(), evt.getOldValue(), evt.getNewValue());
120 					}
121 				});
122 			}
123 
124 			// load fields before initializing allValid_
125 			// in case plugin.validationFailureMessageProperty() initialized by getNode()
126 			tabPane_.getTabs().clear();
127 			List<PreferencesPluginViewI> sortableList = new ArrayList<>();
128 			Comparator<PreferencesPluginViewI> comparator = new Comparator<PreferencesPluginViewI>() {
129 				@Override
130 				public int compare(PreferencesPluginViewI o1, PreferencesPluginViewI o2) {
131 					if (o1.getTabOrder() == o2.getTabOrder()) {
132 						return o1.getName().compareTo(o2.getName());
133 					} else {
134 						return o1.getTabOrder() - o2.getTabOrder();
135 					}
136 				}
137 			};
138 			for (PreferencesPluginViewI plugin : plugins_) {
139 				sortableList.add(plugin);
140 			}
141 			Collections.sort(sortableList, comparator);
142 			for (PreferencesPluginViewI plugin : sortableList) {
143 				logger.debug("Adding PreferencesPluginView tab \"{}\"", plugin.getName());
144 				Label tabLabel = new Label(plugin.getName());
145 				
146 				tabLabel.setMaxHeight(Double.MAX_VALUE);
147 				tabLabel.setMaxWidth(Double.MAX_VALUE);
148 				Tab pluginTab = new Tab();
149 				pluginTab.setGraphic(tabLabel);
150 				Region content = plugin.getContent();
151 				content.setMaxWidth(Double.MAX_VALUE);
152 				content.setMaxHeight(Double.MAX_VALUE);
153 				content.setPadding(new Insets(5.0));
154 				
155 				Label errorMessageLabel = new Label();
156 				errorMessageLabel.textProperty().bind(plugin.validationFailureMessageProperty());
157 				errorMessageLabel.setAlignment(Pos.BOTTOM_CENTER);
158 				TextErrorColorHelper.setTextErrorColor(errorMessageLabel);
159 
160 				VBox vBox = new VBox();
161 				vBox.getChildren().addAll(errorMessageLabel, content);
162 				vBox.setMaxWidth(Double.MAX_VALUE);
163 				vBox.setAlignment(Pos.TOP_CENTER);
164 
165 				plugin.validationFailureMessageProperty().addListener(new ChangeListener<String>() {
166 					@Override
167 					public void changed(
168 							ObservableValue<? extends String> observable,
169 							String oldValue,
170 							String newValue) {
171 						if (newValue != null && ! StringUtils.isEmpty(newValue)) {
172 							TextErrorColorHelper.setTextErrorColor(tabLabel);
173 						} else {
174 							TextErrorColorHelper.clearTextErrorColor(tabLabel);
175 						}
176 					}
177 				});
178 				pluginTab.setContent(vBox);
179 				tabPane_.getTabs().add(pluginTab);
180 			}
181 
182 			allValid_ = new ValidBooleanBinding() {
183 				{
184 					ArrayList<ReadOnlyStringProperty> pluginValidationFailureMessages = new ArrayList<>();
185 					for (PreferencesPluginViewI plugin : plugins_) {
186 						pluginValidationFailureMessages.add(plugin.validationFailureMessageProperty());
187 					}
188 					bind(pluginValidationFailureMessages.toArray(new ReadOnlyStringProperty[pluginValidationFailureMessages.size()]));
189 					setComputeOnInvalidate(true);
190 				}
191 
192 				@Override
193 				protected boolean computeValue() {
194 					for (PreferencesPluginViewI plugin : plugins_) {
195 						if (plugin.validationFailureMessageProperty().get() != null && plugin.validationFailureMessageProperty().get().length() > 0) {
196 							this.setInvalidReason(plugin.validationFailureMessageProperty().get());
197 
198 							logger.debug("Setting PreferencesView allValid_ to false because \"{}\"", this.getReasonWhyInvalid().get());
199 							return false;
200 						}
201 					}
202 
203 					logger.debug("Setting PreferencesView allValid_ to true");
204 
205 					this.clearInvalidReason();
206 					return true;
207 				}
208 			};
209 
210 			okButton_.disableProperty().bind(allValid_.not());
211 			// set focus on default
212 			// Platform.runLater(...);
213 		}
214 		
215 		// Reload persisted values every time view opened
216 		for (PreferencesPluginViewI plugin : plugins_) {
217 			plugin.getContent();
218 		}
219 	}
220 	
221 	private void saveAndExitIfSuccessful() {
222 		logger.debug("performing save...");
223 		
224 		final Map<PreferencesPluginViewI, Exception> caughtExceptions = Collections.synchronizedMap(new WeakHashMap<>());
225 		
226 		for (PreferencesPluginViewI plugin : plugins_) {		
227 			try {
228 				plugin.save();
229 			} catch (IOException e) {
230 				caughtExceptions.put(plugin, e);
231 			}
232 		}
233 		
234 		if (caughtExceptions.size() > 0) {
235 			String msg = "Caught " + caughtExceptions.size() + " exceptions performing save";
236 			StringBuilder builder = new StringBuilder();
237 			for (Map.Entry<PreferencesPluginViewI, Exception> entry : caughtExceptions.entrySet()) {
238 				builder.append("\n" + "Plugin " + entry.getKey().getName() + " (" + entry.getKey().getClass().getName() + ")" + " threw " + entry.getValue().getClass().getName() + " " + entry.getValue().getLocalizedMessage());
239 			
240 				logger.error("Plugin " + entry.getKey().getName() + " (" + entry.getKey().getClass().getName() + ")" + " threw " + entry.getValue().getClass().getName() + " " + entry.getValue().getLocalizedMessage());
241 				entry.getValue().printStackTrace();
242 			}
243 			
244 			AppContext.getCommonDialogs().showErrorDialog("Preferences Save Error", msg, builder.toString(), stage_);
245 		} else {
246 			stage_.close();
247 		}
248 	}
