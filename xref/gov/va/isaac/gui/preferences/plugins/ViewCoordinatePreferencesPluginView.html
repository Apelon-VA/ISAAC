138 	public void save() throws IOException {
139 		
140 			logger.debug("Saving ViewCoordinatePreferencesPluginView data");
141 			UserProfile loggedIn = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
142 			
143 			//Path Property
144 			logger.debug("Setting stored VC path (currently \"{}\") to {}", loggedIn.getViewCoordinatePath(), currentPathProperty().get()); 
145 			loggedIn.setViewCoordinatePath(currentPathProperty().get());
146 			
147 			//Stated/Inferred Policy
148 			logger.debug("Setting stored VC StatedInferredPolicy (currently \"{}\") to {}", loggedIn.getStatedInferredPolicy(), currentStatedInferredOptionProperty().get()); 
149 			loggedIn.setStatedInferredPolicy(currentStatedInferredOptionProperty().get());
150 			
151 			//Time Coordinate Property
152 			logger.debug("Setting stored VC time to :" + currentViewCoordinateTimeProperty().get());
153 			loggedIn.setViewCoordinateTime(currentViewCoordinateTimeProperty().get());
154 		try {
155 			AppContext.getService(UserProfileManager.class).saveChanges(loggedIn);
156 		} catch (InvalidUserException e) {
157 			String msg = "Caught " + e.getClass().getName() + " " + e.getLocalizedMessage() + " attempting to save UserProfile for " + getName();
158 			
159 			logger.error(msg, e);
160 			throw new IOException(msg, e);
161 		}
162 	}

569 	 * 			which populates Time Combo box, the truncTimeToFullTimeMap which maps the truncated times
570 	 * 			im times TreeSet to each times full Long value. The truncTimeToFullTimeMap chooses each time
571 	 * 			up to the second and maps it to the greatest equivalent time up to the milliseconds.
572 	 * 			
573 	 */
574 	protected void setTimeOptions(int path, Long storedTimePref) {
575 		try {
576 			Date startDate = null, finishDate = null;
577 			if(storedTimePref != null) {
578 				StampBdb stampDb = Bdb.getStampDb();
579 				NidSet nidSet = new NidSet(); nidSet.add(path); 
580 				
581 				NidSetBI stamps = null;
582 				if(storedTimePref != getDefaultTime()) {
583 					startDate = getStartOfDay(new Date(storedTimePref)); 
584 					finishDate = getEndOfDay(new Date(storedTimePref));
585 					stamps = stampDb.getSpecifiedStamps(nidSet, startDate.getTime(), finishDate.getTime());
586 				} else {
587 					stamps = stampDb.getSpecifiedStamps(nidSet, Long.MAX_VALUE, Long.MIN_VALUE);
588 				}
589 				
590 				truncTimeToFullTimeMap.clear();
591 				times.clear();
592 
593 				HashSet<Integer> stampSet = stamps.getAsSet();
594 				
595 				if(!stampSet.isEmpty()) {
596 					enableTimeCombo(true);
597 					for(Integer thisStamp : stampSet) {
598 						Long fullTime = null;
599 						Date stampDate;
600 						LocalDate stampInstant = null;
601 						try {
602 							fullTime = stampDb.getPosition(thisStamp).getTime();
603 							stampDate = new Date(fullTime);
604 							stampInstant = stampDate.toInstant().atZone(ZoneId.systemDefault()).toLocalDate();
605 						} catch (Exception e) {
606 							 e.printStackTrace();
607 						}
608 						Calendar cal = Calendar.getInstance();
609 						cal.setTime(new Date(fullTime));
610 						cal.set(Calendar.MILLISECOND, 0); //Strip milliseconds
611 						Long truncTime = cal.getTimeInMillis();
612 						
613 						this.pathDatesList.add(stampInstant); //Build DatePicker
614 						times.add(truncTime); //This can probably go, we don't populate hashmap like this at initialization
615 						timeSelectCombo.getItems().add(truncTime);
616 						
617 						if(truncTimeToFullTimeMap.containsKey(truncTime)) { //Build Truncated Time to Full Time HashMap
618 							//If truncTimeToFullTimeMap has this key, is the value the newest time in milliseconds?
619 							if(new Date(truncTimeToFullTimeMap.get(truncTime)).before(new Date(fullTime))) {
620 								truncTimeToFullTimeMap.put(truncTime, fullTime);
621 							}
622 						} else {
623 							truncTimeToFullTimeMap.put(truncTime, fullTime);
624 						}
625 					}
626 				} else {
627 					disableTimeCombo(true);
628 					logger.error("Could not retreive any Stamps");
629 				}
630 			}
631 		} catch (Exception e) {
632 			logger.error("Error setting the default Time Dropdown");
633 			e.printStackTrace();
634 		}
635 	}

642 		List<UUID> list = new ArrayList<>();
643 
644 		try {
645 			List<ConceptChronicleBI> pathConcepts = OTFUtility.getPathConcepts();
646 			for (ConceptChronicleBI cc : pathConcepts) {
647 				list.add(cc.getPrimordialUuid());
648 			}
649 		} catch (IOException | ContradictionException e) {
650 			logger.error("Failed loading path concepts. Caught {} {}", e.getClass().getName(), e.getLocalizedMessage());
651 			e.printStackTrace();
652 		}
653 		// Add currently-stored value to list of options, if not already there
654 		UUID storedPath = getStoredPath();
655 		if (storedPath != null && ! list.contains(storedPath)) {
656 			list.add(storedPath);
657 		}
658 
659 		//Collections.sort(list);
660 		return list;
661 	}
