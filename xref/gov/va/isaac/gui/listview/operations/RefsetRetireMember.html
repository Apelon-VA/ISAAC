85  			{
86  				if (retireFrom_.getItems().size() > 0)
87  				{
88  					retireFromInvalidString_.set(null);
89  					if (retireFrom_.getSelectionModel().getSelectedItem() == null)

102 			}

104 
105 		onlyRetireIfMatches_.setSelected(true);
106

111 		root_.add(new Label("Options"), 0, 2);
112 		root_.add(onlyRetireIfMatches_, 1, 2);
113 
114 		rvs_.setup(root_, 3);
115 
116 		onlyRetireIfMatches_.setOnAction(new EventHandler<ActionEvent>()
117 		{
118 			@Override
119 			public void handle(ActionEvent event)
120 			{
121 				if (onlyRetireIfMatches_.isSelected())
122 				{
123 					rvs_.setup(root_, 3);
124 				}
125 				else
126 				{
127 					rvs_.remove(root_);
128 				}
129 			}
130 		});
131 
132 		allValid_ = new BooleanBinding()
133 		{
134 			{
135 				bind(rvs_.isValid(), onlyRetireIfMatches_.selectedProperty());
136 			}
137 
138 			@Override
139 			protected boolean computeValue()
140 			{
141 				if (retireFrom_.getItems().size() == 0)
142 				{
143 					return false;
144 				}
145 				if (onlyRetireIfMatches_.isSelected())
146 				{
147 					return rvs_.isValid().get();
148 				}
149 				return true;
150 			}
151 		};
152 
153 		GridPane.setHgrow(wrappedRetireFrom, Priority.ALWAYS);
154 
155 		// remove the optional stuff after calculating col 0 width
156 		FxUtils.preventColCollapse(root_, 0);
157 		onlyRetireIfMatches_.setSelected(false);
158 		rvs_.remove(root_);
159 	}
