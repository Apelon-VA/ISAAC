19  package gov.va.isaac.gui.listview.operations;
20  
21  import gov.va.isaac.gui.SimpleDisplayConcept;
22  import javafx.beans.binding.BooleanExpression;
23  import javafx.collections.ListChangeListener;
24  import javafx.collections.ObservableList;
25  import javafx.scene.Node;
26  import javafx.scene.layout.GridPane;
27  import org.jvnet.hk2.annotations.Contract;
28  
29  /**
30   * {@link Operation}
31   * 
32   * The interface that serves as the basis for batch operations.  Extend this to create a new operation.
33   *
34   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
35   */
36  @Contract
37  public abstract class Operation
38  {
39  	protected ObservableList<SimpleDisplayConcept> conceptList_;
40  	protected GridPane root_;
41  	private boolean initRun = false;
42  	
43  	protected Operation()
44  	{
45  		//For HK2 to create
46  	}
47  	
48  	public synchronized final void initIfNot(ObservableList<SimpleDisplayConcept> conceptList)
49  	{
50  		if(!initRun)
51  		{
52  			initRun = true;
53  			init(conceptList);
54  		}
55  	}
56  	
57  	public void init(ObservableList<SimpleDisplayConcept> conceptList)
58  	{
59  		this.conceptList_ = conceptList;
60  		root_ = new GridPane();
61  		root_.setMaxWidth(Double.MAX_VALUE);
62  		root_.setHgap(5.0);
63  		root_.setVgap(5.0);
64  		
65  		conceptList_.addListener(new ListChangeListener<SimpleDisplayConcept>()
66  		{
67  			@Override
68  			public void onChanged(javafx.collections.ListChangeListener.Change<? extends SimpleDisplayConcept> c)
69  			{
70  				conceptListChanged();
71  			}
72  		});
73  	}
74  	
75  	public Node getNode()
76  	{
77  		return root_;
78  	}
79  	
80  	/**
81  	 * The title of the operation, which will be displayed to the user.
82  	 * @return
83  	 */
84  	public abstract String getTitle();
85  	
86  	/**
87  	 * The description of the operation, which will be displayed to the user
88  	 * @return
89  	 */
90  	public abstract String getOperationDescription();
91  	
92  	/**
93  	 * A convenience method which will be called when the conceptList changes.
94  	 */
95  	protected abstract void conceptListChanged();
96  	
97  	/**
98  	 * Implementers should return a BooleanExpression which evaluates to true when all necessary values
99  	 * for task execution have been provided.
100 	 */
101 	public abstract BooleanExpression isValid();
102 	
103 	/**
104 	 * Implementers should return a task - the task should update the progress value as appropriate, and 
105 	 * update the message value as appropriate as the task progresses.
106 	 * 
107 	 * If the execution fails during the task execution, an Exception should be thrown using the 
108 	 * Task API.  The returned String value should be a summary of what happened during the task execution
109 	 * which will be displayed to end users.
110 	 * 
111 	 * Implementers should also monitor the {@code CustomTask#cancelRequested_} variable, and respond appropriately 
112 	 * if the variable is set to true.
113 	 * 
114 	 * @return a Task implementation that performs the necessary work on the {@code #conceptList_}
115 	 */
116 	//TODO figure out how to return / display formatted output - String probably won't cut it.
117 	public abstract CustomTask<OperationResult> createTask();
118 }
