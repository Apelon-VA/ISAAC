8   import java.util.Map;

10  import javafx.application.Application;
11  import javafx.scene.Scene;
12  import javafx.scene.control.TreeCell;
13  import javafx.scene.control.TreeItem;
14  import javafx.scene.control.TreeView;
15  import javafx.scene.layout.AnchorPane;
16  import javafx.stage.Stage;
17  import javafx.util.Callback;
18  
19  public class DragAndDropTreeCellExample extends Application {
20  	Map<String, NodeDraggable> cache = new HashMap<>();
21  	
22  	public static class TestParentNode extends DraggableParentNode {
23  		private Integer data;
24  		
25  		/**
26  		 * @param data
27  		 */
28  		public TestParentNode(Integer data) {
29  			this.data = data;
30  		}
31  
32  		public String toString() {
33  			return "Parent: " + data.toString() + " (" + getTemporaryUniqueId() + ")";
34  		}
35  	}
36  	public static class TestLeafNode extends DraggableNode {
37  		private Integer data;
38  
39  		/**
40  		 * @param data
41  		 */
42  		public TestLeafNode(Integer data) {
43  			this.data = data;
44  		}
45  		
46  		public String toString() {
47  			return "Leaf: " + data.toString() + " (" + getTemporaryUniqueId() + ")";
48  		}
49  	}
50  	
51  	@Override
52  	public void start(Stage primaryStage) {
53  		TreeItem<NodeDraggable> treeRoot = new TreeItem<>(new TestParentNode(0));
54  		treeRoot.getChildren().add(new TreeItem<>(new TestParentNode(1)));
55  		treeRoot.getChildren().add(new TreeItem<>(new TestParentNode(2)));
56  		treeRoot.getChildren().add(new TreeItem<>(new TestParentNode(3)));
57  		treeRoot.getChildren().add(new TreeItem<>(new TestLeafNode(4)));
58  		treeRoot.getChildren().add(new TreeItem<>(new TestLeafNode(5)));
59  		treeRoot.getChildren().add(new TreeItem<>(new TestLeafNode(6)));
60  		TreeView<NodeDraggable> treeView = new TreeView<>(treeRoot);
61  		treeView.setCellFactory(new Callback<TreeView<NodeDraggable>, TreeCell<NodeDraggable>>() {
62  			@Override
63  			public TreeCell<NodeDraggable> call(TreeView<NodeDraggable> param) {
64  				return new DragAndDropTreeCell<NodeDraggable>(param, cache);
65  			}
66  		});
67  		AnchorPane.setTopAnchor(treeView, 0d);
68  		AnchorPane.setRightAnchor(treeView, 0d);
69  		AnchorPane.setBottomAnchor(treeView, 0d);
70  		AnchorPane.setLeftAnchor(treeView, 0d);
71  		//
72  		AnchorPane root = new AnchorPane();
73  		root.getChildren().add(treeView);
74  		Scene scene = new Scene(root, 300, 250);
75  		//
76  		primaryStage.setTitle("Hello World!");
77  		primaryStage.setScene(scene);
78  		primaryStage.show();
79  	}
