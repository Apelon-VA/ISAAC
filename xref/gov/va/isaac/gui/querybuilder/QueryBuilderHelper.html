97  	public static boolean getUserConfirmToDeleteNodeIfNecessary(TreeItem<NodeDraggable> node) {
98  		boolean delete = true;
99  		if (node.getChildren().size() > 0
100 				|| (node.getValue() != null && (node.getValue() instanceof AssertionNode) && node.getValue().getIsValid())) {
101 			int numDescendants = QueryBuilderHelper.getDescendants(node).size();
102 
103 			DialogResponse response = AppContext.getCommonDialogs().showYesNoDialog("Expression Deletion Confirmation", "Are you sure you want to delete expression\n" + node.getValue().getDescription() + "?" + (numDescendants > 0 ? ("\n\n" + numDescendants + " descendent expression(s) will also be deleted") : ""));
104 			if (response == DialogResponse.YES) {
105 				delete = true;
106 			} else {
107 				delete = false;
108 			}
109 		}
110 		
111 		return delete;
112 	}
113 	public static void initializeQueryNodeTreeView(TreeView<NodeDraggable> tree, GridPane nodeEditorGridPane, BooleanProperty queryNodeTreeViewIsValidProperty) {
114 		tree.getSelectionModel().setSelectionMode(SelectionMode.SINGLE);
115 
116 		tree.getSelectionModel().getSelectedItems().addListener(new ListChangeListener<TreeItem<NodeDraggable>>() {
117 			@Override
118 			public void onChanged(
119 					javafx.collections.ListChangeListener.Change<? extends TreeItem<NodeDraggable>> c) {
120 				
121 				if (tree.getSelectionModel().getSelectedItems().size() == 1) {
122 					QueryBuilderHelper.populateNodeEditorGridPane(nodeEditorGridPane, tree.getSelectionModel().getSelectedItems().get(0).getValue());
123 				}
124 			}
125 		});
126 
127 		tree.rootProperty().addListener(new ChangeListener<TreeItem<NodeDraggable>>() {
128 			@Override
129 			public void changed(
130 					ObservableValue<? extends TreeItem<NodeDraggable>> observable,
131 					TreeItem<NodeDraggable> oldValue,
132 					TreeItem<NodeDraggable> newValue) {
133 				if (newValue != null) {
134 					queryNodeTreeViewIsValidProperty.set(QueryBuilderHelper.isQueryNodeTreeViewValid(tree));
135 				} else {
136 					queryNodeTreeViewIsValidProperty.set(false);
137 				}
138 			}
139 		});
140 	}

142 	private static CheckBox createInversionCheckBox(InvertableNode node) {
143 		CheckBox inversionCheckBox = new CheckBox();
144 		inversionCheckBox.setText("Invert (NOT)");
145 		inversionCheckBox.setSelected(node.getInvert());
146 
147 		inversionCheckBox.selectedProperty().addListener(new ChangeListener<Boolean>() {
148 			@Override
149 			public void changed(
150 					ObservableValue<? extends Boolean> observable,
151 					Boolean oldValue,
152 					Boolean newValue) {
153 				node.setInvert(newValue);
154 			}
155 		});
156 
157 		return inversionCheckBox;
158 	}

568 	public static boolean isValidExpression(TreeItem<NodeDraggable> treeNode) {
569 		if (treeNode == null) {
570 			return false;
571 		}
572 		if (treeNode.getValue() == null) {
573 			return false;
574 		}
575 		if (treeNode.getValue() instanceof AssertionNode) {
576 			return ((AssertionNode)treeNode.getValue()).getIsValid();
577 		} else if (treeNode.getValue() instanceof LogicalNode) {
578 			LogicalNode logicalNode = (LogicalNode)treeNode.getValue();
579 			int numChildren = treeNode.getChildren().size();
580 			if (numChildren > logicalNode.getMaxChildren() || numChildren < logicalNode.getMinimumChildren()) {
581 				return false;
582 			}
583 			Boolean childrenValid = null;
584 			for (TreeItem<NodeDraggable> childTreeNode : treeNode.getChildren()) {
585 				if (isValidExpression(childTreeNode)) {
586 					if (childrenValid == null) {
587 						childrenValid = true;
588 					}
589 				} else {
590 					childrenValid = false;
591 				}
592 			}
593 			
594 			return childrenValid != null ? childrenValid : false;
595 		} else {
596 			logger.warn("isValidExpression() encountered node {} of unexpected type {}.  Expected AssertionNode or LogicalNode.", treeNode.getValue().getDescription(), treeNode.getValue());
597 			return false;
598 		}
599 	}
