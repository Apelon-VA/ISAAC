204 	private void updateMappingItemsList(MappingSet mappingSet)
205 	{
206 		clearMappingItems();
207 		if (mappingSet != null) {
208 			mappingItemTableView.setPlaceholder(new ProgressBar(-1.0));
209 			Utility.execute(() ->
210 			{
211 				ObservableList<MappingItem> mappingItems = FXCollections.observableList(mappingSet.getMappingItems(activeOnlyToggle.isSelected()));
212 				
213 				Platform.runLater(() ->
214 				{
215 					mappingItemTableView.setItems(mappingItems);
216 					mappingItemTableView.setPlaceholder(new Label("The selected Mapping Set contains no Mapping Items."));
217 					
218 					mappingItemListTitleLabel.setText(mappingSet.getName());
219 					plusMappingItemButton.setDisable(false);
220 					minusMappingSetButton.setDisable(false);
221 					editMappingSetButton.setDisable(false);
222 					mappingSetSummaryLabel.setText(mappingSet.getSummary(activeOnlyToggle.isSelected()));
223 				});
224 			});
225 		}

342 				}

383 	private void updateCell(TableCell<?, ?> cell, MappingObject mappingObject) {
384 		if (!cell.isEmpty() && mappingObject != null) {
385 			ContextMenu cm = new ContextMenu();
386 			cell.setContextMenu(cm);
387 			SimpleStringProperty property = null;
388 			int  conceptNid  = 0;
389 			MappingColumnType columnType = (MappingColumnType) cell.getTableColumn().getUserData();
390 
391 			cell.setText(null);
392 			cell.setGraphic(null);
393 
394 			switch (columnType) {
395 			case STATUS_CONDENSED:
396 				StackPane sp = new StackPane();
397 				sp.setPrefSize(25, 25);
398 				String tooltipText = mappingObject.isActive()? "Active" : "Inactive";
399 				ImageView image    = mappingObject.isActive()? Images.BLACK_DOT.createImageView() : Images.GREY_DOT.createImageView();
400 				sizeAndPosition(image, sp, Pos.CENTER);
401 				cell.setTooltip(new Tooltip(tooltipText));
402 				cell.setGraphic(sp);
403 				break;
404 				
405 			case NAME:
406 				property = ((MappingSet)mappingObject).getNameProperty(); 
407 				break;
408 				
409 			case PURPOSE:
410 				property = ((MappingSet)mappingObject).getPurposeProperty(); 
411 				break;
412 				
413 			case DESCRIPTION:
414 				property = ((MappingSet)mappingObject).getDescriptionProperty(); 
415 				break;
416 				
417 			case SOURCE:
418 				property = ((MappingItem)mappingObject).getSourceConceptProperty();
419 				conceptNid = ((MappingItem)mappingObject).getSourceConceptNid();
420 				break;
421 			case TARGET:
422 				property = ((MappingItem)mappingObject).getTargetConceptProperty();
423 				conceptNid = ((MappingItem)mappingObject).getTargetConceptNid();
424 				break;
425 			case QUALIFIER:
426 				property = ((MappingItem)mappingObject).getQualifierConceptProperty();
427 				conceptNid = ((MappingItem)mappingObject).getQualifierConceptNid();
428 				break;
429 			case COMMENTS:
430 				property = ((MappingItem)mappingObject).getCommentsProperty();
431 				break;
432 			case EDITOR_STATUS:
433 				property = mappingObject.getEditorStatusConceptProperty();
434 				conceptNid = mappingObject.getEditorStatusConceptNid();
435 				break;
436 			case STATUS_STRING:
437 				property = mappingObject.getStatusProperty();
438 				break;
439 			case TIME:
440 				property = mappingObject.getTimeProperty();
441 				break;
442 			case AUTHOR:
443 				property = mappingObject.getAuthorProperty();
444 				conceptNid = mappingObject.getAuthorNid();
445 				break;
446 			case MODULE:
447 				property = mappingObject.getModuleProperty();
448 				conceptNid = mappingObject.getModuleNid();
449 				break;
450 			case PATH:
451 				property = mappingObject.getPathProperty();
452 				conceptNid = mappingObject.getPathNid();
453 				break;
454 			default:
455 				// Nothing
456 			}
457 			
458 			if (property != null) {
459 				Text text = new Text();
460 				text.textProperty().bind(property);
461 				text.wrappingWidthProperty().bind(cell.getTableColumn().widthProperty());
462 				cell.setGraphic(text);
463 	
464 				MenuItem mi = new MenuItem("Copy Value");
465 				mi.setOnAction(new EventHandler<ActionEvent>() {
466 					@Override
467 					public void handle(ActionEvent arg0) {
468 						CustomClipboard.set(((Text)cell.getGraphic()).getText());
469 					}
470 				});
471 				mi.setGraphic(Images.COPY.createImageView());
472 				cm.getItems().add(mi);
473 	
474 				if (columnType.isConcept() && conceptNid != 0) {
475 					final int nid = conceptNid;
476 					CommonMenus.addCommonMenus(cm, new CommonMenusNIdProvider() {
477 						@Override
478 						public Collection<Integer> getNIds() {
479 						   return Arrays.asList(new Integer[] {nid});
480 						}
481 					});
482 				}
483 			}
484 		} else {
485 			cell.setText(null);
486 			cell.setGraphic(null);
487 		}
488 	}

686 	public static void sizeAndPosition(Node node, StackPane sp, Pos position)
687 	{
688 		if (node instanceof ImageView)
689 		{
690 			((ImageView)node).setFitHeight(12);
691 			((ImageView)node).setFitWidth(12);
692 		}
693 		Insets insets;
694 		switch (position) {
695 		case TOP_LEFT:
696 			insets = new Insets(0,0,0,0);
697 			break;
698 		case TOP_RIGHT:
699 			insets = new Insets(0,0,0,13);
700 			break;
701 		case BOTTOM_LEFT:
702 			insets = new Insets(13,0,0,0);
703 			break;
704 		case BOTTOM_RIGHT:
705 			insets = new Insets(13,0,0,13);
706 			break;
707 		case CENTER:
708 			insets = new Insets(5,0,0,5);
709 			break;
710 		default:
711 			insets = new Insets(0,0,0,0);
712 		}
713 		StackPane.setMargin(node, insets);
714 		sp.getChildren().add(node);
715 		StackPane.setAlignment(node, Pos.TOP_LEFT);
716 	}

773 	public static void setComboSelection(ComboBox<SimpleDisplayConcept> combo, String selectValue, int defaultIndex) {
774 		boolean found = false;
775 		if (selectValue != null && !selectValue.trim().equals("")) {
776 			for (SimpleDisplayConcept sdc : combo.getItems()) {
777 				if (sdc.getDescription().equals(selectValue)) {
778 					combo.getSelectionModel().select(sdc);
779 					found = true;
780 					break;
781 				}
782 			}
783 		}
784 		if (!found && defaultIndex >= 0 && defaultIndex < combo.getItems().size()) {
785 			combo.getSelectionModel().select(defaultIndex);
786 		}
787 	}
