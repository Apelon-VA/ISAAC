389 	private void doSearch() {
390 		String searchString = criteriaText.getText().trim();
391 		
392 		if (searchString != null && !searchString.equals("")) {
393 			doSearch(searchString);
394 		
395 		} else if (searchObject_ != null) {
396 			doSearch(searchObject_);
397 		}
398 	}

400 	private void doSearch(Object searchObject) {
401 		try {
402 			searchObject_ = searchObject;
403 			SearchHandle searchHandle = null;
404 			scrapeSearchRestriction();
405 			
406 			if (searchObject instanceof ConceptVersionBI) {
407 				searchHandle = MappingUtils.search(
408 						((ConceptVersionBI)searchObject).getNid(), 
409 						null, 
410 						searchRestriction.getDescriptionType(), 
411 						searchRestriction.getAdvancedDescriptionTypeUUID(), 
412 						searchRestriction.getTargetCodeSystemPathNid(), 
413 						searchRestriction.getMemberOfRefsetNid(), 
414 						searchRestriction.getChildOfNid()
415 				);
416 				
417 			} else if (searchObject instanceof String) {
418 				searchHandle = MappingUtils.search(
419 						(String)searchObject, 
420 						null, 
421 						searchRestriction.getDescriptionType(), 
422 						searchRestriction.getAdvancedDescriptionTypeUUID(), 
423 						searchRestriction.getTargetCodeSystemPathNid(), 
424 						searchRestriction.getMemberOfRefsetNid(), 
425 						searchRestriction.getChildOfNid()
426 				);
427 			}
428 			if (searchHandle != null) {
429 				populateSearchResult(searchHandle);	
430 			}
431 		} catch (IOException e) {
432 			LOG.error("Error performing search", e);
433 		}
434 	}

447 	private void populateSearchResult(SearchHandle searchHandle) {
448 		candidatesTableView.getItems().clear();
449 		candidatesTableView.setPlaceholder(LABEL_SEARCHING);
450 		Utility.execute(() ->
451 		{
452 			ObservableList<ConceptVersionBI> conceptList = FXCollections.observableArrayList(); 
453 			try {
454 				Collection<CompositeSearchResult> results = searchHandle.getResults();	
455 				for (CompositeSearchResult result : results) {
456 					ConceptVersionBI concept = result.getContainingConcept();
457 					if (concept != null) {
458 						conceptList.add(concept);
459 					}
460 				}
461 			} catch (Exception e) {
462 				LOG.error("Error populating search results", e);
463 				AppContext.getCommonDialogs().showErrorDialog("There was an error populating search results", e);
464 			}
465 
466 			Platform.runLater(() -> {			
467 				if (conceptList.size() == 0) {
468 					candidatesTableView.setPlaceholder(LABEL_NO_RESULTS);
469 				} else {
470 					candidatesTableView.getItems().addAll(conceptList);	
471 				}
472 			});
473 		});
474 	}

517 	private void updateCell(TableCell<?, ?> cell, ConceptVersionBI concept) {
518 		if (!cell.isEmpty() && concept != null) {
519 			ContextMenu cm = new ContextMenu();
520 			cell.setContextMenu(cm);
521 			final SimpleStringProperty property = new SimpleStringProperty();
522 			int conceptNid  = 0;
523 			MappingColumnType columnType = (MappingColumnType) cell.getTableColumn().getUserData();
524 
525 			cell.setText(null);
526 			cell.setGraphic(null);
527 
528 			switch (columnType) {
529 			case CONCEPT:
530 				property.set("-");
531 				conceptNid = concept.getNid();
532 				Utility.execute(() ->
533 				{
534 					String conceptName = OTFUtility.getDescription(concept);
535 					Platform.runLater(() -> {
536 						property.set(conceptName);
537 					});
538 				});
539 				break;
540 				
541 			case CODE_SYSTEM:
542 				property.set("-");
543 				final int pathNid = concept.getPathNid();
544 				conceptNid = pathNid;
545 				Utility.execute(() ->
546 				{
547 					String pathName = OTFUtility.getDescription(pathNid);
548 					Platform.runLater(() -> {
549 						property.set(pathName);
550 					});
551 				});
552 				break;
553 				
554 			case STATUS_STRING:
555 				try {
556 					property.set(concept.isActive()? "Active":"Inactive");	
557 				} catch (IOException e) {
558 					property.set("-error-");
559 					LOG.error("Error fetching isActive", e);
560 				}
561 				
562 				break;
563 			default:
564 				// Nothing
565 			}
566 			
567 			if (property.get() != null) {
568 				Text text = new Text();
569 				text.textProperty().bind(property);
570 				text.wrappingWidthProperty().bind(cell.getTableColumn().widthProperty());
571 				cell.setGraphic(text);
572 	
573 				MenuItem mi = new MenuItem("Copy Value");
574 				mi.setOnAction(new EventHandler<ActionEvent>() {
575 					@Override
576 					public void handle(ActionEvent arg0) {
577 						CustomClipboard.set(((Text)cell.getGraphic()).getText());
578 					}
579 				});
580 				mi.setGraphic(Images.COPY.createImageView());
581 				cm.getItems().add(mi);
582 	
583 				if (columnType.isConcept() && conceptNid != 0) {
584 					final int nid = conceptNid;
585 					CommonMenus.addCommonMenus(cm, new CommonMenusNIdProvider() {
586 						@Override
587 						public Collection<Integer> getNIds() {
588 						   return Arrays.asList(new Integer[] {nid});
589 						}
590 					});
591 				}
592 
593 				AppContext.getService(DragRegistry.class).setupDragOnly(text, new SingleConceptIdProvider() {
594 					@Override
595 					public String getConceptId() {
596 						return Integer.toString(concept.getNid());
597 					}
598 				});
599 }
600 		} else {
601 			cell.setText(null);
602 			cell.setGraphic(null);
603 		}
604 	}

606 	private void scrapeSearchRestriction() {
607 		if (restrictionsInitialized) {
608 			// Description Type
609 			if (descriptionRestrictionRadio.selectedProperty().get()) {
610 				searchRestriction.setDescriptionType(LuceneDescriptionType.DEFINITION);
611 			
612 			} else if (synonymRestrictionRadio.selectedProperty().get()) {
613 				searchRestriction.setDescriptionType(LuceneDescriptionType.SYNONYM);
614 			
615 			} else if (fsnRestrictionRadio.selectedProperty().get()) {
616 				searchRestriction.setDescriptionType(LuceneDescriptionType.FSN);
617 			
618 			} else {
619 				searchRestriction.setDescriptionType(null);
620 			}
621 			
622 			// Advanced Description Type
623 			SimpleDisplayConcept selectedAdvancedDescription = descriptionRestrictionCombo.getSelectionModel().getSelectedItem();
624 			if (selectedAdvancedDescription != null && selectedAdvancedDescription.getNid() != Integer.MIN_VALUE) {
625 				searchRestriction.setAdvancedDescriptionType(selectedAdvancedDescription);
626 			} else {
627 				searchRestriction.setAdvancedDescriptionType(null);
628 			}
629 			
630 
631 			// Code System
632 			SimpleDisplayConcept selectedCodeSystem = codeSystemRestrictionCombo.getSelectionModel().getSelectedItem();
633 			if (selectedCodeSystem != null && selectedCodeSystem.getNid() != Integer.MIN_VALUE) {
634 				searchRestriction.setTargetCodeSystemPath(selectedCodeSystem);
635 			} else {
636 				searchRestriction.setTargetCodeSystemPath(null);
637 			}
638 
639 			// Refset Member
640 			SimpleDisplayConcept selectedRefset = refsetRestrictionCombo.getSelectionModel().getSelectedItem();
641 			if (selectedRefset != null && selectedRefset.getNid() != Integer.MIN_VALUE) {
642 				searchRestriction.setMemberOfRefset(selectedRefset);
643 			} else {
644 				searchRestriction.setMemberOfRefset(null);
645 			}
646 			
647 			// Child of concept
648 			ConceptVersionBI childOfConcept = childRestrictionConceptNode.getConcept();
649 			if (childOfConcept == null) {
650 				searchRestriction.setChildOf(null);
651 			} else {
652 				searchRestriction.setChildOf(childOfConcept);
653 			}
654 			
655 		}
656 	}

658 	private void paintSearchRestriction() {
659 		// Description Type
660 		if (searchRestriction.descriptionType == null) {
661 			noRestrictionRadio.setSelected(true);
662 		} else {
663 			switch (searchRestriction.descriptionType) {
664 			case DEFINITION:
665 				descriptionRestrictionRadio.setSelected(true);
666 				break;
667 			case SYNONYM:
668 				synonymRestrictionRadio.setSelected(true);
669 				break;
670 			case FSN:
671 				fsnRestrictionRadio.setSelected(true);
672 				break;
673 			default:
674 				noRestrictionRadio.setSelected(true);
675 			}
676 		}
677 		
678 		MappingController.setComboSelection(descriptionRestrictionCombo, searchRestriction.getAdvancedDescriptionType(), 0);
679 		
680 		MappingController.setComboSelection(codeSystemRestrictionCombo, searchRestriction.getTargetCodeSystemPath(), 0);
681 
682 		MappingController.setComboSelection(refsetRestrictionCombo, searchRestriction.getMemberOfRefset(), 0);
683 
684 		resetChildOfRestriction();
685 		if (searchRestriction.getChildOf() != null) {
686 			childRestrictionConceptNode.set(searchRestriction.getChildOf());
687 		}
688 		
689 		restrictionsInitialized = true;
690 	}

695 	}
