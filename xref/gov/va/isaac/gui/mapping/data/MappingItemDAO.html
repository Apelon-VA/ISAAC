30  	/**
31  	 * Construct (and save to the DB) a new MappingItem.  
32  	 * @param sourceConcept - the primary ID of the source concept
33  	 * @param mappingSetID - the primary ID of the mapping type
34  	 * @param targetConcept - the primary ID of the target concept
35  	 * @param qualifierID - (optional) the primary ID of the qualifier concept
36  	 * @param editorStatusID - (optional) the primary ID of the status concept
37  	 * @throws IOException
38  	 */
39  	public static MappingItem createMappingItem(ConceptVersionBI sourceConcept, UUID mappingSetID, ConceptVersionBI targetConcept, UUID qualifierID, UUID editorStatusID) throws IOException
40  	{
41  		try
42  		{
43  			RefexDynamicCAB mappingAnnotation = new RefexDynamicCAB(sourceConcept.getPrimordialUuid(), mappingSetID);
44  			mappingAnnotation.setData(new RefexDynamicDataBI[] {
45  					(targetConcept == null ? null : new RefexDynamicUUID(targetConcept.getPrimordialUuid())),
46  					(qualifierID == null ? null : new RefexDynamicUUID(qualifierID)),
47  					(editorStatusID == null ? null : new RefexDynamicUUID(editorStatusID))}, OTFUtility.getViewCoordinateAllowInactive());
48  			
49  			
50  			UUID mappingItemUUID = UuidT5Generator.get(MappingConstants.MAPPING_NAMESPACE.getPrimodialUuid(), 
51  					sourceConcept.getPrimordialUuid().toString() + "|" 
52  					+ mappingSetID.toString() + "|"
53  					+ ((targetConcept == null)? "" : targetConcept.getPrimordialUuid().toString()) + "|" 
54  					+ ((qualifierID == null)?   "" : qualifierID.toString()));
55  			
56  			if (ExtendedAppContext.getDataStore().hasUuid(mappingItemUUID))
57  			{
58  				throw new IOException("A mapping with the specified source, target and qualifier already exists in this set.  Please edit that mapping.");
59  			}
60  			
61  			mappingAnnotation.setComponentUuidNoRecompute(mappingItemUUID);
62  			OTFUtility.getBuilder().construct(mappingAnnotation);
63  
64  			AppContext.getRuntimeGlobals().disableAllCommitListeners();
65  			ExtendedAppContext.getDataStore().addUncommitted(sourceConcept);
66  			ExtendedAppContext.getDataStore().commit(sourceConcept);
67  			
68  			return new MappingItem((RefexDynamicVersionBI<?>) ExtendedAppContext.getDataStore().getComponent(mappingItemUUID));
69  		}
70  		catch (InvalidCAB | ContradictionException | PropertyVetoException | NoSuchAlgorithmException e)
71  		{
72  			LOG.error("Unexpected", e);
73  			throw new IOException("Invalid mapping. Check Source, Target, and Qualifier.", e);
74  		}
75  		finally
76  		{
77  			AppContext.getRuntimeGlobals().enableAllCommitListeners();
78  		}
79  	}

81  	/**
82  	 * Read all of the mappings items which are defined as part of the specified mapping set.
83  	 * 
84  	 * @param mappingSetID - the mapping set that contains the mapping items
85  	 * @return
86  	 * @throws IOException
87  	 */
88  	public static List<MappingItem> getMappingItems(UUID mappingSetID, boolean activeOnly) throws IOException
89  	{
90  		try
91  		{
92  			ArrayList<MappingItem> result = new ArrayList<>();
93  			boolean hadError = false;
94  			for (SearchResult sr : search(mappingSetID))
95  			{
96  				RefexDynamicVersionBI<?> rc = (RefexDynamicVersionBI<?>) ExtendedAppContext.getDataStore()
97  						.getComponentVersion(OTFUtility.getViewCoordinateAllowInactive(), sr.getNid());
98  				try
99  				{
100 					if (rc != null)
101 					{
102 						if (rc.isActive() || !activeOnly)
103 						{
104 							result.add(new MappingItem(rc));
105 						}
106 					}
107 				}
108 				catch (Exception e)
109 				{
110 					LOG.error("Unexpected error reading mapping " + rc, e);
111 					hadError = true;
112 				}
113 			}
114 			if (hadError)
115 			{
116 				AppContext.getCommonDialogs().showErrorDialog("Internal Error", "Internal Error", "There was an internal error reading all of the mappings.  See logs.");
117 			}
118 			
119 			return result;
120 		}
121 		catch (ContradictionException e)
122 		{
123 			LOG.error("Unexpected error reading comments", e);
124 			throw new IOException("internal error reading comments");
125 		}
126 	}

166 	/**
167 	 * Store the values passed in as a new revision of a mappingItem (the old revision remains in the DB)
168 	 * @param mappingItem - The MappingItem with revisions (contains fields where the setters have been called)
169 	 * @throws IOException
170 	 */
171 	public static void updateMappingItem(MappingItem mappingItem) throws IOException
172 	{
173 		try
174 		{
175 			RefexDynamicVersionBI<?> rdv = readCurrentRefex(mappingItem.getPrimordialUUID());
176 			RefexDynamicCAB mappingItemCab = rdv.makeBlueprint(OTFUtility.getViewCoordinateAllowInactive(), IdDirective.PRESERVE, RefexDirective.EXCLUDE);
177 			mappingItemCab.getData()[2] = (mappingItem.getEditorStatusConcept() != null ? new RefexDynamicUUID(mappingItem.getEditorStatusConcept()) : null);
178 			mappingItemCab.validate(OTFUtility.getViewCoordinateAllowInactive());
179 			RefexDynamicChronicleBI<?> rdc = OTFUtility.getBuilder().construct(mappingItemCab);
180 
181 			ConceptChronicleBI cc = ExtendedAppContext.getDataStore().getConcept(rdc.getConceptNid());
182 			AppContext.getRuntimeGlobals().disableAllCommitListeners();
183 			ExtendedAppContext.getDataStore().addUncommitted(cc);
184 			ExtendedAppContext.getDataStore().commit(cc);
185 		}
186 		catch (InvalidCAB | ContradictionException | PropertyVetoException e)
187 		{
188 			LOG.error("Unexpected!", e);
189 			throw new IOException("Internal error");
190 		}
191 		finally
192 		{
193 			AppContext.getRuntimeGlobals().enableAllCommitListeners();
194 		}
195 	}
