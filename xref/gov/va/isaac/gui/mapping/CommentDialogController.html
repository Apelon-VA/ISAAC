86  		commentsPane.setFitToWidth(true);
87  		commentListVBox.setFillWidth(true);
88  		
89  		saveButton.setOnAction(new EventHandler<ActionEvent>() {
90  			@Override
91  			public void handle(ActionEvent e) {
92  				String commentText = newCommentTextArea.getText().trim();
93  				MappingItemComment newComment = null;
94  				if (commentText != null && !commentText.equals("")) {
95  					try {
96  						//TODO use context?
97  						for (MappingItem mappingItem : mappingItems_) {
98  							newComment = MappingItemCommentDAO.createMappingItemComment(mappingItem.getPrimordialUUID(), commentText, null);
99  							mappingItem.refreshCommentsProperty();
100 						}
101 					} catch (Exception ex) {
102 						LOG.error(ex.toString());
103 						ex.printStackTrace();
104 						String message = (ex.getMessage() == null)? "Unspecified error, possibly use of null" : ex.getMessage();
105 						AppContext.getCommonDialogs().showInformationDialog("Cannot Create Comment", message);
106 					}
107 				}
108 				if (newComment != null) {
109 					if (mappingItems_.size() > 1) {
110 						close();
111 					} else {
112 						refreshComments();
113 						newCommentTextArea.setText("");
114 					}
115 				}
116 			}
117 		});
118 
119 		closeButton.setCancelButton(true);
120 		closeButton.setOnAction(new EventHandler<ActionEvent>() {
121 			@Override
122 			public void handle(ActionEvent e) {
123 				close();
124 			}
125 		});
126 		closeButton.setOnKeyPressed(new EventHandler<KeyEvent>()  {
127 			@Override
128 			public void handle(KeyEvent event) {
129 				if (event.getCode() == KeyCode.ENTER) {
130 					event.consume();
131 					closeButton.fire();
132 				}
133 			}
134 		});
135 	}

149 	public void setMappingSetAndItems(MappingSet mappingSet, ObservableList<MappingItem> mappingItems) {
150 		mappingItems_ = mappingItems;
151 		if (mappingItems.size() > 1) {
152 			double height = newCommentPane.getHeight();
153 			detailPane.setVisible(false);
154 			AnchorPane.setTopAnchor(newCommentPane, 0.0);
155 			mainPane.setPrefHeight(height);
156 			
157 		} else if (mappingItems.size() > 0) {
158 			
159 			mappingSetLabel.setText(mappingSet.getName());
160 			sourceLabel.setText(mappingItems.get(0).getSourceConceptProperty().getValueSafe());
161 			targetLabel.setText(mappingItems.get(0).getTargetConceptProperty().getValueSafe());
162 			qualifierLabel.setText(mappingItems.get(0).getQualifierConceptProperty().getValueSafe());
163 			refreshComments();
164 		}
165 	}

167 	private void addCommentToList(MappingItemComment comment) {
168 		CommentControl cc = new CommentControl();
169 		cc.set(this, mappingItems_.get(0), comment);
170 		cc.setPrefWidth(commentListVBox.getPrefWidth());
171 		commentListVBox.setPrefHeight(commentListVBox.getPrefHeight() + cc.getPrefHeight());
172 		commentListVBox.getChildren().add(cc);
173 	}
174 	
175 	private void clearComments() {
176 		commentListVBox.getChildren().clear();
177 		commentListVBox.setPrefHeight(0);
178 	}
179 	
180 	public void refreshComments() {
181 		try
182 		{
183 			clearComments();
184 			List<MappingItemComment> comments = mappingItems_.get(0).getComments();
185 			for (MappingItemComment comment : comments) {
186 				if (comment.isActive()) {
187 					addCommentToList(comment);
188 				}
189 			}
190 		}
191 		catch (IOException e)
192 		{
193 			// TODO Auto-generated catch block
194 			e.printStackTrace();
195 		}
196 	}
