22  public class CommentControl extends AnchorPane {
23      @FXML private Label timestampLabel;
24      @FXML private Label authorLabel;
25      @FXML private TextArea commentTextArea;
26      @FXML private Button retireButton;
27  
28      @FXML
29      public void initialize() {
30          assert timestampLabel != null : "fx:id=\"timestampLabel\" was not injected: check your FXML file 'CommentControl.fxml'.";
31          assert authorLabel != null : "fx:id=\"authorLabel\" was not injected: check your FXML file 'CommentControl.fxml'.";
32          assert commentTextArea != null : "fx:id=\"commentTextArea\" was not injected: check your FXML file 'CommentControl.fxml'.";
33          assert retireButton != null : "fx:id=\"retireButton\" was not injected: check your FXML file 'CommentControl.fxml'.";
34  
35  		FxUtils.assignImageToButton(retireButton, Images.MINUS.createImageView(), "Retire Comment");
36          
37          retireButton.setOnAction((event) -> {
38          	try {
39  				DialogResponse response = AppContext.getCommonDialogs().showYesNoDialog("Please Confirm", "Are you sure you want to retire this comment?");
40  				if (response == DialogResponse.YES) {
41  	            	MappingItemCommentDAO.retireComment(comment_.getPrimordialUUID());
42  	    			mappingItem_.refreshCommentsProperty();
43  	            	if (dialogController != null) {
44  		            	dialogController.refreshComments();
45  	            	}
46  				}
47  				//this.getScene().getWindow().requestFocus();
48  
49          	} catch (IOException e) {
50          		throw new RuntimeException(e);
51          	}
52          });
53          
54      }
55  
56      private MappingItemComment comment_;
57      private MappingItem mappingItem_;
58      private CommentDialogController dialogController;
59      
60      private static SimpleDateFormat dateTimeFormatShort = new SimpleDateFormat("MM/dd/yy HH:mm");
61      
62      public CommentControl() {
63          FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("CommentControl.fxml"));
64          fxmlLoader.setRoot(this);
65          fxmlLoader.setController(this);
66  
67          try {
68              fxmlLoader.load();
69          } catch (IOException exception) {
70              throw new RuntimeException(exception);
71          }
72  
73      }
74      
75      public void set(CommentDialogController controller, MappingItem mappingItem, MappingItemComment comment) {
76      	dialogController = controller;
77      	mappingItem_ = mappingItem;
78      	comment_ = comment;
79      	commentTextArea.textProperty().set(comment.getCommentText());
80      	authorLabel.textProperty().set(OTFUtility.getDescription(comment.getAuthorName()));
81      	timestampLabel.textProperty().set(dateTimeFormatShort.format(comment.getCreationDate()));
82      }
83  }
