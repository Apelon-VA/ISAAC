102     void addChildren() {
103         LOG.debug("Adding children to " + getConceptUuid());
104         final TaxonomyReferenceWithConcept taxRef = getValue();
105         if (! shouldDisplay()) {
106             // Don't add children to something that shouldn't be displayed
107             LOG.debug("this.shouldDisplay() == false: not adding children to " + this.getConceptUuid());
108         } else if (taxRef == null || taxRef.getConcept() == null) {
109             LOG.debug("addChildren(): taxRef={}, taxRef.getConcept()={}", taxRef, taxRef.getConcept());
110         } else if (taxRef.getConcept() != null) {
111             // Configure a new progress indicator.
112             ProgressIndicator pi = new ProgressIndicator();
113             //TODO (artf231880) Figure out what to do with the progress indicator
114 //            pi.setSkin(new TaxonomyProgressIndicatorSkin(pi));
115             pi.setPrefSize(16, 16);
116             pi.setProgress(-1);
117             setProgressIndicator(pi);
118 
119             // Gather the children
120             ArrayList<SctTreeItem> childrenToProcess = new ArrayList<>();
121 
122             for (RelationshipChronicleDdo r : taxRef.conceptProperty().get().getDestinationRelationships()) {
123                 for (RelationshipVersionDdo rv : r.getVersions()) {
124                     try {
125                         TaxonomyReferenceWithConcept fxtrc = new TaxonomyReferenceWithConcept(rv);
126                         SctTreeItem childItem = new SctTreeItem(fxtrc, displayPolicies);
127                         if (childItem.shouldDisplay()) {
128                             childrenToProcess.add(childItem);
129                         } else {
130                             LOG.debug("item.shouldDisplay() == false: not adding " + childItem.getConceptUuid() + " as child of " + this.getConceptUuid());
131                         }
132                     } catch (IOException | ContradictionException ex) {
133                         LOG.error(null, ex);
134                     }
135                 }
136             }
137 
138             Collections.sort(childrenToProcess);
139             getChildren().addAll(childrenToProcess);
140 
141             FetchConceptsTask fetchChildrenTask = new FetchConceptsTask(childrenToProcess);
142 
143             //pi.progressProperty().bind(fetchChildrenTask.progressProperty());
144             fetchChildrenTask.progressProperty().addListener(new ChangeListener<Number>() {
145                 @Override
146                 public void changed(
147                         ObservableValue<? extends Number> observable,
148                         Number oldValue, Number newValue) {
149 //                    LOG.debug("addChildren(): ProgressIndicator progress for taxRef={} changing from {} to {}", taxRef, oldValue, newValue);
150 
151                     pi.progressProperty().set(newValue.doubleValue());
152                 }
153             });
154             FutureHelper.addFuture(childFetcherService.submit(fetchChildrenTask));
155         }
156     }

158     void addChildrenConceptsAndGrandchildrenItems(ProgressIndicator p1) {
159         ArrayList<SctTreeItem> grandChildrenToProcess = new ArrayList<>();
160 
161         if (! shouldDisplay()) {
162             // Don't add children to something that shouldn't be displayed
163             LOG.debug("this.shouldDisplay() == false: not adding children concepts and grandchildren items to " + this.getConceptUuid());
164         } else {
165             for (TreeItem<TaxonomyReferenceWithConcept> child : getChildren()) {
166                 SctTreeItem tempChild = new SctTreeItem(child.getValue(), this.getDisplayPolicies());
167                 if (tempChild.shouldDisplay()) {
168                     if (child.getChildren().isEmpty() && (child.getValue().getConcept() != null)) {
169                         if (child.getValue().getConcept().getDestinationRelationships().isEmpty()) {
170                             // TODO Why are we removing TaxonomyReferenceWithConcept from childless child,
171                             // computing its graphic,
172                             // then setting it back again?
173                             TaxonomyReferenceWithConcept value = child.getValue();
174                             child.setValue(null);
175                             SctTreeItem noChildItem = (SctTreeItem) child;
176                             noChildItem.computeGraphic();
177                             noChildItem.setValue(value);
178                         } else {
179                             ArrayList<SctTreeItem> grandChildrenToAdd = new ArrayList<>();
180 
181                             for (RelationshipChronicleDdo r :
182                                 child.getValue().conceptProperty().get().getDestinationRelationships()) {
183                                 for (RelationshipVersionDdo rv : r.getVersions()) {
184                                     try {
185                                         TaxonomyReferenceWithConcept taxRef = new TaxonomyReferenceWithConcept(rv);
186                                         SctTreeItem grandChildItem = new SctTreeItem(taxRef, displayPolicies);
187 
188                                         if (grandChildItem.shouldDisplay()) {
189                                             grandChildrenToProcess.add(grandChildItem);
190                                             grandChildrenToAdd.add(grandChildItem);
191                                         } else {
192                                             LOG.debug("grandChildItem.shouldDisplay() == false: not adding " + grandChildItem.getConceptUuid() + " as child of " + tempChild.getConceptUuid());
193                                         }
194                                     } catch (IOException | ContradictionException ex) {
195                                         LOG.error(null, ex);
196                                     }
197                                 }
198                             }
199 
200                             Collections.sort(grandChildrenToAdd);
201                             child.getChildren().addAll(grandChildrenToAdd);
202                         }
203                     } else if (child.getValue().getConcept() == null) {
204                         grandChildrenToProcess.add((SctTreeItem) child);
205                     }
206                 } else {
207                     LOG.debug("childItem.shouldDisplay() == false: not adding " + tempChild.getConceptUuid() + " as child of " + this.getConceptUuid());
208                 }
209             }
210 
211             FetchConceptsTask fetchChildrenTask = new FetchConceptsTask(grandChildrenToProcess);
212 
213             //p1.progressProperty().bind(fetchChildrenTask.progressProperty());
214             fetchChildrenTask.progressProperty().addListener(new ChangeListener<Number>() {
215                 @Override
216                 public void changed(
217                         ObservableValue<? extends Number> observable,
218                         Number oldValue, Number newValue) {
219 //                    LOG.debug("addChildrenConceptsAndGrandchildrenItems(): ProgressIndicator progress for taxRef={} changing from {} to {}", getValue(), oldValue, newValue);
220 
221                     p1.progressProperty().set(newValue.doubleValue());
222                 }
223             });
224             FutureHelper.addFuture(childFetcherService.submit(fetchChildrenTask));
225         }
226     }

311     private static void initExecutorPools() {
312         childFetcherService = Executors.newFixedThreadPool(Math.min(6,
313                 Runtime.getRuntime().availableProcessors() + 1), new NamedThreadFactory(sctTreeItemThreadGroup,
314                 "SctTreeItem child fetcher"));
315         conceptFetcherService = Executors.newFixedThreadPool(Math.min(6,
316                 Runtime.getRuntime().availableProcessors() + 1), new NamedThreadFactory(sctTreeItemThreadGroup,
317                 "SctTreeItem concept fetcher"));
318     }

338     public static String toString(SctTreeItem item) {
339         try {
340             if (item.getValue().getRelationshipVersion() != null) {
341                 if (item.getMultiParentDepth() > 0) {
342                     ComponentReference destRef = item.getValue().getRelationshipVersion().getDestinationReference();
343                     String temp = OTFUtility.getDescription(destRef.getUuid());
344                     if (temp == null) {
345                         return destRef.getText();
346                     } else {
347                         return temp;
348                     }
349                 } else {
350                     ComponentReference originRef = item.getValue().getRelationshipVersion().getOriginReference();
351                     String temp = OTFUtility.getDescription(originRef.getUuid());
352                     if (temp == null) {
353                         return originRef.getText();
354                     } else {
355                         return temp;
356                     }
357                 }
358             }
359 
360             if (item.getValue().conceptProperty().get() != null) {
361                 return OTFUtility.getDescription(item.getValue().conceptProperty().get());
362             }
363 
364             return "root";
365         } catch (RuntimeException re) {
366             LOG.error("Caught {} \"{}\"", re.getClass().getName(), re.getLocalizedMessage());
367             throw re;
368         } catch (Error e) {
369             LOG.error("Caught {} \"{}\"", e.getClass().getName(), e.getLocalizedMessage());
370             throw e;
371         }
372     }

443     public void blockUntilChildrenReady() {
444         if (progressIndicator != null) {
445             synchronized (childFetcherService) {
446                 while (progressIndicator != null) {
447                     try {
448                         childFetcherService.wait();
449                     } catch (InterruptedException e) {
450                         // noop
451                     }
452                 }
453             }
454         }

457     /**
458      * A concrete {@link Task} to fetch concepts.
459      */
460     private final class FetchConceptsTask extends Task<Boolean> {
461 
462         List<SctTreeItem> children;
463 
464         public FetchConceptsTask(List<SctTreeItem> children) {
465             this.children = children;
466         }
467 
468         @Override
469         protected Boolean call() throws Exception {
470             int size = children.size() - 1;
471             int processedCount = 0;
472             List<Future<Boolean>> futureList = new ArrayList<>();
473 
474             for (SctTreeItem childItem : children) {
475                 if (SctTreeView.shutdownRequested) {
476                     return false;
477                 }
478                 if (childItem.getValue().conceptProperty().get() == null) {
479                     GetSctTreeItemConceptCallable fetcher = new GetSctTreeItemConceptCallable(childItem);
480                     futureList.add(conceptFetcherService.submit(fetcher));
481                 } else {
482                     updateProgress(Math.min(processedCount++, size), size);
483                 }
484             }
485 
486             updateProgress(Math.min(processedCount, size), size);
487 
488             for (Future<Boolean> future : futureList) {
489                 try {
490                     future.get();
491                     updateProgress(Math.min(processedCount++, size), size);
492                 } catch (ExecutionException ex) {
493                     if (SctTreeView.shutdownRequested) {
494                         return false;
495                     } else {
496                         LOG.error(null, ex);
497                     }
498                 } catch (InterruptedException ex) {
499                     LOG.error(null, ex);
500                 }
501             }
502 
503             if (SctTreeView.shutdownRequested) {
504                 return false;
505             }
506 
507             updateProgress(1, 1);
508 
509             Platform.runLater(new Runnable() {
510                 @Override
511                 public void run() {
512                     TaxonomyReferenceWithConcept item = SctTreeItem.this.getValue();
513 
514                     SctTreeItem.this.setValue(null);
515                     setProgressIndicator(null);
516                     SctTreeItem.this.setValue(item);
517                 }
518             });
519 
520             return true;
521         }
