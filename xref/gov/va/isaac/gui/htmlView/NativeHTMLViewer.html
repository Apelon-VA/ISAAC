19  package gov.va.isaac.gui.htmlView;
20  
21  import gov.va.isaac.AppContext;
22  import gov.va.isaac.util.Utility;
23  import java.awt.Desktop;
24  import java.io.BufferedWriter;
25  import java.io.File;
26  import java.io.FileWriter;
27  import java.net.URI;
28  import java.util.function.Supplier;
29  
30  /**
31   * {@link NativeHTMLViewer} A utility class that opens HTML in the system default web browser
32   *
33   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
34   */
35  public class NativeHTMLViewer
36  {
37  	/**
38  	 * View the passed in content in the system-native webbrowser.  htmlContent is fetched in a background thread, written to a temp file, 
39  	 * and then opened.
40  	 * @param htmlContent
41  	 */
42  	public static void viewInBrowser(Supplier<String> htmlContent)
43  	{
44  		Utility.execute(() -> 
45  		{
46  			File f = null;
47  			try
48  			{
49  				f = File.createTempFile("isaacHTMLView-", ".html");
50  				BufferedWriter w = new BufferedWriter(new FileWriter(f));
51  				w.write(htmlContent.get());
52  				w.close();
53  				viewInBrowser(f.toURI());
54  				// Sleep long enough for the browser to render, then delete the file
55  				Thread.sleep(10000);
56  			}
57  			catch (InterruptedException e)
58  			{
59  				//noop
60  			}
61  			catch (Exception e)
62  			{
63  				AppContext.getCommonDialogs().showErrorDialog("Error launching browser", "There was an error launching the web browser", null);
64  			}
65  			finally
66  			{
67  				if (f != null)
68  				{
69  					f.delete();
70  				}
71  			}
72  		});
73  	}
74  
75  	/**
76  	 * Just open the URL in the system native browser
77  	 * @param address
78  	 */
79  	public static void viewInBrowser(URI address)
80  	{
81  		try
82  		{
83  			Desktop.getDesktop().browse(address);
84  		}
85  		catch (Exception e)
86  		{
87  			AppContext.getCommonDialogs().showErrorDialog("Error launching browser", "There was an error launching the web browser", null);
88  		}
89  	}
90  }
