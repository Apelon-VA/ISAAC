50  /**
51   * {@link LightWeightDialogs}
52   *
53   * Just a simple login window. Initially used ControlsFX for this... but that broke a bunch of other things - so
54   * reproduced the 'lightweight' login window here.
55   *
56   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
57   */
58  public class LightWeightDialogs
59  {
60  	public static Node buildLoginDialog(Consumer<Boolean> loginSuccessful)
61  	{
62  		BorderPane bp = init("Login", true);
63  
64  		HBox mainContent = new HBox();
65  
66  		ImageView keys = Images.KEYS.createImageView();
67  		HBox.setMargin(keys, new Insets(10));
68  
69  		mainContent.getChildren().add(keys);
70  
71  		StackPane userStack = new StackPane();
72  		userStack.setMaxWidth(Double.MAX_VALUE);
73  		ChoiceBox<String> userName = new ChoiceBox<>();
74  		userName.setMaxWidth(Double.MAX_VALUE);
75  		userName.setItems(ExtendedAppContext.getService(UserProfileManager.class).getUsersWithProfiles());
76  		userName.getSelectionModel().select(0);
77  		userName.setPadding(new Insets(0, 0, 0, 15));
78  		String preselect = ExtendedAppContext.getService(UserProfileManager.class).getLastLoggedInUser();
79  		if (preselect != null)
80  		{
81  			userName.getSelectionModel().select(preselect);
82  		}
83  		userStack.getChildren().add(userName);
84  
85  		ImageView userImage = Images.USER.createImageView();
86  		StackPane.setAlignment(userImage, Pos.CENTER_LEFT);
87  		StackPane.setMargin(userImage, new Insets(0, 0, 0, 3));
88  		userStack.getChildren().add(userImage);
89  
90  		StackPane passwordStack = new StackPane();
91  		PasswordField txPassword = new PasswordField();
92  		txPassword.setPromptText("Password");
93  		txPassword.setPadding(new Insets(5, 0, 5, 20));
94  		final ValidBooleanBinding passwordEmpty = new ValidBooleanBinding()
95  		{
96  			{
97  				bind(txPassword.textProperty());
98  				setComputeOnInvalidate(true);
99  			}
100 
101 			@Override
102 			protected boolean computeValue()
103 			{
104 				if (txPassword.getText().length() == 0)
105 				{
106 					setInvalidReason("Empty passwords are not allowed");
107 					return false;
108 				}
109 				clearInvalidReason();
110 				return true;
111 			}
112 		};
113 
114 		ErrorMarkerUtils.setupErrorMarker(txPassword, passwordStack, passwordEmpty);
115 		ImageView lockImage = Images.LOCK.createImageView();
116 		StackPane.setMargin(lockImage, new Insets(0, 0, 0, 3));
117 		passwordStack.getChildren().add(lockImage);
118 		StackPane.setAlignment(lockImage, Pos.CENTER_LEFT);
119 
120 		Label lbMessage = new Label("");
121 		lbMessage.getStyleClass().addAll("dialog-message-banner");
122 		lbMessage.setVisible(false);
123 		lbMessage.setManaged(false);
124 
125 		final VBox content = new VBox(10);
126 		content.getChildren().add(lbMessage);
127 		content.getChildren().add(userStack);
128 		content.getChildren().add(passwordStack);
129 		HBox.setHgrow(content, Priority.ALWAYS);
130 		HBox.setMargin(content, new Insets(10));
131 
132 		mainContent.getChildren().add(content);
133 		mainContent.setMaxWidth(Double.MAX_VALUE);
134 
135 		bp.setCenter(mainContent);
136 
137 		HBox buttons = new HBox();
138 		buttons.setMaxWidth(Double.MAX_VALUE);
139 		Region r = new Region();
140 		HBox.setHgrow(r, Priority.ALWAYS);
141 		buttons.getChildren().add(r);  //filler
142 		Button cancelButton = new Button("Cancel");
143 		cancelButton.setPadding(new Insets(5, 20, 5, 20));
144 		cancelButton.setCancelButton(true);
145 		//JavaFX is silly:  https://javafx-jira.kenai.com/browse/RT-39145#comment-434189
146 		cancelButton.setOnKeyPressed(new EventHandler<KeyEvent>()
147 		{
148 			@Override
149 			public void handle(KeyEvent event)
150 			{
151 				if (event.getCode() == KeyCode.ENTER)
152 				{
153 					event.consume();
154 					cancelButton.fire();
155 				}
156 			}
157 		});
158 		Button loginButton = new Button("Login");
159 		loginButton.setDefaultButton(true);
160 		loginButton.setPadding(new Insets(5, 20, 5, 20));
161 		loginButton.disableProperty().bind(passwordEmpty.not());
162 
163 		buttons.setPadding(new Insets(10));
164 		buttons.setSpacing(5);
165 		buttons.getChildren().add(cancelButton);
166 		buttons.getChildren().add(loginButton);
167 
168 		bp.setBottom(buttons);
169 
170 		cancelButton.setOnAction((event) -> bp.setVisible(false));
171 
172 		loginButton.setOnAction((event) -> {
173 			if (ExtendedAppContext.getService(UserProfileManager.class).authenticateBoolean(userName.getValue(), txPassword.getText()))
174 			{
175 				lbMessage.setVisible(false);
176 				lbMessage.setManaged(false);
177 				bp.setVisible(false);
178 				loginSuccessful.accept(true);
179 			}
180 			else
181 			{
182 				lbMessage.setVisible(true);
183 				lbMessage.setManaged(true);
184 				lbMessage.setText("Incorrect Password");
185 				loginSuccessful.accept(false);
186 				bp.autosize();
187 			}
188 		});
189 		
190 		txPassword.setOnAction((event) -> loginButton.fire());
191 
192 		Platform.runLater(() -> txPassword.requestFocus());
193 		bp.autosize();
194 		return bp;
195 	}

197 	public static Node buildLoadingDialog()
198 	{
199 		BorderPane bp = init("Please wait", false);
200 
201 		bp.setCenter(new Label("Loading the database"));
202 		ProgressBar pb = new ProgressBar(-1);
203 		
204 		pb.setMaxWidth(Double.MAX_VALUE);
205 		bp.setBottom(pb);
206 		BorderPane.setMargin(bp.getCenter(), new Insets(20));
207 		BorderPane.setMargin(bp.getBottom(), new Insets(0, 20, 20, 20));
208 		bp.autosize();
209 		return bp;
210 	}
211 
212 	private static BorderPane init(String title, boolean showCloseButton)
213 	{
214 		BorderPane bp = new BorderPane();
215 		bp.setMinWidth(400);
216 		bp.setMaxWidth(400);
217 		bp.setMaxHeight(Control.USE_PREF_SIZE);
218 		bp.getStyleClass().add("dialog");
219 
220 		HBox titleBar = new HBox();
221 		titleBar.setPadding(new Insets(5));
222 		titleBar.getStyleClass().add("dialog-window-header");
223 		bp.setTop(titleBar);
224 
225 		Label login = new Label(title);
226 		login.setTextFill(Paint.valueOf("#ebebeb"));
227 		login.setFont(new Font(15));
228 		login.setPadding(new Insets(5, 5, 5, 2));
229 		titleBar.getChildren().add(login);
230 
231 		if (showCloseButton)
232 		{
233 			Region r = new Region();
234 			HBox.setHgrow(r, Priority.ALWAYS);
235 			titleBar.getChildren().add(r);  //filler
236 			
237 			Button b = new Button();
238 			b.setMnemonicParsing(false);
239 			b.setStyle("-fx-cursor:hand");
240 			b.getStyleClass().add("window-close-button");
241 			b.setMinSize(17, 17);
242 			b.setPrefSize(17, 17);
243 			b.setMaxHeight(Double.MAX_VALUE);
244 			b.setPadding(new Insets(4.5));
245 			b.setScaleShape(false);
246 			b.setAlignment(Pos.CENTER);
247 			b.setOnAction((event) -> bp.setVisible(false));
248 			titleBar.getChildren().add(b);
249 		}
250 		return bp;
251 	}
