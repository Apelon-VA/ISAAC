209 		searchInRefex.getConceptProperty().addListener(new InvalidationListener()
210 		{
211 			@Override
212 			public void invalidated(Observable observable)
213 			{
214 				ConceptVersionBI newValue = searchInRefex.getConceptProperty().get();
215 				if (newValue != null)
216 				{
217 					searchInColumnsHolder.getChildren().clear();
218 					try
219 					{
220 						RefexDynamicUsageDescription rdud = RefexDynamicUsageDescription.read(newValue.getNid());
221 						displayIndexConfigMenu_.set(true);
222 						currentlyEnteredAssemblageNid = rdud.getRefexUsageDescriptorNid();
223 						Integer[] indexedColumns = LuceneDynamicRefexIndexerConfiguration.readIndexInfo(currentlyEnteredAssemblageNid);
224 						if (indexedColumns == null || indexedColumns.length == 0)
225 						{
226 							searchInRefex.isValid().setInvalid("Sememe searches can only be performed on indexed columns in the sememe.  The selected "
227 									+ "sememe does not contain any indexed data columns.  Please configure the indexes to search this sememe.");
228 							optionsContentVBox.getChildren().remove(searchInColumnsHolder);
229 						}
230 						else
231 						{
232 							Label l = new Label("Search in Columns");
233 							searchInColumnsHolder.getChildren().add(l);
234 							l.minWidthProperty().bind(((Label)searchInRefexHBox.getChildren().get(0)).widthProperty());
235 							RefexDynamicColumnInfo[] rdci = rdud.getColumnInfo();
236 							if (rdci.length > 0)
237 							{
238 								Arrays.sort(rdci);  //We will depend on them being in the correct order later.
239 								HashSet<Integer> indexedColumnsSet = new HashSet<>(Arrays.asList(indexedColumns));
240 								int indexNumber = 0;
241 								for (RefexDynamicColumnInfo ci : rdci)
242 								{
243 									StackPane cbStack = new StackPane();
244 									CheckBox cb = new CheckBox(ci.getColumnName());
245 									if (ci.getColumnDataType() == RefexDynamicDataType.BYTEARRAY || !indexedColumnsSet.contains(indexNumber))
246 									{
247 										cb.setDisable(true);  //No index on this column... not searchable
248 										Tooltip.install(cbStack, new Tooltip("Column Datatype: " + ci.getColumnDataType().getDisplayName() + " is not indexed"));
249 									}
250 									else
251 									{
252 										cb.setSelected(true);
253 										cb.setTooltip(new Tooltip("Column Datatype: " + ci.getColumnDataType().getDisplayName()));
254 									}
255 									cbStack.getChildren().add(cb);
256 									searchInColumnsHolder.getChildren().add(cbStack);
257 									indexNumber++;
258 								}
259 								optionsContentVBox.getChildren().add(searchInColumnsHolder);
260 							}
261 							else
262 							{
263 								searchInRefex.isValid().setInvalid("Sememe searches can only be performed on the data in the sememe.  The selected "
264 										+ "sememe does not contain any data columns.");
265 								optionsContentVBox.getChildren().remove(searchInColumnsHolder);
266 							}
267 						}
268 					}
269 					catch (Exception e1)
270 					{
271 						searchInRefex.isValid().setInvalid("Sememe searches can only be limited to valid Dynamic Sememe Assemblage concept types."
272 								+ "  The current value is not a Dynamic Sememe Assemblage concept.");
273 						displayIndexConfigMenu_.set(false);
274 						currentlyEnteredAssemblageNid = null;
275 						optionsContentVBox.getChildren().remove(searchInColumnsHolder);
276 						searchInColumnsHolder.getChildren().clear();
277 					}
278 				}
279 				else
280 				{
281 					currentlyEnteredAssemblageNid = null;
282 					optionsContentVBox.getChildren().remove(searchInColumnsHolder);
283 					searchInColumnsHolder.getChildren().clear();
284 				}
285 				
286 			}

335 			@Override
336 			public ListCell<CompositeSearchResult> call(ListView<CompositeSearchResult> arg0)
337 			{
338 				return new ListCell<CompositeSearchResult>()
339 				{
340 					@Override
341 					protected void updateItem(final CompositeSearchResult item, boolean empty)
342 					{
343 						super.updateItem(item, empty);
344 						if (!empty)
345 						{
346 							VBox box = new VBox();
347 							box.setFillWidth(true);
348 							
349 							final ConceptVersionBI wbConcept = item.getContainingConcept();
350 							String preferredText = (wbConcept != null ? OTFUtility.getDescription(wbConcept) : "Not on path!");
351 						
352 							if (item.getMatchingComponents().size() > 0 && item.getMatchingComponents().iterator().next() instanceof RefexDynamicVersionBI<?>)
353 							{
354 								HBox hb = new HBox();
355 								Label concept = new Label("Referenced Concept");
356 								concept.getStyleClass().add("boldLabel");
357 								hb.getChildren().add(concept);
358 								hb.getChildren().add(new Label("  " + preferredText));
359 								
360 								box.getChildren().add(hb);
361 							
362 								for (ComponentVersionBI c : item.getMatchingComponents())
363 								{
364 									if (c instanceof RefexDynamicVersionBI<?>)
365 									{
366 										RefexDynamicVersionBI<?> rv = (RefexDynamicVersionBI<?>)c;
367 										HBox assemblageConBox = new HBox();
368 										Label assemblageCon = new Label("Assemblage Concept");
369 										assemblageCon.getStyleClass().add("boldLabel");
370 										HBox.setMargin(assemblageCon, new Insets(0.0, 0.0, 0.0, 10.0));
371 										assemblageConBox.getChildren().add(assemblageCon);
372 										assemblageConBox.getChildren().add(new Label("  " + OTFUtility.getDescription(rv.getAssemblageNid())));
373 										box.getChildren().add(assemblageConBox);
374 
375 										Label attachedData = new Label("Attached Data");
376 										attachedData.getStyleClass().add("boldLabel");
377 										VBox.setMargin(attachedData, new Insets(0.0, 0.0, 0.0, 10.0));
378 										box.getChildren().add(attachedData);
379 										
380 										try
381 										{
382 											RefexDynamicColumnInfo[] ci = rv.getRefexDynamicUsageDescription().getColumnInfo();
383 											int i = 0;
384 											
385 											for (RefexDynamicDataBI data : rv.getData())
386 											{
387 												Label l = new Label();
388 												if (data == null)  //might be an unset column, if the col is optional
389 												{
390 													continue;
391 												}
392 												if (RefexDynamicDataType.BYTEARRAY == data.getRefexDataType())
393 												{
394 													l.setText(ci[i].getColumnName() +  " - [Binary]");
395 												}
396 												else
397 												{
398 													l.setText(ci[i].getColumnName() + " - " + data.getDataObject().toString());
399 												}
400 												VBox.setMargin(l, new Insets(0.0, 0.0, 0.0, 20.0));
401 												box.getChildren().add(l);
402 												i++;
403 											}
404 										}
405 										catch (Exception e)
406 										{
407 											LOG.error("Unexpected error reading sememe info", e);
408 										}
409 									}
410 									else
411 									{
412 										LOG.warn("Unexpected type on match: {}", c);
413 									}
414 								}
415 							}
416 							else
417 							{
418 								Label concept = new Label(preferredText);
419 								concept.getStyleClass().add("boldLabel");
420 								box.getChildren().add(concept);
421 								for (String s : item.getMatchingStrings())
422 								{
423 									if (s.equals(preferredText))
424 									{
425 										continue;
426 									}
427 									Label matchString = new Label(s);
428 									VBox.setMargin(matchString, new Insets(0.0, 0.0, 0.0, 10.0));
429 									box.getChildren().add(matchString);
430 								}
431 							}
432 							setGraphic(box);
433 
434 							// Also show concept details on double-click.
435 							setOnMouseClicked(new EventHandler<MouseEvent>()
436 							{
437 								@Override
438 								public void handle(MouseEvent mouseEvent)
439 								{
440 									if (mouseEvent.getButton().equals(MouseButton.PRIMARY))
441 									{
442 										if (mouseEvent.getClickCount() == 2)
443 										{
444 											AppContext.getCommonDialogs().showConceptDialog(wbConcept.getUUIDs().get(0));
445 										}
446 									}
447 								}
448 							});

626 	@Override
627 	public void taskComplete(long taskStartTime, Integer taskId)
628 	{
629 
630 		// Run on JavaFX thread.
631 		Platform.runLater(() -> 
632 		{
633 			try
634 			{
635 				if (!ssh.isCancelled())
636 				{
637 					searchResults.getItems().addAll(ssh.getResults());
638 					long time = System.currentTimeMillis() - ssh.getSearchStartTime();
639 					float inSeconds = (float)time / 1000f;
640 					inSeconds = ((float)((int)(inSeconds * 100f)) / 100f);
641 					statusLabel.setText(ssh.getHitCount() + " in " + inSeconds + " seconds");
642 				}
643 				else
644 				{
645 					statusLabel.setText("Search Cancelled");
646 				}
647 			}
648 			catch (Exception ex)
649 			{
650 				String title = "Unexpected Search Error";
651 				LOG.error(title, ex);
652 				AppContext.getCommonDialogs().showErrorDialog(title, "There was an unexpected error running the search", ex.toString());
653 				searchResults.getItems().clear();
654 				statusLabel.setText("Search Failed");
655 			}
656 			finally
657 			{
658 				searchRunning.set(false);
659 			}
660 		});
661 	}

663 	private Integer[] getSearchColumns()
664 	{
665 		if (searchInColumnsHolder.getChildren().size() > 1)
666 		{
667 			ArrayList<Integer> result = new ArrayList<>();
668 			int deselectedCount = 0;
669 			for (int i = 1; i < searchInColumnsHolder.getChildren().size(); i++)
670 			{
671 				CheckBox cb = ((CheckBox)((StackPane)searchInColumnsHolder.getChildren().get(i)).getChildren().get(0));
672 				if (cb.isSelected())
673 				{
674 					result.add(i - 1);
675 				}
676 				else if (!cb.isDisable())
677 				{
678 					deselectedCount++;
679 				}
680 			}
681 			//If they didn't uncheck any, its more efficient to query without the column filter.
682 			return deselectedCount == 0 ? null : result.toArray(new Integer[0]);
683 		}
684 		return null;
685 	}

687 	private synchronized void search()
688 	{
689 		try
690 		{
691 			// Sanity check if search already running.
692 			if (searchRunning.get())
693 			{
694 				return;
695 			}
696 	
697 			searchRunning.set(true);
698 			searchResults.getItems().clear();
699 			// we get called back when the results are ready.
700 			
701 			if (searchIn.getValue() == SearchInOptions.Descriptions)
702 			{
703 				ssh = SearchHandler.descriptionSearch(searchText.getText(), searchLimit.getValue(), this, true);
704 			}
705 			else
706 			{
707 				String searchString = searchText.getText().trim();
708 				try
709 				{
710 					RefexDynamicDataBI data = NumberUtilities.wrapIntoRefexHolder(NumberUtilities.parseNumber(searchString));
711 					LOG.debug("Doing a sememe search with a numeric value");
712 					ssh = SearchHandler.dynamicRefexSearch((indexer) ->
713 					{
714 						try
715 						{
716 							return indexer.query(data, currentlyEnteredAssemblageNid, false, getSearchColumns(), searchLimit.getValue(), null);
717 						}
718 						catch (Exception e)
719 						{
720 							throw new RuntimeException(e);
721 						}
722 					}, this, null, null, null, true);
723 				}
724 				catch (NumberFormatException e)
725 				{
726 					//Not a number...  is it a valid interval?
727 					try
728 					{
729 						Interval interval = new Interval(searchString);
730 						LOG.debug("Doing a sememe search with an interval value");
731 						ssh = SearchHandler.dynamicRefexSearch((indexer) ->
732 						{
733 							try
734 							{
735 								return indexer.queryNumericRange(NumberUtilities.wrapIntoRefexHolder(interval.getLeft()), interval.isLeftInclusive(), 
736 										NumberUtilities.wrapIntoRefexHolder(interval.getRight()), interval.isRightInclusive(), 
737 										currentlyEnteredAssemblageNid, getSearchColumns(), searchLimit.getValue(), null);
738 							}
739 							catch (Exception e1)
740 							{
741 								throw new RuntimeException(e1);
742 							}
743 						}, this, null, null, null, true);
744 					}
745 					catch (NumberFormatException e1) 
746 					{
747 						//run it as a string search
748 						LOG.debug("Doing a sememe search as a string search");
749 						ssh = SearchHandler.dynamicRefexSearch((indexer) ->
750 						{
751 							try
752 							{
753 								return indexer.query(new RefexDynamicString(searchText.getText()), currentlyEnteredAssemblageNid, false, 
754 										getSearchColumns(), searchLimit.getValue(), null);
755 							}
756 							catch (Exception e2)
757 							{
758 								throw new RuntimeException(e2);
759 							}
760 						}, this, null, null, null, true);
761 					}
762 				}
763 			}
764 		}
765 		catch (Exception e)
766 		{
767 			LOG.error("Search imploded unexpectedly...", e);
768 			ssh = null;  //force a null ptr in taskComplete, so an error is displayed.
769 			taskComplete(0, null);
770 		}
771 	}

774 	private void populateDynamicRefexList()
775 	{
776 		Task<Void> t = new Task<Void>()
777 		{
778 			HashSet<SimpleDisplayConcept> dynamicRefexAssemblages = new HashSet<>();
779 
780 			@Override
781 			protected Void call() throws Exception
782 			{
783 				dynamicRefexAssemblages = new HashSet<>();
784 				
785 				LuceneDynamicRefexIndexer indexer = AppContext.getService(LuceneDynamicRefexIndexer.class);
786 				List<SearchResult> refexes = indexer.queryAssemblageUsage(RefexDynamic.REFEX_DYNAMIC_DEFINITION_DESCRIPTION.getNid(), 1000, null);
787 				for (SearchResult sr : refexes)
788 				{
789 					RefexDynamicChronicleBI<?> rc = (RefexDynamicChronicleBI<?>) ExtendedAppContext.getDataStore().getComponent(sr.getNid());
790 					//These are nested refex references - it returns a description component - concept we want is the parent of that.
791 					dynamicRefexAssemblages.add(new SimpleDisplayConcept(
792 							ExtendedAppContext.getDataStore().getComponent(rc.getReferencedComponentNid()).getEnclosingConcept(),
793 							null));
794 					
795 				}
796 				return null;
797 			}
798 
799 			/**
800 			 * @see javafx.concurrent.Task#succeeded()
801 			 */
802 			@Override
803 			protected void succeeded()
804 			{
805 				dynamicRefexList_.clear();
806 				dynamicRefexList_.addAll(dynamicRefexAssemblages);
807 				dynamicRefexList_.sort(new Comparator<SimpleDisplayConcept>()
808 				{
809 					@Override
810 					public int compare(SimpleDisplayConcept o1, SimpleDisplayConcept o2)
811 					{
812 						return o1.getDescription().compareToIgnoreCase(o2.getDescription());
813 					}
814 				});
815 			}
816 		};
817 		
818 		Utility.execute(t);
819 	}
