220 			@Override
221 			public void invalidated(Observable observable)
222 			{
223 				ConceptVersionBI newValue = searchInRefex.getConceptProperty().get();
224 				if (newValue != null)
225 				{
226 					searchInColumnsHolder.getChildren().clear();
227 					try
228 					{
229 						RefexDynamicUsageDescription rdud = RefexDynamicUsageDescription.read(newValue.getNid());
230 						displayIndexConfigMenu_.set(true);
231 						currentlyEnteredAssemblageNid = rdud.getRefexUsageDescriptorNid();
232 						Integer[] indexedColumns = LuceneDynamicRefexIndexerConfiguration.readIndexInfo(currentlyEnteredAssemblageNid);
233 						if (indexedColumns == null || indexedColumns.length == 0)
234 						{
235 							searchInRefex.isValid().setInvalid("Sememe searches can only be performed on indexed columns in the sememe.  The selected "
236 									+ "sememe does not contain any indexed data columns.  Please configure the indexes to search this sememe.");
237 							optionsContentVBox.getChildren().remove(searchInColumnsHolder);
238 						}
239 						else
240 						{
241 							Label l = new Label("Search in Columns");
242 							searchInColumnsHolder.getChildren().add(l);
243 							l.minWidthProperty().bind(((Label)searchInRefexHBox.getChildren().get(0)).widthProperty());
244 							RefexDynamicColumnInfo[] rdci = rdud.getColumnInfo();
245 							if (rdci.length > 0)
246 							{
247 								Arrays.sort(rdci);  //We will depend on them being in the correct order later.
248 								HashSet<Integer> indexedColumnsSet = new HashSet<>(Arrays.asList(indexedColumns));
249 								int indexNumber = 0;
250 								for (RefexDynamicColumnInfo ci : rdci)
251 								{
252 									StackPane cbStack = new StackPane();
253 									CheckBox cb = new CheckBox(ci.getColumnName());
254 									if (ci.getColumnDataType() == RefexDynamicDataType.BYTEARRAY || !indexedColumnsSet.contains(indexNumber))
255 									{
256 										cb.setDisable(true);  //No index on this column... not searchable
257 										Tooltip.install(cbStack, new Tooltip("Column Datatype: " + ci.getColumnDataType().getDisplayName() + " is not indexed"));
258 									}
259 									else
260 									{
261 										cb.setSelected(true);
262 										cb.setTooltip(new Tooltip("Column Datatype: " + ci.getColumnDataType().getDisplayName()));
263 									}
264 									cbStack.getChildren().add(cb);
265 									searchInColumnsHolder.getChildren().add(cbStack);
266 									indexNumber++;
267 								}
268 								if (!optionsContentVBox.getChildren().contains(searchInColumnsHolder))
269 								{
270 									optionsContentVBox.getChildren().add(searchInColumnsHolder);
271 								}
272 							}
273 							else
274 							{
275 								searchInRefex.isValid().setInvalid("Sememe searches can only be performed on the data in the sememe.  The selected "
276 										+ "sememe does not contain any data columns.");
277 								optionsContentVBox.getChildren().remove(searchInColumnsHolder);
278 							}
279 						}
280 					}
281 					catch (Exception e1)
282 					{
283 						searchInRefex.isValid().setInvalid("Sememe searches can only be limited to valid Dynamic Sememe Assemblage concept types."
284 								+ "  The current value is not a Dynamic Sememe Assemblage concept.");
285 						LOG.debug("Exception while checking is sememe concept field in search box was a dynamic sememe", e1);
286 						displayIndexConfigMenu_.set(false);
287 						currentlyEnteredAssemblageNid = null;
288 						optionsContentVBox.getChildren().remove(searchInColumnsHolder);
289 						searchInColumnsHolder.getChildren().clear();
290 					}
291 				}
292 				else
293 				{
294 					currentlyEnteredAssemblageNid = null;
295 					optionsContentVBox.getChildren().remove(searchInColumnsHolder);
296 					searchInColumnsHolder.getChildren().clear();
297 				}

364 			@Override
365 			public ListCell<CompositeSearchResult> call(ListView<CompositeSearchResult> arg0)
366 			{
367 				return new ListCell<CompositeSearchResult>()
368 				{
369 					@Override
370 					protected void updateItem(final CompositeSearchResult item, boolean empty)
371 					{
372 						super.updateItem(item, empty);
373 						if (!empty)
374 						{
375 							VBox box = new VBox();
376 							box.setFillWidth(true);
377 							
378 							final ConceptVersionBI wbConcept = item.getContainingConcept();
379 							String preferredText = (wbConcept != null ? OTFUtility.getDescription(wbConcept) : "Not on path!");
380 						
381 							if (item.getMatchingComponents().size() > 0 && item.getMatchingComponents().iterator().next() instanceof RefexDynamicVersionBI<?>)
382 							{
383 								HBox hb = new HBox();
384 								Label concept = new Label("Referenced Concept");
385 								concept.getStyleClass().add("boldLabel");
386 								hb.getChildren().add(concept);
387 								hb.getChildren().add(new Label("  " + preferredText));
388 								
389 								box.getChildren().add(hb);
390 							
391 								for (ComponentVersionBI c : item.getMatchingComponents())
392 								{
393 									if (c instanceof RefexDynamicVersionBI<?>)
394 									{
395 										RefexDynamicVersionBI<?> rv = (RefexDynamicVersionBI<?>)c;
396 										HBox assemblageConBox = new HBox();
397 										Label assemblageCon = new Label("Assemblage Concept");
398 										assemblageCon.getStyleClass().add("boldLabel");
399 										HBox.setMargin(assemblageCon, new Insets(0.0, 0.0, 0.0, 10.0));
400 										assemblageConBox.getChildren().add(assemblageCon);
401 										assemblageConBox.getChildren().add(new Label("  " + OTFUtility.getDescription(rv.getAssemblageNid())));
402 										box.getChildren().add(assemblageConBox);
403 
404 										Label attachedData = new Label("Attached Data");
405 										attachedData.getStyleClass().add("boldLabel");
406 										VBox.setMargin(attachedData, new Insets(0.0, 0.0, 0.0, 10.0));
407 										box.getChildren().add(attachedData);
408 										
409 										try
410 										{
411 											RefexDynamicColumnInfo[] ci = rv.getRefexDynamicUsageDescription().getColumnInfo();
412 											int i = 0;
413 											
414 											for (RefexDynamicDataBI data : rv.getData())
415 											{
416 												Label l = new Label();
417 												if (data == null)  //might be an unset column, if the col is optional
418 												{
419 													continue;
420 												}
421 												if (RefexDynamicDataType.BYTEARRAY == data.getRefexDataType())
422 												{
423 													l.setText(ci[i].getColumnName() +  " - [Binary]");
424 												}
425 												else
426 												{
427 													l.setText(ci[i].getColumnName() + " - " + data.getDataObject().toString());
428 												}
429 												VBox.setMargin(l, new Insets(0.0, 0.0, 0.0, 20.0));
430 												box.getChildren().add(l);
431 												i++;
432 											}
433 										}
434 										catch (Exception e)
435 										{
436 											LOG.error("Unexpected error reading sememe info", e);
437 										}
438 									}
439 									else
440 									{
441 										LOG.warn("Unexpected type on match: {}", c);
442 									}
443 								}
444 							}
445 							else
446 							{
447 								Label concept = new Label(preferredText);
448 								concept.getStyleClass().add("boldLabel");
449 								box.getChildren().add(concept);
450 								for (String s : item.getMatchingStrings())
451 								{
452 									if (s.equals(preferredText))
453 									{
454 										continue;
455 									}
456 									Label matchString = new Label(s);
457 									VBox.setMargin(matchString, new Insets(0.0, 0.0, 0.0, 10.0));
458 									box.getChildren().add(matchString);
459 								}
460 							}
461 							setGraphic(box);
462 
463 							// Also show concept details on double-click.
464 							setOnMouseClicked(new EventHandler<MouseEvent>()
465 							{
466 								@Override
467 								public void handle(MouseEvent mouseEvent)
468 								{
469 									if (mouseEvent.getButton().equals(MouseButton.PRIMARY))
470 									{
471 										if (mouseEvent.getClickCount() == 2)
472 										{
473 											AppContext.getCommonDialogs().showConceptDialog(wbConcept.getUUIDs().get(0));
474 										}
475 									}
476 								}
477 							});

659 	@Override
660 	public void taskComplete(long taskStartTime, Integer taskId)
661 	{
662 
663 		// Run on JavaFX thread.
664 		Platform.runLater(() -> 
665 		{
666 			try
667 			{
668 				if (!ssh.isCancelled())
669 				{
670 					searchResults.getItems().addAll(ssh.getResults());
671 					long time = System.currentTimeMillis() - ssh.getSearchStartTime();
672 					float inSeconds = (float)time / 1000f;
673 					inSeconds = ((float)((int)(inSeconds * 100f)) / 100f);
674 					statusLabel.setText(ssh.getHitCount() + " in " + inSeconds + " seconds");
675 				}
676 				else
677 				{
678 					statusLabel.setText("Search Cancelled");
679 				}
680 			}
681 			catch (Exception ex)
682 			{
683 				String title = "Unexpected Search Error";
684 				LOG.error(title, ex);
685 				AppContext.getCommonDialogs().showErrorDialog(title, "There was an unexpected error running the search", ex.toString());
686 				searchResults.getItems().clear();
687 				statusLabel.setText("Search Failed");
688 			}
689 			finally
690 			{
691 				searchRunning.set(false);
692 			}
693 		});
694 	}

696 	private Integer[] getSearchColumns()
697 	{
698 		if (searchInColumnsHolder.getChildren().size() > 1)
699 		{
700 			ArrayList<Integer> result = new ArrayList<>();
701 			int deselectedCount = 0;
702 			for (int i = 1; i < searchInColumnsHolder.getChildren().size(); i++)
703 			{
704 				CheckBox cb = ((CheckBox)((StackPane)searchInColumnsHolder.getChildren().get(i)).getChildren().get(0));
705 				if (cb.isSelected())
706 				{
707 					result.add(i - 1);
708 				}
709 				else if (!cb.isDisable())
710 				{
711 					deselectedCount++;
712 				}
713 			}
714 			//If they didn't uncheck any, its more efficient to query without the column filter.
715 			return deselectedCount == 0 ? null : result.toArray(new Integer[0]);
716 		}
717 		return null;
718 	}

720 	private synchronized void search()
721 	{
722 		try
723 		{
724 			// Sanity check if search already running.
725 			if (searchRunning.get())
726 			{
727 				return;
728 			}
729 	
730 			searchRunning.set(true);
731 			searchResults.getItems().clear();
732 			// we get called back when the results are ready.
733 			
734 			if (searchIn.getValue() == SearchInOptions.Descriptions)
735 			{
736 				if (descriptionTypeSelection.getValue().getNid() == Integer.MIN_VALUE)
737 				{
738 					LOG.debug("Doing a description search across all description types");
739 					ssh = SearchHandler.descriptionSearch(searchText.getText(), searchLimit.getValue(), this, true);
740 				}
741 				else if (descriptionTypeSelection.getValue().getNid() == LuceneDescriptionType.FSN.ordinal())
742 				{
743 					LOG.debug("Doing a description search on FSN");
744 					ssh = SearchHandler.descriptionSearch(searchText.getText(), searchLimit.getValue(), false, LuceneDescriptionType.FSN, 
745 							this, null, null, null, true);
746 				}
747 				else if (descriptionTypeSelection.getValue().getNid() == LuceneDescriptionType.DEFINITION.ordinal())
748 				{
749 					LOG.debug("Doing a description search on Definition");
750 					ssh = SearchHandler.descriptionSearch(searchText.getText(), searchLimit.getValue(), false, LuceneDescriptionType.DEFINITION, 
751 							this, null, null, null, true);
752 				}
753 				else if (descriptionTypeSelection.getValue().getNid() == LuceneDescriptionType.SYNONYM.ordinal())
754 				{
755 					LOG.debug("Doing a description search on Synonym");
756 					ssh = SearchHandler.descriptionSearch(searchText.getText(), searchLimit.getValue(), false, LuceneDescriptionType.SYNONYM, 
757 							this, null, null, null, true);
758 				}
759 				else
760 				{
761 					LOG.debug("Doing a description search on the extended type {}", descriptionTypeSelection.getValue().getDescription());
762 					ssh = SearchHandler.descriptionSearch(searchText.getText(), searchLimit.getValue(), false, 
763 							ExtendedAppContext.getDataStore().getUuidPrimordialForNid(descriptionTypeSelection.getValue().getNid()), 
764 							this, null, null, null, true);
765 				}
766 			}
767 			else
768 			{
769 				String searchString = searchText.getText().trim();
770 				try
771 				{
772 					RefexDynamicDataBI data = NumberUtilities.wrapIntoRefexHolder(NumberUtilities.parseNumber(searchString));
773 					LOG.debug("Doing a sememe search with a numeric value");
774 					ssh = SearchHandler.dynamicRefexSearch((indexer) ->
775 					{
776 						try
777 						{
778 							return indexer.query(data, currentlyEnteredAssemblageNid, false, getSearchColumns(), searchLimit.getValue(), null);
779 						}
780 						catch (Exception e)
781 						{
782 							throw new RuntimeException(e);
783 						}
784 					}, this, null, null, null, true);
785 				}
786 				catch (NumberFormatException e)
787 				{
788 					//Not a number...  is it a valid interval?
789 					try
790 					{
791 						Interval interval = new Interval(searchString);
792 						LOG.debug("Doing a sememe search with an interval value");
793 						ssh = SearchHandler.dynamicRefexSearch((indexer) ->
794 						{
795 							try
796 							{
797 								return indexer.queryNumericRange(NumberUtilities.wrapIntoRefexHolder(interval.getLeft()), interval.isLeftInclusive(), 
798 										NumberUtilities.wrapIntoRefexHolder(interval.getRight()), interval.isRightInclusive(), 
799 										currentlyEnteredAssemblageNid, getSearchColumns(), searchLimit.getValue(), null);
800 							}
801 							catch (Exception e1)
802 							{
803 								throw new RuntimeException(e1);
804 							}
805 						}, this, null, null, null, true);
806 					}
807 					catch (NumberFormatException e1) 
808 					{
809 						//run it as a string search
810 						LOG.debug("Doing a sememe search as a string search");
811 						ssh = SearchHandler.dynamicRefexSearch((indexer) ->
812 						{
813 							try
814 							{
815 								return indexer.query(new RefexDynamicString(searchText.getText()), currentlyEnteredAssemblageNid, false, 
816 										getSearchColumns(), searchLimit.getValue(), null);
817 							}
818 							catch (Exception e2)
819 							{
820 								throw new RuntimeException(e2);
821 							}
822 						}, this, null, null, null, true);
823 					}
824 				}
825 			}
826 		}
827 		catch (Exception e)
828 		{
829 			LOG.error("Search imploded unexpectedly...", e);
830 			ssh = null;  //force a null ptr in taskComplete, so an error is displayed.
831 			taskComplete(0, null);
832 		}
833 	}
