39  /**
40   * NonSearchTypeFilter
41   * 
42   * @author <a href="mailto:joel.kniaz@gmail.com">Joel Kniaz</a>
43   *
44   */
45  public abstract class NonSearchTypeFilter<T extends NonSearchTypeFilter<T>> implements Filter<T> {
46  	public abstract Set<Integer> gatherNoSearchTermCaseList(Set<Integer> startList);
47  	abstract IntegerProperty getSingleNid();
48  
49  	protected Set<Integer> getSingleNidNoSearchTermCaseList(Set<Integer> startList) {
50  		Set<Integer> mergedSet = new HashSet<Integer>();
51  
52  		try {
53  			ConceptVersionBI con = OTFUtility.getConceptVersion(getSingleNid().get());
54  			ConceptVersionBI rootCon = OTFUtility.getRootConcept(con);
55  
56  			NativeIdSetBI allConcepts = ExtendedAppContext.getDataStore().getAllConceptNids();
57  			NoSearchTermConcurrentSearcher searcher = new NoSearchTermConcurrentSearcher(allConcepts, rootCon.getConceptNid());
58  			ExtendedAppContext.getDataStore().iterateConceptDataInParallel(searcher);
59  
60  			if (!startList.isEmpty()) {
61  				for (Integer examCon : startList) {
62  					if (searcher.getResults().contains(examCon)) {
63  						mergedSet.add(examCon);
64  					}
65  				}
66  			} else {
67  				mergedSet.addAll(searcher.getResults());
68  			}
69  		} catch (Exception e) {
70  			Log.error("Cannot find calculate the NoSearchTermCaseList", e);
71  		}
72  
73  		return mergedSet;
74  	}
75  }
