95  		//init had to be delayed, because the current init runs way to slow, and hits the DB in the JavaFX thread.
96  		if (svc_ == null)

98  			synchronized (this)
99  			{

101 				{
102 					try
103 					{
104 						svc_ = EnhancedSearchViewController.init();
105 					}
106 					catch (IOException e)
107 					{
108 						LOG.error("Unexpected error initializing the Search View", e);
109 						return new Label("oops - check logs");
110 					}
111 				}
112 			}
113 			
114 		}
115 		return svc_.getRoot();
116 	}
