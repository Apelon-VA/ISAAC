25  package gov.va.isaac.gui.enhancedsearchview;
26  
27  import java.util.HashSet;
28  import java.util.Set;
29  
30  import javafx.scene.Node;
31  import gov.va.isaac.gui.util.Images;
32  import gov.va.isaac.interfaces.gui.views.commonFunctionality.taxonomyView.SctTreeItemDisplayPolicies;
33  import gov.va.isaac.interfaces.gui.views.commonFunctionality.taxonomyView.SctTreeItemI;
34  import javafx.beans.property.BooleanProperty;
35  import javafx.beans.property.SimpleBooleanProperty;
36  import javafx.scene.layout.HBox;
37  
38  /**
39   * SctTreeItemSearchResultsDisplayPolicies
40   * 
41   * @author <a href="mailto:joel.kniaz@gmail.com">Joel Kniaz</a>
42   *
43   */
44  public class SctTreeItemSearchResultsDisplayPolicies implements SctTreeItemDisplayPolicies {
45  	
46  	private final SctTreeItemDisplayPolicies simplePolicies;
47  	
48  	private BooleanProperty shouldFilter = new SimpleBooleanProperty(false);
49  	
50  	Set<Integer> searchResultAncestors = new HashSet<>();
51  	Set<Integer> searchResults = new HashSet<>();
52  	
53  	public SctTreeItemSearchResultsDisplayPolicies(SctTreeItemDisplayPolicies simplePolicies) {
54  		this.simplePolicies = simplePolicies;
55  	}
56  	
57  	public BooleanProperty getFilterMode() {
58  		return shouldFilter;
59  	}
60  	public void setFilterMode(BooleanProperty filterMode) {
61  		this.shouldFilter = filterMode;
62  	}
63  
64  	public Set<Integer> getSearchResultAncestors() {
65  		return searchResultAncestors;
66  	}
67  	public void setSearchResultAncestors(Set<Integer> set) {
68  		searchResultAncestors = set;
69  	}
70  
71  	public Set<Integer> getSearchResults() {
72  		return searchResults;
73  	}
74  	public void setSearchResults(Set<Integer> set) {
75  		searchResults = set;
76  	}
77  	
78  	/* (non-Javadoc)
79  	 * @see gov.va.isaac.interfaces.treeview.SctTreeItemDisplayPolicies#computeGraphic(gov.va.isaac.interfaces.treeview.SctTreeItemI)
80  	 */
81  	@Override
82  	public Node computeGraphic(SctTreeItemI item) {
83  		HBox hbox = new HBox();
84  		
85  		hbox.getChildren().add(simplePolicies.computeGraphic(item));
86  		
87  		if (item.getConceptNid() != null) {
88  			if (getSearchResultAncestors().contains(item.getConceptNid())) {
89  				hbox.getChildren().add(Images.TAXONOMY_SEARCH_RESULT_ANCESTOR.createImageView());
90  			}
91  			if (getSearchResults().contains(item.getConceptNid())) {
92  				hbox.getChildren().add(Images.TAXONOMY_SEARCH_RESULT.createImageView());
93  			}
94  		}
95  		
96  		return hbox;
97  	}
98  
99  	@Override
100 	public boolean shouldDisplay(SctTreeItemI treeItem) {
101 		if (! shouldFilter.get()) {
102 			// if the shouldFilter boolean property is set to false then display
103 			return true;
104 		} 
105 		else if (treeItem.getConceptNid() == null) {
106 			// if the treeItem nid is null, it may be a special node (i.e. hidden, etc), so display
107 			return true;
108 		}
109 		else {
110 			// do filtering
111 			if (getSearchResultAncestors().contains(treeItem.getConceptNid()) || getSearchResults().contains(treeItem.getConceptNid())) {
112 				// if treeItem is in search result ancestor list or is a search result then display
113 				return true;
114 			} else {
115 				// else do not
116 				return false;
117 			}
118 		}
119 	}
120 }
