48  public class EnhancedSearchViewController {
49  	private static final Logger LOG = LoggerFactory.getLogger(EnhancedSearchViewController.class);
50  
51  	@FXML private BorderPane searchBorderPane;
52  	@FXML private SplitPane searchAndTaxonomySplitPane;
53

55  	private EnhancedSearchViewTopPane topPane;
56  	private EnhancedSearchViewBottomPane bottomPane;
57  	private SearchModel searchModel = new SearchModel();
58  	
59  	public static EnhancedSearchViewController init() throws IOException {
60  		// Load FXML
61  		URL resource = EnhancedSearchViewController.class.getResource("EnhancedSearchView.fxml");
62  		LOG.debug("FXML for " + EnhancedSearchViewController.class + ": " + resource);
63  		FXMLLoader loader = new FXMLLoader(resource);
64  		loader.load();
65  		return loader.getController();
66  	}
67  
68  	@FXML
69  	public void initialize() {
70  		//assert searchText != null : "fx:id=\"searchText\" was not injected: check your FXML file 'EnhancedSearchView.fxml'.";
71  		//assert addIsDescendantOfFilterButton != null : "fx:id=\"addIsDescendantOfFilterButton\" was not injected: check your FXML file 'EnhancedSearchView.fxml'.";
72  		assert searchAndTaxonomySplitPane != null : "fx:id=\"searchResultsAndTaxonomySplitPane\" was not injected: check your FXML file 'EnhancedSearchView.fxml'.";
73  
74  		String styleSheet = EnhancedSearchViewController.class.getResource("/isaac-shared-styles.css").toString();
75  		if (! searchAndTaxonomySplitPane.getStylesheets().contains(styleSheet)) {
76  			searchAndTaxonomySplitPane.getStylesheets().add(styleSheet);
77  		}
78  
79  		topPane = new EnhancedSearchViewTopPane();
80  		bottomPane = new EnhancedSearchViewBottomPane(AppContext.getMainApplicationWindow().getPrimaryStage());
81  		
82  		SearchModel.getSearchRunning().addListener(new ChangeListener<Boolean>() {
83  			@Override
84  			public void changed(ObservableValue<? extends Boolean> observable,
85  					Boolean oldValue, Boolean newValue) {
86  				if (newValue) {
87  					searchModel.setSearchRunningPopover(BusyPopover.createBusyPopover("Searching...", topPane.getSearchButton()));
88  				} else {
89  					if (searchModel.getSearchRunningPopover() != null) {
90  						searchModel.getSearchRunningPopover().hide();
91  					}
92  				}
93  			}
94  		});
95  		
96  		//TODO - things that hit the DB (BDB or the Workflow SQL DB) should NOT Be done in the JavaFX foreground thread.  This causes large delays in displaying your GUI.
97  		//this sort of stuff need to be a in  background thread, with an appropriate progress indicator
98  
99  		topPane.getSearchButton().setOnAction((action) -> {
100 			if (SearchModel.getSearchRunning().get() && SearchModel.getSsh() != null) {
101 				SearchModel.getSsh().cancel();
102 			} else {
103 				SearchModel.getSearchTypeSelector().getTypeSpecificModel().search( 
104 						searchModel.getResultsTypeComboBox().getSelectionModel().getSelectedItem(),
105 						searchModel.getMaxResultsCustomTextField());
106 			}
107 		});
108 		SearchModel.getSearchRunning().addListener((observable, oldValue, newValue) -> {
109 			if (SearchModel.getSearchRunning().get()) {
110 				topPane.getSearchButton().setText("Cancel");
111 			} else {
112 				topPane.getSearchButton().setText("Search");
113 			}
114 		});
115 		
116 		ResultsToTaxonomy.setSearchAndTaxonomySplitPane(searchAndTaxonomySplitPane);
117 
118 		searchBorderPane.setTop(topPane.getTopPaneVBox());
119 		searchBorderPane.setBottom(bottomPane.getBottomPaneHBox());
120 		searchBorderPane.setCenter(SearchModel.getSearchResultsTable().getResults());
121 		searchModel.setPanes(bottomPane, searchAndTaxonomySplitPane, ResultsToTaxonomy.getTaxonomyPanelBorderPane());
122 		
123 		SearchModel.getSearchResultsTable().getResults().getItems().addListener(new ListChangeListener<CompositeSearchResult>() {
124 			@Override
125 			public void onChanged(
126 					javafx.collections.ListChangeListener.Change<? extends CompositeSearchResult> c) {
127 				bottomPane.refreshBottomPanel();
128 			}
129 		});
130 	}
131 
132 	public SplitPane getRoot() {
133 		return searchAndTaxonomySplitPane;
134 	}
