377 	private void exportResultsToSearchTaxonomyPanel() {
378 		if (taxonomyView == null) {
379 			taxonomyView = AppContext.getService(TaxonomyViewI.class);
380 
381 			taxonomyDisplayPolicies = new SctTreeItemSearchResultsDisplayPolicies(taxonomyView.getDefaultDisplayPolicies());
382 			taxonomyDisplayPolicies.setFilterMode(taxonomyPanelShouldFilterProperty);
383 
384 			taxonomyView.setDisplayPolicies(taxonomyDisplayPolicies);
385 
386 			taxonomyPanelBorderPane.setCenter(taxonomyView.getView());
387 		}
388 
389 		if (! searchResultsAndTaxonomySplitPane.getItems().contains(taxonomyPanelBorderPane)) {
390 			searchResultsAndTaxonomySplitPane.getItems().add(taxonomyPanelBorderPane);
391 			searchResultsAndTaxonomySplitPane.setDividerPositions(0.6);
392 			LOG.debug("Added taxonomyPanelBorderPane to searchResultsAndTaxonomySplitPane");
393 		}
394 		
395 		if (configureDisplayPoliciesTask != null) {
396 			if (configureDisplayPoliciesTask.isRunning()) {
397 				configureDisplayPoliciesTask.cancel();
398 			}
399 		}
400 		configureDisplayPoliciesTask = new Task<SctTreeItemSearchResultsDisplayPolicies>() {
401 			boolean cancelled = false;
402 			
403 			@Override
404 			protected SctTreeItemSearchResultsDisplayPolicies call() throws Exception {
405 				HashSet<Integer> searchResultAncestors = new HashSet<>();
406 				HashSet<Integer> searchResults = new HashSet<>();
407 
408 				for (CompositeSearchResult c : searchResultsTable.getItems()) {
409 					if (cancelled) {
410 						return taxonomyDisplayPolicies;
411 					}
412 
413 					searchResults.add(c.getContainingConcept().getNid());
414 
415 					Set<ConceptVersionBI> ancestorNids = null;
416 					ancestorNids = WBUtility.getConceptAncestors(c.getContainingConcept().getNid());
417 
418 					for (ConceptVersionBI concept : ancestorNids) {
419 						searchResultAncestors.add(concept.getNid());
420 					}
421 
422 				}
423 				
424 				taxonomyDisplayPolicies.setSearchResultAncestors(searchResultAncestors);
425 				taxonomyDisplayPolicies.setSearchResults(searchResults);
426 				
427 				return taxonomyDisplayPolicies;
428 			}
429 
430 			@Override
431 			protected void succeeded() {
432 				if (! cancelled) {
433 					taxonomyView.refresh();
434 				}
435 			}
436 			
437 			@Override
438 			public boolean cancel(boolean mayInterruptIfRunning) {
439 				cancelled = true;
440 
441 				return super.cancel(mayInterruptIfRunning);
442 			}
443 
444 			@Override
445 			protected void failed() {
446 				Throwable e = getException();
447 				
448 				if (e != null) {
449 					String title = "Failed sending search results to SearchResultsTaxonomy Panel";
450 					String msg = "Failed sending " + searchResultsTable.getItems().size() + " search results to SearchResultsTaxonomy Panel";
451 					String details = "Caught " + e.getClass().getName() + " \"" + e.getLocalizedMessage() + "\".";
452 					AppContext.getCommonDialogs().showErrorDialog(title, msg, details, AppContext.getMainApplicationWindow().getPrimaryStage());
453 
454 					e.printStackTrace();
455 				} else {
456 					LOG.error("Task configureDisplayPoliciesTask FAILED without throwing an exception/throwable");
457 				}
458 			}
459 		};
460 
461 		Utility.execute(configureDisplayPoliciesTask);
462 	}

516 			
517 			switch (taxonomyPanelViewModeComboBox.getSelectionModel().getSelectedItem()) {
