103 	@FXML
104 	public void initialize() {
105 		//assert searchText != null : "fx:id=\"searchText\" was not injected: check your FXML file 'EnhancedSearchView.fxml'.";
106 		//assert addIsDescendantOfFilterButton != null : "fx:id=\"addIsDescendantOfFilterButton\" was not injected: check your FXML file 'EnhancedSearchView.fxml'.";
107 		assert searchAndTaxonomySplitPane != null : "fx:id=\"searchResultsAndTaxonomySplitPane\" was not injected: check your FXML file 'EnhancedSearchView.fxml'.";
108 		assert enhancedSearchMenuBar != null : "fx:id=\"enhancedSearchMenuBar\" was not injected: check your FXML file 'EnhancedSearchView.fxml'.";
109 		
110 		Stage stage = AppContext.getMainApplicationWindow().getPrimaryStage();
111 		
112 		initializeSearchMenus(stage);
113 		
114 		String styleSheet = EnhancedSearchViewController.class.getResource("/isaac-shared-styles.css").toString();
115 		if (! searchAndTaxonomySplitPane.getStylesheets().contains(styleSheet)) {
116 			searchAndTaxonomySplitPane.getStylesheets().add(styleSheet);
117 		}
118 
119 		topPane = new EnhancedSearchViewTopPane();
120 		bottomPane = new EnhancedSearchViewBottomPane(stage);
121 		
122 		SearchModel.getSearchRunning().addListener(new ChangeListener<Boolean>() {
123 			@Override
124 			public void changed(ObservableValue<? extends Boolean> observable,
125 					Boolean oldValue, Boolean newValue) {
126 				if (newValue) {
127 					searchModel.setSearchRunningPopover(BusyPopover.createBusyPopover("Searching...", topPane.getSearchButton()));
128 				} else {
129 					if (searchModel.getSearchRunningPopover() != null) {
130 						searchModel.getSearchRunningPopover().hide();
131 					}
132 				}
133 			}
134 		});
135 		
136 		//TODO (artf231846) - things that hit the DB (BDB or the Workflow SQL DB) should NOT Be done in the JavaFX foreground thread.  This causes large delays in displaying your GUI.
137 		//this sort of stuff need to be a in  background thread, with an appropriate progress indicator
138 
139 		topPane.getSearchButton().setOnAction((action) -> {
140 			if (SearchModel.getSearchRunning().get() && SearchModel.getSsh() != null) {
141 				SearchModel.getSsh().cancel();
142 			} else {
143 				SearchModel.getSearchTypeSelector().getTypeSpecificModel().search( 
144 						searchModel.getResultsTypeComboBox().getSelectionModel().getSelectedItem(),
145 						searchModel.getMaxResultsCustomTextField());
146 			}
147 		});
148 		SearchModel.getSearchRunning().addListener((observable, oldValue, newValue) -> {
149 			if (SearchModel.getSearchRunning().get()) {
150 				topPane.getSearchButton().setText("Cancel");
151 			} else {
152 				topPane.getSearchButton().setText("Search");
153 			}
154 		});
155 		
156 		ResultsToTaxonomy.setSearchAndTaxonomySplitPane(searchAndTaxonomySplitPane);
157 
158 		searchBorderPane.setTop(topPane.getTopPaneVBox());
159 		searchBorderPane.setBottom(bottomPane.getBottomPaneHBox());
160 		searchBorderPane.setCenter(SearchModel.getSearchResultsTable().getResults());
161 		searchModel.setPanes(bottomPane, searchAndTaxonomySplitPane, ResultsToTaxonomy.getTaxonomyPanelBorderPane());
162 		
163 		SearchModel.getSearchResultsTable().getResults().getItems().addListener(new ListChangeListener<CompositeSearchResult>() {
164 			@Override
165 			public void onChanged(
166 					javafx.collections.ListChangeListener.Change<? extends CompositeSearchResult> c) {
167 				bottomPane.refreshBottomPanel();
168 				setSearchMenusDisabled();
169 			}
170 		});
171 		
172 	}

183 	private void initializeSearchMenus(Stage stage) {
184 		ResultsToTaxonomy.initializeTaxonomyPanel();
185 
186 		handleSearchMenu.getItems().clear();
187 		saveSearchMenu.getItems().clear();
188 		
189 		handleSearchMenu.getItems().addAll(resultsToReportMenuItem   ,
190 										   resultsToListMenuItem     ,
191 										   resultsToWorkflowMenuItem ,
192 										   resultsToTaxonomyMenuItem ,
193 										   resultsToSememeMenuItem   ,
194 										   resultsToDroolsMenuItem   );
195 		
196 		saveSearchMenu.getItems().addAll(saveSearchMenuItem, loadSearchMenu);
197 		
198 		enhancedSearchMenuBar.getMenus().addAll(handleSearchMenu, saveSearchMenu);
199 		
200 		resultsToListMenuItem.setOnAction((e) -> resultsToList());
201 		resultsToReportMenuItem.setOnAction((e) -> ResultsToReport.resultsToReport());
202 		resultsToWorkflowMenuItem.setOnAction((e) -> ResultsToWorkflow.multipleResultsToWorkflow());
203 		resultsToTaxonomyMenuItem.setOnAction((e) -> ResultsToTaxonomy.resultsToSearchTaxonomy());
204 		resultsToSememeMenuItem.setOnAction((e) -> createSememe(stage));
205 		resultsToDroolsMenuItem.setOnAction((e) -> ResultsToDrools.createDroolsOnClipboard(searchModel));
206 		
207 		saveSearchMenuItem.setOnAction((e) -> {
208 			topPane.getSearchSaver().saveSearch();
209 			EnhancedSavedSearch.refreshSavedSearchMenu(loadSearchMenu);
210 		});
211 		
212 		EnhancedSavedSearch.refreshSavedSearchMenu(loadSearchMenu);
213 		
214 		//setSearchMenusDisabled(true);
215 	}

217 	public void setSearchMenusDisabled() {
218 		int resultSize = SearchModel.getSearchResultsTable().getResults().getItems().size();
219 
220 		setSearchMenusDisabled(resultSize == 0);
221 
222 		resultsToWorkflowMenuItem.setDisable(resultSize > 0 && resultSize <= 5);
223 	}
224 	
225 	public void setSearchMenusDisabled(boolean isDisabled) {
226 		handleSearchMenu.setDisable(isDisabled);
227 		saveSearchMenuItem.setDisable(isDisabled);
228 	}

230 	private void createSememe(Stage stage) {
231 		try {
232 			String refexName = ResultsToRefset.resultsToRefset(stage, SearchModel.getSearchResultsTable().getResults());
233 			
234 			if (refexName != null) {
235 				AppContext.getCommonDialogs().showInformationDialog("Sememe Successfully Created", "Created and populated new Sememe (" + refexName + ") with all values in results table");
236 			}
237 		} catch (Exception e) {
238 			AppContext.getCommonDialogs().showErrorDialog("Sememe Creation Failure", "Sememe Creation Failure", "Failed to create and populate Sememe with values in results table"); 
239 		}
240 	}
241 
242 	private void resultsToList() {
243 		ListBatchViewI lv = AppContext.getService(ListBatchViewI.class, SharedServiceNames.DOCKED);
244 		AppContext.getMainApplicationWindow().ensureDockedViewIsVisble((DockedViewI) lv);
245 
246 		List<Integer> nids = new ArrayList<>();
247 		for (CompositeSearchResult result : SearchModel.getSearchResultsTable().getResults().getItems()) {
248 			if (! nids.contains(result.getContainingConcept().getNid())) {
249 				nids.add(result.getContainingConcept().getNid());
250 			}
251 		}
252 
253 		lv.addConcepts(nids);
254 	}
