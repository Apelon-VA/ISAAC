233 	private void resultsToList() {
234 		ListBatchViewI lv = AppContext.getService(ListBatchViewI.class, SharedServiceNames.DOCKED);
235 		AppContext.getMainApplicationWindow().ensureDockedViewIsVisble((DockedViewI) lv);
236 
237 		List<Integer> nids = new ArrayList<>();
238 		for (CompositeSearchResult result : searchModel.getSearchResultsTable().getResults().getItems()) {
239 			if (! nids.contains(result.getContainingConcept().getNid())) {
240 				nids.add(result.getContainingConcept().getNid());
241 			}
242 		}
243 
244 		lv.addConcepts(nids);
245 	}
