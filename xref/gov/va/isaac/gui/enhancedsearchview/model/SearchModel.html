112 	private static void initializeResultsTypeOptions() {
113 		// Force single selection
114 		resultsTypeComboBox.getSelectionModel().selectFirst();
115 		resultsTypeComboBox.setCellFactory((p) -> {
116 			final ListCell<ResultsType> cell = new ListCell<ResultsType>() {
117 				@Override
118 				protected void updateItem(ResultsType a, boolean bln) {
119 					super.updateItem(a, bln);
120 
121 					if(a != null){
122 						setText("Display " + a.toString());
123 					}else{
124 						setText(null);
125 					}
126 				}
127 			};
128 
129 			return cell;
130 		});
131 		resultsTypeComboBox.setButtonCell(new ListCell<ResultsType>() {
132 			@Override
133 			protected void updateItem(ResultsType t, boolean bln) {
134 				super.updateItem(t, bln); 
135 				if (bln) {
136 					setText("");
137 				} else {
138 					setText("Display " + t.toString());
139 				}
140 			}
141 		});
142 		resultsTypeComboBox.setOnAction((event) -> {
143 			LOG.trace("aggregationTypeComboBox event (selected: " + resultsTypeComboBox.getSelectionModel().getSelectedItem() + ")");
144 
145 			searchResults.getResults().getItems().clear();
146 			searchResults.initializeSearchResultsTable(searchTypeSelector.getCurrentType(), resultsTypeComboBox.getSelectionModel().getSelectedItem());
147 		});
148 
149 		resultsTypeComboBox.setItems(FXCollections.observableArrayList(ResultsType.values()));
150 		resultsTypeComboBox.getSelectionModel().select(ResultsType.DESCRIPTION);
151 	}
