51  	public boolean isSavableSearch(String errorDialogTitle) {
52  		String validationError = getSearchSavabilityValidationFailureMessage();
53  		if (validationError == null) {
54  			return true;
55  		} else {
56  			String details = "Invalid search type model for save (name=" + getName() + "). " + validationError;
57  			LOG.info(details);
58  
59  			if (errorDialogTitle != null) {
60  				AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogTitle, details, AppContext.getMainApplicationWindow().getPrimaryStage());
61  			}
62  
63  			return false;
64  		}
65  	}

77  			return true;

79  			String details = "Invalid search type model (name=" + getName() + "). " + validationError;
80  			LOG.info(details);

82  			if (errorDialogTitle != null) {
83  				AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogTitle, details, AppContext.getMainApplicationWindow().getPrimaryStage());

168 	public boolean validateSearchTypeModel(String errorDialogTitle) {
169 		if (getViewCoordinate() == null) {
170 			String details = "View coordinate is null: " + this;
171 			LOG.info("Invalid search model (name=" + getName() + "). " + details);
172 
173 			if (errorDialogTitle != null) {

184 	public synchronized void search(ResultsType resultsType, IntegerField maxRequestedResults ) {
185 		// Sanity check if search already running.
186 		if (getSearchRunning().get()) {
187 			return;
188 		}
189 
190 		try {
191 			getSearchRunning().set(true);
192 
193 			SearchModel.getSearchResultsTable().getResults().getItems().clear();
194 			bottomPane.refreshBottomPanel();
195 			bottomPane.refreshTotalResultsSelectedLabel();
196 
197 			if (! validateSearchTypeModel("Cannot execute search")) {
198 				getSearchRunning().set(false);
199 
200 				return;
201 			}
202 			
203 			executeSearch(resultsType, maxRequestedResults.getText());
204 		} catch (Exception e) {
205 			LOG.error("Search failed unexpectedly...", e);
206 			getSearchRunning().set(false);
207 		}
208 	}
