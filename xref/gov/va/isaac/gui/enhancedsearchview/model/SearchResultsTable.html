399 	private class MyTableCellCallback<T> implements Callback<TableColumn<CompositeSearchResult, T>, TableCell<CompositeSearchResult, T>> {
400 		/* (non-Javadoc)
401 		 * @see javafx.util.Callback#call(java.lang.Object)
402 		 */
403 		public TableCell<CompositeSearchResult, T> createNewCell() {
404 			TableCell<CompositeSearchResult, T> cell = new TableCell<CompositeSearchResult, T>() {
405 				@Override
406 				public void updateItem(T item, boolean empty) {
407 					super.updateItem(item, empty);
408 					setText(empty ? null : getString());
409 					setGraphic(null);
410 				}
411 
412 				private String getString() {
413 					return getItem() == null ? "" : getItem().toString();
414 				}
415 			};
416 
417 			return cell;
418 		}
419 
420 		// This method can be overridden to customize cells
421 		public TableCell<CompositeSearchResult, T> modifyCell(TableCell<CompositeSearchResult, T> cell) {
422 			return cell;
423 		}
424 
425 		@Override
426 		public TableCell<CompositeSearchResult, T> call(TableColumn<CompositeSearchResult, T> param) {
427 			TableCell<CompositeSearchResult, T> newCell = createNewCell();
428 			newCell.setUserData(param.getCellData(newCell.getIndex()));
429 
430 			// This event filter adds a concept-specific context menu to all cells based on underlying concept
431 			// It is in this method because it should be common to all cells, even those overriding modifyCell()
432 			newCell.addEventFilter(MouseEvent.MOUSE_CLICKED, new EventHandler<MouseEvent>() {
433 				@Override
434 				public void handle(MouseEvent event) {
435 					bottomPane.refreshTotalResultsSelectedLabel();
436 
437 					if (event.getButton() == MouseButton.SECONDARY) {
438 						@SuppressWarnings("unchecked")
439 						TableCell<CompositeSearchResult, T> c = (TableCell<CompositeSearchResult, T>) event.getSource();
440 
441 						if (c != null && c.getIndex() < c.getTableView().getItems().size()) {
442 							CommonMenusDataProvider dp = new CommonMenusDataProvider() {
443 								@Override
444 								public String[] getStrings() {
445 									List<String> items = new ArrayList<>();
446 									for (Integer index : c.getTableView().getSelectionModel().getSelectedIndices()) {
447 										items.add(c.getTableColumn().getCellData(index).toString());
448 									}
449 
450 									String[] itemArray = items.toArray(new String[items.size()]);
451 
452 									// TODO: determine why we are getting here multiple (2 or 3) times for each selection
453 									//System.out.println("Selected strings: " + Arrays.toString(itemArray));
454 
455 									return itemArray;
456 								}
457 							};
458 							CommonMenusNIdProvider nidProvider = new CommonMenusNIdProvider() {
459 								@Override
460 								public Set<Integer> getNIds() {
461 									Set<Integer> nids = new HashSet<>();
462 									for (CompositeSearchResult r : (ObservableList<CompositeSearchResult>)c.getTableView().getSelectionModel().getSelectedItems()) {
463 										if (resultsType == ResultsType.CONCEPT) {
464 											nids.add(r.getContainingConcept().getNid());
465 										} else if (resultsType == ResultsType.DESCRIPTION) {
466 											nids.add(r.getMatchingDescriptionComponents().iterator().next().getNid());
467 										} else {
468 											LOG.error("Unexpected AggregationType value " + resultsType);
469 											nids.add(r.getContainingConcept().getNid());
470 										}
471 									}
472 
473 									// TODO: determine why we are getting here multiple (2 or 3) times for each selection
474 									//System.out.println("Selected nids: " + Arrays.toString(nids.toArray()));
475 
476 									return nids;
477 								}
478 							};
479 
480 							ContextMenu cm = new ContextMenu();
481 							CommonMenus.addCommonMenus(cm, dp, nidProvider);
482 
483 							c.setContextMenu(cm);
484 						}
485 					}
486 				}
487 			});
488 
489 			return modifyCell(newCell);
490 		}
