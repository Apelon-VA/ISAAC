198 	public void executeSearch(ResultsType resultsType, String modelMaxResults) {
199 
200 		SearchTypeFilter<?> filter = getSearchType();
201 
202 		if (resultsType == ResultsType.DESCRIPTION && (filter.getSearchParameter() == null || filter.getSearchParameterProperty().isEmpty().get())) {
203 			String title = "Search failed";
204 
205 			String msg = "Cannot search on filters and select to return Descriptions.  Must return Concepts instead";
206 			getSearchRunning().set(false);
207 			AppContext.getCommonDialogs().showErrorDialog(title, "Failure to search filters-only", msg, AppContext.getMainApplicationWindow().getPrimaryStage());
208 
209 			return;
210 		} else  if (! (filter instanceof SearchTypeFilter)) {
211 			String title = "Search failed";
212 
213 			String msg = "SearchTypeFilter " + filter.getClass().getName() + " not supported";
214 			getSearchRunning().set(false);
215 			AppContext.getCommonDialogs().showErrorDialog(title, msg, "Only SearchTypeFilter LuceneSearchTypeFilter and RegExpSearchFilter currently supported", AppContext.getMainApplicationWindow().getPrimaryStage());
216 
217 			return;
218 		}
219 
220 		SearchResultsFilter searchResultsFilter = null;
221 		if (getFilters() != null) {
222 			List<SearchResultsFilter> searchResultsFilters = new ArrayList<>();
223 
224 			try {
225 				for (NonSearchTypeFilter<?> nonSearchTypeFilter : getFilters()) {
226 					SearchResultsFilter newSearchResultsFilter = SearchResultsFilterHelper.createSearchResultsFilter(nonSearchTypeFilter);
227 
228 					searchResultsFilters.add(newSearchResultsFilter);
229 				}
230 				LOG.debug("Constructing a new SearchResultsIntersectionFilter with " + searchResultsFilters.size() + " SearchResultsFilter instances: " + Arrays.toString(searchResultsFilters.toArray()));
231 				searchResultsFilter = new SearchResultsIntersectionFilter(searchResultsFilters);
232 
233 				//searchResultsFilter = SearchResultsFilterHelper.createNonSearchTypeFilterSearchResultsIntersectionFilter(getFilters().toArray(new NonSearchTypeFilter[getFilters().size()]));
234 			} catch (SearchResultsFilterException e) {
235 				String title = "Failed creating SearchResultsFilter";
236 				String msg = title + ". Encountered " + e.getClass().getName() + " " + e.getLocalizedMessage();
237 				String details =  msg + " applying " + getFilters().size() + " NonSearchResultFilter filters: " + Arrays.toString(getFilters().toArray());
238 				LOG.error(details);
239 				AppContext.getCommonDialogs().showErrorDialog(title, msg, details, AppContext.getMainApplicationWindow().getPrimaryStage());
240 
241 				ssh.cancel();
242 				taskComplete(0, null);
243 
244 				return;
245 			}
246 		}
247 
248 		// "we get called back when the results are ready."
249 		switch (resultsType) {
250 		case CONCEPT:
251 		{
252 			SearchBuilder builder = SearchBuilder.conceptDescriptionSearchBuilder(filter.getSearchParameter() != null ? filter.getSearchParameter() : "");
253 			builder.setCallback(this);
254 			builder.setTaskId(Tasks.SEARCH.ordinal());
255 			if (searchResultsFilter != null) {
256 				builder.setFilter(searchResultsFilter);
257 			}
258 			if (modelMaxResults != null && modelMaxResults.length() > 0) {
259 				Integer maxResults = Integer.valueOf(modelMaxResults);
260 				if (maxResults != null && maxResults > 0) {
261 					builder.setSizeLimit(maxResults);
262 				}
263 			}
264 			builder.setMergeResultsOnConcept(true);
265 
266 			if (getSearchType().getSearchParameter() != null && !getSearchType().getSearchParameter().isEmpty()) {
267 				ssh = SearchHandler.descriptionSearch(builder);
268 			} else {
269 				builder.setQuery("");
270 				
271 				Supplier<Set<CompositeSearchResult>> filterSupplier = new Supplier<Set<CompositeSearchResult>>()
272 				{
273 					@Override
274 					public Set<CompositeSearchResult> get()
275 					{
276 						Set<Integer> filterList = new HashSet<Integer>();
277 						for (NonSearchTypeFilter<? extends NonSearchTypeFilter<?>> f : filters) {
278 							filterList = f.gatherNoSearchTermCaseList(filterList);
279 						}
280 
281 						Set<CompositeSearchResult> filterCompositeSearchResultList = new HashSet<CompositeSearchResult>();
282 
283 						for (Integer c : filterList) {
284 							filterCompositeSearchResultList.add(new CompositeSearchResult(WBUtility.getConceptVersion(c), 0));
285 						}
286 						
287 						return filterCompositeSearchResultList;
288 					}
289 				};
290 
291 				ssh = SearchHandler.descriptionSearch(builder, filterSupplier);
292 			}
293 			break;
294 		}
295 		case DESCRIPTION:
296 		{
297 			SearchBuilder builder = SearchBuilder.descriptionSearchBuilder(filter.getSearchParameter() != null ? filter.getSearchParameter() : "");
298 			builder.setCallback(this);
299 			builder.setTaskId(Tasks.SEARCH.ordinal());
300 			if (searchResultsFilter != null) {
301 				builder.setFilter(searchResultsFilter);
302 			}
303 			if (modelMaxResults != null && modelMaxResults.length() > 0) {
304 				Integer maxResults = Integer.valueOf(modelMaxResults);
305 				if (maxResults != null && maxResults > 0) {
306 					builder.setSizeLimit(maxResults);
307 				}
308 			}
309 			ssh = SearchHandler.descriptionSearch(builder);
310 			break;
311 		}
312 		default:
313 			String title = "Unsupported Aggregation Type";
314 			String msg = "Aggregation Type " + resultsType + " not supported";
315 			LOG.error(title);
316 			AppContext.getCommonDialogs().showErrorDialog(title, msg, "Aggregation Type must be one of " + Arrays.toString(ResultsType.values()), AppContext.getMainApplicationWindow().getPrimaryStage());
317 
318 			ssh.cancel();
319 			break;
320 		}
