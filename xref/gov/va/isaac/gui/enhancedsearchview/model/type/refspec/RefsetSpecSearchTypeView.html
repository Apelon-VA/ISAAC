77  	private List<QueryNodeType> getSupportedQueryNodeTypes(List<QueryNodeTypeI> unsupportedNodes, QueryNodeType...nodeTypes) {
78  		ArrayList<QueryNodeType> supportedNodeTypes = new ArrayList<>();
79  		
80  		for (QueryNodeType nodeType : nodeTypes) {
81  			if (! unsupportedNodes.contains(nodeType)) {
82  				supportedNodeTypes.add(nodeType);
83  			}
84  		}
85  		
86  		return supportedNodeTypes;
87  	}

90  	protected void loadMenus(RefsetSpecSearchTypeModel refsetSpecModel) {
91  		List<QueryNodeType> groupingNodeTypes = getSupportedQueryNodeTypes(
92  				refsetSpecModel.getUnsupportedQueryNodeTypes(),
93  				QueryNodeType.AND,
94  				QueryNodeType.OR,
95  				QueryNodeType.XOR);
96  		
97  		List<QueryNodeType> conceptAssertionNodeTypes = getSupportedQueryNodeTypes(
98  				refsetSpecModel.getUnsupportedQueryNodeTypes(),
99  				QueryNodeType.CONCEPT_IS,
100 				QueryNodeType.CONCEPT_IS_CHILD_OF,
101 				QueryNodeType.CONCEPT_IS_DESCENDANT_OF,
102 				QueryNodeType.CONCEPT_IS_KIND_OF);
103 		
104 		List<QueryNodeType> descriptionAssertionNodeTypes = getSupportedQueryNodeTypes(
105 				refsetSpecModel.getUnsupportedQueryNodeTypes(),
106 				QueryNodeType.DESCRIPTION_CONTAINS
107 				);
108 		
109 		List<QueryNodeType> relationshipAssertionNodeTypes = getSupportedQueryNodeTypes(
110 				refsetSpecModel.getUnsupportedQueryNodeTypes(),
111 				QueryNodeType.REL_RESTRICTION,
112 				QueryNodeType.REL_TYPE
113 				);
114 		
115 		// Add to dropdown
116 		boolean separatorNeeded = false;
117 		if (groupingNodeTypes.size() > 0) {
118 			refsetSpecModel.getRootNodeTypeComboBox().getItems().addAll(groupingNodeTypes);
119 			separatorNeeded = true;
120 		}
121 		if (conceptAssertionNodeTypes.size() > 0) {
122 			if (separatorNeeded) {
123 				refsetSpecModel.getRootNodeTypeComboBox().getItems().add(new Separator());
124 			}
125 			refsetSpecModel.getRootNodeTypeComboBox().getItems().addAll(conceptAssertionNodeTypes);
126 		}
127 		if (descriptionAssertionNodeTypes.size() > 0) {
128 			if (separatorNeeded) {
129 				refsetSpecModel.getRootNodeTypeComboBox().getItems().add(new Separator());
130 			}
131 			refsetSpecModel.getRootNodeTypeComboBox().getItems().addAll(descriptionAssertionNodeTypes);
132 		}
133 		if (relationshipAssertionNodeTypes.size() > 0) {
134 			if (separatorNeeded) {
135 				refsetSpecModel.getRootNodeTypeComboBox().getItems().add(new Separator());
136 			}
137 			refsetSpecModel.getRootNodeTypeComboBox().getItems().addAll(relationshipAssertionNodeTypes);
138 		}
139 		
140 		if (refsetSpecModel.getQueryNodeTreeView().getContextMenu() != null) {
141 			refsetSpecModel.getQueryNodeTreeView().getContextMenu().getItems().clear();
142 		} else {
143 			refsetSpecModel.getQueryNodeTreeView().setContextMenu(new ContextMenu());
144 		}
145 		refsetSpecModel.addContextMenus(refsetSpecModel.getQueryNodeTreeView().getContextMenu(), refsetSpecModel.getQueryNodeTreeView());
146 	}
