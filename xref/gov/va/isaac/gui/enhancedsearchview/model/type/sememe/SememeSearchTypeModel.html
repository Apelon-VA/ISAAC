269 	@Override
270 	public void executeSearch(ResultsType resultsType, String modelMaxResults) {
271 		try
272 		{
273 			String searchString = searchText.getText().trim();
274 			try
275 			{
276 				RefexDynamicDataBI data = NumberUtilities.wrapIntoRefexHolder(NumberUtilities.parseNumber(searchString));
277 				LOG.debug("Doing a sememe search with a numeric value");
278 				ssh = SearchHandler.dynamicRefexSearch((indexer) ->
279 				{
280 					try
281 					{
282 						return indexer.query(data, currentlyEnteredAssemblageNid, false, getSearchColumns(), Integer.parseInt(modelMaxResults), null);
283 					}
284 					catch (Exception e)
285 					{
286 						throw new RuntimeException(e);
287 					}
288 				}, this, null, null, null, true);
289 			}
290 			catch (NumberFormatException e)
291 			{
292 				//Not a number...  is it a valid interval?
293 				try
294 				{
295 					Interval interval = new Interval(searchString);
296 					LOG.debug("Doing a sememe search with an interval value");
297 					ssh = SearchHandler.dynamicRefexSearch((indexer) ->
298 					{
299 						try
300 						{
301 							return indexer.queryNumericRange(NumberUtilities.wrapIntoRefexHolder(interval.getLeft()), interval.isLeftInclusive(), 
302 									NumberUtilities.wrapIntoRefexHolder(interval.getRight()), interval.isRightInclusive(), 
303 									currentlyEnteredAssemblageNid, getSearchColumns(), Integer.parseInt(modelMaxResults), null);
304 						}
305 						catch (Exception e1)
306 						{
307 							throw new RuntimeException(e1);
308 						}
309 					}, this, null, null, null, true);
310 				}
311 				catch (NumberFormatException e1) 
312 				{
313 					//run it as a string search
314 					LOG.debug("Doing a sememe search as a string search");
315 					ssh = SearchHandler.dynamicRefexSearch((indexer) ->
316 					{
317 						try
318 						{
319 							return indexer.query(new RefexDynamicString(searchText.getText()), currentlyEnteredAssemblageNid, false, 
320 									getSearchColumns(), Integer.parseInt(modelMaxResults), null);
321 						}
322 						catch (Exception e2)
323 						{
324 							throw new RuntimeException(e2);
325 						}
326 					}, this, null, null, null, true);
327 				}
328 			}
329 		}
330 		catch (Exception e)
331 		{
332 			LOG.error("Search imploded unexpectedly...", e);
333 			ssh = null;  //force a null ptr in taskComplete, so an error is displayed.
334 			taskComplete(0, null);
335 		}
336 	}

366 				//searchResultsTable.getItems().clear();

377 		List<CompositeSearchResult> retList = new ArrayList<CompositeSearchResult>();
378 		
379 		for (CompositeSearchResult refConResult : results) {
380 			ConceptVersionBI refCon = refConResult.getContainingConcept();
381 			
382 			for (ComponentVersionBI c : refConResult.getMatchingComponents())
383 			{
384 				if (c instanceof RefexDynamicVersionBI<?>)
385 				{
386 					RefexDynamicVersionBI<?> rv = (RefexDynamicVersionBI<?>)c;
387 					ConceptVersionBI assembCon = WBUtility.getConceptVersion(rv.getAssemblageNid());	
388 					
389 					try
390 					{
391 						RefexDynamicColumnInfo[] ci = rv.getRefexDynamicUsageDescription().getColumnInfo();
392 						int i = 0;
393 						
394 						for (RefexDynamicDataBI data : rv.getData())
395 						{

438 	private void populateDynamicRefexList()
439 	{
440 		Task<Void> t = new Task<Void>()
441 		{
442 			HashSet<SimpleDisplayConcept> dynamicRefexAssemblages = new HashSet<>();
443 
444 			@Override
445 			protected Void call() throws Exception
446 			{
447 				dynamicRefexAssemblages = new HashSet<>();
448 				
449 				LuceneDynamicRefexIndexer indexer = AppContext.getService(LuceneDynamicRefexIndexer.class);
450 				List<SearchResult> refexes = indexer.queryAssemblageUsage(RefexDynamic.REFEX_DYNAMIC_DEFINITION_DESCRIPTION.getNid(), 1000, null);
451 				for (SearchResult sr : refexes)
452 				{
453 					RefexDynamicChronicleBI<?> rc = (RefexDynamicChronicleBI<?>) ExtendedAppContext.getDataStore().getComponent(sr.getNid());
454 					//These are nested refex references - it returns a description component - concept we want is the parent of that.
455 					dynamicRefexAssemblages.add(new SimpleDisplayConcept(
456 							ExtendedAppContext.getDataStore().getComponent(rc.getReferencedComponentNid()).getEnclosingConcept(),
457 							null));
458 					
459 				}
460 				return null;
461 			}
462 
463 			/**
464 			 * @see javafx.concurrent.Task#succeeded()
465 			 */
466 			@Override
467 			protected void succeeded()
468 			{
469 				dynamicRefexList_.clear();
470 				dynamicRefexList_.addAll(dynamicRefexAssemblages);
471 				dynamicRefexList_.sort(new Comparator<SimpleDisplayConcept>()
472 				{
473 					@Override
474 					public int compare(SimpleDisplayConcept o1, SimpleDisplayConcept o2)
475 					{
476 						return o1.getDescription().compareToIgnoreCase(o2.getDescription());
477 					}
478 				});
479 			}
480 		};
481 		
482 		Utility.execute(t);
483 	}
