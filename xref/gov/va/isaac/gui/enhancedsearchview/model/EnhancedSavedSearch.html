242 	public static void refreshSavedSearchComboBox() {
243 		Task<List<SearchDisplayConcept>> loadSavedSearches = new Task<List<SearchDisplayConcept>>() {
244 			private ObservableList<SearchDisplayConcept> searches = FXCollections.observableList(new ArrayList<>());
245 
246 			@Override
247 			protected List<SearchDisplayConcept> call() throws Exception {
248 				Set<ConceptVersionBI> savedSearches = WBUtility.getAllChildrenOfConcept(Search.SEARCH_PERSISTABLE.getNid(), true);
249 
250 				SearchType currentSearchType = SearchModel.getSearchTypeSelector().getSearchTypeComboBox().getSelectionModel().getSelectedItem();
251 				for (ConceptVersionBI concept : savedSearches) {
252 					if (getCachedSearchTypeFromSearchConcept(concept) == currentSearchType) {
253 						boolean addSearchToList = true;
254 						if (currentSearchType == SearchType.TEXT) {
255 							ComponentSearchType currentlyViewedComponentSearchType = TextSearchTypeModel.getCurrentComponentSearchType();
256 							ComponentSearchType loadedComponentSearchType = getCachedComponentSearchTypeFromSearchConcept(concept);
257 							
258 							if (currentlyViewedComponentSearchType != null && loadedComponentSearchType != null) {
259 								if (currentlyViewedComponentSearchType != loadedComponentSearchType) {
260 									addSearchToList = false;
261 								}
262 							}
263 						}
264 						
265 						if (addSearchToList) {
266 							String fsn = WBUtility.getFullySpecifiedName(concept);
267 							String preferredTerm = WBUtility.getConPrefTerm(concept.getNid());
268 							searches.add(new SearchDisplayConcept(fsn, preferredTerm, concept.getNid()));
269 						}
270 					}
271 				}
272 
273 				return searches;
274 			}
275 
276 			@Override
277 			protected void succeeded() {
278 				super.succeeded();
279 
280 				savedSearchesComboBox.setItems(searches);
281 			}
282 		};
283 
284 		savedSearchesComboBox.getItems().clear();
285 		Utility.execute(loadSavedSearches);
286 	}
