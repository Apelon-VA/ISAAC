91  	private static String getTableViewRow(TableView<CompositeSearchResult> table, String delimiter, String lineTerminator, ColumnValueExtractor extractor) {
92  		ObservableList<TableColumn<CompositeSearchResult, ?>> columns = table.getColumns();
93  		StringBuilder row = new StringBuilder();
94  
95  		for (int colIndex = 0; colIndex < columns.size(); ++colIndex) {
96  			TableColumn<CompositeSearchResult, ?> col = columns.get(colIndex);
97  			if (! col.isVisible()) {
98  				// Ensure that newline is written even if column is not
99  				if (colIndex == (columns.size() - 1) && lineTerminator != null) {
100 					// Append newline to row
101 					row.append(lineTerminator);
102 				}
103 
104 				continue;
105 			}
106 			// Extract text or data from column and append to row
107 			row.append(extractor.extract(col));
108 			if (colIndex < (columns.size() - 1)) {
109 				if (delimiter != null) {
110 					// Ensure that delimiter is written only if there are remaining visible columns to be written
111 					boolean hasMoreVisibleCols = false;
112 					for (int remainingColsIndex = colIndex + 1; remainingColsIndex < columns.size(); ++remainingColsIndex) {
113 						if (columns.get(remainingColsIndex).isVisible()) {
114 							hasMoreVisibleCols = true;
115 							break;
116 						}
117 					}
118 					if (hasMoreVisibleCols) {
119 						// Append delimiter to row
120 						row.append(delimiter);
121 					}
122 				}
123 			} else if (colIndex == (columns.size() - 1) && lineTerminator != null) {
124 				// Append newline to row
125 				row.append(lineTerminator);
126 			}
127 		}
128 
129 		return row.toString();
