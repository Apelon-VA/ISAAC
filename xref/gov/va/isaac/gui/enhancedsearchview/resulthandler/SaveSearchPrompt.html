3   import gov.va.isaac.AppContext;
4   import gov.va.isaac.gui.dialog.UserPrompt;
5   import javafx.geometry.HPos;
6   import javafx.geometry.Insets;
7   import javafx.geometry.VPos;
8   import javafx.scene.Node;
9   import javafx.scene.control.Label;
10  import javafx.scene.control.TextField;
11  import javafx.scene.layout.GridPane;
12  import javafx.scene.layout.Priority;
13  
14  public class SaveSearchPrompt extends UserPrompt {
15  	private TextField nameTextField = new TextField();
16  	private TextField descTextField = new TextField();
17  
18  	public SaveSearchPrompt() {
19  		super("Save");
20  	}
21  
22  
23  	protected boolean isSelectedValuesValid() {
24  		return !nameTextField.getText().isEmpty() && 
25  			   !descTextField.getText().isEmpty() ;
26  	}
27  
28  	protected Node createUserInterface() {
29  		GridPane gp = new GridPane();
30  		gp.setHgap(10);
31  		gp.setVgap(10);
32  		gp.setPadding(new Insets(15));
33  		
34  		Label name = createLabel("Name");
35  		Label desc = createLabel("Description");
36  
37  		nameTextField.clear();
38  		descTextField.clear();
39  	    
40  
41  		gp.setConstraints(currentMaxLabel,  0,  0,  1,  1,  HPos.RIGHT,  VPos.CENTER, Priority.NEVER, Priority.ALWAYS);
42  		gp.setConstraints(descTextField,  1,  0,  1,  1,  HPos.CENTER,  VPos.CENTER, Priority.ALWAYS, Priority.ALWAYS);
43  
44  		gp.addRow(0, name, nameTextField);
45  		gp.addRow(1, desc, descTextField);
46  
47  		return gp ;
48  	}
49  
50  
51  	public TextField getNameTextField() {
52  		return nameTextField;
53  	}
54  
55  	public TextField getDescTextField() {
56  		return descTextField;
57  	}
58  
59  
60  	@Override
61  	protected void displayInvalidMessage() {
62  		AppContext.getCommonDialogs().showInformationDialog("Missing Information", "Must select value for each field");
63  	}
64  }
