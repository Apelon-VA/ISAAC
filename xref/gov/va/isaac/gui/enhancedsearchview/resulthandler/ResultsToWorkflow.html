27  	public static void multipleResultsToWorkflow() {
28  		initializeWorkflowServices();
29  
30  		// Use HashSet to ensure that only one workflow is created for each concept
31  		Map<Integer, ComponentVersionBI> conceptsOrComponents = new HashMap<>();
32  		if (searchModel.getResultsTypeComboBox().getSelectionModel().getSelectedItem() == ResultsType.CONCEPT) {
33  			for (CompositeSearchResult result : searchModel.getSearchResultsTable().getResults().getItems()) {
34  				ConceptVersionBI conceptVersion = result.getContainingConcept();
35  				int nid = conceptVersion.getNid();
36  				
37  				if (! conceptsOrComponents.containsKey(nid)) {
38  					conceptsOrComponents.put(nid, conceptVersion);
39  				}
40  			}
41  		} else if (searchModel.getResultsTypeComboBox().getSelectionModel().getSelectedItem() == ResultsType.DESCRIPTION) {
42  			for (CompositeSearchResult result : searchModel.getSearchResultsTable().getResults().getItems()) {
43  				ComponentVersionBI componentVersion = result.getMatchingComponents().iterator().next();
44  				int nid = componentVersion.getNid();
45  				
46  				if (! conceptsOrComponents.containsKey(nid)) {
47  					conceptsOrComponents.put(nid, componentVersion);
48  				}
49  			}
50  		} else {
51  			String title = "Failed exporting search results to workflow";
52  			String msg = "Unsupported AggregationType " + searchModel.getResultsTypeComboBox().getSelectionModel().getSelectedItem();
53  			LOG.error(title + ". " + msg);
54  			AppContext.getCommonDialogs().showErrorDialog(title, msg, title + ". " + msg, AppContext.getMainApplicationWindow().getPrimaryStage());
55  		}
56  
57  		if (conceptsOrComponents.size() > 0) {
58  			DialogResponse response = AppContext.getCommonDialogs().showYesNoDialog("Bulk Workflow Export Confirmation", "Are you sure that you want to generate " + conceptsOrComponents.size() + " new Workflow instance(s)?");
59  
60  			if (response == DialogResponse.YES) {
61  				for (ComponentVersionBI conceptOrComponent : conceptsOrComponents.values()) {
62  					singleResultToWorkflow(conceptOrComponent);
63  				}
64  
65  				conceptWorkflowService.synchronizeWithRemote();
66  			}
67  		}
68  	}

71  	private static void singleResultToWorkflow(ComponentVersionBI componentOrConceptVersion) {
72  		initializeWorkflowServices();
73  
74  		final WorkflowProcess process = WorkflowProcess.REVIEW3;
75  		String preferredDescription = null;
76  		try {
77  			if (componentOrConceptVersion instanceof ConceptVersionBI) {
78  				DescriptionVersionBI<?> desc = ((ConceptVersionBI)componentOrConceptVersion).getPreferredDescription();
79  				preferredDescription = desc.getText();
80  			} else {
81  				preferredDescription = componentOrConceptVersion.toUserString();
82  			}
83  			
84  			LOG.debug("Invoking createNewConceptWorkflowRequest(preferredDescription=\"" + preferredDescription + "\", conceptUuid=\"" 
85  					+ componentOrConceptVersion.getPrimordialUuid().toString() + "\", processName=\"" + process.getText() + "\")");
86  			ProcessInstanceCreationRequestI createdRequest = conceptWorkflowService.createNewComponentWorkflowRequest(preferredDescription, 
87  					componentOrConceptVersion.getPrimordialUuid(), process.getText(), new HashMap<String,String>());
88  			LOG.debug("Created ProcessInstanceCreationRequestI: " + createdRequest);
89  			
90  		} catch (Exception e1) {
91  			String title = "Error sending component to workflow";
92  			String msg = title + ". Unexpected error while sending the component (nid=" + componentOrConceptVersion.getNid() + ", uuid=" 
93  					+ componentOrConceptVersion.getPrimordialUuid().toString() + "): caught " + e1.getClass().getName() + " " + e1.getLocalizedMessage();
94  			LOG.error(msg, e1);
95  			AppContext.getCommonDialogs().showErrorDialog(title, msg, e1.getMessage(), AppContext.getMainApplicationWindow().getPrimaryStage());
96  			
97  			return;
98  		}

100 	}
