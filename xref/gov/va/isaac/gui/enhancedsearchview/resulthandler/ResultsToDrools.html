15  	public static void createDroolsOnClipboard(SearchModel searchModel) {
16  		if (!handleExclusions(searchModel)) {
17  			// Check for EMPTY with isKindOf
18  			String resultStr = null;
19  			
20  			if (1 == 1) {
21  				resultStr = isKindOfBuild(searchModel);
22  			}
23  			
24  			if (resultStr != null) {
25  				CustomClipboard.set(resultStr);
26  			}
27  		}
28  	}

30  	private static boolean handleExclusions(SearchModel searchModel) {
31  		if (searchModel.getSearchTypeSelector().getCurrentType() != SearchType.TEXT) {
32  			AppContext.getCommonDialogs().showInformationDialog("Drools From Search Failure", "Unable to create drools rules from search type: " + searchModel.getSearchTypeSelector().getCurrentType());
33  			return true;
34  		} else {
35  			TextSearchTypeModel model = (TextSearchTypeModel)searchModel.getSearchTypeSelector().getTypeSpecificModel();
36  		
37  			if (model.getSearchType().getSearchParameter() != null && !model.getSearchType().getSearchParameter().isEmpty()) {
38  				AppContext.getCommonDialogs().showInformationDialog("Drools From Search Failure", "Current implementation only allows for creation of drools rule based on empty search string \rsuch that it can be added to nightly QA: " + model.getSearchType().getSearchParameter());
39  				return true;
40  			} else if (model.getFilters().size() != 1) {
41  				AppContext.getCommonDialogs().showInformationDialog("Drools From Search Failure", "Current implementation only allows for creation of drools rule based on one search filter");
42  				return true;
43  			} else if (!(model.getFilters().get(0) instanceof IsDescendantOfFilter) ) {
44  				AppContext.getCommonDialogs().showInformationDialog("Drools From Search Failure", "Current implementation only allows for creation of drools rule based on isDescenedant search filter");
45  				return true;
46  			}
47  		}
48  		
49  		return false;		
50  	}

52  	private static String isKindOfBuild(SearchModel searchModel) {
53  		TextSearchTypeModel model = (TextSearchTypeModel)searchModel.getSearchTypeSelector().getTypeSpecificModel();
54  
55  		StringBuilder strBld = new StringBuilder();
56  		strBld.append(getPackage());
57  		strBld.append("import gov.va.issac.drools.testmodel.DrConcept\r");
58  		strBld.append("import gov.va.issac.drools.helper.templates.ConceptTemplate\r\r");
59  		strBld.append(getGenericImports());
60  		strBld.append("\r");
61  		strBld.append(getGlobals());
62  		strBld.append("\r");
63  		strBld.append(setupRule());
64  		
65  		// Define Variable
66  		strBld.append("\t\t$c : ConceptVersionBI( )\r");
67  		
68  		// Setup Eval
69  		strBld.append("\t\teval(terminologyHelper.isParentOf( ");
70  		
71  		UUID id = OTFUtility.getConceptVersion(((IsDescendantOfFilter)model.getFilters().get(0)).getNidProperty().get()).getPrimordialUuid();
72  		
73  		strBld.append("\"" + id.toString() + "\"");
74  		strBld.append(", $c.dataUUID.toString))\r");
75  		
76  		// Placeholder for then
77  		strBld.append("\tthen\r");
78  		
79  		return strBld.toString();
80  	}
