46  	public static void resultsToSearchTaxonomy() {
47  		initializeTaxonomyPanel();
48  		
49  		if (taxonomyView == null) {
50  			taxonomyView = AppContext.getService(TaxonomyViewI.class);
51  
52  			taxonomyDisplayPolicies = new SctTreeItemSearchResultsDisplayPolicies(taxonomyView.getDefaultDisplayPolicies());
53  			taxonomyDisplayPolicies.setFilterMode(taxonomyPanelShouldFilterProperty);
54  
55  			taxonomyView.setDisplayPolicies(taxonomyDisplayPolicies);
56  
57  			taxonomyPanelBorderPane.setCenter(taxonomyView.getView());
58  		}
59  
60  		if (! searchResultsAndTaxonomySplitPane.getItems().contains(taxonomyPanelBorderPane)) {
61  			searchResultsAndTaxonomySplitPane.getItems().add(taxonomyPanelBorderPane);
62  			searchResultsAndTaxonomySplitPane.setDividerPositions(0.6);
63  			LOG.debug("Added taxonomyPanelBorderPane to searchResultsAndTaxonomySplitPane");
64  		}
65  		
66  		if (configureDisplayPoliciesTask != null) {
67  			if (configureDisplayPoliciesTask.isRunning()) {
68  				configureDisplayPoliciesTask.cancel();
69  			}
70  		}
71  		configureDisplayPoliciesTask = new Task<SctTreeItemSearchResultsDisplayPolicies>() {
72  			boolean cancelled = false;
73  			
74  			@Override
75  			protected SctTreeItemSearchResultsDisplayPolicies call() throws Exception {
76  				HashSet<Integer> searchResultAncestors = new HashSet<>();
77  				HashSet<Integer> searchResults = new HashSet<>();
78  
79  				// TODO this throws ConcurrentModificationException if called before prior export completed
80  				for (CompositeSearchResult c : SearchModel.getSearchResultsTable().getResults().getItems()) {
81  					if (cancelled) {
82  						return taxonomyDisplayPolicies;
83  					}
84  
85  					searchResults.add(c.getContainingConcept().getNid());
86  
87  					Set<ConceptVersionBI> ancestorNids = null;
88  					ancestorNids = OTFUtility.getConceptAncestors(c.getContainingConcept().getNid());
89  
90  					for (ConceptVersionBI concept : ancestorNids) {
91  						searchResultAncestors.add(concept.getNid());
92  					}
93  
94  				}
95  				
96  				taxonomyDisplayPolicies.setSearchResultAncestors(searchResultAncestors);
97  				taxonomyDisplayPolicies.setSearchResults(searchResults);
98  				
99  				return taxonomyDisplayPolicies;
100 			}
101 
102 			@Override
103 			protected void succeeded() {
104 				if (! cancelled) {
105 					taxonomyView.refresh();
106 				}
107 			}
108 			
109 			@Override
110 			public boolean cancel(boolean mayInterruptIfRunning) {
111 				cancelled = true;
112 
113 				return super.cancel(mayInterruptIfRunning);
114 			}
115 
116 			@Override
117 			protected void failed() {
118 				Throwable e = getException();
119 				
120 				if (e != null) {
121 					String title = "Failed sending search results to SearchResultsTaxonomy Panel";
122 					String msg = "Failed sending " + SearchModel.getSearchResultsTable().getResults().getItems().size() + " search results to SearchResultsTaxonomy Panel";
123 					String details = "Caught " + e.getClass().getName() + " \"" + e.getLocalizedMessage() + "\".";
124 					AppContext.getCommonDialogs().showErrorDialog(title, msg, details, AppContext.getMainApplicationWindow().getPrimaryStage());
125 
126 					e.printStackTrace();
127 				} else {
128 					LOG.error("Task configureDisplayPoliciesTask FAILED without throwing an exception/throwable");
129 				}
130 			}
131 		};
132 
133 		Utility.execute(configureDisplayPoliciesTask);
134 
135 	}

156 	private static void initializeTaxonomyViewModeComboBox() {
157 		taxonomyPanelViewModeComboBox = new ComboBox<>();
158 
159 		// Force single selection
160 		taxonomyPanelViewModeComboBox.getSelectionModel().selectFirst();
161 		taxonomyPanelViewModeComboBox.setCellFactory((p) -> {
162 			final ListCell<TaxonomyViewMode> cell = new ListCell<TaxonomyViewMode>() {
163 				@Override
164 				protected void updateItem(TaxonomyViewMode a, boolean bln) {
165 					super.updateItem(a, bln);
166 
167 					if(a != null){
168 						setText(a.toString() + " view mode");
169 					}else{
170 						setText(null);
171 					}
172 				}
173 			};
174 
175 			return cell;
176 		});
177 		taxonomyPanelViewModeComboBox.setButtonCell(new ListCell<TaxonomyViewMode>() {
178 			@Override
179 			protected void updateItem(TaxonomyViewMode t, boolean bln) {
180 				super.updateItem(t, bln); 
181 				if (bln) {
182 					setText("");
183 				} else {
184 					setText(t.toString() + " view mode");
185 				}
186 			}
187 		});
188 		taxonomyPanelViewModeComboBox.setOnAction((event) -> {
189 			LOG.trace("taxonomyPanelViewModeComboBox event (selected: " + taxonomyPanelViewModeComboBox.getSelectionModel().getSelectedItem() + ")");
190 
191 			boolean statusChanged = false;
192 			
193 			switch (taxonomyPanelViewModeComboBox.getSelectionModel().getSelectedItem()) {
194 			case FILTERED:
195 				if (! taxonomyPanelShouldFilterProperty.get()) {
196 					statusChanged = true;
197 				}
198 				taxonomyPanelShouldFilterProperty.set(true);
199 				break;
200 			case UNFILTERED:
201 				if (taxonomyPanelShouldFilterProperty.get()) {
202 					statusChanged = true;
203 				}
204 				taxonomyPanelShouldFilterProperty.set(false);
205 				break;
206 
207 				default:
208 					throw new RuntimeException("Unsupported TaxonomyViewMode value \"" + taxonomyPanelViewModeComboBox.getSelectionModel().getSelectedItem() + "\"");
209 			}
210 			
211 			if (statusChanged) {
212 				taxonomyView.refresh();
213 			}
214 		});
215 
216 		taxonomyPanelViewModeComboBox.setItems(FXCollections.observableArrayList(TaxonomyViewMode.values()));
217 		taxonomyPanelViewModeComboBox.getSelectionModel().select(TaxonomyViewMode.UNFILTERED);
218 	}
