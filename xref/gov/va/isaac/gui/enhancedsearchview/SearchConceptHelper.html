114 	public static ConceptChronicleBI buildSearchConcept(
115 			SearchViewModel model,
116 			String passedSaveConceptFSN,
117 			String passedSaveConceptPT) throws SearchConceptException
118 	{
119 		final String saveConceptFSN = passedSaveConceptFSN != null ? passedSaveConceptFSN : model.getName();
120 		final String saveConceptPT = passedSaveConceptPT != null ? passedSaveConceptPT : model.getDescription();
121 
122 		LOG.debug("buildSearchConcept(): saving model for search fsn=\"" + saveConceptFSN + "\", pt=\"" + saveConceptPT + "\": " + model);
123 		//
124 		// Construct new containing Search Concept
125 		//
126 		// Construct Search Global Attributes refex containing vc (View Coordinate)
127 		//		Populate vc from GUI 
128 		//
129 		// Add Search Global Attributes refex to containing Search Concept
130 		//
131 		// For each Filter spec in GUI
132 		//		Construct appropriate new Search {Lucene|RegExp} Filter refex containing param
133 		//			Populate param from Filter spec in GUI
134 		//			Construct new nested Search Filter Attributes refex containing order
135 		//				Populate nested Search Filter Attributes refex with order from GUI
136 		//			Nest nested Search Filter Attributes refex within Search {Lucene|RegExp} Filter refex
137 		//
138 		//		Add Search {Lucene|RegExp} Filter refex to containing Search Concept
139 		//
140 
141 		try {
142 			ConceptChronicleBI searchConcept = WBUtility.createNewConcept(WBUtility.getConceptVersion(Search.SEARCH_PERSISTABLE.getUuids()[0]), saveConceptFSN, saveConceptPT);
143 			ConceptAttributeAB conceptAttributeBlueprintAmender = new ConceptAttributeAB(searchConcept.getConceptNid(), searchConcept.getVersion(WBUtility.getViewCoordinate()).getConceptAttributesActive().isDefined(), RefexDirective.INCLUDE); //bp.getConceptAttributeAB();
144 
145 			{
146 				// Start with Search Global Attributes
147 				RefexDynamicUsageDescription searchGlobalAttributesRDUD = RefexDynamicUsageDescriptionBuilder.readRefexDynamicUsageDescriptionConcept(Search.SEARCH_GLOBAL_ATTRIBUTES.getNid());
148 
149 				// Add View Coordinate byte[]
150 				RefexDynamicData[] searchGlobalAttributesData = new RefexDynamicData[searchGlobalAttributesRDUD.getColumnInfo().length];
151 
152 				// Serialize passed View Coordinate into byte[]
153 				ByteArrayOutputStream output = new ByteArrayOutputStream();
154 				ObjectOutputStream oos = new ObjectOutputStream(output);
155 				model.getViewCoordinate().writeExternal(oos);
156 				oos.flush();
157 
158 				// Construct and populate RefexDynamicData for View Coordinate
159 				RefexDynamicData viewCoordinateColumnData = new RefexDynamicByteArray(output.toByteArray());
160 				searchGlobalAttributesData[0] = viewCoordinateColumnData;
161 				RefexDynamicData maxResultsColumnData = new RefexDynamicInteger(model.getMaxResults());
162 				searchGlobalAttributesData[1] = maxResultsColumnData;
163 				if (model.getDroolsExpr() != null) {
164 					RefexDynamicData droolsExprColumnData = new RefexDynamicString(model.getDroolsExpr());
165 					searchGlobalAttributesData[2] = droolsExprColumnData;
166 				}
167 
168 				RefexDynamicCAB globalAttributesCAB;
169 				// cab.addAnnotationBlueprint(annotationBlueprint); for nesting
170 				//if (inputType_.getRefex() == null)
171 				//{
172 
173 				// Creates new refex
174 				globalAttributesCAB = new RefexDynamicCAB(searchConcept.getPrimordialUuid(), Search.SEARCH_GLOBAL_ATTRIBUTES.getUuids()[0]);
175 				//}
176 				//else
177 				//{
178 				// This only for editing existing concept
179 				//cab = inputType_.getRefex().makeBlueprint(WBUtility.getViewCoordinate(),IdDirective.PRESERVE, RefexDirective.INCLUDE);
180 				//}
181 				globalAttributesCAB.setData(searchGlobalAttributesData, WBUtility.getViewCoordinate());
182 
183 				conceptAttributeBlueprintAmender.addAnnotationBlueprint(globalAttributesCAB);
184 
185 				TerminologyBuilderBI builder = ExtendedAppContext.getDataStore().getTerminologyBuilder(WBUtility.getEC(), WBUtility.getViewCoordinate());
186 				builder.construct(globalAttributesCAB);
187 			}
188 
189 			// Add search type filter as index 0
190 			addFilterToRefex(searchConcept, conceptAttributeBlueprintAmender, model.getSearchType(), 0);
191 
192 			// Add additional filter as higher indices
193 			for (int filterIndex = 0; filterIndex < model.getFilters().size(); ++filterIndex) {
194 				addFilterToRefex(searchConcept, conceptAttributeBlueprintAmender, model.getFilters().get(filterIndex), filterIndex + 1);
195 			}
196 			
197 			LOG.debug("Displaying newly created save concept refexes");
198 			DynamicRefexHelper.displayDynamicRefexes(searchConcept);
199 			
200 			return searchConcept;
201 		} catch (IOException | InvalidCAB | ContradictionException | PropertyVetoException e) {
202 			throw new SearchConceptException(e.getLocalizedMessage(), e);
203 		}
204 	}

206 	private static void addFilterToRefex(ConceptChronicleBI searchConcept, ConceptAttributeAB conceptAttributeBlueprintAmender, Filter<? extends Filter<?>> currentFilter, int filterIndex) throws ValidationException, PropertyVetoException, IOException, ContradictionException, SearchConceptException, InvalidCAB {
207 		RefexDynamicUsageDescription filterRDUD = null;
208 
209 		if (currentFilter instanceof SearchTypeFilter) {
210 			if (filterIndex > 0) {
211 				throw new SearchConceptException("A SearchTypeFilter Filter must be first in refex list. " + currentFilter.getClass().getName() + " is a SearchTypeFilter filter, but its specified index is " + filterIndex);
212 			}
213 		} else /* if (! (currentFilter instanceof SearchTypeFilter)) */ {
214 			if (filterIndex == 0) {
215 				throw new SearchConceptException("A non-SearchTypeFilter Filter must not be first in refex list. " + currentFilter.getClass().getName() + " is a non-SearchTypeFilter filter, but its specified index is " + filterIndex);
216 			}
217 		}
218 
219 		ConceptSpec filterConceptSpec = null;
220 		if (currentFilter instanceof LuceneSearchTypeFilter) {
221 			filterConceptSpec = Search.SEARCH_LUCENE_FILTER;
222 		} else if (currentFilter instanceof RegExpSearchTypeFilter) {
223 			filterConceptSpec = Search.SEARCH_REGEXP_FILTER;
224 		} else if (currentFilter instanceof IsDescendantOfFilter) {
225 			filterConceptSpec = Search.SEARCH_ISDESCENDANTOF_FILTER;
226 		} else if (currentFilter instanceof IsAFilter) {
227 			filterConceptSpec = Search.SEARCH_ISA_FILTER;
228 		} else {
229 			throw new SearchConceptException("Unsupported Filter type " + currentFilter.getClass().getName());
230 		}
231 
232 		filterRDUD = RefexDynamicUsageDescriptionBuilder.readRefexDynamicUsageDescriptionConcept(filterConceptSpec.getNid());
233 
234 		// First create Filter, which has its own attributes plus common Filter attributes
235 		RefexDynamicData[] filterRefexData = new RefexDynamicData[filterRDUD.getColumnInfo().length];
236 
237 		if (currentFilter instanceof LuceneSearchTypeFilter) {
238 			// Construct and populate RefexDynamicData for search parameter
239 			LuceneSearchTypeFilter filter = (LuceneSearchTypeFilter)currentFilter;
240 
241 			if (filter.getSearchParameter() != null) {
242 
243 				RefexDynamicData searchParameterData = new RefexDynamicString(filter.getSearchParameter());
244 				filterRefexData[0] = searchParameterData;
245 			}
246 		} else if (currentFilter instanceof RegExpSearchTypeFilter) {
247 			// Construct and populate RefexDynamicData for search parameter
248 			RegExpSearchTypeFilter filter = (RegExpSearchTypeFilter)currentFilter;
249 
250 			if (filter.getSearchParameter() != null) {
251 
252 				RefexDynamicData searchParameterData = new RefexDynamicString(filter.getSearchParameter());
253 				filterRefexData[0] = searchParameterData;
254 			}
255 		} else if (currentFilter instanceof IsDescendantOfFilter) {
256 			// Construct and populate RefexDynamicData for search ascendant uuid
257 			IsDescendantOfFilter isDescendantOfFilter = (IsDescendantOfFilter)currentFilter;
258 
259 			if (isDescendantOfFilter.getNid() != 0) {
260 				UUID uuid = WBUtility.getConceptVersion(isDescendantOfFilter.getNid()).getPrimordialUuid();
261 				RefexDynamicData ascendantUuidData = new RefexDynamicUUID(uuid);
262 				filterRefexData[0] = ascendantUuidData;
263 			}
264 		} else if (currentFilter instanceof IsAFilter) {
265 			// Construct and populate RefexDynamicData for search ascendant uuid
266 			IsAFilter isAFilter = (IsAFilter)currentFilter;
267 
268 			if (isAFilter.getNid() != 0) {
269 				UUID uuid = WBUtility.getConceptVersion(isAFilter.getNid()).getPrimordialUuid();
270 				RefexDynamicData matchUuidData = new RefexDynamicUUID(uuid);
271 				filterRefexData[0] = matchUuidData;
272 			}
273 		}
274 
275 		RefexDynamicCAB filterRefexCAB;
276 		// cab.addAnnotationBlueprint(annotationBlueprint); for nesting
277 		//if (inputType_.getRefex() == null)
278 		//{
279 
280 		// Creates new refex
281 		filterRefexCAB = new RefexDynamicCAB(searchConcept.getPrimordialUuid(), filterConceptSpec.getUuids()[0]);
282 		//}
283 		//else
284 		//{
285 		// This only for editing existing concept
286 		//cab = inputType_.getRefex().makeBlueprint(WBUtility.getViewCoordinate(),IdDirective.PRESERVE, RefexDirective.INCLUDE);
287 		//}
288 		filterRefexCAB.setData(filterRefexData, WBUtility.getViewCoordinate());
289 
290 		conceptAttributeBlueprintAmender.addAnnotationBlueprint(filterRefexCAB);
291 
292 		// Handle Search Filter Attributes for Filter
293 		RefexDynamicUsageDescription nestedFilterAttributesRDUD = RefexDynamicUsageDescriptionBuilder.readRefexDynamicUsageDescriptionConcept(Search.SEARCH_FILTER_ATTRIBUTES.getNid());
294 
295 		// Common Filter attributes
296 		RefexDynamicData[] nestedFilterAttributesRefexData = new RefexDynamicData[nestedFilterAttributesRDUD.getColumnInfo().length];
297 
298 		// Construct and populate RefexDynamicData for search order
299 		RefexDynamicData filterOrderData = new RefexDynamicInteger(filterIndex);
300 		nestedFilterAttributesRefexData[0] = filterOrderData;
301 		// If relevant, populate with invert boolean
302 		if (currentFilter instanceof Invertable) {
303 			RefexDynamicData filterInvertData = new RefexDynamicBoolean(((Invertable)currentFilter).getInvert());
304 			nestedFilterAttributesRefexData[1] = filterInvertData;
305 		}
306 
307 		RefexDynamicCAB nestedFilterAttributesCAB;
308 		// cab.addAnnotationBlueprint(annotationBlueprint); for nesting
309 		//if (inputType_.getRefex() == null)
310 		//{
311 
312 		// Creates new refex
313 		nestedFilterAttributesCAB = new RefexDynamicCAB(filterRefexCAB.getComponentUuid(), Search.SEARCH_FILTER_ATTRIBUTES.getUuids()[0]);
314 		//}
315 		//else
316 		//{
317 		// This only for editing existing concept
318 		//cab = inputType_.getRefex().makeBlueprint(WBUtility.getViewCoordinate(),IdDirective.PRESERVE, RefexDirective.INCLUDE);
319 		//}
320 		nestedFilterAttributesCAB.setData(nestedFilterAttributesRefexData, WBUtility.getViewCoordinate());
321 
322 		filterRefexCAB.addAnnotationBlueprint(nestedFilterAttributesCAB);
323 
324 		TerminologyBuilderBI builder = ExtendedAppContext.getDataStore().getTerminologyBuilder(WBUtility.getEC(), WBUtility.getViewCoordinate());
325 		builder.construct(filterRefexCAB);
326 	}

356 	private static <T extends NonSearchTypeFilter<T>> void loadEmbeddedSearchFilterAttributes(RefexDynamicVersionBI<?> refex, Filter<?> newFilter, Map<Integer, Collection<T>> filterOrderMap) throws InvalidNameException, IndexOutOfBoundsException, IOException, ContradictionException, SearchConceptException {
357 		LOG.debug("Loading data into model from embedded Search Filter Attributes refex");
358 
359 		SearchTypeFilter<?> searchTypeFilter = null;
360 		T nonSearchTypeFilter = null;
361 		
362 		if (newFilter instanceof SearchTypeFilter) {
363 			searchTypeFilter = (SearchTypeFilter<?>)newFilter;
364 			if (filterOrderMap != null) {
365 				throw new SearchConceptException("Cannot call this method with a non-null filterOrderMap when passing a SearchTypeFilter. " + newFilter.getClass().getName() + " is a SearchTypeFilter filter.");
366 			}
367 		} else if (newFilter instanceof NonSearchTypeFilter) {
368 			nonSearchTypeFilter = (T)newFilter;
369 			if (filterOrderMap == null) {
370 				throw new SearchConceptException("Cannot call this method with a null filterOrderMap when passing a non-SearchTypeFilter. " + newFilter.getClass().getName() + " is a non-SearchTypeFilter filter.");
371 			}
372 		} else {
373 			throw new SearchConceptException("Unsupported Filter type " + newFilter.getClass().getName() + ". Must be of either " + SearchTypeFilter.class.getName() + " or " + NonSearchTypeFilter.class.getName());
374 		}
375 		
376 		// Now read SEARCH_FILTER_ATTRIBUTES refex column
377 		for (RefexDynamicVersionBI<?> embeddedRefex : refex.getRefexesDynamicActive(WBUtility.getViewCoordinate())) {
378 			DynamicRefexHelper.displayDynamicRefex(embeddedRefex);
379 
380 			RefexDynamicUsageDescription embeddedRefexDUD = null;
381 			try {
382 				embeddedRefexDUD = embeddedRefex.getRefexDynamicUsageDescription();
383 			} catch (IOException | ContradictionException e) {
384 				LOG.error("Failed performing getRefexDynamicUsageDescription() on embedded refex: caught " + e.getClass().getName() + " \"" + e.getLocalizedMessage() + "\"", e);
385 
386 				return;
387 			}
388 
389 			if (embeddedRefexDUD.getRefexName().equals(Search.SEARCH_FILTER_ATTRIBUTES.getDescription() /*"Search Filter Attributes"*/)) {
390 				RefexDynamicIntegerBI filterOrderCol = (RefexDynamicIntegerBI)embeddedRefex.getData(Search.ORDER_COLUMN.getDescription());
391 
392 				if (newFilter instanceof SearchTypeFilter) {
393 					if (filterOrderCol != null && filterOrderCol.getDataInteger() > 0) {
394 						throw new SearchConceptException("A SearchTypeFilter Filter must be first in list. " + newFilter.getClass().getName() + " is a SearchTypeFilter filter, but its stored index is " + filterOrderCol.getDataInteger());
395 					}
396 				} else /* if (! (currentFilter instanceof SearchTypeFilter)) */ {
397 					if (filterOrderMap.get(filterOrderCol.getDataInteger()) == null) {
398 						filterOrderMap.put(filterOrderCol.getDataInteger(), new ArrayList<>());
399 					}
400 					filterOrderMap.get(filterOrderCol.getDataInteger()).add(nonSearchTypeFilter);
401 
402 					if (filterOrderCol == null || filterOrderCol.getDataInteger() == 0) {
403 						throw new SearchConceptException("A non-SearchTypeFilter Filter must not be first in list. " + newFilter.getClass().getName() + " is a non-SearchTypeFilter filter, but its stored index is " + (filterOrderCol != null ? filterOrderCol.getDataInteger() : filterOrderCol));
404 					}
405 				}
406 
407 				LOG.debug("Read Integer filter order from " + embeddedRefexDUD.getRefexName() + " refex: \"" + (filterOrderCol != null ? filterOrderCol.getDataInteger() : filterOrderCol) + "\"");
408 
409 				RefexDynamicBooleanBI filterInvertCol = (RefexDynamicBooleanBI)embeddedRefex.getData(Search.FILTER_INVERT_COLUMN.getDescription());
410 				if (filterInvertCol != null) {
411 					if (newFilter instanceof Invertable) {
412 						((Invertable) newFilter).setInvert(filterInvertCol.getDataBoolean());
413 						LOG.debug("Read Boolean filter invert from " + embeddedRefexDUD.getRefexName() + " refex: \"" + filterInvertCol.getDataBoolean() + "\"");
414 
415 					} else if (filterInvertCol.getDataBoolean()) {
416 						LOG.error("Cannot make invertable non-invertable Filter of type " + newFilter.getClass().getName());
417 
418 						throw new SearchConceptException("Cannot make invertable non-invertable Filter of type " + newFilter.getClass().getName());
419 					}
420 				}
421 			} else {
422 				LOG.warn("Encountered unexpected embedded refex \"" + embeddedRefexDUD.getRefexName() + "\". Ignoring...");
423 			}
424 		}
425 	}

427 	public static SearchViewModel loadSavedSearch(SearchDisplayConcept displayConcept) throws SearchConceptException {
428 		LOG.info("loadSavedSearch(" + displayConcept + ")");
429 
430 		SearchViewModel model = null;
431 
432 		try {
433 			ConceptVersionBI matchingConcept = WBUtility.getConceptVersion(displayConcept.getNid());
434 
435 			if (matchingConcept != null) {
436 				LOG.debug("loadSavedSearch(): savedSearchesComboBox has concept: " + matchingConcept);
437 
438 				Map<Integer, Collection<NonSearchTypeFilter>> filterOrderMap = new TreeMap<>();
439 
440 				model = new SearchViewModel();
441 
442 				model.setName(WBUtility.getFullySpecifiedName(matchingConcept));
443 				model.setDescription(WBUtility.getConPrefTerm(matchingConcept.getNid()));
444 
445 				try {
446 					LOG.debug("loadSavedSearch(): concept \"" + displayConcept + "\" all refexes: " +  matchingConcept.getRefexes().size());
447 					LOG.debug("loadSavedSearch(): concept \"" + displayConcept + "\" all dynamic refexes: " +  matchingConcept.getRefexesDynamic().size());
448 					LOG.debug("loadSavedSearch(): concept \"" + displayConcept + "\" active dynamic refexes (StandardViewCoordinates.getWbAuxiliary()): " +  matchingConcept.getRefexesDynamicActive(StandardViewCoordinates.getWbAuxiliary()).size());
449 					LOG.debug("loadSavedSearch(): concept \"" + displayConcept + "\" active dynamic refexes (WBUtility.getViewCoordinate()): " +  matchingConcept.getRefexesDynamicActive(WBUtility.getViewCoordinate()).size());
