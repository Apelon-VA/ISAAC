19  package gov.va.isaac.mojos.dbBuilder;
20  
21  import gov.va.isaac.config.users.GenerateUsers;
22  import java.io.File;
23  import org.apache.maven.plugin.AbstractMojo;
24  import org.apache.maven.plugin.MojoExecutionException;
25  
26  /**
27   * Goal which processes a users.xml file which is formatted according to the UserGenerationSchema.xsd
28   * stored in otf-util.  See {@link GenerateUsers} for more details.
29   * 
30   * @goal create-users
31   * 
32   * @phase process-sources
33   */
34  public class GenerateUsersMojo extends AbstractMojo
35  {
36  
37  	/**
38  	 * The location of the users.xml file to process. If not provided,
39  	 * will read "/users.xml" from the classpath.
40  	 * 
41  	 * @parameter
42  	 * @optional
43  	 */
44  	File usersFileLocation = null;
45  
46  	/**
47  	 * To execute this mojo, you need to first have run the "Setup" mojo against
48  	 * the same database. Here, we assume the data store is ready to go and we can
49  	 * acquire it simply as shown in the createPath method below.
50  	 * 
51  	 * If not yet initialized, this will fail.
52  	 */
53  	@Override
54  	public void execute() throws MojoExecutionException
55  	{
56  		try
57  		{
58  			getLog().info("Processing the users file " + (usersFileLocation == null ? "(classpath) /users.xml" : usersFileLocation.getAbsolutePath()));
59  			if (usersFileLocation != null)
60  			{
61  				GenerateUsers.generateUsers(usersFileLocation);
62  			}
63  			else
64  			{
65  				GenerateUsers.generateUsers();
66  			}
67  			getLog().info("Done Processing the users file.");
68  		}
69  		catch (Exception e)
70  		{
71  			throw new MojoExecutionException("Unexpected error processing the users file", e);
72  		}
73  	}	
74  }
