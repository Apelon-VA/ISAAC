19  package gov.va.isaac.mojos.dbBuilder;
20  
21  import gov.va.isaac.AppContext;
22  import gov.va.isaac.models.api.BdbInformationModelService;
23  
24  import org.apache.maven.plugin.AbstractMojo;
25  import org.apache.maven.plugin.MojoExecutionException;
26  import org.ihtsdo.otf.tcc.api.store.TerminologyStoreDI;
27  import org.ihtsdo.otf.tcc.datastore.BdbPathManager;
28  
29  /**
30   * Goal which creates information model refset metadata in the ISAAC environment
31   * 
32   * @goal create-metadata
33   * 
34   * @phase process-sources
35   */
36  public class InfoModelMetadata extends AbstractMojo {
37  
38    /** The data store. */
39    private TerminologyStoreDI dataStore = null;
40  
41    /**
42     * To execute this mojo, you need to first have run the "Setup" mojo against
43     * the same database. Here, we assume the data store is ready to go and we can
44     * acquire it simply as shown in the createPath method below.
45     * 
46     * If not yet initialized, this will fail.
47     */
48    @Override
49    public void execute() throws MojoExecutionException {
50      try {
51        getLog().info("Creating metadata");
52        BdbPathManager.get().resetPathMap();  //TODO dan doesn't know what this does... really the only thing preventing the killing of this mojo.
53        createMetadata();
54        getLog().info("Done creating new path.");
55      } catch (Exception e) {
56        e.printStackTrace();
57        throw new MojoExecutionException("Unexpected failure when creating path",
58            e);
59      }
60    }
61  
62    /**
63     * Creates the information model metadata.
64     *
65     * @throws Exception the exception
66     */
67    private void createMetadata() throws Exception {
68  
69      // Obtain already-open datastore and use it
70      dataStore = AppContext.getService(TerminologyStoreDI.class);
71      BdbInformationModelService service = new BdbInformationModelService(
72          dataStore);
73      //TODO kill this MOJO - it is no longer necessary.  The metadata should be properly defined in isaac-constants, within the InformationModels class.
74      //This would also let us get rid of the pom-level dependency on the import-export module.
75      service.createMetadataConcepts();
76    }
77  }
