19  package gov.va.isaac.mojos.changesets;
20  
21  import gov.va.isaac.config.changesets.ChangesetProcessor;
22  import java.io.File;
23  import org.apache.maven.plugin.AbstractMojo;
24  import org.apache.maven.plugin.MojoExecutionException;
25  
26  /**
27   * Goal which recursively processes any changeset (*.eccs) files found in the specified folder.
28   * 
29   * @goal process-changesets
30   * @phase compile
31   */
32  
33  public class ProcessChangesets extends AbstractMojo
34  {
35  
36  	/**
37  	 * The location of the (already existing) profiles folder which contains changesets to process
38  	 * 
39  	 * @parameter
40  	 * @required
41  	 */
42  	File userProfileFolderLocation = null;
43  
44  	/**
45  	 * @throws MojoExecutionException
46  	 */
47  	public ProcessChangesets() throws MojoExecutionException
48  	{
49  		super();
50  	}
51  
52  	@Override
53  	public void execute() throws MojoExecutionException
54  	{
55  		try
56  		{
57  			getLog().info("Processing " + userProfileFolderLocation.getAbsolutePath() + " for changesets");
58  			ChangesetProcessor.processChangeSets(userProfileFolderLocation);
59  			getLog().info("Done Processing changesets");
60  		}
61  		catch (Exception e)
62  		{
63  			throw new MojoExecutionException("Unexpected error configuring SCM for the profiles", e);
64  		}
65  	}
66  }
