45  /**
46   * {@link ProfilesMojoBase}
47   * 
48   * This allows authentication to be passed in via system property, parameter, or, will 
49   * prompt for the username/password (if allowed by the system property 'profileSyncNoPrompt')
50   * IN THAT ORDER.  System properties have the highest priority.
51   * 
52   * To prevent prompting during automated runs - set the system property 'profileSyncNoPrompt=true'
53   * To set the username via system property - set 'profileSyncUsername=username'
54   * To set the password via system property - set 'profileSyncPassword=password'
55   * 
56   * To enable authentication without prompts, using public keys - set both of the following 
57   *   'profileSyncUsername=username'
58   *   'profileSyncNoPrompt=true' 
59   *   
60   * This will cause a public key authentication to be attempted using the ssh credentials found 
61   * in the current users .ssh folder (in their home directory)
62   *
63   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
64   */
65  public abstract class ProfilesMojoBase extends AbstractMojo
66  {
67  	// For disabling Profile Sync entirely
68  	public static final String PROFILE_SYNC_DISABLE = "profileSyncDisable";
69  	
70  	// For preventing command line prompts for credentials during automated runs - set this system property to true.
71  	public static final String PROFILE_SYNC_NO_PROMPTS = "profileSyncNoPrompt";
72  	
73  	// Allow setting the username via a system property
74  	public static final String PROFILE_SYNC_USERNAME_PROPERTY = "profileSyncUsername";
75  	
76  	// Allow setting the password via a system property
77  	public static final String PROFILE_SYNC_PASSWORD_PROPERTY = "profileSyncPassword";
78  	
79  	private boolean disableHintGiven = false;
80  	
81  	/**
82  	 * The location of the (already existing) profiles folder which should be shared via SCM.
83  	 * @parameter
84  	 * @required
85  	 */
86  	File userProfileFolderLocation = null;
87  	
88  	/**
89  	 * The location of the (already existing) app.xml file which contains the SCM connection information.
90  	 * @parameter
91  	 * @required
92  	 */
93  	File appXMLFile = null;
94  	
95  	/**
96  	 * The username to use for remote operations
97  	 * @parameter
98  	 * @optional
99  	 */
100 	private String profileSyncUsername = null;
101 	
102 	/**
103 	 * The password to use for remote operations
104 	 * @parameter
105 	 * @optional
106 	 */
107 	private String profileSyncPassword = null;
108 	
109 	private IsaacAppConfig config_;
110 	private static String username = null;
111 	private static String password = null;
112 
113 	public ProfilesMojoBase() throws MojoExecutionException
114 	{
115 		super();
116 	}
117 	
118 	/**
119 	 * @see org.apache.maven.plugin.Mojo#execute()
120 	 */
121 	@Override
122 	public void execute() throws MojoExecutionException
123 	{
124 		if (appXMLFile == null || !appXMLFile.isFile())
125 		{
126 			throw new MojoExecutionException("The file specified in the appXMLFile parameter must exist, and the parameter must be specified.");
127 		}
128 		
129 		try
130 		{
131 			config_ = IsaacAppConfigWrapper.unmarshallStream(new FileInputStream(appXMLFile));
132 		}
133 		catch (IOException | JAXBException | SAXException e)
134 		{
135 			throw new MojoExecutionException("Failure reading " + appXMLFile, e);
136 		}
137 		if (StringUtils.isNotBlank(config_.getChangeSetUrl()))
138 		{
139 			if (config_.getChangeSetUrlType() == null)
140 			{
141 				throw new MojoExecutionException("If the 'changeSetUrl' parameter is provided, then you must provide the 'changeSetUrlType' parameter");
142 			}
143 		}
144 	}
145 
146 	protected boolean skipRun()
147 	{
148 		if (Boolean.getBoolean(PROFILE_SYNC_DISABLE))
149 		{
150 			return true;
151 		}
152 		if (StringUtils.isBlank(config_.getChangeSetUrl()))
153 		{
154 			getLog().info("No SCM configuration will be done - no 'changeSetUrl' parameter was provided");
155 			return true;
156 		}
157 		else
158 		{
159 			return false;
160 		}
161 	}
162 	
163 	protected ProfileSyncI getProfileSyncImpl() throws MojoExecutionException
164 	{
165 		if (config_.getChangeSetUrlType() == ChangeSetSCMType.GIT)
166 		{
167 			return new SyncServiceGIT(userProfileFolderLocation);
168 		}
169 		else if (config_.getChangeSetUrlType() == ChangeSetSCMType.SVN)
170 		{
171 			return new SyncServiceSVN(userProfileFolderLocation);
172 		}
173 		else
174 		{
175 			throw new MojoExecutionException("Unsupported change set URL Type");
176 		}
177 	}
178 	
179 	/**
180 	 * Does the necessary substitution to put the contents of getUserName() into the URL, if a known pattern needing substitution is found.
181 	 *  ssh://someuser@csfe.aceworkspace.net:29418/... for example needs to become:
182 	 *  ssh://<getUsername()>@csfe.aceworkspace.net:29418/...
183 	 * @throws MojoExecutionException 
184 	 */
185 	protected String getURL() throws MojoExecutionException
186 	{
187 		return getProfileSyncImpl().substituteURL(config_.getChangeSetUrl(), getUsername());
188 	}
189 	
190 	protected String getUsername() throws MojoExecutionException
191 	{
192 		if (username == null)
193 		{
194 			username = System.getProperty(PROFILE_SYNC_USERNAME_PROPERTY);
195 			
196 			//still blank, try property
197 			if (StringUtils.isBlank(username))
198 			{
199 				username = profileSyncUsername;
200 			}
201 			
202 			//still no username, prompt if allowed
203 			if (StringUtils.isBlank(username) && !Boolean.getBoolean(PROFILE_SYNC_NO_PROMPTS))
204 			{
205 				Callable<Void> callable = new Callable<Void>()
206 				{
207 					@Override
208 					public Void call() throws Exception
209 					{
210 						if (!disableHintGiven)
211 						{
212 							System.out.println("To disable remote sync during build, add '-D" + PROFILE_SYNC_DISABLE + "=true' to your maven command");
213 							disableHintGiven = true;
214 						}
215 						
216 						try
217 						{
218 							System.out.println("Enter the " + config_.getChangeSetUrlType().name() + " username for the Profiles/Changset remote store (" +
219 									config_.getChangeSetUrl() + "):");
220 							BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
221 							username = br.readLine();
222 						}
223 						catch (IOException e)
224 						{
225 							throw new MojoExecutionException("Error reading username from console");
226 						}
227 						return null;
228 					}
229 				};
230 				
231 				try
232 				{
233 					Executors.newSingleThreadExecutor(new ThreadFactory()
234 					{
235 						@Override
236 						public Thread newThread(Runnable r)
237 						{
238 							Thread t = new Thread(r, "User Prompt Thread");
239 							t.setDaemon(true);
240 							return t;
241 						}
242 					}).submit(callable).get(2, TimeUnit.MINUTES);
243 				}
244 				catch (TimeoutException | InterruptedException e)
245 				{
246 					throw new MojoExecutionException("Username not provided within timeout");
247 				}
248 				catch (ExecutionException ee)
249 				{
250 					throw (ee.getCause() instanceof MojoExecutionException ? (MojoExecutionException)ee.getCause() : 
251 						new MojoExecutionException("Unexpected", ee.getCause()));
252 				}
253 			}
254 		}
255 		return username;
256 	}
257 	
258 	protected String getPassword() throws MojoExecutionException
259 	{
260 		if (password == null)
261 		{
262 			password = System.getProperty(PROFILE_SYNC_PASSWORD_PROPERTY);
263 			
264 			//still blank, try the passed in param
265 			if (StringUtils.isBlank(password))
266 			{
267 				password = profileSyncPassword;
268 			}
269 			
270 			//still no password, prompt if allowed
271 			if (StringUtils.isBlank(password) && !Boolean.getBoolean(PROFILE_SYNC_NO_PROMPTS))
272 			{
273 				Callable<Void> callable = new Callable<Void>()
274 				{
275 					@Override
276 					public Void call() throws Exception
277 					{
278 						try
279 						{
280 							if (!disableHintGiven)
281 							{
282 								System.out.println("To disable remote sync during build, add '-D" + PROFILE_SYNC_DISABLE + "=true' to your maven command");
283 								disableHintGiven = true;
284 							}
285 							System.out.println("Enter the " + config_.getChangeSetUrlType().name() + " password for the Profiles/Changset remote store: (" +
286 									config_.getChangeSetUrl() + "):");
287 							
288 							//Use console if available, for password masking
289 							Console console = System.console();
290 							if (console != null)
291 							{
292 								password = new String(console.readPassword());
293 							}
294 							else
295 							{
296 								BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
297 								password = br.readLine();
298 							}
299 						}
300 						catch (IOException e)
301 						{
302 							throw new MojoExecutionException("Error reading password from console");
303 						}
304 						return null;
305 					}
306 				};
307 				
308 				try
309 				{
310 					Executors.newSingleThreadExecutor(new ThreadFactory()
311 					{
312 						@Override
313 						public Thread newThread(Runnable r)
314 						{
315 							Thread t = new Thread(r, "User Password Prompt Thread");
316 							t.setDaemon(true);
317 							return t;
318 						}
319 					}).submit(callable).get(2, TimeUnit.MINUTES);
320 				}
321 				catch (TimeoutException | InterruptedException e)
322 				{
323 					throw new MojoExecutionException("Password not provided within timeout");
324 				}
325 				catch (ExecutionException ee)
326 				{
327 					throw (ee.getCause() instanceof MojoExecutionException ? (MojoExecutionException)ee.getCause() : 
328 						new MojoExecutionException("Unexpected", ee.getCause()));
329 				}
330 			}
331 		}
332 		return password;
333 	}
