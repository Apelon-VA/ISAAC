235 	private void sync()
236 	{
237 		cancelRequested_ = false;
238 		try
239 		{
240 			UserProfile up = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
241 			if (syncService_.isRootLocationConfiguredForSCM())
242 			{
243 				addLine("Setting Remote Address");
244 				
245 				boolean successful = false;
246 				while (!successful)
247 				{
248 					try
249 					{
250 						syncService_.relinkRemote(url_, up.getSyncUsername(), up.getSyncPassword());
251 						successful = true;
252 					}
253 					catch (AuthenticationException ae)
254 					{
255 						try
256 						{
257 							promptUserForCreds(up);
258 						}
259 						catch (InterruptedException e)
260 						{
261 							//noop
262 						}
263 						
264 						if (cancelRequested_)
265 						{
266 							return;
267 						}
268 					}
269 					catch (Exception e)
270 					{
271 						log.error("Sync failure", e);
272 						AppContext.getCommonDialogs().showErrorDialog("Sync Error", "Sync error setting up remote address", e.getMessage());
273 						return;
274 					}
275 				}
276 			}
277 			else
278 			{
279 				//Note, this operation is not expected to be run by real end users - hence, no attempt to gather a new password if necessary.
280 				//this is mostly for test purposes.  This is typically done during bundle build.
281 				addLine("Initial Sync Beginning");
282 				try
283 				{
284 					syncService_.linkAndFetchFromRemote(url_, up.getSyncUsername(), up.getSyncPassword());
285 					addLine("*****************");
286 					addLine("INITIAL SYNC PERFORMED - PLEASE RESTART ISAAC AFTER THIS OPERATION COMPLETES!");
287 				}
288 				catch (Exception e)
289 				{
290 					log.error("Sync failure", e);
291 					AppContext.getCommonDialogs().showErrorDialog("Sync Error", "Sync error performing initial sync", e.getMessage());
292 					return;
293 				}
294 			}
295 			
296 			if (cancelRequested_)
297 			{
298 				addLine("Cancelled");
299 				return;
300 			}
301 			
302 			Set<String> changedFiles = new HashSet<>();
303 			
304 			//recover from a previous merge fail state that didn't get resolved for whatever reason
305 			try
306 			{
307 				Set<String> conflictFiles = syncService_.getFilesInMergeConflict();
308 				if (conflictFiles.size() > 0)
309 				{
310 					MergeFailure mf = new MergeFailure(conflictFiles, new HashSet<String>());
311 					changedFiles.addAll(resolveMergeFailure(mf));
312 				}
313 			}
314 			catch (Exception e)
315 			{
316 				log.error("Sync failure", e);
317 				AppContext.getCommonDialogs().showErrorDialog("Sync Error", "Error checking for aborted merge conflicts", e.getMessage());
318 				return;
319 			}
320 			
321 			if (cancelRequested_)
322 			{
323 				addLine("Cancelled");
324 				return;
325 			}
326 			addLine("Adding new local files");
327 			
328 			try
329 			{
330 				syncService_.addUntrackedFiles();
331 			}
332 			catch (Exception e)
333 			{
334 				log.error("Sync failure", e);
335 				AppContext.getCommonDialogs().showErrorDialog("Sync Error", "Error adding new files", e.getMessage());
336 				return;
337 			}
338 			
339 			if (cancelRequested_)
340 			{
341 				addLine("Cancelled");
342 				return;
343 			}
344 			addLine("Performing remote sync");
345 			try
346 			{
347 				int modifiedFileCount = syncService_.getLocallyModifiedFileCount();
348 				addLine(modifiedFileCount + " local modifications to be sent");
349 				
350 				commitMessage_ = null;
351 				
352 				if (modifiedFileCount > 0)
353 				{
354 					CountDownLatch await = new CountDownLatch(1);
355 					
356 					
357 					Platform.runLater(() ->
358 					{
359 						new CommitMessage(root_.getScene().getWindow()).getMessage(result ->
360 						{
361 							commitMessage_ = result;
362 							await.countDown();
363 						});
364 					});
365 					
366 					await.await();
367 					
368 					if (StringUtils.isBlank(commitMessage_))
369 					{
370 						addLine("Commit message is required.  Cancelling.");
371 						return;
372 					}
373 				}
374 				
375 				boolean successful = false;
376 				while (!successful)
377 				{
378 					try
379 					{
380 						//If we didn't collect a commit message above, we had no outgoing changes - just sync down
381 						if (commitMessage_ == null)
382 						{
383 							//in theory, it should be impossible to get a merge failure here - but if we do - just run them through the normal 
384 							//resolution process.
385 							addLine("No local changes - checking for changes on server");
386 							changedFiles.addAll(syncService_.updateFromRemote(up.getSyncUsername(), up.getSyncPassword(), MergeFailOption.FAIL));
387 							successful = true;
388 						}
389 						else
390 						{
391 							//sync down and push back
392 							addLine("Sending changes and checking for changes on server");
393 							changedFiles.addAll(syncService_.updateCommitAndPush(commitMessage_, up.getSyncUsername(), up.getSyncPassword(), 
394 									MergeFailOption.FAIL, (String[])null));
395 							successful = true;
396 						}
397 					}
398 					catch (MergeFailure mf)
399 					{
400 						if (commitMessage_ == null)
401 						{
402 							//We (somehow) got a merge failure when only doing an update, even though we had no changes to push.
403 							//put in a commit message, so on the next loop of the code, we do a push as well, to push the resolution.
404 							commitMessage_ = "Merge Failure Resolution";
405 						}
406 						changedFiles.addAll(resolveMergeFailure(mf));
407 					}
408 					catch (AuthenticationException ae)
409 					{
410 						promptUserForCreds(up);
411 						
412 						if (cancelRequested_)
413 						{
414 							return;
415 						}
416 					}
417 				}
418 			}
419 			catch (Exception e)
420 			{
421 				log.error("Sync failure", e);
422 				AppContext.getCommonDialogs().showErrorDialog("Sync Error", "Error adding new files", e.getMessage());
423 				return;
424 			}
425 			
426 			
427 			//Process the changed files list
428 			addLine("Processing the changed files (" + changedFiles.size() + ")");
429 			StringBuilder errorsDuringProcess = new StringBuilder();
430 			try
431 			{
432 				AppContext.getService(BdbTerminologyStore.class).suspendChangeNotifications();
433 				for (String s : changedFiles)
434 				{
435 					log.debug("Post processing {} after change during sync", s);
436 					File f = new File(syncService_.getRootLocation(), s);
437 					if (f.getName().equals(UserProfileManager.PREFS_FILE_NAME) && f.getParentFile().getName().equals(ExtendedAppContext.getCurrentlyLoggedInUser()))
438 					{
439 						addLine("Rereading current user profile");
440 						try
441 						{
442 							AppContext.getService(UserProfileManager.class).rereadProfile();
443 						}
444 						catch (IOException e)
445 						{
446 							log.error("Error rereading changed user profile!", e);
447 							AppContext.getCommonDialogs().showErrorDialog("Unexpected error reading updated user profile", e);
448 						}
449 					}
450 					else if (f.getName().toLowerCase().endsWith(".eccs"))
451 					{
452 						try
453 						{
454 							addLine("Processing changeset " + f.getName());
455 							ChangeSetReader csr = new ChangeSetReader();
456 							csr.setChangeSetFile(f);
457 							Set<ConceptChronicleBI> indexedAnnotationConcepts = new HashSet<>();
458 							csr.read(indexedAnnotationConcepts);
459 							if (indexedAnnotationConcepts.size() > 0)
460 							{
461 								log.info("Dan doesn't know what to do with this after change set processing: {}", indexedAnnotationConcepts);
462 							}
463 						}
464 	
465 						catch (Exception e)
466 						{
467 							log.error("Error processing change set file " + f.getAbsolutePath(), e);
468 							errorsDuringProcess.append("Error processing change set file " + f.getName() + "/r");
469 						}
470 					}
471 					else
472 					{
473 						log.info("No processing done for changed file {}", f.getAbsolutePath());
474 					}
475 				}
476 			}
477 			finally
478 			{
479 				AppContext.getService(BdbTerminologyStore.class).resumeChangeNotifications();
480 			}
481 			if (errorsDuringProcess.length() > 0)
482 			{
483 				AppContext.getCommonDialogs().showErrorDialog("Errors processing changesets", "Errors processing changesets:", errorsDuringProcess.toString());
484 			}
485 			addLine("Syncronization complete!");
486 		}
487 		finally
488 		{
489 			Platform.runLater(() ->
490 			{
491 				pb_.setProgress(0.0);
492 				running_.set(false);
493 			});
494 		}
495 	}

497 	private void promptUserForCreds(UserProfile up) throws InterruptedException
498 	{
499 		CountDownLatch awaitCreds = new CountDownLatch(1);
500 		
501 		Platform.runLater(() ->
502 		{
503 			AppContext.getService(CredentialsPromptDialog.class).showView(up.getSyncUsername(), up.getSyncPassword(), 
504 					"Please provide the Sync credentials", credentials ->
505 			{
506 				if (credentials == null)
507 				{
508 					addLine("Cancelling");
509 					cancelRequested_ = true;
510 				}
511 				else
512 				{
513 					
514 					if (!up.getSyncUsername().equals(credentials.getUsername()))
515 					{
516 						try
517 						{
518 							syncService_.relinkRemote(syncService_.substituteURL(AppContext.getAppConfiguration().getChangeSetUrl(), 
519 									credentials.getUsername()), credentials.getUsername(), credentials.getPassword());
520 						}
521 						catch (AuthenticationException e)
522 						{
523 							log.info("Sync credentials still incorrect", e);
524 						}
525 						catch (Exception e)
526 						{
527 							//highly unlikely.. don't care, it will just fail - will work next time.
528 							log.error("Unexpected", e);
529 						}
530 					}
531 					up.setSyncUsername(credentials.getUsername());
532 					up.setSyncPassword(credentials.getPassword());
533 					
534 					try
535 					{
536 						AppContext.getService(UserProfileManager.class).saveChanges(up);
537 					}
538 					catch (Exception e)
539 					{
540 						//doesn't really matter - just a pw change, no big deal if we can't save it.
541 						log.error("Unexpected error changing profile change", e);
542 					}
543 				}
544 				awaitCreds.countDown();
545 			});
546 		});
547 		
548 		awaitCreds.await();
549 	}

551 	private Set<String> resolveMergeFailure(MergeFailure mf) throws IllegalArgumentException, IOException
552 	{
553 		Set<String> changedFiles = mf.getFilesChangedDuringMergeAttempt();
554 		
555 		CountDownLatch cdl = new CountDownLatch(1);
556 		HashMap<String, MergeFailOption> resolutions = new HashMap<String, MergeFailOption>();
557 
558 		Platform.runLater(() ->
559 		{
560 			new ResolveConflicts(root_.getScene().getWindow(), mf.getMergeFailures(), new Consumer<HashMap<String, MergeFailOption>>()
561 			{
562 				@Override
563 				public void accept(HashMap<String, MergeFailOption> t)
564 				{
565 					resolutions.putAll(t);
566 					cdl.countDown();
567 				}
568 			});
569 		});
570 		
571 		try
572 		{
573 			cdl.await();
574 		}
575 		catch (InterruptedException e)
576 		{
577 			log.info("Interrupted during wait for resolutions");
578 		}
579 		
580 		try
581 		{
582 			syncService_.resolveMergeFailures(resolutions);
583 		}
584 		catch (MergeFailure nestedMF)
585 		{
586 			changedFiles.addAll(resolveMergeFailure(nestedMF));
587 		}
588 		
589 		return changedFiles;
590 	}

617 	private void syncRequested()
618 	{
619 		//Check and see if we have a sync implementation present
620 		ProfileSyncI syncImpl = AppContext.getService(ProfileSyncI.class);
621 		if (syncImpl == null)
622 		{
623 			log.error("No implementation of ProfileSyncI is available on the classpath.  Datastore sync is not available.");
624 			AppContext.getCommonDialogs().showInformationDialog("Sync not installed in bundle", 
625 					"This ISAAC bundle has not been configured for Sync.\nPlease consult with the administrators for this bundle.");
626 		}
627 		else
628 		{
629 			String url = AppContext.getAppConfiguration().getChangeSetUrl();
630 			String urlType = AppContext.getAppConfiguration().getChangeSetUrlTypeName();
631 			
632 			if (StringUtils.isBlank(url) || StringUtils.isBlank(urlType))
633 			{
634 				log.error("Sync URL and/or Sync URL Type is missing or blank.  Cannot sync.");
635 				AppContext.getCommonDialogs().showInformationDialog("Sync not configured correctly in bundle", 
636 						"This ISAAC bundle has not been configured for Sync.\nPlease consult with the administrators for this bundle.");
637 			}
638 			else
639 			{
640 				syncService_ = AppContext.getService(ProfileSyncI.class, urlType);
641 				
642 				if (syncService_ == null)
643 				{
644 					log.error("Sync implementation for {} is not available. Cannot sync.", urlType);
645 					AppContext.getCommonDialogs().showInformationDialog("Sync not configured correctly in bundle", 
646 							"This ISAAC bundle has not been configured for Sync using " + urlType + ".\nPlease consult with the administrators for this bundle.");
647 				}
648 				else
649 				{
650 					syncService_.setRootLocation(AppContext.getService(UserProfileManager.class).getProfilesFolder());
651 					showView(AppContext.getMainApplicationWindow().getPrimaryStage());
652 				}
653 			}
654 		}
655 	}
