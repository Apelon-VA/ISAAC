45  /**
46   * 
47   * {@link ResolveConflicts}
48   *
49   * A Simple dialog to the user decision for handeling merge conflicts
50   *
51   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
52   */
53  
54  public class ResolveConflicts extends Stage
55  {
56  	private BorderPane root_;
57  	private Consumer<HashMap<String, MergeFailOption>> sendResultTo_;
58  	private ValidBooleanBinding allValid ;
59  	private HashMap<String, ToggleGroup> selections = new HashMap<>();
60  	
61  	protected ResolveConflicts(Window parent, Set<String> filesWithConflicts, Consumer<HashMap<String, MergeFailOption>> sendResultTo)
62  	{
63  		super();
64  		sendResultTo_ = sendResultTo;
65  		setTitle("Resolve Merge Conflicts");
66  		setResizable(true);
67  
68  		initOwner(parent);
69  		initModality(Modality.WINDOW_MODAL);
70  		initStyle(StageStyle.UTILITY);
71  
72  		setOnCloseRequest((event) -> {
73  			event.consume();
74  		});
75  
76  		setupGUI(filesWithConflicts);
77  		
78  		Scene scene = new Scene(root_);
79  		scene.getStylesheets().add(ResolveConflicts.class.getResource("/isaac-shared-styles.css").toString());
80  		setScene(scene);
81  		sizeToScene();
82  		show();
83  	}
84  
85  	private void setupGUI(Set<String> filesWithConflicts)
86  	{
87  		VBox top = new VBox();
88  		top.setAlignment(Pos.CENTER);
89  		Label title = new Label("Resolve Merge Conflicts");
90  		title.getStyleClass().add("headerBackground");
91  		title.getStyleClass().add("titleLabel");
92  		title.setMaxWidth(Double.MAX_VALUE);
93  		title.setPrefHeight(40.0);
94  		title.setAlignment(Pos.CENTER);
95  		
96  		top.getChildren().add(title);
97  		
98  		Label details = new Label("Please select an action for each conflict");
99  		details.setPadding(new Insets(5.0));
100 		top.getChildren().add(details);
101 		
102 		root_ = new BorderPane();
103 		root_.setPrefWidth(500);
104 		
105 		root_.setTop(top);
106 		
107 		
108 		GridPane gp = new GridPane();
109 		gp.setPadding(new Insets(5.0));
110 		gp.setMaxWidth(Double.MAX_VALUE);
111 		gp.setHgap(10.0);
112 		gp.setVgap(5.0);
113 		
114 		int row = 0;
115 		for (String s : filesWithConflicts)
116 		{
117 			Label l = new Label(s);
118 			GridPane.setHgrow(l, Priority.ALWAYS);
119 			gp.add(l, 0, row);
120 			
121 			ToggleGroup tg = new ToggleGroup();
122 			RadioButton local = new RadioButton("Keep Local Version");
123 			local.setUserData(MergeFailOption.KEEP_LOCAL);
124 			local.setToggleGroup(tg);
125 			
126 			gp.add(local, 1, row);
127 			
128 			RadioButton remote = new RadioButton("Use Server Version");
129 			remote.setUserData(MergeFailOption.KEEP_REMOTE);
130 			remote.setToggleGroup(tg);
131 			
132 			gp.add(remote, 2, row++);
133 			
134 			selections.put(s, tg);
135 		}
136 		
137 		ScrollPane sp = new ScrollPane(gp);
138 		sp.setFitToWidth(true);
139 		root_.setCenter(sp);
140 		
141 		allValid = new ValidBooleanBinding()
142 		{
143 			{
144 				setComputeOnInvalidate(true);
145 				for (Entry<String, ToggleGroup> tg : selections.entrySet())
146 				{
147 					for (Toggle t : tg.getValue().getToggles())
148 					{
149 						bind(t.selectedProperty());
150 					}
151 				}
152 			}
153 			
154 			@Override
155 			protected boolean computeValue()
156 			{
157 				for (Entry<String, ToggleGroup> tg : selections.entrySet())
158 				{
159 					if (tg.getValue().getSelectedToggle() == null)
160 					{
161 						return false;
162 					}
163 				}
164 				return true;
165 			}
166 		};
167 		
168 		VBox vbox = new VBox();
169 		vbox.setAlignment(Pos.CENTER);
170 		vbox.setMaxWidth(Double.MAX_VALUE);
171 		vbox.setFillWidth(true);
172 		vbox.setPadding(new Insets(10.0));
173 		
174 		Button ok = new Button("Ok");
175 		ok.disableProperty().bind(allValid.not());
176 		ok.setOnAction(event ->
177 		{
178 			HashMap<String, MergeFailOption> result = new HashMap<>();
179 			
180 			for (Entry<String, ToggleGroup> tg : selections.entrySet())
181 			{
182 				result.put(tg.getKey(), (MergeFailOption)tg.getValue().getSelectedToggle().getUserData());
183 			}
184 			
185 			if (sendResultTo_ != null)
186 			{
187 				sendResultTo_.accept(result);
188 			}
189 			this.hide();
190 		});
191 		
192 		vbox.getChildren().add(ok);
193 		
194 		root_.setBottom(vbox);
195 	}
