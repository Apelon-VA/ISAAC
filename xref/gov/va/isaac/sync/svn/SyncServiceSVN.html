127 	/**
128 	 * @throws AuthenticationException 
129 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#linkAndFetchFromRemote(java.io.File, java.lang.String, java.lang.String, java.lang.String)
130 	 */
131 	@Override
132 	public void linkAndFetchFromRemote(String remoteAddress, String username, String password) throws IllegalArgumentException, IOException, AuthenticationException
133 	{
134 		log.info("linkAndFetchFromRemote called - folder: {}, remoteAddress: {}, username: {}", localFolder_, remoteAddress, username);
135 		setupCredentials(username, password);
136 		try
137 		{
138 			File svnFolder = new File(localFolder_, ".svn");
139 			
140 			if (svnFolder.isDirectory())
141 			{
142 				log.debug("Root folder already contains a .svn folder.  Checking to see if it can be relinked to remote.");
143 				
144 				String id = getSvn().getWCClient().doInfo(SVNURL.parseURIEncoded(SVNEncodingUtil.autoURIEncode(remoteAddress)), SVNRevision.HEAD, SVNRevision.HEAD)
145 						.getRepositoryUUID();
146 				
147 				log.debug("Remote repository ID: {}", id);
148 				
149 				String localId = getSvn().getStatusClient().doStatus(localFolder_, false).getRepositoryUUID();
150 				log.debug("Local repository ID: {}", localId);
151 				
152 				if (id.equals(localId))
153 				{
154 					log.info("Ok to re-link local to remote - they have the same ID");
155 					relinkRemote(remoteAddress, username, password);
156 					
157 					//But, if we have any pre-existing merge conflicts.....
158 					Set<String> conflicts = getFilesInMergeConflict();
159 					if (conflicts.size() > 0)
160 					{
161 						log.info("Local repository was left in a conflicted state.  Will discared local repository information, and check out clean (preserving local files)");
162 						
163 						for (String s : conflicts)
164 						{
165 							File mine = new File(localFolder_, s + ".mine");
166 							File conflicted = new File(localFolder_, s);
167 							
168 							if (mine.isFile() && conflicted.isFile())
169 							{
170 								log.info("Deleting pre-merged conflicted file " + conflicted.getAbsolutePath() + " replacing with " + mine.getAbsolutePath());
171 								if (conflicted.delete())
172 								{
173 									mine.renameTo(conflicted);
174 								}
175 							}
176 							else
177 							{
178 								log.error("Unresolved conflict being left - user will have to manually resolve before next commit!");
179 							}
180 						}
181 						
182 						deleteRecursive(svnFolder.toPath());
183 					}
184 				}
185 				else
186 				{
187 					log.info("Local and remote are not the same repository.  Deleting local repository information - will check out clean (preserving local files).");
188 					deleteRecursive(svnFolder.toPath());
189 				}
190 			}
191 			
192 			//we might have deleted it above, so this isn't a simple else case.
193 			if (!svnFolder.isDirectory())
194 			{
195 				log.debug("Root folder does not contain a .svn subfolder.  Will checkout from remote.");
196 				long rev = getSvn(true).getUpdateClient().doCheckout(SVNURL.parseURIEncoded(SVNEncodingUtil.autoURIEncode(remoteAddress)),
197 						localFolder_, SVNRevision.HEAD, SVNRevision.HEAD, SVNDepth.INFINITY, true);
198 				log.info("Checkout out revision " + rev);
199 			}
200 			else
201 			{
202 				//Just do an update
203 				try
204 				{
205 					updateFromRemote(username, password, MergeFailOption.KEEP_LOCAL);
206 				}
207 				catch (MergeFailure e)
208 				{
209 					// merge failure should be impossible with the settings above.  We have already cleared any pre-existing local conflicts above, with a 
210 					//metadata delete, and keep_local prevents conflicts on update.
211 					throw new IOException("Should be impossible", e);
212 				}
213 			}
214 			
215 			List<String> modifiedFiles = makeInitialFilesAsNecessary(localFolder_);
216 
217 			ArrayList<File> filesToCommit = new ArrayList<>();
218 			for (String s : modifiedFiles)
219 			{
220 				File f = new File(localFolder_, s);
221 				if (f.isFile())
222 				{
223 					addFiles(f.getName());
224 					filesToCommit.add(f);
225 				}
226 			}
227 			
228 			//See if ignore is set up property for 'lastUser.txt'
229 			SVNPropertyData prop = getSvn().getWCClient().doGetProperty(localFolder_, "svn:ignore", SVNRevision.HEAD, SVNRevision.HEAD);
230 			if (prop == null || prop.getValue() == null || prop.getValue().getString() == null || prop.getValue().getString().length() == 0)
231 			{
232 				log.debug("Configuring initial ignore settings");
233 				getSvn().getWCClient().doSetProperty(localFolder_, "svn:ignore", SVNPropertyValue.create("lastUser.txt"), false, SVNDepth.EMPTY,
234 						null, null);
235 				filesToCommit.add(localFolder_);
236 			}
237 			
238 			if (filesToCommit.size() > 0)
239 			{
240 				log.debug("Committing initial repository files");
241 				getSvn().getCommitClient().doCommit(filesToCommit.toArray(new File[filesToCommit.size()]), false, "Adding initial repository files", 
242 						null, null, false, true, SVNDepth.EMPTY);
243 			}
244 			
245 			log.info("Status after init: " + statusToString());
246 			
247 		}
248 		catch (SVNAuthenticationException e) 
249 		{
250 			log.info("Auth fail", e);
251 			throw new AuthenticationException("Auth fail");
252 		}
253 		catch (SVNException e)
254 		{
255 			log.error("Unexpected", e);
256 			throw new IOException("Internal error", e);
257 		}
258 	}

346 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#addUntrackedFiles(java.io.File)
347 	 */
348 	@Override
349 	public void addUntrackedFiles() throws IllegalArgumentException, IOException
350 	{
351 		log.info("Add Untracked files called");
352 		try
353 		{
354 			HashSet<String> result = new HashSet<>();
355 			HashSet<String> conflicts = new HashSet<>();
356 			getSvn().getStatusClient().doStatus(localFolder_, SVNRevision.BASE, SVNDepth.INFINITY, false, false, false, false, new ISVNStatusHandler()
357 			{
358 
359 				@Override
360 				public void handleStatus(SVNStatus status) throws SVNException
361 				{
362 					if (status.getNodeStatus() == SVNStatusType.STATUS_UNVERSIONED)
363 					{
364 						result.add(getPathRelativeToRoot(status.getFile()));
365 					}
366 					else if (status.getNodeStatus() == SVNStatusType.STATUS_CONFLICTED)
367 					{
368 						conflicts.add(getPathRelativeToRoot(status.getFile()));
369 					}
370 				}
371 			}, new ArrayList<String>());
372 
373 			//don't try to add conflict marker files
374 			for (String s : conflicts)
375 			{
376 				Iterator<String> toAdd = result.iterator();
377 				while (toAdd.hasNext())
378 				{
379 					String item = toAdd.next();
380 					if (item.equals(s + ".mine") || item.startsWith(s + ".r"))
381 					{
382 						toAdd.remove();
383 					}
384 				}
385 			}
386 
387 			addFiles(result.toArray(new String[result.size()]));
388 		}
389 		catch (SVNException e)
390 		{
391 			log.error("Unexpected", e);
392 			throw new IOException("Internal error", e);
393 		}
394 	}

396 	/**
397 	 * @throws MergeFailure
398 	 * @throws AuthenticationException
399 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#updateCommitAndPush(java.io.File, java.lang.String, java.lang.String, java.lang.String,
400 	 * java.lang.String[])
401 	 */
402 	@Override
403 	public Set<String> updateCommitAndPush(String commitMessage, String username, String password, MergeFailOption mergeFailOption, String... fileNamesToCommit)
404 			throws IllegalArgumentException, IOException, MergeFailure, AuthenticationException
405 	{
406 		log.info("Commit Files called {}", (fileNamesToCommit == null ? "-null-" : Arrays.toString(fileNamesToCommit)));
407 		setupCredentials(username, password);
408 		try
409 		{
410 			Set<String> conflicting = getFilesInMergeConflict();
411 			if (conflicting.size() > 0)
412 			{
413 				log.info("Previous merge failure not yet resolved");
414 				throw new MergeFailure(conflicting, new HashSet<>());
415 			}
416 
417 			if (fileNamesToCommit == null)
418 			{
419 				fileNamesToCommit = getLocallyModifiedFiles().toArray(new String[0]);
420 				log.info("Will commit the uncommitted files {}", Arrays.toString(fileNamesToCommit));
421 			}
422 
423 			if (commitMessage == null || commitMessage.trim().length() == 0 && fileNamesToCommit.length > 0)
424 			{
425 				throw new IllegalArgumentException("The commit message is required when files are specified");
426 			}
427 
428 			//perform update
429 			Set<String> updatedFiles = updateFromRemote(username, password, mergeFailOption);
430 
431 			if (fileNamesToCommit.length > 0)
432 			{
433 				File[] filesToCommit = new File[fileNamesToCommit.length];
434 				for (int i = 0; i < fileNamesToCommit.length; i++)
435 				{
436 					filesToCommit[i] = new File(localFolder_, fileNamesToCommit[i]);
437 				}
438 
439 				SVNCommitInfo result = getSvn().getCommitClient().doCommit(filesToCommit, false, commitMessage, null, null, false, false, SVNDepth.EMPTY);
440 
441 				if (result.getErrorMessage() != null)
442 				{
443 					//I don't know if/when this would actually ever happen - seems most things result in a SVNException being thrown.
444 					log.error("Error during commit! " + result.getErrorMessage().toString());
445 					throw new IOException("Failed to commit.  Try updating first.");
446 				}
447 
448 				log.debug("Commit completed - now at " + result.getNewRevision());
449 			}
450 
451 			log.info("commit and push complete.  Current status: " + statusToString());
452 			return updatedFiles;
453 		}
454 
455 		catch (SVNAuthenticationException e) 
456 		{
457 			log.info("Auth fail", e);
458 			throw new AuthenticationException("Auth fail");
459 		}
460 		catch (SVNException e)
461 		{
462 			if (e.getErrorMessage() != null && SVNErrorCode.WC_NOT_UP_TO_DATE == e.getErrorMessage().getErrorCode())
463 			{
464 				log.error("Local checkout not up to date", e);
465 				throw new IOException("Local copy out of date with server.  Please update again");
466 			}
467 			log.error("Unexpected", e);
468 			throw new IOException("Internal error", e);
469 		}
470 	}

472 	/**
473 	 * @throws MergeFailure
474 	 * @throws AuthenticationException
475 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#updateFromRemote(java.io.File, java.lang.String, java.lang.String,
476 	 * gov.va.isaac.interfaces.sync.MergeFailOption)
477 	 */
478 	@Override
479 	public Set<String> updateFromRemote(String username, String password, MergeFailOption mergeFailOption) throws IllegalArgumentException, IOException, MergeFailure,
480 			AuthenticationException
481 	{
482 		log.info("update from remote called ");
483 		setupCredentials(username, password);
484 		try
485 		{
486 			log.debug("Fetching from remote");
487 			final Set<String> filesChangedDuringPull = new HashSet<>();
488 			Set<String> mergeConflicts = getFilesInMergeConflict();
489 			if (mergeConflicts.size() > 0)
490 			{
491 				log.info("Previous merge failure not yet resolved");
492 				throw new MergeFailure(mergeConflicts, new HashSet<>());
493 			}
494 
495 			SVNUpdateClient client = getSvn().getUpdateClient();
496 			client.setEventHandler(new ISVNEventHandler()
497 			{
498 
499 				@Override
500 				public void checkCancelled() throws SVNCancelException
501 				{
502 					// noop
503 				}
504 
505 				@Override
506 				public void handleEvent(SVNEvent event, double progress) throws SVNException
507 				{
508 					//dont care about directory updates
509 					if (!event.getFile().isDirectory())
510 					{
511 						if (event.getAction() == SVNEventAction.UPDATE_UPDATE || event.getAction() == SVNEventAction.UPDATE_ADD
512 								|| event.getAction() == SVNEventAction.UPDATE_DELETE || event.getAction() == SVNEventAction.UPDATE_REPLACE)
513 						{
514 							filesChangedDuringPull.add(getPathRelativeToRoot(event.getFile()));
515 						}
516 						else
517 						{
518 							log.error("Unhandeled update action! {}", event.getAction());
519 						}
520 					}
521 				}
522 			});
523 
524 			log.debug("Running Update");
525 			client.doUpdate(localFolder_, SVNRevision.HEAD, SVNDepth.INFINITY, false, false);
526 
527 			log.info("After update:" + statusToString());
528 
529 			Set<String> conflicted = getFilesInMergeConflict();
530 
531 			if (conflicted.size() > 0)
532 			{
533 				if (mergeFailOption == null || MergeFailOption.FAIL == mergeFailOption)
534 				{
535 					throw new MergeFailure(conflicted, filesChangedDuringPull);
536 				}
537 				else if (MergeFailOption.KEEP_LOCAL == mergeFailOption || MergeFailOption.KEEP_REMOTE == mergeFailOption)
538 				{
539 					HashMap<String, MergeFailOption> resolutions = new HashMap<>();
540 					for (String s : conflicted)
541 					{
542 						resolutions.put(s, mergeFailOption);
543 					}
544 					log.debug("Resolving merge failures with option {}", mergeFailOption);
545 					filesChangedDuringPull.addAll(resolveMergeFailures(resolutions));
546 				}
547 				else
548 				{
549 					throw new IllegalArgumentException("Unexpected option");
550 				}
551 			}
552 
553 			log.info("Files changed during updateFromRemote: {}", filesChangedDuringPull);
554 			return filesChangedDuringPull;
555 		}
556 
557 		catch (SVNAuthenticationException e) 
558 		{
559 			log.info("Auth fail", e);
560 			throw new AuthenticationException("Auth fail");
561 		}
562 		catch (SVNException e)
563 		{
564 			log.error("Unexpected", e);
565 			throw new IOException("Internal error", e);
566 		}
567 	}

569 	/**
570 	 * @throws MergeFailure
571 	 * @throws NoWorkTreeException
572 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#resolveMergeFailures(java.io.File, java.util.Map)
573 	 */
574 	@Override
575 	public Set<String> resolveMergeFailures(Map<String, MergeFailOption> resolutions) throws IllegalArgumentException, IOException, MergeFailure
576 	{
577 		log.info("resolve merge failures called - resolutions: {}", resolutions);
578 		try
579 		{
580 			Set<String> conflicting = getFilesInMergeConflict();
581 			if (conflicting.size() == 0)
582 			{
583 				throw new IllegalArgumentException("You do not appear to have any conflicting files");
584 			}
585 			if (conflicting.size() != resolutions.size())
586 			{
587 				throw new IllegalArgumentException("You must provide a resolution for each conflicting file.  Files in conflict: " + conflicting);
588 			}
589 			for (String s : conflicting)
590 			{
591 				if (!resolutions.containsKey(s))
592 				{
593 					throw new IllegalArgumentException("No conflit resolution specified for file " + s + ".  Resolutions must be specified for all files");
594 				}
595 			}
596 
597 			ArrayList<File> filesToCommit = new ArrayList<>();
598 
599 			for (Entry<String, MergeFailOption> r : resolutions.entrySet())
600 			{
601 				if (MergeFailOption.FAIL == r.getValue())
602 				{
603 					throw new IllegalArgumentException("MergeFailOption.FAIL is not a valid option");
604 				}
605 				else if (MergeFailOption.KEEP_LOCAL == r.getValue())
606 				{
607 					log.debug("Keeping our local file for conflict {}", r.getKey());
608 					getSvn().getWCClient().doResolve(new File(localFolder_, r.getKey()), SVNDepth.EMPTY, SVNConflictChoice.MINE_FULL);
609 				}
610 				else if (MergeFailOption.KEEP_REMOTE == r.getValue())
611 				{
612 					log.debug("Keeping remote file for conflict {}", r.getKey());
613 					getSvn().getWCClient().doResolve(new File(localFolder_, r.getKey()), SVNDepth.EMPTY, SVNConflictChoice.THEIRS_FULL);
614 				}
615 				else
616 				{
617 					throw new IllegalArgumentException("MergeFailOption is required");
618 				}
619 
620 				filesToCommit.add(new File(localFolder_, r.getKey()));
621 			}
622 
623 			log.info("resolve merge failures Complete.  Current status: " + statusToString());
624 
625 			return resolutions.keySet();
626 		}
627 		catch (SVNException e)
628 		{
629 			log.error("Unexpected", e);
630 			throw new IOException("Internal error", e);
631 		}
632 	}

670 		getSvn().getStatusClient().doStatus(localFolder_, SVNRevision.BASE, SVNDepth.INFINITY, false, true, true, false, new ISVNStatusHandler()

681 				}
682 				String path = status.getRepositoryRelativePath();
683 				if (path == null || path.isEmpty())
684 				{

686 					if (path.isEmpty())
687 					{

693 		}, new ArrayList<String>());
694 
695 		for (Entry<SVNStatusType, ArrayList<String>> x : result.entrySet())
696 		{
697 			sb.append(x.getKey().toString() + " - " + x.getValue() + eol);

826 	private static void deleteRecursive(Path path) throws IOException
827 	{
828 		Files.walkFileTree(path, new SimpleFileVisitor<Path>()
829 		{
830 			@Override
831 			public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException
832 			{
833 				Files.delete(file);
834 				return FileVisitResult.CONTINUE;
835 			}
836 
837 			@Override
838 			public FileVisitResult visitFileFailed(Path file, IOException exc) throws IOException
839 			{
840 				// try to delete the file anyway, even if its attributes could not be read, since delete-only access is theoretically possible
841 				Files.delete(file);
842 				return FileVisitResult.CONTINUE;
843 			}
844 
845 			@Override
846 			public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException
847 			{
848 				if (exc == null)
849 				{
850 					Files.delete(dir);
851 					return FileVisitResult.CONTINUE;
852 				}
853 				else
854 				{
855 					// directory iteration failed; propagate exception
856 					throw exc;
857 				}
858 			}
859 		});
860 	}
