170 	/**
171 	 * @throws AuthenticationException 
172 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#linkAndFetchFromRemote(java.io.File, java.lang.String, java.lang.String, java.lang.String)
173 	 */
174 	@Override
175 	public void linkAndFetchFromRemote(String remoteAddress, String username, String password) throws IllegalArgumentException, IOException, AuthenticationException
176 	{
177 		log.info("linkAndFetchFromRemote called - folder: {}, remoteAddress: {}, username: {}", localFolder, remoteAddress, username);
178 		try
179 		{
180 			File gitFolder = new File(localFolder, ".git");
181 			Repository r = new FileRepository(gitFolder);
182 
183 			if (!gitFolder.isDirectory())
184 			{
185 				log.debug("Root folder does not contain a .git subfolder.  Creating new git repository.");
186 				r.create();
187 			}
188 
189 			relinkRemote(remoteAddress, username, password);
190 
191 			Git git = new Git(r);
192 
193 			CredentialsProvider cp = new SSHFriendlyUsernamePasswordCredsProvider(username, password);
194 
195 			log.debug("Fetching");
196 			FetchResult fr = git.fetch().setCheckFetchedObjects(true).setCredentialsProvider(cp).call();
197 			log.debug("Fetch messages: {}", fr.getMessages());
198 
199 			boolean remoteHasMaster = false;
200 			Collection<Ref> refs = git.lsRemote().setCredentialsProvider(cp).call();
201 			for (Ref ref : refs)
202 			{
203 				if ("refs/heads/master".equals(ref.getName()))
204 				{
205 					remoteHasMaster = true;
206 					log.debug("Remote already has 'heads/master'");
207 					break;
208 				}
209 			}
210 
211 			if (remoteHasMaster)
212 			{
213 				//we need to fetch and (maybe) merge - get onto origin/master.
214 
215 				log.debug("Fetching from remote");
216 				String fetchResult = git.fetch().setCredentialsProvider(cp).call().getMessages();
217 				log.debug("Fetch Result: {}", fetchResult);
218 
219 				log.debug("Resetting to origin/master");
220 				git.reset().setMode(ResetType.MIXED).setRef("origin/master").call();
221 
222 				//Get the files from master that we didn't have in our working folder
223 				log.debug("Checking out missing files from origin/master");
224 				for (String missing : git.status().call().getMissing())
225 				{
226 					log.debug("Checkout {}", missing);
227 					git.checkout().addPath(missing).call();
228 				}
229 
230 				for (String newFile : makeInitialFilesAsNecessary(localFolder))
231 				{
232 					log.debug("Adding and committing {}", newFile);
233 					git.add().addFilepattern(newFile).call();
234 					git.commit().setMessage("Adding " + newFile).setAuthor(username, "42").call();
235 
236 					for (PushResult pr : git.push().setCredentialsProvider(cp).call())
237 					{
238 						log.debug("Push Message: {}", pr.getMessages());
239 					}
240 				}
241 			}
242 			else
243 			{
244 				//just push
245 				//make sure we have something to push
246 				for (String newFile : makeInitialFilesAsNecessary(localFolder))
247 				{
248 					log.debug("Adding and committing {}", newFile);
249 					git.add().addFilepattern(newFile).call();
250 					git.commit().setMessage("Adding readme file").setAuthor(username, "42").call();
251 				}
252 
253 				log.debug("Pushing repository");
254 				for (PushResult pr : git.push().setCredentialsProvider(cp).call())
255 				{
256 					log.debug("Push Result: {}", pr.getMessages());
257 				}
258 			}
259 
260 			log.info("linkAndFetchFromRemote Complete.  Current status: " + statusToString(git.status().call()));
261 		}
262 		catch (TransportException te)
263 		{
264 			if (te.getMessage().contains("Auth fail"))
265 			{
266 				log.info("Auth fail", te);
267 				throw new AuthenticationException("Auth fail");
268 			}
269 			else
270 			{
271 				log.error("Unexpected", te);
272 				throw new IOException("Internal error", te);
273 			}
274 		}
275 		catch (GitAPIException e)
276 		{
277 			log.error("Unexpected", e);
278 			throw new IOException("Internal error", e);
279 		}
280 	}

286 	public void relinkRemote(String remoteAddress, String username, String password) throws IllegalArgumentException, IOException
287 	{
288 		log.debug("Configuring remote URL and fetch defaults to {}", remoteAddress);
289 		StoredConfig sc = getGit().getRepository().getConfig();
290 		sc.setString("remote", "origin", "url", remoteAddress);
291 		sc.setString("remote", "origin", "fetch", "+refs/heads/*:refs/remotes/origin/*");
292 		sc.save();
293 	}

381 	 * @throws MergeFailure 
382 	 * @throws AuthenticationException 
383 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#updateCommitAndPush(java.io.File, java.lang.String, java.lang.String, java.lang.String,
384 	 * java.lang.String[])
385 	 */
386 	@Override
387 	public Set<String> updateCommitAndPush(String commitMessage, String username, String password, MergeFailOption mergeFailOption, String... files)
388 			throws IllegalArgumentException, IOException, MergeFailure, AuthenticationException
389 	{
390 		try
391 		{
392 			log.info("Commit Files called {}", (files == null ? "-null-" : Arrays.toString(files)));
393 			Git git = getGit();
394 			
395 			if (git.status().call().getConflicting().size() > 0)
396 			{
397 				log.info("Previous merge failure not yet resolved");
398 				throw new MergeFailure(git.status().call().getConflicting(), new HashSet<>());
399 			}
400 			
401 			if (files == null)
402 			{
403 				files = git.status().call().getUncommittedChanges().toArray(new String[0]);
404 				log.info("Will commit the uncommitted files {}", Arrays.toString(files));
405 			}
406 			
407 			if (StringUtils.isEmptyOrNull(commitMessage) && files.length > 0)
408 			{
409 				throw new IllegalArgumentException("The commit message is required when files are specified");
410 			}
411 
412 			if (files.length > 0)
413 			{
414 				CommitCommand commit = git.commit();
415 				for (String file : files)
416 				{
417 					commit.setOnly(file);
418 				}
419 
420 				commit.setAuthor(username, "42");
421 				commit.setMessage(commitMessage);
422 				RevCommit rv = commit.call();
423 				log.debug("Local commit completed: " + rv.getFullMessage());
424 			}
425 
426 			//need to merge origin/master into master now, prior to push
427 			Set<String> result = updateFromRemote(username, password, mergeFailOption);
428 
429 			log.debug("Pushing");
430 			CredentialsProvider cp = new SSHFriendlyUsernamePasswordCredsProvider(username, password);
431 
432 			Iterable<PushResult> pr = git.push().setCredentialsProvider(cp).call();
433 			pr.forEach(new Consumer<PushResult>()
434 			{
435 				@Override
436 				public void accept(PushResult t)
437 				{
438 					log.debug("Push Result Messages: " + t.getMessages());
439 				}
440 			});
441 
442 			log.info("commit and push complete.  Current status: " + statusToString(git.status().call()));
443 			return result;
444 		}
445 		catch (TransportException te)
446 		{
447 			if (te.getMessage().contains("Auth fail"))
448 			{
449 				log.info("Auth fail", te);
450 				throw new AuthenticationException("Auth fail");
451 			}
452 			else
453 			{
454 				log.error("Unexpected", te);
455 				throw new IOException("Internal error", te);
456 			}
457 		}
458 		catch (GitAPIException e)
459 		{
460 			log.error("Unexpected", e);
461 			throw new IOException("Internal error", e);
462 		}
463 	}

465 	/**
466 	 * @throws MergeFailure
467 	 * @throws AuthenticationException 
468 	 * @see gov.va.isaac.interfaces.sync.ProfileSyncI#updateFromRemote(java.io.File, java.lang.String, java.lang.String,
469 	 * gov.va.isaac.interfaces.sync.MergeFailOption)
470 	 */
471 	@Override
472 	public Set<String> updateFromRemote(String username, String password, MergeFailOption mergeFailOption) throws IllegalArgumentException, IOException,
473 			MergeFailure, AuthenticationException
474 	{
475 		Set<String> filesChangedDuringPull;
476 		try
477 		{
478 			log.info("update from remote called ");
479 
480 			Git git = getGit();
481 			
482 			log.debug("Fetching from remote");
483 			
484 			if (git.status().call().getConflicting().size() > 0)
485 			{
486 				log.info("Previous merge failure not yet resolved");
487 				throw new MergeFailure(git.status().call().getConflicting(), new HashSet<>());
488 			}
489 			
490 			CredentialsProvider cp = new SSHFriendlyUsernamePasswordCredsProvider(username, password);
491 			log.debug("Fetch Message" + git.fetch().setCredentialsProvider(cp).call().getMessages());
492 			
493 			ObjectId masterIdBeforeMerge = git.getRepository().getRef("master").getObjectId();
494 			if (git.getRepository().getRef("refs/remotes/origin/master").getObjectId().getName().equals(masterIdBeforeMerge.getName()))
495 			{
496 				log.info("No changes to merge");
497 				return new HashSet<String>();
498 			}
499 
500 			RevCommit stash = null;
501 			if (git.status().call().getUncommittedChanges().size() > 0)
502 			{
503 				log.info("Stashing uncommitted changes");
504 				stash = git.stashCreate().call();
505 			}
506 
507 			{
508 				log.debug("Merging from remotes/origin/master");
509 				MergeResult mr = git.merge().include(git.getRepository().getRef("refs/remotes/origin/master")).call();
510 				AnyObjectId headAfterMergeID = mr.getNewHead();
511 				
512 				if (!mr.getMergeStatus().isSuccessful())
513 				{
514 					if (mergeFailOption == null || MergeFailOption.FAIL == mergeFailOption)
515 					{
516 						addNote(NOTE_FAILED_MERGE_HAPPENED_ON_REMOTE + (stash == null ? ":NO_STASH" : STASH_MARKER + stash.getName()), git);
517 						//We can use the status here - because we already stashed the stuff that they had uncommitted above.
518 						throw new MergeFailure(mr.getConflicts().keySet(), git.status().call().getUncommittedChanges());
519 					}
520 					else if (MergeFailOption.KEEP_LOCAL == mergeFailOption || MergeFailOption.KEEP_REMOTE == mergeFailOption)
521 					{
522 						HashMap<String, MergeFailOption> resolutions = new HashMap<>();
523 						for (String s : mr.getConflicts().keySet())
524 						{
525 							resolutions.put(s, mergeFailOption);
526 						}
527 						log.debug("Resolving merge failures with option {}", mergeFailOption);
528 						filesChangedDuringPull = resolveMergeFailures(MergeFailType.REMOTE_TO_LOCAL, (stash == null ? null : stash.getName()), resolutions);
529 					}
530 					else
531 					{
532 						throw new IllegalArgumentException("Unexpected option");
533 					}
534 				}
535 				else
536 				{
537 					//Conflict free merge - or perhaps, no merge at all.
538 					if (masterIdBeforeMerge.getName().equals(headAfterMergeID.getName()))
539 					{
540 						log.debug("Merge didn't result in a commit - no incoming changes");
541 						filesChangedDuringPull = new HashSet<>();
542 					}
543 					else
544 					{
545 						filesChangedDuringPull = listFilesChangedInCommit(git.getRepository(), masterIdBeforeMerge, headAfterMergeID);
546 					}
547 				}
548 			}
549 
550 			if (stash != null)
551 			{
552 				log.info("Replaying stash");
553 				try
554 				{
555 					git.stashApply().setStashRef(stash.getName()).call();
556 					log.debug("stash applied cleanly, dropping stash");
557 					git.stashDrop().call();
558 				}
559 				catch (StashApplyFailureException e)
560 				{
561 					log.debug("Stash failed to merge");
562 					if (mergeFailOption == null || MergeFailOption.FAIL == mergeFailOption)
563 					{
564 						addNote(NOTE_FAILED_MERGE_HAPPENED_ON_STASH, git);
565 						throw new MergeFailure(git.status().call().getConflicting(), filesChangedDuringPull);
566 					}
567 					
568 					else if (MergeFailOption.KEEP_LOCAL == mergeFailOption || MergeFailOption.KEEP_REMOTE == mergeFailOption)
569 					{
570 						HashMap<String, MergeFailOption> resolutions = new HashMap<>();
571 						for (String s : git.status().call().getConflicting())
572 						{
573 							resolutions.put(s, mergeFailOption);
574 						}
575 						log.debug("Resolving stash apply merge failures with option {}", mergeFailOption);
576 						resolveMergeFailures(MergeFailType.STASH_TO_LOCAL, null, resolutions);
577 						//When we auto resolve to KEEP_LOCAL - these files won't have really changed, even though we recorded a change above.
578 						for (Entry<String, MergeFailOption> r : resolutions.entrySet())
579 						{
580 							if (MergeFailOption.KEEP_LOCAL == r.getValue())
581 							{
582 								filesChangedDuringPull.remove(r.getKey());
583 							}
584 						}
585 					}
586 					else
587 					{
588 						throw new IllegalArgumentException("Unexpected option");
589 					}
590 				}
591 			}
592 			log.info("Files changed during updateFromRemote: {}", filesChangedDuringPull);
593 			return filesChangedDuringPull;
594 		}
595 		catch (CheckoutConflictException e)
596 		{
597 			log.error("Unexpected", e);
598 			throw new IOException("A local file exists (but is not yet added to source control) which conflicts with a file from the server."
599 					+ "  Either delete the local file, or call addFile(...) on the offending file prior to attempting to update from remote.", e);
600 		}
601 		catch (TransportException te)
602 		{
603 			if (te.getMessage().contains("Auth fail"))
604 			{
605 				log.info("Auth fail", te);
606 				throw new AuthenticationException("Auth fail");
607 			}
608 			else

689 	private Set<String> resolveMergeFailures(MergeFailType mergeFailType, String stashIDToApply, Map<String, MergeFailOption> resolutions) 
690 			throws IllegalArgumentException, IOException, MergeFailure
691 	{
692 		log.debug("resolve merge failures called - mergeFailType: {} stashIDToApply: {} resolutions: {}", mergeFailType, stashIDToApply, resolutions);
693 		try
694 		{
695 			Git git = getGit();
696 			
697 			//We unfortunately, must know the mergeFailType option, because the resolution mechanism here uses OURS and THEIRS - but the 
698 			//meaning of OURS and THEIRS reverse, depending on if you are recovering from a merge failure, or a stash apply failure.
699 			
700 			for (Entry<String, MergeFailOption> r : resolutions.entrySet())
701 			{
702 				if (MergeFailOption.FAIL == r.getValue())
703 				{
704 					throw new IllegalArgumentException("MergeFailOption.FAIL is not a valid option");
705 				}
706 				else if (MergeFailOption.KEEP_LOCAL == r.getValue())
707 				{
708 					log.debug("Keeping our local file for conflict {}", r.getKey());
709 					git.checkout().addPath(r.getKey()).setStage(MergeFailType.REMOTE_TO_LOCAL == mergeFailType ? Stage.OURS : Stage.THEIRS).call();
710 				}
711 				else if (MergeFailOption.KEEP_REMOTE == r.getValue())
712 				{
713 					log.debug("Keeping remote file for conflict {}", r.getKey());
714 					git.checkout().addPath(r.getKey()).setStage(MergeFailType.REMOTE_TO_LOCAL == mergeFailType ? Stage.THEIRS : Stage.OURS).call();
715 				}
716 				else
717 				{
718 					throw new IllegalArgumentException("MergeFailOption is required");
719 				}
720 				
721 				log.debug("calling add to mark merge resolved");
722 				git.add().addFilepattern(r.getKey()).call();
723 			}
724 			
725 			if (mergeFailType == MergeFailType.STASH_TO_LOCAL)
726 			{
727 				//clean up the stash
728 				log.debug("Dropping stash");
729 				git.stashDrop().call();
730 			}
731 			
732 			
733 			RevWalk walk = new RevWalk(git.getRepository());
734 			Ref head = git.getRepository().getRef("refs/heads/master");
735 			RevCommit commitWithPotentialNote = walk.parseCommit(head.getObjectId());
736 			
737 			log.info("resolve merge failures Complete.  Current status: " + statusToString(git.status().call()));
738 			
739 			RevCommit rc = git.commit().setMessage("Merging with user specified merge failure resolution for files " + resolutions.keySet()).call();
740 			
741 			git.notesRemove().setObjectId(commitWithPotentialNote).call();
742 			Set<String> filesChangedInCommit = listFilesChangedInCommit(git.getRepository(), commitWithPotentialNote.getId(), rc);
743 			
744 			//When we auto resolve to KEEP_REMOTE - these will have changed - make sure they are in the list.
745 			//seems like this shouldn't really be necessary - need to look into the listFilesChangedInCommit algorithm closer.
746 			//this might already be fixed by the rework on 11/12/14, but no time to validate at the moment. - doesn't do any harm.
747 			for (Entry<String, MergeFailOption> r : resolutions.entrySet())
748 			{
749 				if (MergeFailOption.KEEP_REMOTE == r.getValue())
750 				{
751 					filesChangedInCommit.add(r.getKey());
752 				}
753 				if (MergeFailOption.KEEP_LOCAL == r.getValue())
754 				{
755 					filesChangedInCommit.remove(r.getKey());
756 				}
757 			}
758 			
759 			if (!StringUtils.isEmptyOrNull(stashIDToApply))
760 			{
761 				log.info("Replaying stash identified in note");
762 				try
763 				{
764 					git.stashApply().setStashRef(stashIDToApply).call();
765 					log.debug("stash applied cleanly, dropping stash");
766 					git.stashDrop().call();
767 				}
768 				catch (StashApplyFailureException e)
769 				{
770 					log.debug("Stash failed to merge");
771 					addNote(NOTE_FAILED_MERGE_HAPPENED_ON_STASH, git);
772 					throw new MergeFailure(git.status().call().getConflicting(), filesChangedInCommit);
773 				}
774 			}
775 			
776 			return filesChangedInCommit;
777 		}
778 		catch (GitAPIException e)
779 		{
780 			log.error("Unexpected", e);
781 			throw new IOException("Internal error", e);
782 		}
783 	}

794 	private HashSet<String> listFilesChangedInCommit(Repository repository, AnyObjectId beforeID, AnyObjectId afterID) throws MissingObjectException, IncorrectObjectTypeException, IOException
795 	{
796 		log.info("calculating files changed in commit");
797 		HashSet<String> result = new HashSet<>();
798 		RevWalk rw = new RevWalk(repository);
799 		RevCommit commitBefore = rw.parseCommit(beforeID);
800 		RevCommit commitAfter = rw.parseCommit(afterID);
801 		DiffFormatter df = new DiffFormatter(DisabledOutputStream.INSTANCE);
802 		df.setRepository(repository);
803 		df.setDiffComparator(RawTextComparator.DEFAULT);
804 		df.setDetectRenames(true);
805 		List<DiffEntry> diffs = df.scan(commitBefore.getTree(), commitAfter.getTree());
806 		for (DiffEntry diff : diffs)
807 		{
808 			result.add(diff.getNewPath());
809 		}
810 		log.debug("Files changed between commits commit: {} and {} - {}", beforeID.getName(), afterID, result);
811 		return result;
812 	}
