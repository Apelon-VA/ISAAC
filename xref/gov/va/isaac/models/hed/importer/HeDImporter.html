103   /*
104    * (non-Javadoc)
105    * 
106    * @see gov.va.isaac.ie.ImportHandler#importModel(java.io.File)
107    */
108   @Override
109   public InformationModel importModel(File file) throws Exception {
110     LOG.info("Preparing to import CEM model from: " + file.getName());
111     LOG.info("Ending import of CEM model from: " + file.getName());
112     KnowledgeDocument document =
113         (KnowledgeDocument) utils
114             .getGraphForFile(file, KnowledgeDocument.class);
115     // Parse into HeD model.
116     HeDInformationModel infoModel = createInformationModel(document);
117 
118     // Save the information model
119     // if (service.exists(infoModel)) {
120     // throw new IOException("Model already imported.");
121     // }
122     InformationModelService service = getInformationModelService();
123     service.saveInformationModel(infoModel);
124     LOG.info("Ending import of HeD model from stream");
125     return infoModel;
126   }

133   @Override

135     KnowledgeDocument document =

139     // Parse into HeD model.
140     HeDInformationModel infoModel = createInformationModel(document);

142     // Save the information model
143     // if (service.exists(infoModel)) {
144     // throw new IOException("Model already imported.");
145     // }
146     InformationModelService service = getInformationModelService();
147     service.saveInformationModel(infoModel);
148     LOG.info("Ending import of HeD model from stream");
149     return infoModel;
150 
151   }

153   /**
154    * Creates the information model from the XML DOM {@link Node}.
155    *
156    * @param rootNode the root node
157    * @return the HeD information model
158    * @throws IOException
159    * @throws JAXBException
160    * @throws TransformerConfigurationException
161    * @throws UnsupportedEncodingException
162    * @throws ParserConfigurationException
163    * @throws TransformerException
164    */
165   private HeDInformationModel createInformationModel(KnowledgeDocument kd)
166     throws JAXBException, TransformerConfigurationException,
167     UnsupportedEncodingException {
168 
169     // Create the model
170     HeDInformationModel infoModel = new HeDInformationModel();
171     infoModel.setType(InformationModelType.HeD);
172 
173     String key =
174         kd.getMetadata().getIdentifiers().getIdentifier().get(0).getRoot();
175     LOG.debug("      key = " + key);
176     // Key may contain things indexer can't handle, URL encode
177     infoModel.setKey(URLEncoder.encode(key, StandardCharsets.UTF_8.name()));
178 
179     Identifiers identifiers = kd.getMetadata().getIdentifiers();
180     LOG.debug("      identifiers = " + identifiers);
181     infoModel.setIdentifiers(identifiers);
182 
183     String name = kd.getMetadata().getTitle().getValue();
184     LOG.debug("      name = " + name);
185     infoModel.setName(name);
186 
187     String artifactType =
188         kd.getMetadata().getArtifactType().getValue().toString();
189     LOG.debug("      artifactType = " + artifactType);
190     infoModel.setArtifactType(artifactType);
191 
192     String schemaIdentifier = kd.getMetadata().getSchemaIdentifier().getRoot();
193     LOG.debug("      schemaIdentifier = " + schemaIdentifier);
194     infoModel.setSchemaIdentifier(schemaIdentifier);
195 
196     TemplateIds templateIds = kd.getMetadata().getTemplateIds();
197     LOG.debug("      templateIds = " + templateIds);
198     infoModel.setTemplateIds(templateIds);
199 
200     DataModels dataModels = kd.getMetadata().getDataModels();
201     LOG.debug("      dataModels.ct = " + dataModels.getModelReference().size());
202     infoModel.setDataModels(dataModels);
203 
204     Libraries libraries = kd.getMetadata().getLibraries();
205     LOG.debug("      libraries = " + libraries);
206     infoModel.setLibraries(libraries);
207 
208     if (kd.getMetadata().getDescription() != null) {
209       String description = kd.getMetadata().getDescription().getValue();
210       LOG.debug("      description = " + description);
211       infoModel.setDescription(description);
212     }
213 
214     List<String> keyTerms = new ArrayList<>();
215     if (kd.getMetadata().getKeyTerms() != null) {
216       for (CD keyTerm : kd.getMetadata().getKeyTerms().getTerm()) {
217         keyTerms.add(keyTerm.getOriginalText().getValue());
218       }
219       LOG.debug("      keyTerms.ct = " + keyTerms.size());
220       if (keyTerms.size() > 0) {
221         infoModel.setKeyTerms(keyTerms);
222       }
223     }
224 
225     String status = kd.getMetadata().getStatus().getValue().toString();
226     LOG.debug("      status = " + status);
227     infoModel.setStatus(status);
228 
229     InlineResource documentation = kd.getMetadata().getDocumentation();
230     LOG.debug("      documentation = " + documentation);
231     infoModel.setDocumentation(documentation);
232 
233     RelatedResources relatedResources = kd.getMetadata().getRelatedResources();
234     LOG.debug("      relatedResources = " + relatedResources);
235     infoModel.setRelatedResources(relatedResources);
236 
237     SupportingEvidence supportingEvidence =
238         kd.getMetadata().getSupportingEvidence();
239     LOG.debug("      supportingEvidence = " + supportingEvidence);
240     infoModel.setSupportingEvidence(supportingEvidence);
241 
242     Applicability applicability = kd.getMetadata().getApplicability();
243     LOG.debug("      applicability = " + applicability);
244     infoModel.setApplicability(applicability);
245 
246     Categories categories = kd.getMetadata().getCategories();
247     LOG.debug("      categories = " + categories);
248     infoModel.setCategories(categories);
249 
250     EventHistory eventHistory = kd.getMetadata().getEventHistory();
251     LOG.debug("      event history = " + eventHistory);
252     infoModel.setEventHistory(eventHistory);
253 
254     Contributions contributions = kd.getMetadata().getContributions();
255     LOG.debug("      contributions = " + contributions);
256     infoModel.setContributions(contributions);
257 
258     Publishers publishers = kd.getMetadata().getPublishers();
259     LOG.debug("      publishers = " + publishers);
260     infoModel.setPublishers(publishers);
261 
262     UsageTerms usageTerms = kd.getMetadata().getUsageTerms();
263     LOG.debug("      usageTerms = " + usageTerms);
264     infoModel.setUsageTerms(usageTerms);
265 
266     // Content
267 
268     ExternalData externalData = kd.getExternalData();
269     LOG.debug("      externalData = " + externalData);
270     infoModel.setExternalData(externalData);
271 
272     Conditions conditions = kd.getConditions();
273     LOG.debug("      conditions = " + conditions);
274     infoModel.setConditions(conditions);
275 
276     ActionGroup actionGroup = kd.getActionGroup();
277     LOG.debug("      actionGroup = " + actionGroup);
278     infoModel.setActionGroup(actionGroup);
279 
280     handleEnumerations(externalData);
281     handleEnumerations(conditions);
282     handleEnumerations(actionGroup);
283     
284     return infoModel;
285   }

287 	private void handleEnumerations(ActionGroup actionGroup) {
288 		if (actionGroup != null) {
289 			for (ActionBase subElement : actionGroup.getSubElements().getSimpleActionOrActionGroupOrActionRef()) {
290 				handleEnumerations((subElement.getConditions()));
291 				
292 				if (subElement.getActors() != null) {
293 					for (Actor actor : subElement.getActors().getActor()) {
294 						handleEnumerations(actor.getActor());
295 					}
296 				}
297 				
298 				if (subElement instanceof ActionGroup) {
299 					handleEnumerations((ActionGroup)subElement);
300 				}
301 			}
302 		}		
303 	}
304 	
305 	private void handleEnumerations(Conditions conditions) {
306 		if (conditions != null) {
307 			for (Condition condition : conditions.getCondition()) {
308 				handleEnumerations(condition.getLogic());
309 			}
310 		}		
311 	}
312 	
313 	private void handleEnumerations(ExternalData externalData) {
314 		if (externalData != null && externalData.getDef() != null) {
315 			for (ExpressionDef expDef : externalData.getDef()) {
316 				handleEnumerations(expDef.getExpression());
317 			}
318 		}		
319 	}
320 
321 	private void handleEnumerations(Expression exp) {
322 		if (exp != null) {
323 			if (exp instanceof ClinicalRequest) {
324 				if (((ClinicalRequest)exp).getCodes() != null && 
325 					((ClinicalRequest)exp).getCodes() instanceof ValueSet) {
326 					// Have Value Set
327 					ValueSet vs = (ValueSet)((ClinicalRequest)exp).getCodes();
328 					
329 					// Find Values
330 					String auth = vs.getAuthority();
331 					String id = vs.getId();
332 					String vers = vs.getVersion();
333 
334 					// Create Name
335 					String value = auth + "'s " + id + " " + " v" + vers + " HeD Referenced Enumeration";
336 			
337 					// Only create if doesn't already exist
338 					if (!ExtendedAppContext.getDataStore().hasUuid(OTFUtility.getUuidForFsn(value, value))) {
339 						// Create Refex's Description
340 						StringBuilder enumerationDesc = new StringBuilder();
341 						enumerationDesc.append("Enumeration Sememe for " + auth + "'s " + id);
342 						if (vs.getDescription() != null) {
343 							enumerationDesc.append("\r\n" + vs.getDescription());
344 						}
345 						enumerationDesc.append("\r\n");					
346 						enumerationDesc.append("Based on version: " + vers);	
347 
348 						// create Refex
349 						try {
350 							// Create Enumeration
351 							AppContext.getRuntimeGlobals().disableAllCommitListeners();
352 							RefexDynamicUsageDescriptionBuilder.createNewRefexDynamicUsageDescriptionConcept(value, value, enumerationDesc.toString(), 
353 																											 new RefexDynamicColumnInfo[] {},
354 																											 InformationModels.HED_ENUMERATIONS.getUuids()[0], 
355 																											 false, null);
356 						} catch (IOException | ContradictionException | InvalidCAB
357 								| PropertyVetoException e) {
358 							LOG.error("Unable to create HED Enumeration for " + value);
359 						}
360 						finally
361 						{
362 							AppContext.getRuntimeGlobals().enableAllCommitListeners();
363 						}
364 					}
365 				}
366 			}
367 			
368 		}
