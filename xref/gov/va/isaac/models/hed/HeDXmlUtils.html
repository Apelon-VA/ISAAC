58    /** The transformer for DOM -> XML. */
59    private Transformer transformer;
60  
61    /**
62     * Instantiates an empty {@link HeDXmlUtils}.
63     *
64     * @throws TransformerConfigurationException the transformer configuration
65     *           exception
66     */
67    public HeDXmlUtils() throws TransformerConfigurationException {
68      TransformerFactory factory = TransformerFactory.newInstance();
69      transformer = factory.newTransformer();
70      // Indent output.
71      transformer.setOutputProperty(OutputKeys.INDENT, "yes");
72      transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount",
73          "4");
74      // Skip XML declaration header.
75      transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
76    }

78    /**
79     * Returns the graph for string.
80     *
81     * @param xml the xml
82     * @param graphClass the graph class
83     * @return the graph for string
84     * @throws JAXBException the JAXB exception
85     */
86    @SuppressWarnings("static-method")
87    public Object getGraphForString(String xml, Class<?> graphClass)
88      throws JAXBException {
89      JAXBContext context = JAXBContext.newInstance(graphClass);
90      Unmarshaller unmarshaller = context.createUnmarshaller();
91      return unmarshaller.unmarshal(new StreamSource(new StringReader(xml)));
92    }
93  
94    /**
95     * Returns the graph for file.
96     *
97     * @param file the file
98     * @param graphClass the graph class
99     * @return the graph for file
100    * @throws FileNotFoundException the file not found exception
101    * @throws JAXBException the JAXB exception
102    */
103   @SuppressWarnings("resource")
104   public Object getGraphForFile(File file, Class<?> graphClass)
105     throws FileNotFoundException, JAXBException {
106     return getGraphForString(new Scanner(file, "UTF-8").useDelimiter("\\A")
107         .next(), graphClass);
108   }

119   @SuppressWarnings("resource")
120   public Object getGraphForStream(InputStream in, Class<?> graphClass)

126   /**
127    * Returns the XML string for for graph object.
128    *
129    * @param object the object
130    * @return the string for for graph
131    * @throws JAXBException the JAXB exception
132    */
133   @SuppressWarnings("static-method")
134   public String getStringForGraph(Object object) throws JAXBException {
135     StringWriter writer = new StringWriter();
136     JAXBContext jaxbContext = null;
137     jaxbContext = JAXBContext.newInstance(object.getClass());
138     Marshaller jaxbMarshaller = jaxbContext.createMarshaller();
139     jaxbMarshaller.marshal(object, writer);
140     return writer.toString();
141   }

143   /**
144    * Returns the node for string.
145    *
146    * @param xml the xml
147    * @return the node for string
148    * @throws ParserConfigurationException the parser configuration exception
149    * @throws SAXException the SAX exception
150    * @throws IOException Signals that an I/O exception has occurred.
151    */
152   @SuppressWarnings("static-method")
153   public Node getNodeForString(String xml) throws ParserConfigurationException,
154     SAXException, IOException {
155 
156     InputStream in =
157         new ByteArrayInputStream(xml.getBytes(StandardCharsets.UTF_8));
158     // Parse XML file.
159     DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
160     DocumentBuilder db = dbf.newDocumentBuilder();
161     Document document = db.parse(in);
162     Node rootNode = document.getFirstChild();
163     return rootNode;
164   }
165 
166   /**
167    * Returns the node for file.
168    *
169    * @param file the file
170    * @return the node for file
171    * @throws ParserConfigurationException the parser configuration exception
172    * @throws SAXException the SAX exception
173    * @throws IOException Signals that an I/O exception has occurred.
174    */
175   @SuppressWarnings("static-method")
176   public Node getNodeForFile(File file) throws ParserConfigurationException,
177     SAXException, IOException {
178     InputStream in = new FileInputStream(file);
179     // Parse XML file.
180     DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
181     DocumentBuilder db = dbf.newDocumentBuilder();
182     Document document = db.parse(in);
183     Node rootNode = document.getFirstChild();
184     in.close();
185     return rootNode;
186   }

188   /**
189    * Returns the string for node.
190    *
191    * @param root the root node
192    * @return the string for node
193    * @throws TransformerException the transformer exception
194    * @throws ParserConfigurationException the parser configuration exception
195    */
196   public String getStringForNode(Node root) throws TransformerException,
197     ParserConfigurationException {
198     DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
199     DocumentBuilder builder = factory.newDocumentBuilder();
200     Document document = builder.newDocument();
201     document.appendChild(document.importNode(root, true));
202     DOMSource source = new DOMSource(document);
203     StringWriter out = new StringWriter();
204     StreamResult result = new StreamResult(out);
205     transformer.transform(source, result);
206     return out.toString();
207   }

240   /**
241    * Pretty format.
242    *
243    * @param input the input
244    * @param indent the indent
245    * @return the string
246    */
247   @SuppressWarnings("static-method")
248   public String prettyFormat(String input, int indent) {
249     try {
250       Source xmlInput = new StreamSource(new StringReader(input));
251       StringWriter stringWriter = new StringWriter();
252       StreamResult xmlOutput = new StreamResult(stringWriter);
253       TransformerFactory transformerFactory = TransformerFactory.newInstance();
254       transformerFactory.setAttribute("indent-number", indent);
255       Transformer transformer = transformerFactory.newTransformer();
256       transformer.setOutputProperty(OutputKeys.INDENT, "yes");
257       transformer.transform(xmlInput, xmlOutput);
258       return xmlOutput.getWriter().toString();
259     } catch (Exception e) {
260       throw new RuntimeException(e); // simple exception handling, please review
261                                      // it
262     }
263   }
