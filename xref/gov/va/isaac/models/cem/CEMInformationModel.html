45    /**

53    /**
54     * Sets the data type.
55     *
56     * @param dataType the data type
57     */
58    public void setDataType(String dataType) {
59      // Remove any "data" properties
60      removePropertiesByLabel("data");
61  
62      // Add new one
63      InformationModelProperty dataTypeProperty =
64          new DefaultInformationModelProperty();
65      dataTypeProperty.setLabel("data");
66      dataTypeProperty.setType(dataType);
67      addProperty(dataTypeProperty);
68    }

98     *

102     InformationModelProperty property = getPropertyByLabel("data");
103     if (property == null)
104       return null;
105     return property.getType();
106   }

108   /**
109    * Adds the component.
110    *
111    * @param component the component
112    */
113   private void addComponent(CEMComponent component) {
114     InformationModelProperty property = new DefaultInformationModelProperty();
115     property.setLabel(component.getComponentType().getCemlTagName());
116     property.setType(component.getDateTypeRef());
117     property.setName(component.getName());
118     property.setValue(component.getValue() == null ? "" : component.getValue());
119     property.setCardinalityMin(component.getCardinalityMin());
120     property.setCardinalityMax(component.getCardinalityMax());
121     addProperty(property);
122   }
123

125    * Removes the component.

127    * @param component the component

132     property.setType(component.getDateTypeRef());

140   /**
141    * Returns the components.
142    *
143    * @param type the type
144    * @return the components
145    */
146   private Set<CEMComponent> getComponents(CEMComponentType type) {
147     Set<CEMComponent> components = new HashSet<>();
148     for (InformationModelProperty property : properties) {
149       if (property.getLabel().equals(type.getCemlTagName())) {
150         CEMComponent component = new CEMComponent();
151         component.setComponentType(type);
152         component.setDateTypeRef(property.getType());
153         component.setName(property.getName());
154         property.setValue(component.getValue() == null ? "" : component.getValue());
155         component.setCardinalityMin(property.getCardinalityMin());
156         component.setCardinalityMax(property.getCardinalityMax());
157         components.add(component);
158       }
159     }
160     return components;
161   }

294   /**
295    * Returns the constraints.
296    *
297    * @return the constraints
298    */
299   public Set<CEMConstraint> getConstraints() {
300     Set<CEMConstraint> constraints = new HashSet<>();
301     for (InformationModelProperty property : properties) {
302       if (property.getLabel().equals(CEMConstraint.getCemlTagName())) {
303         CEMConstraint constraint = new CEMConstraint();
304         constraint.setPath(property.getName());
305         constraint.setValue(property.getValue());
306         constraints.add(constraint);
307       }
308     }
309     return constraints;
310   }

312   /**
313    * Returns the constraints for component.
314    *
315    * @param component the component
316    * @return the constraints for component
317    */
318   public Set<CEMConstraint> getConstraintsForComponent(CEMComponent component) {
319     Set<CEMConstraint> constraints = new HashSet<>();
320     for (InformationModelProperty property : properties) {
321       // Check "constraint" properties
322       if (property.getLabel().equals(CEMConstraint.getCemlTagName())) {
323 
324         // e.g. "qual.methodDevice"
325         // e.g. "data.pq"
326         if (property.getName().startsWith(
327             component.getComponentType().getCemlTagName() + "."
328                 + component.getName() + ".")
329             ||
330 
331             property.getName().startsWith(
332                 component.getComponentType().getCemlTagName() + "."
333                     + component.getDateTypeRef() + ".")) {
334 
335           CEMConstraint constraint = new CEMConstraint();
336           constraint.setPath(property.getName());
337           constraint.setValue(property.getValue());
338           constraints.add(constraint);
339         }
340       }
341     }
342     return constraints;
