79    /**
80     * Export model.
81     *
82     * @param uuid the concept uuid
83     * @throws Exception the exception
84     */
85    public void exportModel(UUID uuid) throws Exception {
86      LOG.info("Starting export of CEM model");
87      LOG.debug("  UUID = " + uuid);
88  
89      InformationModelService service = getInformationModelService();
90  
91      // Parse into CEM model.
92      InformationModel model = service.getInformationModel(uuid);
93      // Abort if not available.
94      if (model == null) {
95        LOG.warn("No CEM model to export on " + uuid);
96        return;
97      }
98      CEMInformationModel infoModel = new CEMInformationModel(model);
99  
100     // Build a DOM tree in the style of CEM.
101     this.document = buildDom();
102     Element root = buildCemTree(infoModel);
103     document.appendChild(root);
104 
105     // Transform DOM tree into stream.
106     Transformer transformer = buildTransformer();
107     DOMSource source = new DOMSource(document);
108     StreamResult result = new StreamResult(outputStream);
109     transformer.transform(source, result);
110 
111     LOG.info("Ending export of CEM model");
112   }

124   /**
125    * Builds the cem tree.
126    *
127    * @param infoModel the info model
128    * @return the element
129    * @throws ValidationException the validation exception
130    * @throws IOException Signals that an I/O exception has occurred.
131    * @throws ContradictionException the contradiction exception
132    */
133   private Element buildCemTree(CEMInformationModel infoModel)
134     throws ValidationException, IOException, ContradictionException {
135     LOG.debug("  Build Cem Tree");
136     LOG.debug("    Handle " + CEML);
137     Element root = document.createElement(CEML);
138 
139     // CETYPE element.
140     Element cetype = buildCetypeElement(infoModel);
141     root.appendChild(cetype);
142 
143     return root;
144   }

146   /**
147    * The spec for this model is the DiastolicBloodPressureMeas.xml document in
148    * ISAAC/import-export/cem.
149    *
150    * @param infoModel the info model
151    * @return the element
152    * @throws ValidationException the validation exception
153    * @throws IOException Signals that an I/O exception has occurred.
154    * @throws ContradictionException the contradiction exception
155    */
156   private Element buildCetypeElement(CEMInformationModel infoModel)
157     throws ValidationException, IOException, ContradictionException {
158     LOG.debug("    Handle " + CEM);
159     Element cetype = document.createElement(CEM);
160 
161     // Name attribute (1).
162     String name = infoModel.getName();
163     LOG.debug("      name = " + name);
164     Attr nameAttr = buildNameAttr(name);
165     cetype.setAttributeNode(nameAttr);
166 
167     // Key element (0-1).
168     String key = infoModel.getKey();
169     LOG.debug("      key = " + key);
170     Element keyElement = buildKeyElement(key);
171     cetype.appendChild(keyElement);
172 
173     // Data element (0-1).
174     String dataType = infoModel.getDataType();
175     LOG.debug("      dataType = " + dataType);
176     if (dataType != null) {
177       Element dataElement = buildDataElement(dataType);
178       cetype.appendChild(dataElement);
179     }
180 
181     // Qual elements (0-M).
182     LOG.debug("      quals");
183     Set<CEMComponent> quals = infoModel.getQualifiers();
184     for (CEMComponent qual : quals) {
185       Element qualElement = buildCompositionElement(QUAL, qual);
186       cetype.appendChild(qualElement);
187       LOG.debug("        " + qual.getName());
188     }
189 
190     // Mod elements (0-M).
191     LOG.debug("      mods");
192     Set<CEMComponent> mods = infoModel.getModifiers();
193     for (CEMComponent mod : mods) {
194       Element modElement = buildCompositionElement(MOD, mod);
195       cetype.appendChild(modElement);
196       LOG.debug("        " + mod.getName());
197     }
198 
199     // Att elements (0-M).
200     LOG.debug("      atts");
201     Set<CEMComponent> atts = infoModel.getAttributions();
202     for (CEMComponent att : atts) {
203       Element attElement = buildCompositionElement(ATT, att);
204       cetype.appendChild(attElement);
205       LOG.debug("        " + att.getName());
206     }
207 
208     // Constraint elements (0-M).
209     LOG.debug("      constraints");
210     Set<CEMConstraint> constraints = infoModel.getConstraints();
211     for (CEMConstraint constraint : constraints) {
212       Element constraintElement = buildConstraintElement(constraint);
213       cetype.appendChild(constraintElement);
214       LOG.debug("        " + constraint.getPath());
215     }
216 
217     return cetype;
218   }

220   /**
221    * Builds the constraint element.
222    *
223    * @param constraint the constraint
224    * @return the element
225    */
226   private Element buildConstraintElement(CEMConstraint constraint) {
227     Element e = document.createElement(CONSTRAINT);
228 
229     // Path attribute (1).
230     Attr pathAttr = document.createAttribute(PATH);
231     String path = constraint.getPath();
232     if (path != null) {
233       pathAttr.setNodeValue(path);
234     }
235     e.setAttributeNode(pathAttr);
236 
237     // Value attribute (1).
238     Attr valueAttr = document.createAttribute(VALUE);
239     String value = constraint.getValue();
240     if (value != null) {
241       valueAttr.setNodeValue(value);
242     }
243     e.setAttributeNode(valueAttr);
244 
245     return e;
246   }

248   /**
249    * Builds the composition element.
250    *
251    * @param elementName the element name
252    * @param component the composition
253    * @return the element
254    */
255   private Element buildCompositionElement(String elementName,
256     CEMComponent component) {
257     Element e = document.createElement(elementName);
258 
259     // Type attribute.
260     Attr typeAttr = document.createAttribute(TYPE);
261     String type = component.getDateTypeRef();
262     typeAttr.setNodeValue(type);
263     e.setAttributeNode(typeAttr);
264 
265     // Name attribute.
266     Attr nameAttr = document.createAttribute(ID);
267     String name = component.getName();
268     nameAttr.setNodeValue(name);
269     e.setAttributeNode(nameAttr);
270 
271     // Constraint attribute (0-1).
272     if (!component.getCardinalityMin().isEmpty()) {
273       Attr constraintAttr = document.createAttribute("card");
274       constraintAttr.setNodeValue(component.getCardinalityMin() + "-"
275           + component.getCardinalityMax());
276       e.setAttributeNode(constraintAttr);
277     }
278 
279     // Value element - only really here for demo puropses
280     String value = component.getValue();
281     if (value != null) {
282       e.setTextContent(value);
283     }
284 
285     return e;
286   }

349   /**
350    * Builds the transformer.
351    *
352    * @return the transformer
353    * @throws Exception the exception
354    */
355   private Transformer buildTransformer() throws Exception {
356     TransformerFactory factory = TransformerFactory.newInstance();
357     Transformer transformer = factory.newTransformer();
358 
359     // Indent output.
360     transformer.setOutputProperty(OutputKeys.INDENT, "yes");
361     transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount",
362         "4");
363 
364     // Skip XML declaration header.
365     transformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION, "yes");
366 
367     return transformer;
368   }

370   /**
371    * Utility method to take a string and convert it to normal Java variable name
372    * capitalization. This normally means converting the first character from
373    * upper case to lower case, but in the (unusual) special case when there is
374    * more than one character and both the first and second characters are upper
375    * case, we leave it alone.
376    * 
377    * Thus "FooBah" becomes "fooBah" and "X" becomes "x", but "URL" stays as
378    * "URL".
379    * 
380    * Parameters
381    * @param name The string to be decapitalized. Returns:
382    * @return The decapitalized version of the string.
383    * 
384    *         Note, this was copied from 1.7_40 release of the JDK, as it was
385    *         removed from com.sun.xml.internal.ws.util.StringUtils in 1.8.
386    */
387 
388   public static String decapitalize(String name) {
389     if (name == null || name.length() == 0) {
390       return name;
391     }
392     if (name.length() > 1 && Character.isUpperCase(name.charAt(1))
393         && Character.isUpperCase(name.charAt(0))) {
394       return name;
395     }
396     char chars[] = name.toCharArray();
397     chars[0] = Character.toLowerCase(chars[0]);
398     return new String(chars);
399   }
