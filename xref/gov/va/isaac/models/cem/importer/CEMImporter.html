77    @Override
78    public InformationModel importModel(File file) throws Exception {
79      FileInputStream in = null;
80      try {
81        LOG.info("Preparing to import CEM model from: " + file.getName());
82        in = new FileInputStream(file);
83        LOG.info("Ending import of CEM model from: " + file.getName());
84        InformationModel model = importModel(in);
85        in.close();
86        return model;
87      } catch (Exception e) {
88        in.close();
89        throw e;
90      }
91    }

93    @Override
94    public InformationModel importModel(InputStream in) throws Exception {
95      LOG.info("Preparing to import CEM model from stream");
96      // Make sure in background thread.
97      FxUtils.checkBackgroundThread();
98  
99      // Obtain service
100     InformationModelService service = getInformationModelService();
101 
102     // Load DOM tree from file.
103     Node domRoot = loadModel(in);
104 
105     // Parse into CEM model.
106     CEMInformationModel infoModel = createInformationModel(domRoot);
107 
108     // TODO: this should be made dynamic within the app
109     // Sample connection to a terminology concept
110     // This is to retain functionality from before
111     UUID conceptUuid = UUID.fromString("215fd598-e21d-3e27-a0a2-8e23b1b36dfc");
112     infoModel.addAssociatedConceptUuid(conceptUuid);
113 
114     // Save the information model
115     // if (service.exists(infoModel)) {
116     // throw new IOException(
117     // "Model already imported");
118     // }
119     service.saveInformationModel(infoModel);
120     LOG.info("Ending import of CEM model from stream");
121     return infoModel;
122   }

124   /**
125    * Creates the information model from the XML DOM {@link Node}.
126    *
127    * @param rootNode the root node
128    * @return the CEM information model
129    * @throws IOException
130    */
131   private CEMInformationModel createInformationModel(Node rootNode)
132     throws IOException {
133 
134     // Look for CEM child node.
135     Node cemNode = null;
136     NodeList childNodes = rootNode.getChildNodes();
137     for (int nodeCount = 0; nodeCount < childNodes.getLength(); nodeCount++) {
138       Node childNode = childNodes.item(nodeCount);
139       LOG.debug("childNode : " + nodeCount + " - " + childNode.getNodeName());
140 
141       if (childNode.getNodeName().equals(CEM)) {
142         cemNode = childNode;
143         break;
144       }
145     }
146 
147     // Sanity check.
148     Preconditions.checkNotNull(cemNode, "No CEM child node in XML file!");
149     LOG.debug("cem: " + cemNode.getNodeName());
150 
151     // Create the model
152     CEMInformationModel infoModel = new CEMInformationModel();
153     infoModel.setType(InformationModelType.CEM);
154 
155     // Parse CEM node attributes.
156     String name = cemNode.getAttributes().getNamedItem(NAME).getTextContent();
157     infoModel.setName(name);
158     LOG.info("name: " + name);
159 
160     // Iterate through CEM node children and process.
161     NodeList cemChildren = cemNode.getChildNodes();
162     for (int nodeCount = 0; nodeCount < cemChildren.getLength(); nodeCount++) {
163       Node loopNode = cemChildren.item(nodeCount);
164       LOG.debug("loopNode : " + nodeCount + " - " + loopNode.getNodeName());
165 
166       switch (loopNode.getNodeName()) {
167         case KEY:
168           String key =
169               loopNode.getAttributes().getNamedItem(CODE).getTextContent();
170           LOG.info("key: " + key);
171           infoModel.setKey(key);
172           break;
173 
174         case DATA:
175           String data =
176               loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
177           LOG.info("data: " + data);
178           infoModel.setDataType(data);
179           break;
180 
181         case QUAL:
182           String qualName =
183               loopNode.getAttributes().getNamedItem(ID).getTextContent();
184           String qualType =
185               loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
186           String qualCard =
187               loopNode.getAttributes().getNamedItem(CARD).getTextContent();
188           String qualCompValue = loopNode.getTextContent();
189           LOG.info(String.format("mod: %s %s %s %s", qualName, qualType,
190               qualCard, qualCompValue));
191           CEMComponent qual = new CEMComponent();
192           qual.setComponentType(CEMComponentType.QUALIFIER);
193           qual.setDateTypeRef(qualType);
194           qual.setName(qualName);
195           if (qualCard.length() == 3) {
196             qual.setCardinalityMin(qualCard.substring(0, 1));
197             qual.setCardinalityMax(qualCard.substring(2, 3));
198           } else if (qualCard.length() == 0) {
199             // do nothing
200           } else {
201             throw new IOException("Unable to handle cardinality value: "
202                 + qualCard);
203           }
204           infoModel.addQualifier(qual);
205           break;
206 
207         case MOD:
208           String modName =
209               loopNode.getAttributes().getNamedItem(ID).getTextContent();
210           String modType =
211               loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
212           String modCard =
213               loopNode.getAttributes().getNamedItem(CARD).getTextContent();
214           String modCompValue = loopNode.getTextContent();
215           LOG.info(String.format("mod: %s %s %s %s", modName, modType, modCard,
216               modCompValue));
217           CEMComponent mod = new CEMComponent();
218           mod.setComponentType(CEMComponentType.MODIFIER);
219           mod.setDateTypeRef(modType);
220           mod.setName(modName);
221           if (modCard.length() == 3) {
222             mod.setCardinalityMin(modCard.substring(0, 1));
223             mod.setCardinalityMax(modCard.substring(2, 3));
224           } else if (modCard.length() == 0) {
225             // do nothing
226           } else {
227             throw new IOException("Unable to handle cardinality value: "
228                 + modCard);
229           }
230           infoModel.addModifier(mod);
231           break;
232 
233         case ATT:
234           String attName =
235               loopNode.getAttributes().getNamedItem(ID).getTextContent();
236           String attType =
237               loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
238           String attCard =
239               loopNode.getAttributes().getNamedItem(CARD).getTextContent();
240           String attCompValue = loopNode.getTextContent();
241           LOG.info(String.format("att: %s %s %s %s", attName, attType, attCard,
242               attCompValue));
243 
244           CEMComponent att = new CEMComponent();
245           att.setComponentType(CEMComponentType.ATTRIBUTION);
246           att.setDateTypeRef(attType);
247           att.setName(attName);
248           if (attCard.length() == 3) {
249             att.setCardinalityMin(attCard.substring(0, 1));
250             att.setCardinalityMax(attCard.substring(2, 3));
251           } else if (attCard.length() == 0) {
252             // do nothing
253           } else {
254             throw new IOException("Unable to handle cardinality value: "
255                 + attCard);
256           }
257           infoModel.addAttribution(att);
258           break;
259 
260         case CONSTRAINT:
261           String path =
262               loopNode.getAttributes().getNamedItem(PATH).getTextContent();
263           String value =
264               loopNode.getAttributes().getNamedItem(VALUE).getTextContent();
265           LOG.info(String.format("constraint: %s %s", path, value));
266 
267           CEMConstraint constraint = new CEMConstraint();
268           constraint.setPath(path);
269           constraint.setValue(value);
270           constraint.handleConstraintEnumerationRefex();
271           infoModel.addConstraint(constraint);
272           break;
273 
274         default:
275 
276           break;
277       }
278     }
279 
280     LOG.info("CHECK FOR INFO");
281     // Handle the definition
282     childNodes = rootNode.getChildNodes();
283     for (int nodeCount = 0; nodeCount < childNodes.getLength(); nodeCount++) {
284       Node childNode = childNodes.item(nodeCount);
285       if (childNode.getNodeName().equals(INFO)) {
286         LOG.info("FOUND info");
287         NodeList infoChildren = childNode.getChildNodes();
288         for (int chdNodeCount = 0; chdNodeCount < infoChildren.getLength(); chdNodeCount++) {
289           Node defNode = infoChildren.item(chdNodeCount);
290           if (defNode.getNodeName().equals(DEFINITION)) {
291             String definition = defNode.getTextContent();
292             LOG.info("definition - " + definition);
293             infoModel.setDefinition(definition);
294           }
295         }
296         break;
297       }
298     }
299 
300     return infoModel;
301   }

303   /**
304    * Load model from XML file into a {@link Node}.
305    *
306    * @param file the file
307    * @return the node
308    * @throws ParserConfigurationException the parser configuration exception
309    * @throws SAXException the SAX exception
310    * @throws IOException Signals that an I/O exception has occurred.
311    */
312   private Node loadModel(InputStream in) throws ParserConfigurationException,
313     SAXException, IOException {
314     // Parse XML file.
315     DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
316     DocumentBuilder db = dbf.newDocumentBuilder();
317     Document document = db.parse(in);
318 
319     Node rootNode = document.getFirstChild();
320     LOG.info("rootNode: " + rootNode.getNodeName());
321 
322     // Sanity check.
323     Preconditions.checkState(rootNode.getNodeName().toLowerCase().equals(CEML),
324         "No CEML root node");
325 
326     return rootNode;
327   }
