64      @SuppressWarnings("unused")
65      public ConceptChronicleBI importModel(File file) throws Exception {
66          LOG.info("Preparing to import CEM model from: " + file.getName());
67  
68          // Make sure in background thread.
69          FxUtils.checkBackgroundThread();
70  
71          // Parse XML file.
72          DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
73          DocumentBuilder db = dbf.newDocumentBuilder();
74          Document document = db.parse(file);
75  
76          Node rootNode = document.getFirstChild();
77          LOG.info("rootNode: " + rootNode.getNodeName());
78  
79          // Sanity check.
80          Preconditions.checkState(rootNode.getNodeName().toLowerCase().equals(CEML),
81                  "No CEML root node in XML file! " + file.getName());
82  
83          LOG.info("File OK: " + file.getName());
84  
85          // Look for CETYPE child node.
86          Node cetypeNode = null;
87          NodeList childNodes = rootNode.getChildNodes();
88          for (int nodeCount = 0; nodeCount < childNodes.getLength(); nodeCount++) {
89              Node childNode = childNodes.item(nodeCount);
90              LOG.debug("childNode : " + nodeCount + " - " + childNode.getNodeName());
91  
92              if (childNode.getNodeName().equals(CETYPE)) {
93                  cetypeNode = childNode;
94                  break;
95              }
96          }
97  
98          // Sanity check.
99          Preconditions.checkNotNull(cetypeNode, "No CETYPE child node in XML file! " + file.getName());
100 
101         LOG.info("cetype: " + cetypeNode.getNodeName());
102 
103         // Get focus concept.
104         String focusConceptUuid = "215fd598-e21d-3e27-a0a2-8e23b1b36dfc";
105         ConceptChronicleBI focusConcept = getDataStore().getConcept(UUID.fromString(focusConceptUuid));
106         LOG.info("focusConcept: " + focusConcept.toString());
107 
108         // Throw exception if import already performed.
109         ComponentVersionBI latestVersion = focusConcept.getVersion(getVC());
110         Collection<? extends RefexChronicleBI<?>> annotations = latestVersion.getAnnotations();
111         for (RefexChronicleBI<?> annotation : annotations) {
112             Preconditions.checkState(annotation.getAssemblageNid() != CEMMetadataBinding.CEM_TYPE_REFSET.getNid(),
113                     "CEM import has already been performed on " + focusConceptUuid);
114         }
115 
116         // Parse CETYPE node attributes.
117         String type = cetypeNode.getAttributes().getNamedItem(NAME).getTextContent();
118         addRefexInStrExtensionRefset(focusConcept, CEMMetadataBinding.CEM_TYPE_REFSET, type);
119         LOG.info("type: " + type);
120 
121         // Iterate through CETYPE node children and process.
122         NodeList cetypeChildren = cetypeNode.getChildNodes();
123         for (int nodeCount = 0; nodeCount < cetypeChildren.getLength(); nodeCount++) {
124             Node loopNode = cetypeChildren.item(nodeCount);
125             LOG.debug("loopNode : " + nodeCount + " - " + loopNode.getNodeName());
126 
127             switch (loopNode.getNodeName()) {
128             case KEY:
129                 String key = loopNode.getAttributes().getNamedItem(CODE).getTextContent();
130                 addRefexInStrExtensionRefset(focusConcept, CEMMetadataBinding.CEM_KEY_REFSET, key);
131                 LOG.info("key: " + key);
132                 break;
133             case DATA:
134                 String data = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
135                 LOG.info("data: " + data);
136                 switch (data.toUpperCase()) {
137                 case PQ:
138                     addMemberInDataRefset(focusConcept, CEMMetadataBinding.CEM_PQ);
139                     break;
140                 case CD:
141                     addMemberInDataRefset(focusConcept, CEMMetadataBinding.CEM_CD);
142                     break;
143                 case CWE:  // Another way of specifying CEM CD data type.
144                     addMemberInDataRefset(focusConcept, CEMMetadataBinding.CEM_CD);
145                     break;
146                     //TODO: add the rest of data types
147                 }
148                 break;
149             case QUAL:
150                 String qualName = loopNode.getAttributes().getNamedItem(NAME).getTextContent();
151                 String qualType = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
152                 String qualCard = loopNode.getAttributes().getNamedItem(CARD).getTextContent();
153                 RefexChronicleBI qual = addMemberInCompositionRefset(focusConcept, CEMMetadataBinding.CEM_QUAL, qualType);
154 
155                 // Simulate String-String refset for CEM Constraints.
156                 RefexChronicleBI qualConstraint = addMemberInConstraintsRefset(qual);
157                 RefexChronicleBI qualPath = addRefexInStrExtensionRefset(qualConstraint,
158                         CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, CARD);
159                 RefexChronicleBI qualValue = addRefexInStrExtensionRefset(qualConstraint,
160                         CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, qualCard);
161 
162                 String qualCompValue = loopNode.getTextContent();
163                 if (qualCompValue != null && qualCompValue.length() > 0) {
164                     RefexChronicleBI qualCompRefex = addMemberInValueRefset(qual, qualCompValue);
165                 }
166 
167                 LOG.info(String.format("mod: %s %s %s %s", qualName, qualType, qualCard, qualCompValue));
168                 break;
169             case MOD:
170                 String modName = loopNode.getAttributes().getNamedItem(NAME).getTextContent();
171                 String modType = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
172                 String modCard = loopNode.getAttributes().getNamedItem(CARD).getTextContent();
173                 RefexChronicleBI mod = addMemberInCompositionRefset(focusConcept, CEMMetadataBinding.CEM_MOD, modType);
174 
175                 // Simulate String-String refset for CEM Constraints.
176                 RefexChronicleBI modConstraint = addMemberInConstraintsRefset(mod);
177                 RefexChronicleBI modPath = addRefexInStrExtensionRefset(modConstraint,
178                         CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, CARD);
179                 RefexChronicleBI modValue = addRefexInStrExtensionRefset(modConstraint,
180                         CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, modCard);
181 
182                 String modCompValue = loopNode.getTextContent();
183                 if (modCompValue != null && modCompValue.length() > 0) {
184                     RefexChronicleBI modCompRefex = addMemberInValueRefset(mod, modCompValue);
185                 }
186 
187                 LOG.info(String.format("mod: %s %s %s %s", modName, modType, modCard, modCompValue));
188                 break;
189             case ATT:
190                 String attName = loopNode.getAttributes().getNamedItem(NAME).getTextContent();
191                 String attType = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
192                 String attCard = loopNode.getAttributes().getNamedItem(CARD).getTextContent();
193                 RefexChronicleBI att = addMemberInCompositionRefset(focusConcept, CEMMetadataBinding.CEM_ATTR, attType);
194 
195                 // Simulate String-String refset for CEM Constraints.
196                 RefexChronicleBI attConstraint = addMemberInConstraintsRefset(att);
197                 RefexChronicleBI attPath = addRefexInStrExtensionRefset(attConstraint,
198                         CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, CARD);
199                 RefexChronicleBI attValue = addRefexInStrExtensionRefset(attConstraint,
200                         CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, attCard);
201 
202                 String attCompValue = loopNode.getTextContent();
203                 if (attCompValue != null && attCompValue.length() > 0) {
204                     RefexChronicleBI attCompRefex = addMemberInValueRefset(att, attCompValue);
205                 }
206 
207                 LOG.info(String.format("att: %s %s %s %s", attName, attType, attCard, attCompValue));
208                 break;
209             case CONSTRAINT:
210                 String path = loopNode.getAttributes().getNamedItem(PATH).getTextContent();
211                 String value = loopNode.getAttributes().getNamedItem(VALUE).getTextContent();
212 
213                 // Simulate String-String refset for CEM Constraints.
214                 RefexChronicleBI constraint = addMemberInConstraintsRefset(focusConcept);
215                 RefexChronicleBI constraintPath = addRefexInStrExtensionRefset(constraint,
216                         CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, path);
217                 RefexChronicleBI constraintValue = addRefexInStrExtensionRefset(constraint,
218                         CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, value);
219 
220                 LOG.info(String.format("constraint: %s %s", path, value));
221                 break;
222             }
223         }
224 
225         getDataStore().addUncommitted(focusConcept);
226         getDataStore().commit();
227 
228         LOG.debug("Long form after commit:" + focusConcept.toLongString());
229         LOG.info("Ending import of CEM model from: " + file.getName());
230 
231         return focusConcept;
232     }

234     private RefexChronicleBI addMemberInValueRefset(RefexChronicleBI focusComponent, String value)
235             throws IOException, InvalidCAB, ContradictionException {
236         return addRefexInStrExtensionRefset(focusComponent, CEMMetadataBinding.CEM_VALUE_REFSET,
237                 value);
238     }
239 
240     private RefexChronicleBI addMemberInDataRefset(ConceptChronicleBI focusConcept,
241             ConceptSpec conceptExtension)
242             throws IOException, InvalidCAB, ContradictionException {
243         return addRefexInCidExtensionRefset(focusConcept, CEMMetadataBinding.CEM_DATA_REFSET,
244                 conceptExtension);
245     }
246 
247     private RefexChronicleBI addMemberInCompositionRefset(ConceptChronicleBI focusConcept,
248             ConceptSpec componentExtension, String stringExtension)
249             throws IOException, InvalidCAB, ContradictionException {
250         return addRefexInCidStrExtensionRefset(focusConcept, CEMMetadataBinding.CEM_COMPOSITION_REFSET,
251                 componentExtension, stringExtension);
252     }
253 
254     private RefexChronicleBI addMemberInConstraintsRefset(ComponentChronicleBI focusComponent)
255             throws IOException, InvalidCAB, ContradictionException {
256         return addRefexInMemberRefset(focusComponent, CEMMetadataBinding.CEM_CONSTRAINTS_REFSET);
257     }
