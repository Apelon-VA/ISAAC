70      @Override
71      public ConceptChronicleBI importModel(File file) throws Exception {
72          LOG.info("Preparing to import CEM model from: " + file.getName());
73  
74          // Make sure in background thread.
75          FxUtils.checkBackgroundThread();
76  
77          // Get focus concept.
78          String focusConceptUuid = "215fd598-e21d-3e27-a0a2-8e23b1b36dfc";
79          ConceptChronicleBI focusConcept = getDataStore().getConcept(UUID.fromString(focusConceptUuid));
80          LOG.info("focusConcept: " + focusConcept.toString());
81  
82          // Throw exception if import already performed.
83          ComponentVersionBI latestVersion = focusConcept.getVersion(getVC());
84          Collection<? extends RefexChronicleBI<?>> annotations = latestVersion.getAnnotations();
85          for (RefexChronicleBI<?> annotation : annotations) {
86              Preconditions.checkState(annotation.getAssemblageNid() != CEMMetadataBinding.CEM_TYPE_REFSET.getNid(),
87                      "CEM import has already been performed on " + focusConceptUuid);
88          }
89  
90          // Load DOM tree from file.
91          Node domRoot = loadModel(file);
92  
93          // Parse into CEM model.
94          CEMInformationModel infoModel = createInformationModel(domRoot);
95  
96          // Annotate focusConcept with Refset members.
97          annotateWithRefsets(focusConcept, infoModel);
98  
99          getDataStore().addUncommitted(focusConcept);
100         getDataStore().commit();
101 
102         LOG.debug("Long form after commit:" + focusConcept.toLongString());
103         LOG.info("Ending import of CEM model from: " + file.getName());
104 
105         return focusConcept;
106     }

108     @SuppressWarnings("unused")
109     private void annotateWithRefsets(ConceptChronicleBI focusConcept,
110             CEMInformationModel infoModel)
111             throws IOException, InvalidCAB, ContradictionException {
112 
113         // CEM Type refset.
114         String type = infoModel.getName();
115         addMemberInTypeRefset(focusConcept, type);
116 
117         // CEM Key refset.
118         String key = infoModel.getKey();
119         addMemberInKeyRefset(focusConcept, key);
120 
121         // CEM Data refset.
122         ConceptSpec dataType = infoModel.getDataType();
123         addMemberInDataRefset(focusConcept, dataType);
124 
125         // CEM Composition refset : Quals.
126         List<Composition> quals = infoModel.getQualComponents();
127         for (Composition qual : quals) {
128             String component = qual.getComponent();
129             RefexChronicleBI qualRefex = addMemberInCompositionRefset(focusConcept, CEMMetadataBinding.CEM_QUAL, component);
130 
131             // Constraint?
132             Constraint constraint = qual.getConstraint();
133             if (constraint !=  null) {
134                 String path = constraint.getPath();
135                 String value = constraint.getValue();
136 
137                 // Simulate String-String refset for CEM Constraints.
138                 RefexChronicleBI constraintRefex = addMemberInConstraintsRefset(qualRefex);
139                 RefexChronicleBI pathRefex = addRefexInStrExtensionRefset(constraintRefex,
140                         CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, path);
141                 RefexChronicleBI qalueRefex = addRefexInStrExtensionRefset(constraintRefex,
142                         CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, value);
143             }
144 
145             // Value?
146             String value = qual.getValue();
147             if (value != null) {
148                 RefexChronicleBI qualCompRefex = addMemberInValueRefset(qualRefex, value);
149             }
150         }
151 
152         // CEM Composition refset : Mods.
153         List<Composition> mods = infoModel.getModComponents();
154         for (Composition mod : mods) {
155             String component = mod.getComponent();
156             RefexChronicleBI modRefex = addMemberInCompositionRefset(focusConcept, CEMMetadataBinding.CEM_MOD, component);
157 
158             // Constraint?
159             Constraint constraint = mod.getConstraint();
160             if (constraint !=  null) {
161                 String path = constraint.getPath();
162                 String value = constraint.getValue();
163 
164                 // Simulate String-String refset for CEM Constraints.
165                 RefexChronicleBI constraintRefex = addMemberInConstraintsRefset(modRefex);
166                 RefexChronicleBI pathRefex = addRefexInStrExtensionRefset(constraintRefex,
167                         CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, path);
168                 RefexChronicleBI qalueRefex = addRefexInStrExtensionRefset(constraintRefex,
169                         CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, value);
170             }
171 
172             // Value?
173             String value = mod.getValue();
174             if (value != null) {
175                 RefexChronicleBI modCompRefex = addMemberInValueRefset(modRefex, value);
176             }
177         }
178 
179         // CEM Composition refset : Atts.
180         List<Composition> atts = infoModel.getAttComponents();
181         for (Composition att : atts) {
182             String component = att.getComponent();
183             RefexChronicleBI attRefex = addMemberInCompositionRefset(focusConcept, CEMMetadataBinding.CEM_ATT, component);
184 
185             // Constraint?
186             Constraint constraint = att.getConstraint();
187             if (constraint !=  null) {
188                 String path = constraint.getPath();
189                 String value = constraint.getValue();
190 
191                 // Simulate String-String refset for CEM Constraints.
192                 RefexChronicleBI constraintRefex = addMemberInConstraintsRefset(attRefex);
193                 RefexChronicleBI pathRefex = addRefexInStrExtensionRefset(constraintRefex,
194                         CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, path);
195                 RefexChronicleBI qalueRefex = addRefexInStrExtensionRefset(constraintRefex,
196                         CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, value);
197             }
198 
199             // Value?
200             String value = att.getValue();
201             if (value != null) {
202                 RefexChronicleBI attCompRefex = addMemberInValueRefset(attRefex, value);
203             }
204         }
205 
206         // CEM Constraints refset.
207         List<Constraint> constraints = infoModel.getConstraints();
208         for (Constraint constraint : constraints) {
209             String path = constraint.getPath();
210             String value = constraint.getValue();
211 
212             // Simulate String-String refset for CEM Constraints.
213             RefexChronicleBI constraintRefex = addMemberInConstraintsRefset(focusConcept);
214             RefexChronicleBI constraintPathRefex = addRefexInStrExtensionRefset(constraintRefex,
215                     CEMMetadataBinding.CEM_CONSTRAINTS_PATH_REFSET, path);
216             RefexChronicleBI constraintValueRefex = addRefexInStrExtensionRefset(constraintRefex,
217                     CEMMetadataBinding.CEM_CONSTRAINTS_VALUE_REFSET, value);
218         }
219     }

221     private CEMInformationModel createInformationModel(Node rootNode) {
222 
223         // Look for CETYPE child node.
224         Node cetypeNode = null;
225         NodeList childNodes = rootNode.getChildNodes();
226         for (int nodeCount = 0; nodeCount < childNodes.getLength(); nodeCount++) {
227             Node childNode = childNodes.item(nodeCount);
228             LOG.debug("childNode : " + nodeCount + " - " + childNode.getNodeName());
229 
230             if (childNode.getNodeName().equals(CETYPE)) {
231                 cetypeNode = childNode;
232                 break;
233             }
234         }
235 
236         // Sanity check.
237         Preconditions.checkNotNull(cetypeNode, "No CETYPE child node in XML file!");
238         LOG.debug("cetype: " + cetypeNode.getNodeName());
239 
240         // Parse CETYPE node attributes.
241         String name = cetypeNode.getAttributes().getNamedItem(NAME).getTextContent();
242         LOG.info("name: " + name);
243 
244         CEMInformationModel infoModel = new CEMInformationModel(name);
245 
246         // Iterate through CETYPE node children and process.
247         NodeList cetypeChildren = cetypeNode.getChildNodes();
248         for (int nodeCount = 0; nodeCount < cetypeChildren.getLength(); nodeCount++) {
249             Node loopNode = cetypeChildren.item(nodeCount);
250             LOG.debug("loopNode : " + nodeCount + " - " + loopNode.getNodeName());
251 
252             switch (loopNode.getNodeName()) {
253             case KEY:
254                 String key = loopNode.getAttributes().getNamedItem(CODE).getTextContent();
255                 LOG.info("key: " + key);
256                 infoModel.setKey(key);
257                 break;
258             case DATA:
259                 String data = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
260                 LOG.info("data: " + data);
261                 switch (data.toUpperCase()) {
262                 case PQ:
263                     infoModel.setDataType(CEMMetadataBinding.CEM_PQ);
264                     break;
265                 case CD:
266                     infoModel.setDataType(CEMMetadataBinding.CEM_CD);
267                     break;
268                 case CWE:  // Another way of specifying CEM CD data type.
269                     infoModel.setDataType(CEMMetadataBinding.CEM_CD);
270                     break;
271                     //TODO: add the rest of data types
272                 }
273                 break;
274             case QUAL:
275                 String qualName = loopNode.getAttributes().getNamedItem(NAME).getTextContent();
276                 String qualType = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
277                 String qualCard = loopNode.getAttributes().getNamedItem(CARD).getTextContent();
278                 String qualCompValue = loopNode.getTextContent();
279                 LOG.info(String.format("mod: %s %s %s %s", qualName, qualType, qualCard, qualCompValue));
280 
281                 Composition qual = infoModel.addQualComponent(qualType);
282 
283                 // Cardinality constraint.
284                 if (qualCard != null && qualCard.length() > 0) {
285                     Constraint constraint = new Constraint(CARD, qualCard);
286                     qual.setConstraint(constraint);
287                 }
288 
289                 // Hack for Stan Huff demo.
290                 if (qualCompValue != null && qualCompValue.length() > 0) {
291                     qual.setValue(qualCompValue);
292                 }
293 
294                 break;
295             case MOD:
296                 String modName = loopNode.getAttributes().getNamedItem(NAME).getTextContent();
297                 String modType = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
298                 String modCard = loopNode.getAttributes().getNamedItem(CARD).getTextContent();
299                 String modCompValue = loopNode.getTextContent();
300                 LOG.info(String.format("mod: %s %s %s %s", modName, modType, modCard, modCompValue));
301 
302                 Composition mod = infoModel.addModComponent(modType);
303 
304                 // Cardinality constraint.
305                 if (modCard != null && modCard.length() > 0) {
306                     Constraint constraint = new Constraint(CARD, modCard);
307                     mod.setConstraint(constraint);
308                 }
309 
310                 // Hack for Stan Huff demo.
311                 if (modCompValue != null && modCompValue.length() > 0) {
312                    mod.setValue(modCompValue);
313                 }
314 
315                 break;
316             case ATT:
317                 String attName = loopNode.getAttributes().getNamedItem(NAME).getTextContent();
318                 String attType = loopNode.getAttributes().getNamedItem(TYPE).getTextContent();
319                 String attCard = loopNode.getAttributes().getNamedItem(CARD).getTextContent();
320                 String attCompValue = loopNode.getTextContent();
321                 LOG.info(String.format("att: %s %s %s %s", attName, attType, attCard, attCompValue));
322 
323                 Composition att = infoModel.addAttComponent(attType);
324 
325                 // Cardinality constraint.
326                 if (attCard != null && attCard.length() > 0) {
327                     Constraint constraint = new Constraint(CARD, attCard);
328                     att.setConstraint(constraint);
329                 }
330 
331                 // Hack for Stan Huff demo.
332                if (attCompValue != null && attCompValue.length() > 0) {
333                     att.setValue(attCompValue);
334                 }
335 
336                 break;
337             case CONSTRAINT:
338                 String path = loopNode.getAttributes().getNamedItem(PATH).getTextContent();
339                 String value = loopNode.getAttributes().getNamedItem(VALUE).getTextContent();
340                 LOG.info(String.format("constraint: %s %s", path, value));
341 
342                 Constraint constraint = new Constraint(path, value);
343                 infoModel.addConstraint(constraint);
344 
345                 break;
346             }
347         }
348 
349         return infoModel;
350     }

391     private Node loadModel(File file) throws ParserConfigurationException, SAXException, IOException {
392 
393         // Parse XML file.
394         DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
395         DocumentBuilder db = dbf.newDocumentBuilder();
396         Document document = db.parse(file);
397 
398         Node rootNode = document.getFirstChild();
399         LOG.info("rootNode: " + rootNode.getNodeName());
400 
401         // Sanity check.
402         Preconditions.checkState(rootNode.getNodeName().toLowerCase().equals(CEML),
403                 "No CEML root node in XML file! " + file.getName());
404 
405         LOG.info("File OK: " + file.getName());
406 
407         return rootNode;
408     }
