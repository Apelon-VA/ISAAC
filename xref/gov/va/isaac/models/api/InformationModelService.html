34  /**
35   * Generically represents a service for interacting with information models.
36   *
37   * @author bcarlsenca
38   */
39  public interface InformationModelService {
40  
41    /**
42     * Indicates whether or not the information model exists.
43     *
44     * @param model the model
45     * @return true, if successful
46     * @throws IOException if anything goes wrong
47     */
48    public boolean exists(InformationModel model) throws IOException;
49  
50    /**
51     * Returns the information model for the specified uuid. Returns null if there
52     * is no corresponding information model.
53     *
54     * @param uuid the uuid
55     * @return the information model
56     * @throws IOException if anything goes wrong
57     * @throws ContradictionException if anything goes wrong
58     */
59    public InformationModel getInformationModel(UUID uuid) throws IOException,
60      ContradictionException;
61  
62    /**
63     * Returns the information model for the specified model-specific key. Returns
64     * null if there is no corresponding information model.
65     * 
66     * For example, this takes an information model type, like
67     * InformationModelType.CEM, and an identifier or unique name in that type,
68     * like DiastolicBloodPressure_KEY_ECID, and returns the corresponding model
69     * object. This is a way of finding a model when you do not know its ISAAC
70     * UUID.
71     * 
72     * In the background, this is implemented by using the lucene index.
73     * @param type the type
74     * @param key the key
75     * @return the information model
76     * @throws IOException if anything goes wrong
77     * @throws ContradictionException if anything goes wrong
78     */
79    public InformationModel getInformationModel(InformationModelType type,
80      String key) throws IOException, ContradictionException;
81  
82    /**
83     * Returns the information model children.
84     *
85     * @param model the model
86     * @return the information model children
87     * @throws ContradictionException 
88     * @throws IOException 
89     * @throws ValidationException 
90     */
91    public Set<InformationModel> getInformationModelChildren(InformationModel model) throws ValidationException, IOException, ContradictionException;
92      
93  
94    /**
95     * Adds or saves changes to the information model.
96     *
97     * @param model the model
98     * @throws IOException if anything goes wrong
99     * @throws ContradictionException if anything goes wrong
100    * @throws InvalidCAB if anything goes wrong
101    * @throws NoSuchAlgorithmException if anything goes wrong
102    * @throws PropertyVetoException 
103    */
104   public void saveInformationModel(InformationModel model) throws IOException,
105     InvalidCAB, ContradictionException, NoSuchAlgorithmException, PropertyVetoException;
106 
107   /**
108    * Creates the metadata concepts.
109    *
110    * @throws IOException Signals that an I/O exception has occurred.
111    * @throws InvalidCAB the invalid cab
112    * @throws ContradictionException the contradiction exception
113    * @throws PropertyVetoException the property veto exception
114    */
115   void createMetadataConcepts() throws IOException, InvalidCAB,
116     ContradictionException, PropertyVetoException;
117 }
