109    * gov.va.isaac.models.api.InformationModelService#exists(gov.va.isaac.models
110    * .InformationModel)

112   @Override
113   public boolean exists(InformationModel model) throws IOException {
114     LOG.info("Check whether information model exists: "
115         + model.getType().getDisplayName() + ", " + model.getKey());
116 
117     if (model.getType() == null) {
118       throw new IOException("Model unexpectedly has null type");
119     }
120     if (model.getKey() == null) {
121       throw new IOException("Model unexpectedly has null key");
122     }
123     try {
124       boolean flag =
125           getInformationModel(model.getType(), model.getKey()) != null;
126       LOG.debug("  " + flag);
127       return flag;
128     } catch (ContradictionException e) {
129       e.printStackTrace();
130       throw new IOException(
131           "Unexpected inability to acquire UUID for type and key, "
132               + model.getType() + ", " + model.getKey());
133     }
134   }

156   /*
157    * (non-Javadoc)
158    * 
159    * @see
160    * gov.va.isaac.models.api.InformationModelService#getInformationModel(java
161    * .lang.String, java.lang.String)
162    */
163   @Override
164   public InformationModel getInformationModel(InformationModelType type,
165     String key) throws IOException, ContradictionException {
166     if (type == null) {
167       throw new IllegalArgumentException("Unexpected null type parameter.");
168     }
169     LOG.info("Get information model: " + type.getDisplayName() + ", " + key);
170     if (key == null) {
171       throw new IllegalArgumentException("Unexpected null key parameter.");
172     }
173     try {
174       LOG.debug("  Lucene Search: '" + key + "'");
175       LuceneDescriptionIndexer descriptionIndexer =
176           AppContext.getService(LuceneDescriptionIndexer.class);
177       if (descriptionIndexer == null) {
178         throw new IOException("No description indexer found, aborting.");
179       }
180       // Look for description matches.
181       ComponentProperty field = ComponentProperty.DESCRIPTION_TEXT;
182       int limit = 10;
183       List<SearchResult> searchResults;
184       searchResults =
185           descriptionIndexer.query(key, false, field, limit, Long.MIN_VALUE);
186 
187       // Results are descriptions, need to look up concepts
188       for (SearchResult result : searchResults) {
189         int conceptNid =
190             dataStore.getComponent(result.getNid()).getConceptNid();
191         ConceptVersionBI conceptVersion =
192             OTFUtility.getConceptVersion(conceptNid);
193         LOG.debug("    Check " + conceptVersion.getPrimordialUuid() + ", "
194             + conceptVersion.getPreferredDescription().getText());
195 
196         if (conceptVersion.getPreferredDescription().getText().equals(key)
197             || conceptVersion.getFullySpecifiedDescription().getText()
198                 .equals(key)) {
199           LOG.debug("    Found match, compute information model type");
200           InformationModelType computedType =
201               computeInformationModelType(conceptVersion);
202           if (computedType == null) {
203             LOG.debug("      type = null");
204             continue;
205           } else {
206             LOG.debug("      type = " + computedType.getDisplayName());
207           }
208           if (type == computedType) {
209             InformationModel model =
210                 getInformationModel(conceptVersion.getPrimordialUuid());
211             LOG.debug("  Model found, UUID = " + model.getUuid());
212             return model;
213           }
214         }
215       }
216       LOG.debug("  Model not found");
217       return null;
218     } catch (ParseException e) {
219       throw new IOException("Unable to parse key for lucene search, " + key, e);
220     }
221 
222   }

233     InformationModel model) throws ValidationException, IOException,
234     ContradictionException {
235     ConceptVersionBI modelConcept =
236         OTFUtility.getConceptVersion(model.getUuid());
237     Set<InformationModel> models = new HashSet<>();
238     for (RelationshipChronicleBI rel : modelConcept.getRelationshipsIncoming()) {
239       RelationshipVersionBI<?> relVersion =
240           rel.getVersion(OTFUtility.getViewCoordinate());
241       // Look for matching typeId and "active" flag
242       if (relVersion.getTypeNid() == Snomed.IS_A.getLenient().getNid()
243           && relVersion.isActive()) {
244         // Add the model
245         models.add(getInformationModel(OTFUtility.getConceptVersion(
246             relVersion.getDestinationNid()).getPrimordialUuid()));
247       }
248     }
249     return models;
250   }

279   /**
280    * Concept to information model.
281    *
282    * @param modelConcept the model concept
283    * @return the information model
284    * @throws ContradictionException
285    * @throws IOException
286    */
287   private InformationModel conceptToInformationModel(
288     ConceptChronicleBI modelConcept) throws IOException, ContradictionException {
289     if (modelConcept == null) {
290       throw new IOException("Model concept unexpectedly null");
291     }
292 
293     LOG.info("Convert concept to information model: "
294         + modelConcept.getPrimordialUuid());
295 
296     // Key is the FN
297     String key = OTFUtility.getFullySpecifiedName(modelConcept);
298     LOG.debug("  key = " + key);
299     // Name is the PT
300     String name = OTFUtility.getConPrefTerm(modelConcept.getNid());
301     LOG.debug("  name = " + name);
302     // UUID of the concept
303     UUID uuid = modelConcept.getPrimordialUuid();
304     LOG.debug("  uuid = " + uuid);
305     // information model type - obtain by walking up the tree
306     InformationModelType type = computeInformationModelType(modelConcept);
307     LOG.debug("  type = " + type.getDisplayName());
308     InformationModel model = new DefaultInformationModel(key, name, uuid, type);
309 
310     // Assemble Metadata
311     // Need to look at stamp nid for concept attributes instead of concept
312     int stampNid =
313         modelConcept.getConceptAttributes()
314             .getVersion(OTFUtility.getViewCoordinate()).getStamp();
315     InformationModelMetadata metadata =
316         InformationModelMetadata.newInstance(stampNid, dataStore,
317             OTFUtility.getViewCoordinate());
318     model.setMetadata(metadata);
319 
320     // Check super-model
321     // Build associated concept UUIDs from relationships
322     for (RelationshipChronicleBI rel : modelConcept.getRelationshipsOutgoing()) {
323       RelationshipVersionBI<?> relVersion =
324           rel.getVersion(OTFUtility.getViewCoordinate());
325       // Look for matching typeId and "active" flag
326       if (relVersion.getTypeNid() == Snomed.IS_A.getLenient().getNid()
327           && relVersion.isActive()) {
328         UUID superModelUuid =
329             OTFUtility.getConceptVersion(relVersion.getDestinationNid())
330                 .getPrimordialUuid();
331         // Avoid assigning uuid if parent is the info model type concept
332         if (!superModelUuid.equals(model.getType().getUuid())) {
333           // set the super model uuid unless already set
334           if (model.getSuperModelUuid() != null) {
335             throw new IOException(
336                 "Unexpected multiple lineage in information model concept: "
337                     + model.getUuid());
338           } else {
339             model.setSuperModelUuid(superModelUuid);
340           }
341         }
342       }
343     }
344 
345     // Get property information from dynamic refset members
346     // Create refex entries for properties
347     RefexDynamicUsageDescription propertyRefset =
348         RefexDynamicUsageDescriptionBuilder
349             .readRefexDynamicUsageDescriptionConcept(InformationModelAux.INFORMATION_MODEL_PROPERTIES_REFSET
350                 .getLenient().getNid());
351     LOG.debug("  property refset = " + propertyRefset.getRefexName());
352     for (RefexDynamicChronicleBI<?> refex : modelConcept
353         .getRefexDynamicAnnotations()) {
354       RefexDynamicVersionBI<?> refexVersion =
355           refex.getVersion(OTFUtility.getViewCoordinate());
356       LOG.debug("  sememe = " + refex.toUserString());
357       // Look for matching refex id and "active" flag
358       if (refexVersion != null
359           && refex.getAssemblageNid() == propertyRefset
360               .getRefexUsageDescriptorNid() && refexVersion.isActive()) {
361         // Create properties for each annotation
362         InformationModelProperty property =
363             new DefaultInformationModelProperty();
364         RefexDynamicDataBI[] data = refexVersion.getData();
365         if (data[0] != null)
366           property.setLabel(((RefexDynamicString) data[0]).getDataString());
367         if (data[1] != null)
368           property.setType(((RefexDynamicString) data[1]).getDataString());
369         if (data[2] != null)
370           property.setName(((RefexDynamicString) data[2]).getDataString());
371         if (data[3] != null)
372           property.setDefaultValue(((RefexDynamicString) data[3])
373               .getDataString());
374         if (data[4] != null)
375           property.setValue(((RefexDynamicString) data[4]).getDataString());
376         if (data[5] != null)
377           property.setCardinalityMin(((RefexDynamicString) data[5])
378               .getDataString());
379         if (data[6] != null)
380           property.setCardinalityMax(((RefexDynamicString) data[6])
381               .getDataString());
382         if (data[7] != null)
383           property
384               .setVisibility(((RefexDynamicString) data[7]).getDataString());
385         LOG.debug("    property " + property.getLabel() + ", "
386             + property.getName());
387         model.addProperty(property);
388       }
389     }
390 
391     // Build associated concept UUIDs from relationships
392     for (RelationshipChronicleBI rel : modelConcept.getRelationshipsOutgoing()) {
393       RelationshipVersionBI<?> relVersion =
394           rel.getVersion(OTFUtility.getViewCoordinate());
395       // Look for matching typeId and "active" flag
396       if (relVersion.getTypeNid() == InformationModelAux.HAS_TERMINOLOGY_CONCEPT
397           .getLenient().getNid() && relVersion.isActive()) {
398         // Add the destination UUID
399         model.addAssociatedConceptUuid(OTFUtility.getConceptVersion(
400             relVersion.getDestinationNid()).getPrimordialUuid());
401       }
402     }
403 
404     return model;
405   }

407   /**
408    * Returns the information model type. Walks up the tree from the
409    * corresponding concept to the level just below "information models" root and
410    * returns the type corresponding with the preferred name of the concept at
411    * that level.
412    *
413    * @param modelConcept the model concept
414    * @return the information model type
415    * @throws ContradictionException
416    * @throws IOException
417    */
418   private InformationModelType computeInformationModelType(
419     ConceptChronicleBI modelConcept) throws IOException, ContradictionException {
420     if (modelConcept == null) {
421       throw new IOException("Model concept unexpectedly null");
422     }
423     LOG.info("Compute information model type: "
424         + modelConcept.getPrimordialUuid());
425     ConceptVersionBI concept =
426         OTFUtility.getConceptVersion(modelConcept.getNid());
427     if (concept == null) {
428       throw new IOException("Model concept version unexpectedly null "
429           + modelConcept.getPrimordialUuid());
430     }
431     if (!concept.isActive()) {
432       throw new IOException("Model concept is inactive "
433           + modelConcept.getPrimordialUuid());
434     }
435 
436     InformationModelType[] values = InformationModelType.values();
437     // Walk up tree until we encounter a "type" concept
438     while (true) {
439       // Look for match (the pref name will match an information model type)
440       String prefName = OTFUtility.getConPrefTerm(concept.getNid());
441       if (prefName == null) {
442         throw new IOException("Concept preferred name unexepectedly null "
443             + concept.getPrimordialUuid());
444       }
445       for (InformationModelType type : values) {
446         if (prefName.equals(type.getDisplayName())) {
447           LOG.debug("  FOUND TYPE: " + type.getDisplayName());
448           return type;
449         }
450       }
451 
452       // continue walking
453       @SuppressWarnings("rawtypes")
454       Collection<? extends RelationshipVersionBI> rels =
455           concept.getRelationshipsOutgoingActiveIsa();
456       if (rels.size() < 1) {
457         break;
458       }
459       concept =
460           OTFUtility.getConceptVersion(rels.iterator().next()
461               .getDestinationNid());
462       LOG.debug("  " + concept.getPrimordialUuid());
463     }
464     throw new IOException("Information model type not found for "
465         + modelConcept.getPrimordialUuid());
466   }

468   /**
469    * Synchronizes the current state of the information model with the underlying
470    * strored concept representation. If it does not exist, it creates it. If it
471    * has changed, it updates it.
472    *
473    * TODO: to properly implement sync, we should take the old version of the
474    * model and the new version of the model and compare each field to see
475    * whether its changed. either retire or add the appropriate underlying data
476    * element.
477    * @param modelConcept the model concept
478    * @return the information model
479    * @throws ContradictionException
480    * @throws InvalidCAB
481    * @throws IOException
482    * @throws UnsupportedEncodingException
483    * @throws NoSuchAlgorithmException
484    * @throws PropertyVetoException
485    */
486   private ConceptChronicleBI syncInformationModelConcept(InformationModel model)
487     throws NoSuchAlgorithmException, UnsupportedEncodingException, IOException,
488     InvalidCAB, ContradictionException, PropertyVetoException {
489     if (model == null) {
490       throw new IOException("Model unexpectedly null");
491     }
492     LOG.info("Convert information model to concept");
493 
494     // Determine parent - either model type or super model uuid
495     ConceptChronicleBI parent = determineParent(model);
496 
497     // Acquire concept blueprint - either load from DB or create new one
498     LOG.debug("  parent = " + parent.getPrimordialUuid());
499     LOG.debug("  FN = " + model.getKey());
500     LOG.debug("  PT = " + model.getName());
501     ConceptCB modelConceptCB = null;
502     ConceptChronicleBI modelConcept = null;
503     // handle case where model concept exists already
504     if (exists(model)) {
505       // Obtain the UUID
506       UUID modelUuid =
507           this.getInformationModel(model.getType(), model.getKey()).getUuid();
508       LOG.debug("  Model exists already, UUID = " + modelUuid);
509       // Look it up by UUID0
510       modelConcept = dataStore.getConcept(modelUuid);
511       ConceptVersionBI modelConceptVersion =
512           modelConcept.getVersion(OTFUtility.getViewCoordinate());
513       modelConceptCB =
514           modelConceptVersion.makeBlueprint(OTFUtility.getViewCoordinate(),
515               IdDirective.PRESERVE, RefexDirective.EXCLUDE);
516       // synchronize descriptions (in case name or key changed)
517       syncDescriptions(modelConceptCB, model);
518 
519       // Sync super model UUID
520       syncSuperModel(modelConceptCB, model, parent);
521 
522       // Sync "has terminology concept" relationships
523       syncConnectedTerminologyConcepts(modelConceptCB, model);
524 
525     }
526     // handle case where model concept does not exist
527     else {
528       LOG.debug("  Model concept does not exist yet, create it");
529       modelConceptCB =
530           createNewConceptBlueprint(parent, model.getKey(), model.getName());
531       // Apply concept changes
532       modelConcept = OTFUtility.getBuilder().construct(modelConceptCB);
533       model.setUuid(modelConcept.getPrimordialUuid());
534       LOG.debug("    UUID = " + modelConcept.getPrimordialUuid());
535     }
536 
537     // Apply refex changes - modelConcept has been set above
538     syncRefexes(modelConcept, model);
539 
540     LOG.debug("  add uncommitted");
541     OTFUtility.addUncommitted(modelConcept);
542     LOG.debug("  commit");
543     OTFUtility.commit();
544 
545     return modelConcept;
546   }

548   /**
549    * Determines the parent concept. (there will be only one)
550    *
551    * @param model the model
552    * @return the concept chronicle bi
553    * @throws Exception the exception
554    */
555   private ConceptChronicleBI determineParent(InformationModel model)
556     throws IOException {
557     // Determine parent - either model type or super model uuid
558     InformationModelType type = model.getType();
559     ConceptChronicleBI parent = null;
560     if (model.getSuperModelUuid() != null) {
561       parent = dataStore.getConcept(model.getSuperModelUuid());
562     } else {
563       parent = dataStore.getConcept(type.getUuid());
564     }
565     if (parent == null) {
566       throw new IOException("Something went wrong determining parent concept "
567           + type.getUuid() + ", " + model.getSuperModelUuid());
568     }
569     return parent;
570   }

572   /**
573    * Sync descriptions. Retire descriptions no longer matching key or name.
574    * @param modelConceptCB the model concept cb
575    * @param model the model
576    * @throws ContradictionException
577    * @throws InvalidCAB
578    * @throws IOException
579    */
580   private void syncDescriptions(ConceptCB modelConceptCB, InformationModel model)
581     throws IOException, InvalidCAB, ContradictionException {
582     for (DescriptionCAB descCAB : modelConceptCB.getDescriptionCABs()) {
583       if (descCAB.getText().equals(model.getKey())) {
584         continue;
585       }
586       if (descCAB.getText().equals(model.getName())) {
587         continue;
588       }
589       descCAB.setRetired();
590       OTFUtility.getBuilder().constructIfNotCurrent(descCAB);
591     }
592   }

594   /**
595    * Sync super model relationships
596    *
597    * @param modelConceptCB the model concept cb
598    * @param model the model
599    * @throws IOException
600    * @throws ContradictionException
601    * @throws InvalidCAB
602    */
603   private void syncSuperModel(ConceptCB modelConceptCB, InformationModel model,
604     ConceptChronicleBI parent) throws IOException, InvalidCAB,
605     ContradictionException {
606 
607     LOG.debug("  Handle super-model UUID");
608     LOG.debug("    parent UUID = " + parent.getPrimordialUuid());
609 
610     // Retire any active Isa relationships to non-type UUIDs that are not the
611     // super model UUID
612     boolean found = false;
613     for (RelationshipCAB relCAB : modelConceptCB.getRelationshipCABs()) {
614       // LOG.debug("    rel cab type nid = " + relCAB.getTypeNid());
615       // LOG.debug("    TermAux.IS_A = " + TermAux.IS_A.getLenient().getNid());
616       // LOG.debug("    Snomed.IS_A = " + Snomed.IS_A.getLenient().getNid());
617       // LOG.debug("    SnomedRelType.IS_A = " +
618       // SnomedRelType.IS_A.getLenient().getNid());
619 
620       // Look for matching typeId flag
621       if (relCAB.getTypeNid() == Snomed.IS_A.getLenient().getNid()) {
622 
623         UUID uuid =
624             OTFUtility.getConceptVersion(relCAB.getTargetNid())
625                 .getPrimordialUuid();
626         LOG.debug("    target UUID = " + uuid);
627 
628         // If the target of this rel doesn't match the computed parent, retire
629         // it
630         if (!parent.getPrimordialUuid().equals(uuid)) {
631           relCAB.setRetired();
632           OTFUtility.getBuilder().constructIfNotCurrent(relCAB);
633         } else {
634           found = true;
635         }
636       }
637     }
638 
639     // if we didn't find it, create a new rel
640     if (!found) {
641       RelationshipCAB relCAB =
642           new RelationshipCAB(modelConceptCB.getComponentUuid(),
643               Snomed.IS_A.getUuids()[0], parent.getPrimordialUuid(), 0,
644               RelationshipType.STATED_ROLE, IdDirective.GENERATE_HASH);
645       OTFUtility.getBuilder().constructIfNotCurrent(relCAB);
646     }

649   /**
650    * Sync connected terminology concepts.
651    *
652    * @param modelConceptCB the model concept cb
653    * @param model the model
654    * @throws IOException Signals that an I/O exception has occurred.
655    * @throws ContradictionException
656    * @throws InvalidCAB
657    */
658   private void syncConnectedTerminologyConcepts(ConceptCB modelConceptCB,
659     InformationModel model) throws IOException, InvalidCAB,
660     ContradictionException {
661 
662     LOG.debug("  Iterate through associated UUIDs");
663     Set<UUID> associatedConceptUuids = model.getAssociatedConceptUuids();
664     // Retire any active relationships to UUIDs no longer in this set
665     for (RelationshipCAB relCAB : modelConceptCB.getRelationshipCABs()) {
666       // Look for matching typeId and "active" flag
667       if (relCAB.getTypeNid() == InformationModelAux.HAS_TERMINOLOGY_CONCEPT
668           .getLenient().getNid()) {
669         UUID uuid =
670             OTFUtility.getConceptVersion(relCAB.getTargetNid())
671                 .getPrimordialUuid();
672 
673         // If the UUID is not in range, retire the rel
674         if (!associatedConceptUuids.contains(uuid)) {
675           LOG.debug("    Found relationship to retire - "
676               + OTFUtility.getConPrefTerm(relCAB.getTargetNid()));
677           relCAB.setRetired();
678           OTFUtility.getBuilder().constructIfNotCurrent(relCAB);
679         }
680 
681         // Otherwise, remove from list, no need to create it
682         else {
683           associatedConceptUuids.remove(uuid);
684         }
685       }
686     }
687 
688     // Create rels for any UUIDs not accounted for, make new rels
689     for (UUID destinationUuid : associatedConceptUuids) {
690       LOG.debug("  Create relationship for "
691           + modelConceptCB.getComponentUuid() + " => " + destinationUuid);
692       UUID typeUid =
693           InformationModelAux.HAS_TERMINOLOGY_CONCEPT.getLenient()
694               .getPrimordialUuid();
695       RelationshipCAB relCAB =
696           new RelationshipCAB(modelConceptCB.getComponentUuid(), typeUid,
697               destinationUuid, 0, RelationshipType.STATED_ROLE,
698               IdDirective.GENERATE_HASH);
699       OTFUtility.getBuilder().constructIfNotCurrent(relCAB);
700       modelConceptCB.setRelationshipCAB(relCAB);
701     }
702   }

704   /**
705    * Sync refexes. Determine which one should exist, add them if they do not,
706    * retire them if they do.
707    * @param modelConceptCB the model concept cb
708    * @param modelCB the model cb
709    * @throws ContradictionException
710    * @throws IOException
711    * @throws ValidationException
712    * @throws InvalidCAB
713    * @throws PropertyVetoException
714    */
715   private void syncRefexes(ConceptChronicleBI modelConcept,
716     InformationModel model) throws ValidationException, IOException,
717     ContradictionException, InvalidCAB, PropertyVetoException {
718     // Create refex entries for properties
719     RefexDynamicUsageDescription propertyRefset =
720         RefexDynamicUsageDescriptionBuilder
721             .readRefexDynamicUsageDescriptionConcept(InformationModelAux.INFORMATION_MODEL_PROPERTIES_REFSET
722                 .getLenient().getNid());
723     LOG.debug("  Found " + propertyRefset.getRefexName());
724     // Iterate through information model properties and add refexes
725     LOG.debug("  Iterate through properties");
726     // Create a dynamic refex CAB for each entry
727     // If these already exist, then the construct will have no practical effect
728     Set<UUID> refexUuids = new HashSet<>();
729     for (InformationModelProperty property : model.getProperties()) {
730       LOG.debug("    " + property.getLabel() + ", " + property.getName());
731       // Need better understanding of what IdDirective to use here
732       RefexDynamicCAB refexBlueprint =
733           new RefexDynamicCAB(modelConcept.getNid(),
734               propertyRefset.getRefexUsageDescriptorNid());
735 
736       // The order of these columns is tightly bound to the definition,
737       // if the definition changes, this has to be updated as well.
738       if (propertyRefset.getColumnInfo().length != 8) {
739         throw new IOException(
740             "Information model properties refset has unexpected number of columns");
741       }
742       RefexDynamicDataBI[] data =
743           new RefexDynamicDataBI[propertyRefset.getColumnInfo().length];
744       data[0] = new RefexDynamicString(property.getLabel());
745       data[1] = new RefexDynamicString(property.getType());
746       data[2] = new RefexDynamicString(property.getName());
747       data[3] = new RefexDynamicString(property.getDefaultValue());
748       data[4] = new RefexDynamicString(property.getValue());
749       data[5] = new RefexDynamicString(property.getCardinalityMin());
750       data[6] = new RefexDynamicString(property.getCardinalityMax());
751       data[7] = new RefexDynamicString(property.getVisibility());
752       refexBlueprint.setData(data, OTFUtility.getViewCoordinate());
753 
754       // Construct and wire the dynamic refex
755       RefexDynamicChronicleBI<?> refex =
756           OTFUtility.getBuilder().construct(refexBlueprint);
757       modelConcept.addDynamicAnnotation(refex);
758       refexUuids.add(refex.getPrimordialUuid());
759       LOG.debug("    UUID = " + refex.getPrimordialUuid());
760     }
761 
762     // Now iterate through all refexes and retire those that do not
763     // have uuids from the code above
764     for (RefexDynamicChronicleBI<?> refex : modelConcept
765         .getRefexDynamicAnnotations()) {
766       RefexDynamicVersionBI<?> refexVersion =
767           refex.getVersion(OTFUtility.getViewCoordinate());
768       if (refexVersion != null
769           && !refexUuids.contains(refex.getPrimordialUuid())) {
770         RefexDynamicCAB refexBlueprint =
771             refexVersion.makeBlueprint(OTFUtility.getViewCoordinate(),
772                 IdDirective.PRESERVE, RefexDirective.EXCLUDE);
773         refexBlueprint.setRetired();
774         RefexDynamicChronicleBI<?> refex2 =
775             OTFUtility.getBuilder().construct(refexBlueprint);
776         LOG.debug("  Retire sememe UUID = " + refex2.getPrimordialUuid());
777       }
778     }
779   }

781   /**
782    * Creates the new concept blueprint
783    *
784    * @param parent the parent
785    * @param fsn the fsn
786    * @param prefTerm the pref term
787    * @return the concept cb
788    * @throws ValidationException the validation exception
789    * @throws IOException Signals that an I/O exception has occurred.
790    * @throws InvalidCAB the invalid cab
791    * @throws ContradictionException the contradiction exception
792    */
793   private ConceptCB createNewConceptBlueprint(ConceptChronicleBI parent,
794     String fsn, String prefTerm) throws ValidationException, IOException,
795     InvalidCAB, ContradictionException {
796     if (parent == null) {
797       throw new IOException("Parent is unexpectedly null");
798     }
799     if (fsn == null) {
800       throw new IOException("FN is unexpectedly null");
801     }
802     if (prefTerm == null) {
803       throw new IOException("PT is unexpectedly null");
804     }
805     // Assume language US English
806     LanguageCode lc = LanguageCode.EN_US;
807     // Assume hierarchy is represented as IS_A rel
808     UUID isA = Snomed.IS_A.getUuids()[0];
809     // Compute UUIDs in standard way
810     IdDirective idDir = IdDirective.GENERATE_HASH;
811     // Assume module should be SNOMED core module (this may be something to
812     // change)
813     UUID module = Snomed.CORE_MODULE.getLenient().getPrimordialUuid();
814     // Create single parent
815     UUID parentUUIDs[] = new UUID[1];
816     parentUUIDs[0] = parent.getPrimordialUuid();
817     return new ConceptCB(fsn, prefTerm, lc, isA, idDir, module, parentUUIDs);
818   }
