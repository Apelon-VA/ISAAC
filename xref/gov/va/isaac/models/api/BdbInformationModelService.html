112   @Override
113   public boolean exists(InformationModel model) throws IOException {
114     LOG.info("Check whether information model exists: "
115         + model.getType().getDisplayName() + ", " + model.getKey());
116 
117     if (model.getType() == null) {
118       throw new IOException("Model unexpectedly has null type");
119     }
120     if (model.getKey() == null) {
121       throw new IOException("Model unexpectedly has null key");
122     }
123     try {
124       boolean flag =
125           getInformationModel(model.getType(), model.getKey()) != null;
126       LOG.debug("  " + flag);
127       return flag;
128     } catch (ContradictionException e) {
129       e.printStackTrace();
130       throw new IOException(
131           "Unexpected inability to acquire UUID for type and key, "
132               + model.getType() + ", " + model.getKey());
133     }
134   }

145     ContradictionException {
146     if (uuid == null) {
147       throw new IOException("UUID unexpectedly null");
148     }

152     LOG.debug("  " + model.getType().getDisplayName() + ", " + model.getKey());
153     return model;

156   /*
157    * (non-Javadoc)
158    * 
159    * @see
160    * gov.va.isaac.models.api.InformationModelService#getInformationModel(java
161    * .lang.String, java.lang.String)
162    */
163   @Override
164   public InformationModel getInformationModel(InformationModelType type,
165     String key) throws IOException, ContradictionException {
166     if (type == null) {
167       throw new IllegalArgumentException("Unexpected null type parameter.");
168     }
169     LOG.info("Get information model: " + type.getDisplayName() + ", " + key);
170     if (key == null) {
171       throw new IllegalArgumentException("Unexpected null key parameter.");
172     }
173     try {
174       LOG.debug("  Lucene Search: '" + key + "'");
175       LuceneDescriptionIndexer descriptionIndexer =
176           AppContext.getService(LuceneDescriptionIndexer.class);
177       if (descriptionIndexer == null) {
178         throw new IOException("No description indexer found, aborting.");
179       }
180       // Look for description matches.
181       ComponentProperty field = ComponentProperty.DESCRIPTION_TEXT;
182       int limit = 10;
183       List<SearchResult> searchResults;
184       searchResults =
185           descriptionIndexer.query(key, false, field, limit, Long.MIN_VALUE);
186 
187       // Results are descriptions, need to look up concepts
188       for (SearchResult result : searchResults) {
189         int conceptNid =
190             dataStore.getComponent(result.getNid()).getConceptNid();
191         ConceptVersionBI conceptVersion =
192             WBUtility.getConceptVersion(conceptNid);
193         LOG.debug("    Check " + conceptVersion.getPrimordialUuid() + ", "
194             + conceptVersion.getPreferredDescription().getText());
195 
196         if (conceptVersion.getPreferredDescription().getText().equals(key)
197             || conceptVersion.getFullySpecifiedDescription().getText()
198                 .equals(key)) {
199           LOG.debug("    Found match, compute information model type");
200           InformationModelType computedType =
201               computeInformationModelType(conceptVersion);
202           if (computedType == null) {
203             LOG.debug("      type = null");
204             continue;
205           } else {
206             LOG.debug("      type = " + computedType.getDisplayName());
207           }
208           if (type == computedType) {
209             InformationModel model =
210                 getInformationModel(conceptVersion.getPrimordialUuid());
211             LOG.debug("  Model found, UUID = " + model.getUuid());
212             return model;
213           }
214         }
215       }
216       LOG.debug("  Model not found");
217       return null;
218     } catch (ParseException e) {
219       throw new IOException("Unable to parse key for lucene search, " + key, e);
220     }
221 
222   }

237     Set<InformationModel> models = new HashSet<>();
238     for (RelationshipChronicleBI rel : modelConcept.getRelationshipsIncoming()) {

241       // Look for matching typeId and "active" flag
242       if (relVersion.getTypeNid() == Snomed.IS_A.getLenient().getNid()
243           && relVersion.isActive()) {
244         // Add the model
245         models.add(getInformationModel(WBUtility.getConceptVersion(
246             relVersion.getDestinationNid()).getPrimordialUuid()));
247       }
248     }
249     return models;
250   }

271   /**
272    * Concept to information model.
273    *
274    * @param modelConcept the model concept
275    * @return the information model
276    * @throws ContradictionException
277    * @throws IOException
278    */
279   private InformationModel conceptToInformationModel(
280     ConceptChronicleBI modelConcept) throws IOException, ContradictionException {
281     if (modelConcept == null) {
282       throw new IOException("Model concept unexpectedly null");
283     }
284 
285     LOG.info("Convert concept to information model: "
286         + modelConcept.getPrimordialUuid());
287 
288     // Key is the FN
289     String key = WBUtility.getFullySpecifiedName(modelConcept);
290     LOG.debug("  key = " + key);
291     // Name is the PT
292     String name = WBUtility.getConPrefTerm(modelConcept.getNid());
293     LOG.debug("  name = " + name);
294     // UUID of the concept
295     UUID uuid = modelConcept.getPrimordialUuid();
296     LOG.debug("  uuid = " + uuid);
297     // information model type - obtain by walking up the tree
298     InformationModelType type = computeInformationModelType(modelConcept);
299     LOG.debug("  type = " + type.getDisplayName());
300     InformationModel model = new DefaultInformationModel(key, name, uuid, type);
301 
302     // Assemble Metadata
303     // Need to look at stamp nid for concept attributes instead of concept
304     int stampNid =
305         modelConcept.getConceptAttributes()
306             .getVersion(WBUtility.getViewCoordinate()).getStamp();
307     InformationModelMetadata metadata =
308         InformationModelMetadata.newInstance(stampNid, dataStore,
309             WBUtility.getViewCoordinate());
310     model.setMetadata(metadata);
311 
312     // Check super-model
313     // Build associated concept UUIDs from relationships
314     for (RelationshipChronicleBI rel : modelConcept.getRelationshipsOutgoing()) {
315       RelationshipVersionBI<?> relVersion =
316           rel.getVersion(WBUtility.getViewCoordinate());
317       // Look for matching typeId and "active" flag
318       if (relVersion.getTypeNid() == Snomed.IS_A.getLenient().getNid()
319           && relVersion.isActive()) {
320         UUID superModelUuid =
321             WBUtility.getConceptVersion(relVersion.getDestinationNid())
322                 .getPrimordialUuid();
323         // Avoid assigning uuid if parent is the info model type concept
324         if (!superModelUuid.equals(model.getType().getUuid())) {
325           // set the super model uuid unless already set
326           if (model.getSuperModelUuid() != null) {
327             throw new IOException(
328                 "Unexpected multiple lineage in information model concept: "
329                     + model.getUuid());
330           } else {
331             model.setSuperModelUuid(superModelUuid);
332           }
333         }
334       }
335     }
336 
337     // Get property information from dynamic refset members
338     // Create refex entries for properties
339     RefexDynamicUsageDescription propertyRefset =
340         RefexDynamicUsageDescriptionBuilder
341             .readRefexDynamicUsageDescriptionConcept(InformationModelAux.INFORMATION_MODEL_PROPERTIES_REFSET
342                 .getLenient().getNid());
343     LOG.debug("  property refset = " + propertyRefset.getRefexName());
344     for (RefexDynamicChronicleBI<?> refex : modelConcept
345         .getRefexDynamicAnnotations()) {
346       RefexDynamicVersionBI<?> refexVersion =
347           refex.getVersion(WBUtility.getViewCoordinate());
348       LOG.debug("  refex = " + refex.toUserString());
349       // Look for matching refex id and "active" flag
350       if (refexVersion != null
351           && refex.getAssemblageNid() == propertyRefset
352               .getRefexUsageDescriptorNid() && refexVersion.isActive()) {
353         // Create properties for each annotation
354         InformationModelProperty property =
355             new DefaultInformationModelProperty();
356         RefexDynamicDataBI[] data = refexVersion.getData();
357         if (data[0] != null)
358           property.setLabel(((RefexDynamicString) data[0]).getDataString());
359         if (data[1] != null)
360           property.setType(((RefexDynamicString) data[1]).getDataString());
361         if (data[2] != null)
362           property.setName(((RefexDynamicString) data[2]).getDataString());
363         if (data[3] != null)
364           property.setDefaultValue(((RefexDynamicString) data[3])
365               .getDataString());
366         if (data[4] != null)
367           property.setValue(((RefexDynamicString) data[4]).getDataString());
368         if (data[5] != null)
369           property.setCardinalityMin(((RefexDynamicString) data[5])
370               .getDataString());
371         if (data[6] != null)
372           property.setCardinalityMax(((RefexDynamicString) data[6])
373               .getDataString());
374         if (data[7] != null)
375           property
376               .setVisibility(((RefexDynamicString) data[7]).getDataString());
377         LOG.debug("    property " + property.getLabel() + ", "
378             + property.getName());
379         model.addProperty(property);
380       }
381     }
382 
383     // Build associated concept UUIDs from relationships
384     for (RelationshipChronicleBI rel : modelConcept.getRelationshipsOutgoing()) {
385       RelationshipVersionBI<?> relVersion =
386           rel.getVersion(WBUtility.getViewCoordinate());
387       // Look for matching typeId and "active" flag
388       if (relVersion.getTypeNid() == InformationModelAux.HAS_TERMINOLOGY_CONCEPT
389           .getLenient().getNid() && relVersion.isActive()) {
390         // Add the destination UUID
391         model.addAssociatedConceptUuid(WBUtility.getConceptVersion(
392             relVersion.getDestinationNid()).getPrimordialUuid());
393       }
394     }
395 
396     return model;
397   }

399   /**
400    * Returns the information model type. Walks up the tree from the
401    * corresponding concept to the level just below "information models" root and
402    * returns the type corresponding with the preferred name of the concept at
403    * that level.
404    *
405    * @param modelConcept the model concept
406    * @return the information model type
407    * @throws ContradictionException
408    * @throws IOException
409    */
410   private InformationModelType computeInformationModelType(
411     ConceptChronicleBI modelConcept) throws IOException, ContradictionException {
412     if (modelConcept == null) {
413       throw new IOException("Model concept unexpectedly null");
414     }
415     LOG.info("Compute information model type: "
416         + modelConcept.getPrimordialUuid());
417     ConceptVersionBI concept =
418         WBUtility.getConceptVersion(modelConcept.getNid());
419     if (concept == null) {
420       throw new IOException("Model concept version unexpectedly null "
421           + modelConcept.getPrimordialUuid());
422     }
423     if (!concept.isActive()) {
424       throw new IOException("Model concept is inactive "
425           + modelConcept.getPrimordialUuid());
426     }
427 
428     InformationModelType[] values = InformationModelType.values();
429     // Walk up tree until we encounter a "type" concept
430     while (true) {
431       // Look for match (the pref name will match an information model type)
432       String prefName = WBUtility.getConPrefTerm(concept.getNid());
433       if (prefName == null) {
434         throw new IOException("Concept preferred name unexepectedly null "
435             + concept.getPrimordialUuid());
436       }
437       for (InformationModelType type : values) {
438         if (prefName.equals(type.getDisplayName())) {
439           LOG.debug("  FOUND TYPE: " + type.getDisplayName());
440           return type;
441         }
442       }
443 
444       // continue walking
445       @SuppressWarnings("rawtypes")
446       Collection<? extends RelationshipVersionBI> rels =
447           concept.getRelationshipsOutgoingActiveIsa();
448       if (rels.size() < 1) {
449         break;
450       }
451       concept =
452           WBUtility.getConceptVersion(rels.iterator().next()
453               .getDestinationNid());
454       LOG.debug("  " + concept.getPrimordialUuid());
455     }
456     throw new IOException("Information model type not found for "
457         + modelConcept.getPrimordialUuid());
458   }

460   /**
461    * Synchronizes the current state of the information model with the underlying
462    * strored concept representation. If it does not exist, it creates it. If it
463    * has changed, it updates it.
464    *
465    * TODO: to properly implement sync, we should take the old version of the
466    * model and the new version of the model and compare each field to see
467    * whether its changed. either retire or add the appropriate underlying data
468    * element.
469    * @param modelConcept the model concept
470    * @return the information model
471    * @throws ContradictionException
472    * @throws InvalidCAB
473    * @throws IOException
474    * @throws UnsupportedEncodingException
475    * @throws NoSuchAlgorithmException
476    * @throws PropertyVetoException
477    */
478   private ConceptChronicleBI syncInformationModelConcept(InformationModel model)
479     throws NoSuchAlgorithmException, UnsupportedEncodingException, IOException,
480     InvalidCAB, ContradictionException, PropertyVetoException {
481     if (model == null) {
482       throw new IOException("Model unexpectedly null");
483     }
484     LOG.info("Convert information model to concept");
485 
486     // Determine parent - either model type or super model uuid
487     ConceptChronicleBI parent = determineParent(model);
488 
489     // Acquire concept blueprint - either load from DB or create new one
490     LOG.debug("  parent = " + parent.getPrimordialUuid());
491     LOG.debug("  FN = " + model.getKey());
492     LOG.debug("  PT = " + model.getName());
493     ConceptCB modelConceptCB = null;
494     ConceptChronicleBI modelConcept = null;
495     // handle case where model concept exists already
496     if (exists(model)) {
497       // Obtain the UUID
498       UUID modelUuid =
499           this.getInformationModel(model.getType(), model.getKey()).getUuid();
500       LOG.debug("  Model exists already, UUID = " + modelUuid);
501       // Look it up by UUID0
502       modelConcept = dataStore.getConcept(modelUuid);
503       ConceptVersionBI modelConceptVersion =
504           modelConcept.getVersion(WBUtility.getViewCoordinate());
505       modelConceptCB =
506           modelConceptVersion.makeBlueprint(WBUtility.getViewCoordinate(),
507               IdDirective.PRESERVE, RefexDirective.EXCLUDE);
508       // synchronize descriptions (in case name or key changed)
509       syncDescriptions(modelConceptCB, model);
510 
511       // Sync super model UUID
512       syncSuperModel(modelConceptCB, model, parent);
513 
514       // Sync "has terminology concept" relationships
515       syncConnectedTerminologyConcepts(modelConceptCB, model);
516 
517     }
518     // handle case where model concept does not exist
519     else {
520       LOG.debug("  Model concept does not exist yet, create it");
521       modelConceptCB =
522           createNewConceptBlueprint(parent, model.getKey(), model.getName());
523       // Apply concept changes
524       modelConcept = WBUtility.getBuilder().construct(modelConceptCB);
525       model.setUuid(modelConcept.getPrimordialUuid());
526       LOG.debug("    UUID = " + modelConcept.getPrimordialUuid());
527     }
528 
529     // Apply refex changes - modelConcept has been set above
530     syncRefexes(modelConcept, model);
531 
532     LOG.debug("  add uncommitted");
533     WBUtility.addUncommitted(modelConcept);
534     LOG.debug("  commit");
535     WBUtility.commit();
536 
537     return modelConcept;
538   }

540   /**
541    * Determines the parent concept. (there will be only one)
542    *
543    * @param model the model
544    * @return the concept chronicle bi
545    * @throws Exception the exception
546    */
547   private ConceptChronicleBI determineParent(InformationModel model)
548     throws IOException {
549     // Determine parent - either model type or super model uuid
550     InformationModelType type = model.getType();
551     ConceptChronicleBI parent = null;
552     if (model.getSuperModelUuid() != null) {
553       parent = dataStore.getConcept(model.getSuperModelUuid());
554     } else {
555       parent = dataStore.getConcept(type.getUuid());
556     }
557     if (parent == null) {
558       throw new IOException("Something went wrong determining parent concept "
559           + type.getUuid() + ", " + model.getSuperModelUuid());
560     }
561     return parent;
562   }

564   /**
565    * Sync descriptions. Retire descriptions no longer matching key or name.
566    * @param modelConceptCB the model concept cb
567    * @param model the model
568    * @throws ContradictionException
569    * @throws InvalidCAB
570    * @throws IOException
571    */
572   private void syncDescriptions(ConceptCB modelConceptCB, InformationModel model)
573     throws IOException, InvalidCAB, ContradictionException {
574     for (DescriptionCAB descCAB : modelConceptCB.getDescriptionCABs()) {
575       if (descCAB.getText().equals(model.getKey())) {
576         continue;
577       }
578       if (descCAB.getText().equals(model.getName())) {
579         continue;
580       }
581       descCAB.setRetired();
582       WBUtility.getBuilder().constructIfNotCurrent(descCAB);
583     }
584   }

586   /**
587    * Sync super model relationships
588    *
589    * @param modelConceptCB the model concept cb
590    * @param model the model
591    * @throws IOException
592    * @throws ContradictionException
593    * @throws InvalidCAB
594    */
595   private void syncSuperModel(ConceptCB modelConceptCB, InformationModel model,
596     ConceptChronicleBI parent) throws IOException, InvalidCAB,
597     ContradictionException {
598 
599     LOG.debug("  Handle super-model UUID");
600     LOG.debug("    parent UUID = " + parent.getPrimordialUuid());
601 
602     // Retire any active Isa relationships to non-type UUIDs that are not the
603     // super model UUID
604     boolean found = false;
605     for (RelationshipCAB relCAB : modelConceptCB.getRelationshipCABs()) {
606       // LOG.debug("    rel cab type nid = " + relCAB.getTypeNid());
607       // LOG.debug("    TermAux.IS_A = " + TermAux.IS_A.getLenient().getNid());
608       // LOG.debug("    Snomed.IS_A = " + Snomed.IS_A.getLenient().getNid());
609       // LOG.debug("    SnomedRelType.IS_A = " +
610       // SnomedRelType.IS_A.getLenient().getNid());
611 
612       // Look for matching typeId flag
613       if (relCAB.getTypeNid() == Snomed.IS_A.getLenient().getNid()) {
614 
615         UUID uuid =

696   /**
697    * Sync refexes. Determine which one should exist, add them if they do not,
698    * retire them if they do.
699    * @param modelConceptCB the model concept cb
700    * @param modelCB the model cb
701    * @throws ContradictionException
702    * @throws IOException
703    * @throws ValidationException
704    * @throws InvalidCAB
705    * @throws PropertyVetoException
706    */
707   private void syncRefexes(ConceptChronicleBI modelConcept,
708     InformationModel model) throws ValidationException, IOException,
709     ContradictionException, InvalidCAB, PropertyVetoException {
710     // Create refex entries for properties
711     RefexDynamicUsageDescription propertyRefset =
712         RefexDynamicUsageDescriptionBuilder
713             .readRefexDynamicUsageDescriptionConcept(InformationModelAux.INFORMATION_MODEL_PROPERTIES_REFSET
714                 .getLenient().getNid());
715     LOG.debug("  Found " + propertyRefset.getRefexName());
716     // Iterate through information model properties and add refexes
717     LOG.debug("  Iterate through properties");
718     // Create a dynamic refex CAB for each entry
719     // If these already exist, then the construct will have no practical effect
720     Set<UUID> refexUuids = new HashSet<>();
721     for (InformationModelProperty property : model.getProperties()) {
722       LOG.debug("    " + property.getLabel() + ", " + property.getName());
723       // Need better understanding of what IdDirective to use here
724       RefexDynamicCAB refexBlueprint =
725           new RefexDynamicCAB(modelConcept.getNid(),
726               propertyRefset.getRefexUsageDescriptorNid());
727 
728       // The order of these columns is tightly bound to the definition,
729       // if the definition changes, this has to be updated as well.
730       if (propertyRefset.getColumnInfo().length != 8) {
731         throw new IOException(
732             "Information model properties refset has unexpected number of columns");
733       }
734       RefexDynamicDataBI[] data =
735           new RefexDynamicDataBI[propertyRefset.getColumnInfo().length];
736       data[0] = new RefexDynamicString(property.getLabel());
737       data[1] = new RefexDynamicString(property.getType());
738       data[2] = new RefexDynamicString(property.getName());
739       data[3] = new RefexDynamicString(property.getDefaultValue());
740       data[4] = new RefexDynamicString(property.getValue());
741       data[5] = new RefexDynamicString(property.getCardinalityMin());
742       data[6] = new RefexDynamicString(property.getCardinalityMax());
743       data[7] = new RefexDynamicString(property.getVisibility());
744       refexBlueprint.setData(data, WBUtility.getViewCoordinate());
745 
746       // Construct and wire the dynamic refex
747       RefexDynamicChronicleBI<?> refex =
748           WBUtility.getBuilder().construct(refexBlueprint);
749       modelConcept.addDynamicAnnotation(refex);
750       refexUuids.add(refex.getPrimordialUuid());
751       LOG.debug("    UUID = " + refex.getPrimordialUuid());
752     }
753 
754     // Now iterate through all refexes and retire those that do not
755     // have uuids from the code above
756     for (RefexDynamicChronicleBI<?> refex : modelConcept
757         .getRefexDynamicAnnotations()) {
758       RefexDynamicVersionBI<?> refexVersion =
759           refex.getVersion(WBUtility.getViewCoordinate());
760       if (refexVersion != null
761           && !refexUuids.contains(refex.getPrimordialUuid())) {
762         RefexDynamicCAB refexBlueprint =
763             refexVersion.makeBlueprint(WBUtility.getViewCoordinate(),
764                 IdDirective.PRESERVE, RefexDirective.EXCLUDE);
765         refexBlueprint.setRetired();
766         RefexDynamicChronicleBI<?> refex2 =
767             WBUtility.getBuilder().construct(refexBlueprint);
768         LOG.debug("  Retire refex UUID = " + refex2.getPrimordialUuid());
769       }
770     }
