35  /**
36   * Abstract superclass for metadata creators.
37   *
38   * @author ocarlsen
39   */
40  public abstract class MetadataCreatorBase extends CommonBase {
41  
42      protected MetadataCreatorBase() {
43          super();
44      }
45  
46      protected static final String REFSET_CONCEPT = "7e38cd2d-6f1a-3a81-be0b-21e6090573c2";
47      protected static final String REFSET_ATTRIBUTE_CONCEPT = "7e52203e-8a35-3121-b2e7-b783b34d97f2";
48  
49      /**
50       * Concrete subclasses should implement to create metadata if it does not
51       * already exist.  Return value should indicate as appropriate.
52       * @return {@code true} if metadata was created, or {@code false} if metadata
53       * already existed and so creation was bypassed.
54       * @throws Exception if there is a problem creating metadata.
55       */
56      public abstract boolean createMetadata() throws Exception;
57  
58      /**
59       * @return The {@link ConceptChronicleBI} for the specified {@link UUID},
60       * or {@code null} if it has not yet been created.
61       */
62      protected ConceptChronicleBI getConcept(UUID uuid) throws IOException {
63          ConceptChronicleBI concept = getDataStore().getConcept(uuid);
64          if (concept.getDescriptions().size() == 0) {
65              return null;
66          }
67          return concept;
68      }
69  
70      /**
71       * Copied from Jesse's RefsetCreation demo.
72       */
73      protected ConceptChronicleBI createNewRefsetConcept(ConceptChronicleBI parent,
74              String fsn, String prefTerm, boolean isAnnotated)
75              throws IOException, InvalidCAB, ContradictionException {
76          ConceptCB newConCB = createNewConceptBlueprint(parent, fsn, prefTerm);
77  
78          ConceptChronicleBI newCon = WBUtility.getBuilder().construct(newConCB);
79  
80          newCon.setAnnotationStyleRefex(isAnnotated);
81  
82          getDataStore().addUncommitted(newCon);
83  
84          return newCon;
85      }
86  
87      protected ConceptChronicleBI createNewConcept(ConceptChronicleBI parent, String fsn,
88              String prefTerm) throws IOException, InvalidCAB, ContradictionException {
89          ConceptCB newConCB = createNewConceptBlueprint(parent, fsn, prefTerm);
90  
91          ConceptChronicleBI newCon = WBUtility.getBuilder().construct(newConCB);
92  
93          getDataStore().addUncommitted(newCon);
94  
95          return newCon;
96      }
97  
98      protected ConceptCB createNewConceptBlueprint(ConceptChronicleBI parent, String fsn, String prefTerm)
99              throws ValidationException, IOException, InvalidCAB,
100             ContradictionException {
101         LanguageCode lc = LanguageCode.EN_US;
102         UUID isA = Snomed.IS_A.getUuids()[0];
103         IdDirective idDir = IdDirective.GENERATE_HASH;
104         UUID module = Snomed.CORE_MODULE.getLenient().getPrimordialUuid();
105         UUID parentUUIDs[] = new UUID[1];
106         parentUUIDs[0] = parent.getPrimordialUuid();
107         return new ConceptCB(fsn, prefTerm, lc, isA, idDir, module, parentUUIDs);
108     }
