87      public void exportModel(UUID modelUUID) throws Exception {
88          LOG.info("Starting export of FHIM model");
89  
90          // Get chronicle for component.
91          ComponentChronicleBI<?> focusComponent = getDataStore().getComponent(modelUUID);
92          LOG.debug("focusComponent="+focusComponent);
93  
94          // Abort if not available.
95          if (focusComponent == null) {
96              LOG.warn("No model found: " + modelUUID);
97              return;
98          }
99  
100         // Expect a RefexChronicle.
101         if (! (focusComponent instanceof RefexChronicleBI<?>)) {
102             LOG.warn("Expected FHIM model to get a sememe: " + modelUUID);
103             return;
104         }
105         RefexChronicleBI<?> modelRefex = (RefexChronicleBI<?>) focusComponent;
106 
107         // Parse into FHIM model.
108         FHIMInformationModel infoModel = createInformationModel(modelRefex);
109 
110         // Abort if not available.
111         if (infoModel == null) {
112             LOG.warn("No model found: " + modelUUID);
113             return;
114         }
115 
116         // Convert to UML model in the style of FHIM.
117         Model2UMLConverter converter = new Model2UMLConverter();
118         @SuppressWarnings("unused")
119         Package umlModel = converter.createUMLModel(infoModel);
120 
121         // Write UML model to OutputStream.
122         List<Package> pkgs = converter.getTopLevelPackages();
123         saveModel(pkgs);
124 
125         LOG.info("Ending export of FHIM model");
126     }

133     private void saveModel(List<Package> pkgs) throws IOException {
134 
135         // Create a resource-set to contain the resource(s) that we are saving
136         ResourceSet resourceSet = new ResourceSetImpl();
137 
138         // Initialize registrations of resource factories, library models,
139         // profiles, Ecore metadata, and other dependencies required for
140         // serializing and working with UML resources. This is only necessary in
141         // applications that are not hosted in the Eclipse platform run-time, in
142         // which case these registrations are discovered automatically from
143         // Eclipse extension points.
144         UMLResourcesUtil.init(resourceSet);
145 
146         // Create the output resource and add our model package to it.
147 //        Resource resource = resourceSet.createResource(uri);
148         Resource resource = new XMIResourceImpl();
149         for (Package pkg : pkgs) {
150             resource.getContents().add(pkg);
151         }
152 
153         // And save.
154         Map<?, ?> options = null;   // No save options needed.
155         resource.save(outputStream, options);
156         outputStream.close();
157     }

456     private Type getTypeForNid(int nid) {
457         Type t = null;
458 
459         // Class?
460         t = getClass(nid);
461         if (t != null) {
462             return t;
463         }
464 
465         // Enumeration?
466         t = getEnumeration(nid);
467         if (t != null) {
468             return t;
469         }
470 
471         // Attribute?
472         t = getAttribute(nid);
473         if (t != null) {
474             return t;
475         }
476 
477         // External?
478         t = getExternal(nid);
479         if (t != null) {
480             return t;
481         }
482 
483         // Something wrong!
484         throw new IllegalArgumentException("Could not find type for " + nid);
485     }
