91      public void exportModel(UUID modelUUID) throws Exception {
92          LOG.info("Starting export of FHIM model");
93  
94          // Get chronicle for component.
95          ComponentChronicleBI<?> focusComponent = getDataStore().getComponent(modelUUID);
96          LOG.debug("focusComponent="+focusComponent);
97  
98          // Abort if not available.
99          if (focusComponent == null) {
100             LOG.warn("No model found: " + modelUUID);
101             return;
102         }
103 
104         // Expect a RefexChronicle.
105         if (! (focusComponent instanceof RefexChronicleBI<?>)) {
106             LOG.warn("Expected FHIM model to ge a refex: " + modelUUID);
107             return;
108         }
109         RefexChronicleBI<?> modelRefex = (RefexChronicleBI<?>) focusComponent;
110 
111         // Parse into FHIM model.
112         FHIMInformationModel infoModel = createInformationModel(modelRefex);
113 
114         // Abort if not available.
115         if (infoModel == null) {
116             LOG.warn("No model found: " + modelUUID);
117             return;
118         }
119 
120         // Convert to UML model in the style of FHIM.
121         Model2UMLConverter converter = new Model2UMLConverter();
122         @SuppressWarnings("unused")
123         Package umlModel = converter.createUMLModel(infoModel);
124 
125         // Write UML model to OutputStream.
126         List<Package> pkgs = converter.getTopLevelPackages();
127         saveModel(pkgs);
128 
129         LOG.info("Ending export of FHIM model");
130     }

137     private void saveModel(List<Package> pkgs) throws IOException {
138 
139         // Create a resource-set to contain the resource(s) that we are saving
140         ResourceSet resourceSet = new ResourceSetImpl();
141 
142         // Initialize registrations of resource factories, library models,
143         // profiles, Ecore metadata, and other dependencies required for
144         // serializing and working with UML resources. This is only necessary in
145         // applications that are not hosted in the Eclipse platform run-time, in
146         // which case these registrations are discovered automatically from
147         // Eclipse extension points.
148         UMLResourcesUtil.init(resourceSet);
149 
150         // Create the output resource and add our model package to it.
151 //        Resource resource = resourceSet.createResource(uri);
152         Resource resource = new XMIResourceImpl();
153         for (Package pkg : pkgs) {
154             resource.getContents().add(pkg);
155         }
156 
157         // And save.
158         Map<?, ?> options = null;   // No save options needed.
159         resource.save(outputStream, options);
160         outputStream.close();
161     }

163     private FHIMInformationModel createInformationModel(RefexChronicleBI<?> modelRefex)
164             throws ValidationException, IOException, ContradictionException {
165 
166         // Model name, UUID.
167         StringMember modelAnnotation = (StringMember) modelRefex;
168         String modelName = modelAnnotation.getString1();
169         UUID modelUUID = modelRefex.getPrimordialUuid();
170         FHIMInformationModel infoModel = new FHIMInformationModel(modelName, modelUUID);
171         LOG.debug("Model name: " + modelName);
172         LOG.debug("Model UUID: " + modelUUID);
173 
174         // Get all annotations on the model annotation.
175         Collection<? extends RefexChronicleBI<?>> modelAnnotations = getLatestAnnotations(modelRefex);
176 
177         // Enumerations.
178         List<StringMember> enumAnnotations = filterAnnotations(modelAnnotations,
179                 FHIMMetadataBinding.FHIM_ENUMERATIONS_REFSET, StringMember.class);
180         if (enumAnnotations.isEmpty()) {
181             LOG.info("No FHIM_ENUMERATIONS_REFSET member found.");
182         } else {
183             for (StringMember enumAnnotation : enumAnnotations) {
184                 Enumeration enumeration = buildEnumeration(enumAnnotation);
185                 infoModel.addEnumeration(enumeration);
186             }
187         }
188 
189         // Stub Classes first, they will be used as types.
190         List<StringMember> classAnnotations = filterAnnotations(modelAnnotations,
191                 FHIMMetadataBinding.FHIM_CLASSES_REFSET, StringMember.class);
192         if (classAnnotations.isEmpty()) {
193             LOG.info("No FHIM_CLASSES_REFSET member found.");
194         } else {
195             for (StringMember classAnnotation : classAnnotations) {
196 
197                 // Stub for now, we will flesh out later.
198                 Class clazz = new Class(classAnnotation.getString1());
199 
200                 // Keep track for later.
201                 nidClassMap.put(classAnnotation.getNid(), clazz);
202             }
203         }
204 
205         // Now flesh out Classes.
206         for (StringMember classAnnotation : classAnnotations) {
207             Class c = buildClass(classAnnotation);
208             infoModel.addClass(c);
209         }
210 
211         // Dependencies.
212         List<NidNidStringMember> dependencyAnnotations = filterAnnotations(modelAnnotations,
213                 FHIMMetadataBinding.FHIM_DEPENDENCIES_REFSET, NidNidStringMember.class);
214         if (dependencyAnnotations.isEmpty()) {
215             LOG.info("No FHIM_DEPENDENCIES_REFSET member found.");
216         } else {
217             for (NidNidStringMember dependencyAnnotation : dependencyAnnotations) {
218                 Dependency d = buildDependency(dependencyAnnotation);
219                 infoModel.addDependency(d);
220             }
221         }
222 
223         // Create model-scoped Attributes, which will be used by Associations.
224         Collection<NidStringMember> attributeAnnotations = filterAnnotations(
225                 modelAnnotations,
226                 FHIMMetadataBinding.FHIM_ATTRIBUTES_REFSET,
227                 NidStringMember.class);
228         for (NidStringMember attributeAnnotation : attributeAnnotations) {
229             buildAttribute(attributeAnnotation);
230         }
231 
232         // Associations.
233         List<StringMember> associationAnnotations = filterAnnotations(modelAnnotations,
234                 FHIMMetadataBinding.FHIM_ASSOCIATIONS_REFSET, StringMember.class);
235         if (associationAnnotations.isEmpty()) {
236             LOG.info("No FHIM_ASSOCIATIONS_REFSET member found.");
237         } else {
238             for (StringMember associationAnnotation : associationAnnotations) {
239                 Association a = buildAssociation(associationAnnotation);
240                 infoModel.addAssociation(a);
241             }
242         }
243 
244         return infoModel;
245     }

247     private Association buildAssociation(StringMember associationAnnotation)
248             throws ValidationException, IOException, ContradictionException {
249         String name = associationAnnotation.getString1();
250         Association a = new Association(name);
251         LOG.debug("Association: " + name);
252 
253         // Association Ends.
254         List<NidBooleanMember> associationEndAnnotations = filterAnnotations(
255                 getLatestAnnotations(associationAnnotation),
256                 FHIMMetadataBinding.FHIM_ASSOCIATIONENDS_REFSET, NidBooleanMember.class);
257         if (associationEndAnnotations.isEmpty()) {
258             LOG.info("No FHIM_ASSOCIATIONENDS_REFSET member found.");
259         } else {
260             for (NidBooleanMember associationEndAnnotation : associationEndAnnotations) {
261                 int memberEndNid = associationEndAnnotation.getNid1();
262                 boolean owned = associationEndAnnotation.getBoolean1();
263 
264                 Attribute memberEnd = getAttribute(memberEndNid);
265                 if (memberEnd == null) {
266                     LOG.warn("No memberEnd for " + memberEndNid);
267                 } else {
268                     a.addMemberEnd(memberEnd);
269                     a.setOwned(memberEnd, owned);
270                 }
271             }
272         }
273 
274         return a;
275     }

277     private Dependency buildDependency(NidNidStringMember dependencyAnnotation) {
278         String name = dependencyAnnotation.getStrValue();
279         LOG.debug("Dependency: " + name);
280 
281         // Client.
282         int clientNid = dependencyAnnotation.getC1Nid();
283         Type client = getTypeForNid(clientNid);
284         LOG.debug("    client: " + client.getName());
285 
286         // Supplier.
287         int supplierNid = dependencyAnnotation.getC2Nid();
288         Type supplier = getTypeForNid(supplierNid);
289         LOG.debug("    supplier: " + supplier.getName());
290 
291         return new Dependency(name, client, supplier);

294     private Class buildClass(StringMember classAnnotation)
295             throws ValidationException, IOException, ContradictionException {
296         int nid = classAnnotation.getNid();
297         Class c = getClass(nid);
298         LOG.debug("Class: " + c.getName());
299 
300         // Attributes.
301         Collection<NidStringMember> attributeAnnotations = filterAnnotations(
302                 getLatestAnnotations(classAnnotation),
303                 FHIMMetadataBinding.FHIM_ATTRIBUTES_REFSET,
304                 NidStringMember.class);
305         for (NidStringMember attributeAnnotation : attributeAnnotations) {
306             Attribute a = buildAttribute(attributeAnnotation);
307             c.addAttribute(a);
308         }
309 
310         // Generalizations.
311         Collection<NidMember> generalizationAnnotations = filterAnnotations(
312                 getLatestAnnotations(classAnnotation),
313                 FHIMMetadataBinding.FHIM_GENERALIZATIONS_REFSET,
314                 NidMember.class);
315         for (NidMember generalizationAnnotation : generalizationAnnotations) {
316             Generalization g = buildGeneralization(classAnnotation, generalizationAnnotation);
317             c.addGeneralization(g);
318         }
319 
320         return c;
321     }

339     private Attribute buildAttribute(NidStringMember attributeAnnotation)
340             throws IOException, ContradictionException {
341         String name = attributeAnnotation.getString1();
342 
343         // Type.
344         int typeNid = attributeAnnotation.getNid1();
345         Type type = getTypeForNid(typeNid);
346 
347         Attribute a = new Attribute(name, type);
348         LOG.debug("Attribute: " + a.getName());
349 
350         // Get all annotations on the attribute annotation.
351         Collection<? extends RefexChronicleBI<?>> attributeAnnotations = getLatestAnnotations(attributeAnnotation);
352 
353         // DefaultValue.
354         String defaultValue = getDefaultValue(attributeAnnotations);
355         if (defaultValue != null) {
356             LOG.debug("    defaultValue: " + defaultValue);
357             a.setDefaultValue(defaultValue);
358         }
359 
360         // Multiplicity
361         Multiplicity multiplicity = buildMultiplicity(attributeAnnotations);
362         if (multiplicity != null) {
363             LOG.debug("    multiplicity: " + multiplicity);
364             a.setMultiplicity(multiplicity);
365         }
366 
367         // Visibility
368         VisibilityKind visibility = getVisibility(attributeAnnotations);
369         if (visibility != null) {
370             LOG.debug("    visibility: " + visibility);
371             a.setVisibility(visibility);
372         }
373 
374         // Keep track for later.
375         nidAttributeMap.put(attributeAnnotation.getNid(), a);
376 
377         return a;
378     }

394     private Multiplicity buildMultiplicity(
395             Collection<? extends RefexChronicleBI<?>> attributeAnnotations)
396             throws IOException {
397         List<NidIntMember> multiplicityAnnotations = filterAnnotations(attributeAnnotations,
398                 FHIMMetadataBinding.FHIM_MULTIPLICITY_REFSET, NidIntMember.class);
399 
400         // If no multiplicity annotations, abort.
401         int multiplicityCount = multiplicityAnnotations.size();
402         if (multiplicityCount == 0) {
403             return null;
404         }
405 
406         // Otherwise, expect exactly two multiplicity annotations: upper and lower.
407         if (multiplicityCount != 2) {
408             LOG.warn("Expected 2 multiplicity annotations, found " + multiplicityCount);
409         }
410 
411         int upper = LiteralUnlimitedNatural.UNLIMITED;
412         int lower = LiteralUnlimitedNatural.UNLIMITED;
413 
414         for (NidIntMember multiplicityAnnotation : multiplicityAnnotations) {
415             int nid = multiplicityAnnotation.getC1Nid();
416             if (nid == FHIMMetadataBinding.FHIM_LOWER.getNid()) {
417                 lower = multiplicityAnnotation.getInt1();
418             } else if (nid == FHIMMetadataBinding.FHIM_UPPER.getNid()) {
419                 upper = multiplicityAnnotation.getInt1();
420             } else {
421                 LOG.warn("Unexpected multiplicity metadata: " + multiplicityAnnotation);
422             }
423         }
424 
425         return new Multiplicity(lower, upper);
426     }

428     private String getDefaultValue(
429             Collection<? extends RefexChronicleBI<?>> attributeAnnotations)
430             throws ValidationException, IOException {
431         List<StringMember> defaultValueAnnotations = filterAnnotations(attributeAnnotations,
432                 FHIMMetadataBinding.FHIM_DEFAULTVALUES_REFSET, StringMember.class);
433 
434         // If no default value annotations, abort.
435         int defaultValueCount = defaultValueAnnotations.size();
436         if (defaultValueCount == 0) {
437             return null;
438         }
439 
440         // Otherwise, expect exactly one default value annotation.
441         if (defaultValueCount != 1) {
442             LOG.warn("Expected 1 default value annotation, found " + defaultValueCount);
443         }
444 
445         StringMember defaultValueAnnotation = defaultValueAnnotations.get(0);
446         return defaultValueAnnotation.getString1();
447     }

449     private Type getTypeForNid(int nid) {
450         Type t = null;
451 
452         // Class?
453         t = getClass(nid);
454         if (t != null) {
455             return t;
456         }
457 
458         // Enumeration?
459         t = getEnumeration(nid);
460         if (t != null) {
461             return t;
462         }
463 
464         // Attribute?
465         t = getAttribute(nid);
466         if (t != null) {
467             return t;
468         }
469 
470         // External?
471         t = getExternal(nid);
472         if (t != null) {
473             return t;
474         }
475 
476         // Something wrong!
477         throw new IllegalArgumentException("Could not find type for " + nid);
478     }

496     private Enumeration buildEnumeration(StringMember enumAnnotation)
497             throws IOException, ContradictionException {
498         String name = enumAnnotation.getString1();
499         Enumeration e = new Enumeration(name);
500         LOG.debug("Attribute: " + e.getName());
501 
502         // Enumeration values.
503         Collection<StringMember> valueAnnotations = filterAnnotations(
504                 getLatestAnnotations(enumAnnotation),
505                 FHIMMetadataBinding.FHIM_ENUMERATIONVALUES_REFSET,
506                 StringMember.class);
507         for (StringMember valueAnnotation : valueAnnotations) {
508             String value = valueAnnotation.getString1();
509             e.addLiteral(value);
510         }
511 
512         // Keep track for later.
513         int nid = enumAnnotation.getNid();
514         nidEnumerationMap.put(nid, e);
515 
516         return e;
517     }
