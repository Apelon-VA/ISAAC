21  import gov.va.isaac.gui.util.FxUtils;
22  import gov.va.isaac.models.util.MetadataCreatorBase;

24  import java.io.IOException;
25  import java.util.UUID;
26  
27  import org.ihtsdo.otf.tcc.api.concept.ConceptChronicleBI;
28  import org.ihtsdo.otf.tcc.api.spec.ValidationException;
29  import org.slf4j.Logger;
30  import org.slf4j.LoggerFactory;

32  /**
33   * Class to create FHIM metadata concepts.
34   *
35   * @author ocarlsen
36   */
37  public class FHIMMetadataCreator extends MetadataCreatorBase {
38  
39      private static final Logger LOG = LoggerFactory.getLogger(FHIMMetadataCreator.class);
40  
41      public FHIMMetadataCreator() throws ValidationException, IOException {
42          super();
43      }
44  
45      @Override
46      @SuppressWarnings("unused")
47      public boolean createMetadata() throws Exception {
48  
49          // Make sure NOT in application thread.
50          FxUtils.checkBackgroundThread();
51  
52          // Check if metadata already created.
53          ConceptChronicleBI FHIMRefsets = getConcept(FHIMMetadataBinding.FHIM_REFSET.getUuids()[0]);
54          if (FHIMRefsets != null) {
55              LOG.info("FHIM metadata already created.");
56              return false;
57          }
58  
59          LOG.info("Preparing to create FHIM metadata.");
60  
61          ConceptChronicleBI refsetsRoot = getDataStore().getConcept(UUID.fromString(REFSET_CONCEPT));
62          LOG.debug("Refsets root:" + refsetsRoot.toString());
63  
64          FHIMRefsets = createNewConcept(refsetsRoot, "FHIM reference sets (foundation metadata concept)", "FHIM reference set");
65  
66          ConceptChronicleBI FHIMTypeRefset = createNewConcept(FHIMRefsets, "FHIM type reference set (foundation metadata concept)", "FHIM type reference set");
67          ConceptChronicleBI FHIMCodeRefset = createNewConcept(FHIMRefsets, "FHIM code reference set (foundation metadata concept)", "FHIM code reference set");
68          ConceptChronicleBI FHIMDataRefset = createNewConcept(FHIMRefsets, "FHIM data reference set (foundation metadata concept)", "FHIM data reference set");
69          ConceptChronicleBI FHIMCompositionRefset = createNewConcept(FHIMRefsets, "FHIM composition reference set (foundation metadata concept)", "FHIM composition reference set");
70          ConceptChronicleBI FHIMConstraintsRefset = createNewConcept(FHIMRefsets, "FHIM constraints reference set (foundation metadata concept)", "FHIM constraints reference set");
71          ConceptChronicleBI FHIMConstraintPath = createNewConcept(FHIMRefsets, "FHIM constraints path reference set (foundation metadata concept)", "FHIM constraint path");
72          ConceptChronicleBI FHIMConstraintValue = createNewConcept(FHIMRefsets, "FHIM constraints value reference set (foundation metadata concept)", "FHIM constraint value");
73  
74          ConceptChronicleBI attributesRoot = getDataStore().getConcept(UUID.fromString(REFSET_ATTRIBUTE_CONCEPT));
75          LOG.debug("Attributes root:" + attributesRoot.toString());
76  
77          ConceptChronicleBI FHIMAttributes = createNewConcept(attributesRoot, "FHIM attributes (foundation metadata concept)", "FHIM attributes");
78  
79          ConceptChronicleBI FHIMDataTypes = createNewConcept(FHIMAttributes, "FHIM data types (foundation metadata concept)", "FHIM data types");
80          ConceptChronicleBI FHIMPysicalQuantityDataType = createNewConcept(FHIMDataTypes, "FHIM PysicalQuantity data type (foundation metadata concept)", "FHIM PysicalQuantity data type");
81          // TODO: Other data types as necessary.
82  
83          ConceptChronicleBI FHIMComponentTypes = createNewConcept(FHIMAttributes, "FHIM component types (foundation metadata concept)", "FHIM component types");
84          ConceptChronicleBI FHIMQual = createNewConcept(FHIMComponentTypes, "FHIM qualifier (foundation metadata concept)", "FHIM quaifier");
85          ConceptChronicleBI FHIMMod = createNewConcept(FHIMComponentTypes, "FHIM modifier (foundation metadata concept)", "FHIM modifier");
86          ConceptChronicleBI FHIMAttr = createNewConcept(FHIMComponentTypes, "FHIM attribution (foundation metadata concept)", "FHIM attribution");
87  
88          for (ConceptChronicleBI loopUc : getDataStore().getUncommittedConcepts()) {
89              LOG.debug("Uncommitted concept:" + loopUc.toString() + " - " + loopUc.getPrimordialUuid());
90          }
91  
92          getDataStore().commit();
93  
94          LOG.info("FHIM metadata creation finished.");
95          return true;
96      }
97  }
