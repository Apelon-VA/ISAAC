83      @Override
84      public ConceptChronicleBI importModel(File file) throws Exception {
85          LOG.info("Preparing to import FHIM model from: " + file.getName());
86  
87          // Make sure in background thread.
88          FxUtils.checkBackgroundThread();
89  
90          // Get focus concept.
91          String focusConceptUuid = "215fd598-e21d-3e27-a0a2-8e23b1b36dfc";
92          ConceptChronicleBI focusConcept = getDataStore().getConcept(UUID.fromString(focusConceptUuid));
93          LOG.info("focusConcept: " + focusConcept.toString());
94  
95          // Throw exception if import already performed.
96          ComponentVersionBI latestVersion = focusConcept.getVersion(getVC());
97          Collection<? extends RefexChronicleBI<?>> annotations = latestVersion.getAnnotations();
98          for (RefexChronicleBI<?> annotation : annotations) {
99              Preconditions.checkState(annotation.getAssemblageNid() != FHIMMetadataBinding.FHIM_MODELS_REFSET.getNid(),
100                     "FHIM import has already been performed on " + focusConceptUuid);
101         }
102 
103         // Load UML model from file.
104         String packageName = "VitalSigns";
105         Package umlModel = loadModel(file, packageName);
106 
107         // Abort if not available.
108         if (umlModel == null) {
109             throw new IllegalStateException("No UML Package found: " + packageName);
110         }
111 
112         // Locate "BloodPressure" package.
113         packageName = "BloodPressure";
114         Package bloodPressurePackage = umlModel.getNestedPackage(packageName);
115 
116         // Abort if not available.
117         if (bloodPressurePackage == null) {
118             throw new IllegalStateException("No UML Package found: " + packageName);
119         }
120 
121         // Parse into FHIM model.
122         UML2ModelConverter converter = new UML2ModelConverter();
123         FHIMInformationModel infoModel = converter.createInformationModel(bloodPressurePackage);
124 
125         // Annotate focusConcept with Refset members.
126         annotateWithRefsets(focusConcept, infoModel);
127 
128         getDataStore().addUncommitted(focusConcept);
129         getDataStore().commit();
130 
131         LOG.debug("Long form after commit:" + focusConcept.toLongString());
132         LOG.info("Ending import of CEM model from: " + file.getName());
133 
134         return focusConcept;
135     }

137     private void annotateWithRefsets(ComponentChronicleBI<?> focusComponent,
138             FHIMInformationModel infoModel)
139             throws IOException, InvalidCAB, ContradictionException {
140 
141         // FHIM Models refset.
142         String modelName = infoModel.getName();
143         LOG.debug("Adding refex for model: " + modelName);
144         RefexChronicleBI<?> modelRefex = addRefexInStrExtensionRefset(focusComponent,
145                 FHIMMetadataBinding.FHIM_MODELS_REFSET, modelName);
146 
147         // Persist Classes and Enumerations first, they will be used as types later.
148 
149         // FHIM Enumerations refset.
150         List<Enumeration> enumerations = infoModel.getEnumerations();
151         for (Enumeration enumeration : enumerations) {
152             RefexChronicleBI<?> enumRefex = addEnumerationsRefsetMember(
153                     modelRefex, enumeration);
154 
155             // FHIM EnumerationValues refset.
156             List<String> values = enumeration.getLiterals();
157             for (String value : values) {
158                 addEnumerationValuesRefsetMember(enumRefex, value);
159             }
160         }
161 
162         // FHIM Classes refset.
163         List<Class> classes = infoModel.getClasses();
164         for (Class clazz : classes) {
165             addClassesRefsetMember(modelRefex, clazz);
166         }
167 
168         // Now go back through and finish the Generalizations & Attributes.
169         for (Class clazz : classes) {
170             RefexChronicleBI<?> classRefex = Preconditions.checkNotNull(getClassRefex(clazz));
171 
172             // FHIM Generalizations refset.
173             List<Generalization> generalizations = clazz.getGeneralizations();
174             for (Generalization generalization : generalizations) {
175                 addGeneralizationsRefsetMember(classRefex, generalization);
176             }
177 
178             // FHIM Attributes refset.
179             List<Attribute> attributes = clazz.getAttributes();
180             for (Attribute attribute : attributes) {
181                 annotateWithAttributeRefsets(classRefex, attribute);
182             }
183         }
184 
185         // Dependencies.
186         List<Dependency> dependencies = infoModel.getDependencies();
187         for (Dependency dependency : dependencies) {
188             addDependenciesRefsetMember(modelRefex, dependency);
189         }
190 
191         // Associations.
192         List<Association> associations = infoModel.getAssociations();
193         for (Association association : associations) {
194             RefexChronicleBI<?> associationRefex = addAssociationsRefsetMember(modelRefex, association);
195 
196             // Association Ends.
197             for (Attribute memberEnd : association.getMemberEnds()) {
198                 RefexChronicleBI<?> memberEndRefex = getAttributeRefex(memberEnd);
199 
200                 // Sometimes the owned ends are orphaned and no attribute refset member
201                 // would have been created by this point.  Create one now.
202                 if (memberEndRefex == null) {
203                     memberEndRefex = annotateWithAttributeRefsets(modelRefex, memberEnd);
204                 }
205 
206                 boolean owned = association.isOwned(memberEnd);
207                 addAssociationEndsRefsetMember(associationRefex, memberEnd, owned);
208             }
209         }
210     }

212     private RefexChronicleBI<?> annotateWithAttributeRefsets(RefexChronicleBI<?> focusComponent,
213             Attribute attribute) throws ValidationException, IOException,
214             InvalidCAB, ContradictionException {
215         RefexChronicleBI<?> attributeRefex = addAttributesRefsetMember(
216                 focusComponent, attribute);
217 
218         // FHIM DefaultValue refset.
219         String defaultValue = attribute.getDefaultValue();
220         if (defaultValue != null) {
221             addDefaultValueRefsetMember(attributeRefex, defaultValue);
222         }
223 
224         // FHIM Multiplicity refset.
225         Multiplicity multiplicity = attribute.getMultiplicity();
226         if (multiplicity != null) {
227             int lower = multiplicity.getLower();
228             addMultiplicityRefsetMember(attributeRefex, FHIMMetadataBinding.FHIM_LOWER, lower);
229             int upper = multiplicity.getUpper();
230             addMultiplicityRefsetMember(attributeRefex, FHIMMetadataBinding.FHIM_UPPER, upper);
231         }
232 
233         // FHIM Visibility refset.
234         VisibilityKind visibility = attribute.getVisibility();
235         if (visibility != null) {
236             addVisibilityRefsetMember(attributeRefex, visibility);
237         }
238 
239         return attributeRefex;
240     }

242     private RefexChronicleBI<?> addAssociationEndsRefsetMember(RefexChronicleBI<?> focusComponent,
243             Attribute memberEnd, boolean owned)
244             throws IOException, InvalidCAB, ContradictionException {
245         String memberEndName = memberEnd.getName();
246         LOG.debug("Adding refex for memberEnd: " + memberEndName);
247 
248         RefexChronicleBI<?> memberEndRefex = getAttributeRefex(memberEnd);
249         int memberEndNid = memberEndRefex.getNid();
250 
251         return addRefexInCidBooleanExtensionRefset(focusComponent,
252                 FHIMMetadataBinding.FHIM_ASSOCIATIONENDS_REFSET, memberEndNid, owned);
253     }

268         String dependencyName = dependency.getName();
269         LOG.debug("Adding refex for dependency: " + dependencyName);
270 
271         Type client = dependency.getClient();
272         Type supplier = dependency.getSupplier();
273 
274         // Need to find appropriate NIDs.
275         int clientNid = getNidForType(client);

277 
278         return addRefexInCidCidStrExtensionRefset(focusComponent,
279                 FHIMMetadataBinding.FHIM_DEPENDENCIES_REFSET, clientNid, supplierNid, dependencyName);

381     private int getNidForType(Type type) throws ValidationException,
382             IOException {
383         if (type instanceof External) {
384             return ((External) type).getConceptSpec().getNid();
385         } else if (type instanceof Class) {
386             Class c = (Class) type;
387             RefexChronicleBI<?> typeRefex = Preconditions.checkNotNull(getClassRefex(c));
388             return typeRefex.getNid();
389         } else if (type instanceof Enumeration) {
390             Enumeration e = (Enumeration) type;
391             RefexChronicleBI<?> typeRefex = Preconditions.checkNotNull(getEnumerationRefex(e));
392             return typeRefex.getNid();
393         } else if (type instanceof Attribute) {
394             Attribute a = (Attribute) type;
395             RefexChronicleBI<?> attributeRefex = Preconditions.checkNotNull(getAttributeRefex(a));
396             return attributeRefex.getNid();
397         } else {
398             throw new IllegalStateException("Unexpected target: " + type.getClass());
399         }
400     }

414     private Package loadModel(File file, String packageName) throws IOException {
415 
416         URI fileURI = URI.createFileURI(file.getAbsolutePath());
417 
418         // Create a resource-set to contain the resource(s) that we are saving
419         ResourceSet resourceSet = new ResourceSetImpl();
420 
421         // Initialize registrations of resource factories, library models,
422         // profiles, Ecore metadata, and other dependencies required for
423         // serializing and working with UML resources. This is only necessary in
424         // applications that are not hosted in the Eclipse platform run-time, in
425         // which case these registrations are discovered automatically from
426         // Eclipse extension points.
427         UMLResourcesUtil.init(resourceSet);
428 
429         Resource resource = resourceSet.createResource(fileURI);
430 
431         // And load.
432         Map<?, ?> options = null;   // No load options needed.
433         resource.load(options);
434 
435         // Look for Package with specified name.
436         EList<EObject> contents = resource.getContents();
437         for (EObject content : contents) {
438             if (content instanceof Package) {
439                 Package model = (Package) content;
440                 if (model.getName().equals(packageName)) {
441                     LOG.info("Loaded '" + model.getQualifiedName() + "' from '" + fileURI + "'.");
442                     return model;
443                 }
444             }
445         }
446 
447         return null;
448     }
