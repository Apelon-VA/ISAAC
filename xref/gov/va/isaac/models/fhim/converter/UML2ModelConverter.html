71      public FHIMInformationModel createInformationModel(Package pkg) {
72          FHIMInformationModel infoModel = new FHIMInformationModel(pkg.getName(), null);
73  
74          // Gather Classes & Enumerations first, they will be used as types later.
75          EList<PackageableElement> elements = pkg.getPackagedElements();
76          for (Iterator<PackageableElement> i = elements.iterator(); i.hasNext();) {
77              PackageableElement element = i.next();
78              if (element instanceof Enumeration) {
79                  Enumeration enumeration = (Enumeration) element;
80                  LOG.debug("Enumeration: " + enumeration.getName());
81  
82                  FHIMInformationModel.Enumeration enumerationModel = createEnumerationModel(enumeration);
83                  infoModel.addEnumeration(enumerationModel);
84  
85                  // Keep track of it for re-use later.
86                  nameEnumerationMap.put(enumerationModel.getName(), enumerationModel);
87  
88                  // Remove from packaged elements, it is fully modeled.
89                  i.remove();
90              } else if (element instanceof Class) {
91                  Class clazz = (Class) element;
92  
93                  // Stub for now, we will flesh out later.
94                  FHIMInformationModel.Class classModel = new FHIMInformationModel.Class(clazz.getName());
95  
96                  // Keep track of it for re-use later.
97                  nameClassMap.put(classModel.getName(), classModel);
98              }
99          }
100 
101         // Associations will create some Attributes that are shared by Classes.
102         for (Iterator<PackageableElement> i = elements.iterator(); i.hasNext();) {
103             PackageableElement element = i.next();
104             if (element instanceof Association) {
105                 Association association = (Association) element;
106                 FHIMInformationModel.Association associationModel = buildAssociationModel(association);
107                 infoModel.addAssociation(associationModel);
108 
109                 // Remove from packaged elements, it is fully modeled.
110                 i.remove();
111             }
112         }
113 
114         // Now flesh out Classes, it will create the remaining Attributes.
115         for (Iterator<PackageableElement> i = elements.iterator(); i.hasNext();) {
116             PackageableElement element = i.next();
117             if (element instanceof Class) {
118                 Class clazz = (Class) element;
119                 FHIMInformationModel.Class classModel = buildClassModel(clazz);
120                 infoModel.addClass(classModel);
121 
122                 // Remove from packaged elements, it is fully modeled.
123                 i.remove();
124             }
125         }
126 
127         // Dependencies.
128         for (Iterator<PackageableElement> i = elements.iterator(); i.hasNext();) {
129             PackageableElement element = i.next();
130             if (element instanceof Dependency) {
131                 Dependency dependency = (Dependency) element;
132                 FHIMInformationModel.Dependency dependencyModel = buildDependencyModel(dependency);
133                 infoModel.addDependency(dependencyModel);
134 
135                 // Remove from packaged elements, it is fully modeled.
136                 i.remove();
137             }
138         }
139 
140         // Shouldn't be any elements left.
141         for (PackageableElement element : elements) {
142             LOG.warn("Unrecognized element: " + element);
143         }
144 
145         return infoModel;
146     }

148     private FHIMInformationModel.Association buildAssociationModel(Association association) {
149         String name = association.getName();
150         FHIMInformationModel.Association associationModel = new FHIMInformationModel.Association(name);
151         LOG.debug("Association: " + name);
152 
153         // Expect binary associations.
154         Preconditions.checkArgument(association.isBinary());
155 
156         // "Owned" ends.
157         EList<Property> ownedEnds = association.getOwnedEnds();
158 
159         // Member ends.
160         EList<Property> memberEnds = association.getMemberEnds();
161         for (Property memberEnd : memberEnds) {
162             LOG.debug("    memberEnd: " + memberEnd.getName());
163             FHIMInformationModel.Attribute memberEndModel = getAttributeModel(memberEnd);
164             associationModel.addMemberEnd(memberEndModel);
165 
166             boolean owned = ownedEnds.contains(memberEnd);
167             LOG.debug("    owned: " + owned);
168             associationModel.setOwned(memberEndModel, owned);
169         }
170 
171         return associationModel;
172     }

174     private FHIMInformationModel.Dependency buildDependencyModel(Dependency dependency) {
175         String name = dependency.getName();
176         LOG.debug("Dependency: " + name);
177 
178         // Expect exactly one client.
179         EList<NamedElement> clients = dependency.getClients();
180         int clientCount = clients.size();
181         if (clientCount != 1) {
182             LOG.warn("Expected 1 client, found " + clientCount);
183         }
184         NamedElement client = clients.get(0);
185         String clientName = client.getName();
186         LOG.debug("    client: " + clientName);
187         FHIMInformationModel.Type clientModel = getTypeModel(clientName);
188 
189         // Expect exactly one supplier.
190         EList<NamedElement> suppliers = dependency.getSuppliers();
191         int supplierCount = suppliers.size();
192         if (supplierCount != 1) {
193             LOG.warn("Expected 1 supplier, found " + supplierCount);
194         }
195         NamedElement supplier = suppliers.get(0);
196         String supplierName = supplier.getName();
197         LOG.debug("    supplier: " + supplierName);
198         FHIMInformationModel.Type supplierModel = getTypeModel(supplierName);
199 
200         return new FHIMInformationModel.Dependency(name, clientModel, supplierModel);
201     }

203     private FHIMInformationModel.Class buildClassModel(Class clazz) {
204         String name = clazz.getName();
205         LOG.debug("Class: " + name);
206 
207         FHIMInformationModel.Class classModel = nameClassMap.get(name);
208 
209         // Generalizations.
210         EList<Generalization> generalizations = clazz.getGeneralizations();
211         for (Generalization generalization : generalizations) {
212             FHIMInformationModel.Generalization generalizationModel = createGeneralizationModel(generalization);
213             classModel.addGeneralization(generalizationModel);
214         }
215 
216         // Attributes.
217         EList<Property> attributes = clazz.getOwnedAttributes();
218         for (Property attribute : attributes) {
219             FHIMInformationModel.Attribute attributeModel = getAttributeModel(attribute);
220             classModel.addAttribute(attributeModel);
221         }
222 
223         return classModel;
224     }

226     private FHIMInformationModel.Generalization createGeneralizationModel(Generalization generalization) {
227         LOG.debug("    generalization: ");
228 
229         // Expect exactly one source of type Class.
230         EList<Element> sources = generalization.getSources();
231         int sourceCount = sources.size();
232         if (sourceCount != 1) {
233             LOG.warn("Expected 1 source, found " + sourceCount);
234         }
235         Class source = (Class) sources.get(0);
236         String sourceName = source.getName();
237         LOG.debug("        source: " + sourceName);
238         FHIMInformationModel.Type sourceModel = getTypeModel(sourceName);
239 
240         // Expect exactly one target of type Class.
241         EList<Element> targets = generalization.getTargets();
242         int targetCount = targets.size();
243         if (targetCount != 1) {
244             LOG.warn("Expected 1 target, found " + targetCount);
245         }
246         Class target = (Class) targets.get(0);
247         String targetName = target.getName();
248         LOG.debug("        target: " + targetName);
249         FHIMInformationModel.Type targetModel = getTypeModel(targetName);
250 
251         return new FHIMInformationModel.Generalization(sourceModel, targetModel);
252     }

254     private Attribute getAttributeModel(Property property) {
255         Attribute attribute = propertyAttributeMap.get(property);
256         if (attribute == null) {
257             attribute = createAttributeModel(property);
258             propertyAttributeMap.put(property, attribute);
259         } else {
260             LOG.trace("Cache hit: " + property);
261         }
262         return attribute;
263     }

265     private Attribute createAttributeModel(Property property) {
266         String name = property.getName();
267         LOG.debug("Attribute: " + name);
268 
269         // Attribute type.
270         Type type = property.getType();
271         String typeName = type.getName();
272         FHIMInformationModel.Type typeModel = getTypeModel(typeName);
273 
274         Attribute attributeModel = new Attribute(name, typeModel);
275 
276         // Default value.
277         ValueSpecification valueSpec = property.getDefaultValue();
278         String defaultValue = (valueSpec != null ? valueSpec.stringValue() : null);
279         if (defaultValue != null) {
280             attributeModel.setDefaultValue(defaultValue);
281         }
282 
283         // Multiplicity.
284         int lower = property.getLower();
285         int upper = property.getUpper();
286         Multiplicity multiplicity = new Multiplicity(lower, upper);
287         attributeModel.setMultiplicity(multiplicity);
288 
289         // Visibility.
290         VisibilityKind visibility = property.getVisibility();
291         attributeModel.setVisibility(visibility);
292 
293         return attributeModel;
294     }

296     private FHIMInformationModel.Enumeration createEnumerationModel(Enumeration enumeration) {
297         String name = enumeration.getName();
298         FHIMInformationModel.Enumeration enumerationModel = new FHIMInformationModel.Enumeration(name);
299 
300         EList<EnumerationLiteral> literals = enumeration.getOwnedLiterals();
301         for (EnumerationLiteral literal : literals) {
302             String literalName = literal.getName();
303             LOG.debug("    literal: " + literalName);
304             enumerationModel.addLiteral(literalName);
305         }
306         return enumerationModel;
307     }

309     private FHIMInformationModel.Type getTypeModel(String typeName) {
310         FHIMInformationModel.Type modelType = null;
311 
312         // Try Enumerations first.
313         modelType = nameEnumerationMap.get(typeName);
314         if (modelType != null) {
315             return modelType;
316         }
317 
318         // Try Classes next.
319         modelType = nameClassMap.get(typeName);
320         if (modelType != null) {
321             return modelType;
322         }
323 
324         // Try OTF metadata concept type.
325         modelType = nameExternalMap.get(typeName);
326         if (modelType != null) {
327             return modelType;
328         }
329 
330         // Something wrong!
331         throw new IllegalArgumentException("Could not find type for '" + typeName + "'");
332     }
