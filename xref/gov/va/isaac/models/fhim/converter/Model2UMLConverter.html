86      public Package createUMLModel(FHIMInformationModel infoModel) {
87          String name = infoModel.getName();
88          Package pkg = vitalSignsPkg.createNestedPackage(name);
89  
90          // Enumerations.
91          for (FHIMInformationModel.Enumeration enumerationModel : infoModel.getEnumerations()) {
92              Enumeration e = createEnumeration(pkg, enumerationModel);
93  
94              // Keep track of it for re-use later.
95              modelEnumerationMap.put(enumerationModel, e);
96          }
97  
98          // Stub Classes, they will be used as types.
99          for (FHIMInformationModel.Class classModel : infoModel.getClasses()) {
100 
101             // Stub for now, we will flesh out later.
102             String className = classModel.getName();
103             Class c = pkg.createOwnedClass(className, false);  // Not abstract.
104 
105             // Keep track of it for re-use later.
106             modelClassMap.put(classModel, c);
107         }
108 
109         // Associations will create some Properties that are shared by Classes.
110         for (FHIMInformationModel.Association asociationModel : infoModel.getAssociations()) {
111             createAssociation(pkg, asociationModel);
112         }
113 
114         // Now flesh out Classes.
115         for (FHIMInformationModel.Class classModel : infoModel.getClasses()) {
116             createClass(pkg, classModel);
117         }
118 
119         // Dependencies.
120         for (FHIMInformationModel.Dependency dependencyModel : infoModel.getDependencies()) {
121             createDependency(pkg, dependencyModel);
122         }
123 
124         return vitalSignsPkg;
125     }

127     private Association createAssociation(Package pkg, FHIMInformationModel.Association associationModel) {
128         String name = associationModel.getName();
129         LOG.debug("Association: " + name);
130 
131         // To create a binary Association, we need two ends.
132         // Use the "owned" end to create an Association to the "unowned" end.
133         // Expect one of each.
134         Attribute ownedEndModel = null;
135         Attribute unownedEndModel = null;
136         for (FHIMInformationModel.Attribute memberEnd : associationModel.getMemberEnds()) {
137             boolean owned = associationModel.isOwned(memberEnd);
138             if (owned) {
139                 if (ownedEndModel == null) {
140                     ownedEndModel = memberEnd;
141                 } else {
142                     LOG.warn("Expected one ownedEndModel, found another: " + ownedEndModel);
143                 }
144             } else {
145                 if (unownedEndModel == null) {
146                     unownedEndModel = memberEnd;
147                 } else {
148                     LOG.warn("Expected one unownedEndModel, found another: " + unownedEndModel);
149                 }
150             }
151         }
152 
153         // Sanity check.
154         Preconditions.checkNotNull(ownedEndModel, "Expected to find an ownedEndModel!");
155         Preconditions.checkNotNull(unownedEndModel, "Expected to find an unownedEndModel!");
156 
157         // Use "owned" end for end1.
158         boolean end1IsNavigable = false;  // false=owned
159         AggregationKind end1Aggregation = AggregationKind.NONE_LITERAL;
160         String end1Name = ownedEndModel.getName();
161         int end1Lower = ownedEndModel.getMultiplicity().getLower();
162         int end1Upper = ownedEndModel.getMultiplicity().getUpper();
163         Type end1Type = getTypeForModel(ownedEndModel.getType());
164 
165         // Use "unowned" end for end2.
166         boolean end2IsNavigable = true;  // true=unowned
167         AggregationKind end2Aggregation = AggregationKind.NONE_LITERAL;
168         String end2Name = unownedEndModel.getName();
169         int end2Lower = unownedEndModel.getMultiplicity().getLower();
170         int end2Upper = unownedEndModel.getMultiplicity().getUpper();
171         Type end2Type = getTypeForModel(unownedEndModel.getType());
172 
173         // The Type#createAssociation API is awkward...
174         Association association = end1Type.createAssociation(
175                 end2IsNavigable, end2Aggregation, end2Name, end2Lower, end2Upper,
176                 end2Type, end1IsNavigable, end1Aggregation, end1Name, end1Lower, end1Upper);
177 
178         // Two Properties should have been created, one "owned" and one "unowned".
179         EList<Property> ownedEnds = association.getOwnedEnds();
180         EList<Property> memberEnds = association.getMemberEnds();
181         Preconditions.checkState(ownedEnds.size() == 1);
182         Preconditions.checkState(memberEnds.size() == 2);
183 
184         // Save the Properties that were created for later.
185         for (Property memberEnd : memberEnds) {
186             if (ownedEnds.contains(memberEnd)) {
187                 setDefaultValue(ownedEndModel, memberEnd);
188                 setVisibility(ownedEndModel, memberEnd);
189                 modelPropertyMap.put(ownedEndModel, memberEnd);
190             } else {
191                 setDefaultValue(unownedEndModel, memberEnd);
192                 setVisibility(unownedEndModel, memberEnd);
193                 modelPropertyMap.put(unownedEndModel, memberEnd);
194             }
195         }
196 
197         return association;
198     }

200     private Dependency createDependency(Package pkg, FHIMInformationModel.Dependency dependencyModel) {
201         String name = dependencyModel.getName();
202         LOG.debug("Dependency: " + name);
203 
204         // Client.
205         FHIMInformationModel.Type clientModel = dependencyModel.getClient();
206         Type client = getTypeForModel(clientModel);
207         LOG.debug("    client: " + client.getName());
208 
209         // Supplier.
210         FHIMInformationModel.Type supplierModel = dependencyModel.getSupplier();
211         Type supplier = getTypeForModel(supplierModel);
212         LOG.debug("    supplier: " + supplier.getName());
213 
214         Dependency dependency = client.createDependency(supplier);
215         dependency.setName(name);
216 
217         return dependency;
218     }

220     private Class createClass(Package pkg, FHIMInformationModel.Class classModel) {
221         Class clazz = modelClassMap.get(classModel);
222 
223         LOG.debug("Class: " + clazz.getName());
224 
225         // Attributes.
226         for (Attribute attributeModel : classModel.getAttributes()) {
227             getProperty(clazz, attributeModel);
228         }
229 
230         // Generalizations.
231         for (FHIMInformationModel.Generalization generalizationModel : classModel.getGeneralizations()) {
232             createGeneralization(clazz, generalizationModel);
233         }
234 
235         return clazz;
236     }

238     private Generalization createGeneralization(Class clazz,
239             FHIMInformationModel.Generalization generalizationModel) {
240         FHIMInformationModel.Type typeModel = generalizationModel.getTarget();
241         Type type = getTypeForModel(typeModel);
242 
243         // Expect type to be an instance of Class.
244         if (type instanceof Class) {
245             Class general = (Class) type;
246 
247             LOG.debug("Generalization: " + general.getName());
248 
249             return clazz.createGeneralization(general);
250         }
251 
252         // Something wrong!
253         throw new IllegalArgumentException("Unexpected type: " + type);
254     }

256     private Property getProperty(Class clazz, Attribute attributeModel) {
257         Property property = modelPropertyMap.get(attributeModel);
258         if (property == null) {
259             property = createProperty(clazz, attributeModel);
260             modelPropertyMap.put(attributeModel, property);
261         } else {
262             LOG.trace("Cache hit: " + attributeModel);
263         }
264         return property;

267     private Property createProperty(Class clazz, Attribute attributeModel) {
268         String name = attributeModel.getName();
269 
270         // Type.
271         FHIMInformationModel.Type typeModel = attributeModel.getType();
272         Type type = getTypeForModel(typeModel);
273         Property property = clazz.createOwnedAttribute(name, type);
274 
275         LOG.debug("Property: " + property.getName());
276 
277         // DefaultValue.
278         setDefaultValue(attributeModel, property);
279 
280         // Multiplicity.
281         setMultiplicity(attributeModel, property);
282 
283         // Visibility.
284         setVisibility(attributeModel, property);
285 
286         return property;
287     }

298         FHIMInformationModel.Multiplicity multiplicity = attributeModel.getMultiplicity();

300             LOG.debug("    multiplicity: " + multiplicity);
301             int upper = multiplicity.getUpper();

348     private Type getTypeForModel(FHIMInformationModel.Type typeModel) {
349         Type type = null;
350 
351         // Try Enumerations first.
352         type = modelEnumerationMap.get(typeModel);
353         if (type != null) {
354             return type;
355         }
356 
357         // Try Classes next.
358         type = modelClassMap.get(typeModel);
359         if (type != null) {
360             return type;
361         }
362 
363         // Try OTF metadata concept type.
364         if (typeModel instanceof External) {
365             ConceptSpec conceptSpec = ((External) typeModel).getConceptSpec();
366             type = conceptSpecClassMap.get(conceptSpec);
367             if (type != null) {
368                 return type;
369             }
370         }
371 
372         // Something wrong!
373         throw new IllegalArgumentException("Could not find type for '" + typeModel + "'");
374     }
