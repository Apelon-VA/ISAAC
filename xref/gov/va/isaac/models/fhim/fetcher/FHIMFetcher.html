37  /**
38   * Class for fetching FHIM models.
39   *
40   * @author ocarlsen
41   */
42  public class FHIMFetcher extends ExporterBase {
43  
44      private static final Logger LOG = LoggerFactory.getLogger(FHIMFetcher.class);
45  
46      public FHIMFetcher() {
47          super();
48      }
49  
50      public FHIMInformationModel fetchFHIMModel(UUID focusConceptUUID) throws Exception {
51          LOG.info("Starting fetch of FHIM model type");
52          LOG.debug("focusConceptUUID="+focusConceptUUID);
53  
54          // Make sure NOT in application thread.
55          FxUtils.checkBackgroundThread();
56  
57          // Get chronicle for concept.
58          ConceptChronicleBI focusConcept = getDataStore().getConcept(focusConceptUUID);
59          LOG.debug("focusConcept="+focusConcept);
60  
61          // Get all annotations on the specified concept.
62          Collection<? extends RefexChronicleBI<?>> focusConceptAnnotations =
63                  getLatestAnnotations(focusConcept);
64  
65          // FHIM Models refset (1).
66          StringMember modelAnnotation = getSingleAnnotation(focusConceptAnnotations,
67                  FHIMMetadataBinding.FHIM_MODELS_REFSET, StringMember.class);
68  
69          // Abort if not found.
70          if (modelAnnotation == null) {
71              return  null;
72          }
73  
74          String modelName = modelAnnotation.getString1();
75          FHIMInformationModel informationModel = new FHIMInformationModel(modelName);
76  
77          Metadata metadata = Metadata.newInstance(modelAnnotation.getStamp(),
78                  getDataStore(), getVC());
79          informationModel.setMetadata(metadata);
80  
81          ConceptVersionBI focusConceptVersion = focusConcept.getVersion(getVC());
82          String focusConceptName = focusConceptVersion.getFullySpecifiedDescription().getText();
83          informationModel.setFocusConceptName(focusConceptName);
84          informationModel.setFocusConceptUUID(focusConceptUUID);
85  
86          LOG.debug("informationModel="+informationModel);
87          LOG.info("Ending fetch of FHIM model type");
88  
89          return informationModel;
90      }
91  
92      @Override
93      protected Logger getLogger() {
94          return LOG;
95      }
96  }
