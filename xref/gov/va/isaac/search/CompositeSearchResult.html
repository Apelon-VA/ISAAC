19  package gov.va.isaac.search;
20  
21  import gov.va.isaac.util.OTFUtility;
22  import java.util.ArrayList;
23  import java.util.Collections;
24  import java.util.HashSet;
25  import java.util.List;
26  import java.util.Set;
27  import org.ihtsdo.otf.tcc.api.chronicle.ComponentVersionBI;
28  import org.ihtsdo.otf.tcc.api.concept.ConceptVersionBI;
29  import org.ihtsdo.otf.tcc.api.description.DescriptionAnalogBI;
30  
31  /**
32   * Encapsulates a data store search result.
33   * <p>
34   * Logic has been mostly copied from LEGO {@code SnomedSearchResult}.
35   *
36   * @author ocarlsen
37   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
38   */
39  public class CompositeSearchResult {
40  
41  	private ConceptVersionBI containingConcept = null;
42  	private final Set<ComponentVersionBI> matchingComponents = new HashSet<>();
43  	private int matchingComponentNid;
44  	private float bestScore; // best score, rather than score, as multiple matches may go into a SearchResult
45  
46  	public CompositeSearchResult(ComponentVersionBI matchingComponent, float score) {
47  		this.matchingComponents.add(matchingComponent);
48  		this.bestScore = score;
49  		//matchingComponent may be null, if the match is not on our view path...
50  		if (matchingComponent == null)
51  		{
52  			throw new RuntimeException("Please call the constructor that takes a nid, if matchingComponent is null...");
53  		}
54  		this.containingConcept = OTFUtility.getConceptVersion(matchingComponent.getConceptNid());
55  	}
56  	public CompositeSearchResult(int matchingComponentNid, float score) {
57  		this.bestScore = score;
58  		//matchingComponent may be null, if the match is not on our view path...
59  		this.containingConcept = null;
60  		this.matchingComponentNid = matchingComponentNid;
61  		
62  	}
63  	
64  	protected void adjustScore(float newScore) {
65  		bestScore = newScore;
66  	}
67  
68  	public float getBestScore() {
69  		return bestScore;
70  	}
71  	
72  	/**
73  	 * This may return null, if the concept and/or matching component was not on the path
74  	 */
75  	public ConceptVersionBI getContainingConcept() {
76  		return containingConcept;
77  	}
78  
79  	/**
80  	 * A convenience method to get string values from the matching Components
81  	 */
82  	public List<String> getMatchingStrings() {
83  		ArrayList<String> strings = new ArrayList<>();
84  		if (matchingComponents.size() == 0)
85  		{
86  			if (containingConcept == null)
87  			{
88  				strings.add("Match to NID (not on path):" + matchingComponentNid);
89  			}
90  			else
91  			{
92  				throw new RuntimeException("Unexpected");
93  			}
94  		}
95  		for (ComponentVersionBI cc : matchingComponents)
96  		{
97  			if (cc instanceof DescriptionAnalogBI)
98  			{
99  				strings.add(((DescriptionAnalogBI<?>) cc).getText());
100 			}
101 			else if (cc instanceof ConceptVersionBI)
102 			{
103 				//This means they matched on a UUID or other ID lookup.
104 				//Return UUID for now - matches on other ID types will be handled differently 
105 				//in the near future - so ignore the SCTID case for now.
106 				strings.add(cc.getPrimordialUuid().toString());
107 			}
108 			else
109 			{
110 				strings.add("ERROR: No string extractor available for " + cc.getClass().getName());
111 			}
112 		}
113 		return strings;
114 	}
115 
116 	public Set<ComponentVersionBI> getMatchingComponents() {
117 		return matchingComponents;
118 	}
119 	
120 	/**
121 	 * Convenience method to return a filtered list of matchingComponents such that it only returns
122 	 * Description type components
123 	 */
124 	public Set<DescriptionAnalogBI<?>> getMatchingDescriptionComponents() {
125 		Set<DescriptionAnalogBI<?>> setToReturn = new HashSet<>();
126 		for (ComponentVersionBI comp : matchingComponents) {
127 			if (comp instanceof DescriptionAnalogBI) {
128 				setToReturn.add((DescriptionAnalogBI<?>)comp);
129 			}
130 		}
131 		
132 		return Collections.unmodifiableSet(setToReturn);
133 	}
134 	
135 	protected void merge(CompositeSearchResult other)
136 	{
137 		if (containingConcept.getNid() !=  other.containingConcept.getNid())
138 		{
139 			throw new RuntimeException("Unmergeable!");
140 		}
141 		if (other.bestScore > bestScore)
142 		{
143 			bestScore = other.bestScore;
144 		}
145 		matchingComponents.addAll(other.getMatchingComponents());
146 	}
147 }
