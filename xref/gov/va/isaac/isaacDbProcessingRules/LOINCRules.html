54  /**
55   * {@link LOINCRules}
56   * 
57   * A Transformer that implements various rules for LOINC transformations.
58   * 
59   * See docs/initial LOINC Rules.xlsx for the details on these that have been implemented so far.
60   *
61   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
62   */
63  @Service
64  @Named(value = "LOINC representation rules")
65  public class LOINCRules implements TransformConceptIterateI
66  {
67  	private AtomicInteger generatedRels = new AtomicInteger();
68  	private AtomicInteger mergedConcepts = new AtomicInteger();
69  	private AtomicInteger examinedConcepts = new AtomicInteger();
70  	
71  	private HashMap<Integer, AtomicInteger> ruleHits = new HashMap<>();
72  	
73  	//private final Logger log = LoggerFactory.getLogger(this.getClass());

76  	private int loincPathNid;
77  	private ViewCoordinate vc;
78

80  	//SNOMED integer id	108252007	Laboratory procedure
81  	private final UUID sctLabProcedure = UUID.fromString("42bd9b4b-f6da-39d9-b6a7-bb0d4690d6f9");
82  	
83  	//SNOMED integer id	489004	Ferritin measurement
84  	private final UUID sctFerritinMeasurement = UUID.fromString("0fcef5eb-8229-3edf-87ef-d71927ae4d20");
85  	
86  	//SNOMED integer id	121278003	Drug measurement
87  	private final UUID sctDrugMeasurement = UUID.fromString("4ddf00b1-dc17-38b1-a445-2eecbc53913e");
88  	
89  	//SNOMED integer id	388469008	Ampicillin RAST
90  	private final UUID sctAmpicillin = UUID.fromString("7f798b0f-ceab-3c3f-b55a-17538aaa82ef");
91  	
92  	//Ampicillin IgE RAST Ql - 15533-3
93  	private final UUID loincAmpicillin = UUID.fromString("e284b1f7-61bf-5ca0-b2a2-f0fd838e064b");
94  	
95  	//CLASSTYPE
96  	private final UUID classtype = UUID.fromString("537869e6-a36e-5bd5-8e5b-dad90e9f4015");

99  	private final UUID orderObs = UUID.fromString("a77932ee-55ea-56c6-9c7d-e93ccf6620a7");

102 	private final  UUID hasComponent = UUID.fromString("481bb791-103a-5216-946c-b630aa95d322");

116 	/**
117 	 * @throws IOException 
118 	 * @see gov.va.isaac.mojos.dbTransforms.TransformI#configure(java.io.File, org.ihtsdo.otf.tcc.api.store.TerminologyStoreDI)
119 	 */
120 	@Override
121 	public void configure(File configFile, TerminologyStoreDI ts) throws IOException
122 	{
123 		loincPathNid = ts.getConcept(LOINC_PATH).getNid();
124 		classTypeNid = ts.getConcept(classtype).getNid();
125 		orderObsNid = ts.getConcept(orderObs).getNid();
126 		hasComponentNid = ts.getConcept(hasComponent).getNid();
127 		
128 		//Some LOINC constants
129 		//Ferritin | Bld-Ser-Plas
130 		loincFerritinMeasurementNid = ts.getNidForUuids(UUID.fromString("e1064652-d641-5736-9281-f68a9fef42c0"));
131 		
132 		//Drug allergens
133 		loincDrugAllergensNid = ts.getNidForUuids(UUID.fromString("a20d28ab-15c3-5a0e-ae53-147c0e33ea29"));
134 		
135 		vc = StandardViewCoordinates.getSnomedStatedLatest();
136 		
137 		// Start with standard view coordinate and override the path setting to use the LOINC path
138 		Position position = ts.newPosition(ts.getPath(loincPathNid), Long.MAX_VALUE);
139 		vc.setViewPosition(position);
140 		
141 		for (int i = 1000; i < 1004; i++)
142 		{
143 			ruleHits.put(i, new AtomicInteger());
144 		}
145 	}
146 
147 	/**
148 	 * 
149 	 * @see gov.va.isaac.mojos.dbTransforms.TransformConceptIterateI#transform(org.ihtsdo.otf.tcc.api.store.TerminologyStoreDI, 
150 	 *  org.ihtsdo.otf.tcc.api.concept.ConceptChronicleBI)
151 	 */
152 	@Override
153 	public boolean transform(TerminologyStoreDI ts, ConceptChronicleBI cc) throws Exception
154 	{
155 		examinedConcepts.incrementAndGet();
156 		ConceptAttributeVersionBI<?> latest = OTFUtility.getLatestAttributes(cc.getConceptAttributes().getVersions());
157 		if (latest.getPathNid() == loincPathNid)
158 		{
159 			//Rule for all other rules:
160 			if (classTypeIs("1", cc) && (orderIs("Order", cc) || orderIs("Both", cc)))
161 			{
162 				//Rule 1000 - everything that matches this is a lab procedure
163 				addRel(cc, sctLabProcedure, ts);
164 				ruleHits.get(1000).getAndIncrement();
165 				
166 				//Rule 1001
167 				if (ts.isKindOf(cc.getConceptNid(), loincFerritinMeasurementNid, vc))
168 				{
169 					addRel(cc, sctFerritinMeasurement, ts);
170 					ruleHits.get(1001).getAndIncrement();
171 				}
172 				
173 				//Rule 1002
174 				if (ts.isKindOf(cc.getConceptNid(), loincDrugAllergensNid, vc) && (!componentContains("Ab.", cc, ts) && !componentContains("lgE", cc, ts)))
175 				{
176 					addRel(cc, sctDrugMeasurement, ts);
177 					ruleHits.get(1002).getAndIncrement();
178 				}
179 				
180 				//Rule 1003
181 				if (cc.getPrimordialUuid().equals(loincAmpicillin))
182 				{
183 					mergeConcepts(cc, sctAmpicillin, ts);
184 					ruleHits.get(1003).getAndIncrement();
185 				}
186 				return true;
187 			}
188 		}
189 		
190 		return false;
191 	}
192 	
193 	private boolean classTypeIs(String classTypeValue, ConceptChronicleBI cc) throws IOException
194 	{
195 		for (RefexVersionBI<?> rv : cc.getAnnotationsActive(vc))
196 		{
197 			if (rv.getAssemblageNid() == classTypeNid)
198 			{
199 				if (((RefexStringVersionBI<?>)rv).getString1().equals(classTypeValue))
200 				{
201 					return true;
202 				}
203 			}
204 		}
205 		return false;
206 	}
207 	
208 	private boolean orderIs(String orderValue, ConceptChronicleBI cc) throws IOException
209 	{
210 		for (RefexVersionBI<?> rv : cc.getAnnotationsActive(vc))
211 		{
212 			if (rv.getAssemblageNid() == orderObsNid)
213 			{
214 				if (((RefexStringVersionBI<?>)rv).getString1().equals(orderValue))
215 				{
216 					return true;
217 				}
218 			}
219 		}
220 		return false;
221 	}
222 	
223 	private boolean componentContains(String matchString, ConceptChronicleBI cc, TerminologyStoreDI ts) throws IOException, ContradictionException
224 	{
225 		
226 		for (RelationshipVersionBI<?> rel :  cc.getVersion(vc).getRelationshipsOutgoingActive())
227 		{
228 			if (rel.getTypeNid() == hasComponentNid)
229 			{
230 				ConceptVersionBI target = ts.getConceptVersion(vc, rel.getDestinationNid());
231 				for (DescriptionVersionBI<?> d : target.getDescriptionsActive())
232 				{
233 					if (d.getText().contains(matchString))
234 					{
235 						return true;
236 					}
237 				}
238 			}
239 		}
240 		return false;
241 	}
242 	
243 	private void addRel(ConceptChronicleBI source, UUID target, TerminologyStoreDI ts) throws ValidationException, IOException, InvalidCAB, ContradictionException
244 	{
245 		RelationshipCAB rCab = new RelationshipCAB(source.getPrimordialUuid(), Snomed.IS_A.getUuids()[0], target, 0, RelationshipType.STATED_ROLE, 
246 				IdDirective.GENERATE_HASH);
247 		
248 		ts.getTerminologyBuilder(new EditCoordinate(TermAux.USER.getLenient().getConceptNid(), TermAux.UNSPECIFIED_MODULE.getLenient().getNid(), 
249 				loincPathNid), StandardViewCoordinates.getWbAuxiliary()).construct(rCab);
250 		ts.addUncommitted(source);
251 		
252 		generatedRels.getAndIncrement();
253 	}
254 	
255 	private void mergeConcepts(ConceptChronicleBI source, UUID target, TerminologyStoreDI ts) throws IOException, InvalidCAB, ContradictionException
256 	{
257 		ConceptAttributeVersionBI<?> sourceAttrib = source.getVersion(vc).getConceptAttributesActive();
258 		ConceptAttributeAB cab = new ConceptAttributeAB(source.getConceptNid(), sourceAttrib.isDefined(), RefexDirective.EXCLUDE);
259 		cab.setComponentUuidNoRecompute(target);
260 		
261 		ts.getTerminologyBuilder(new EditCoordinate(TermAux.USER.getLenient().getConceptNid(), TermAux.UNSPECIFIED_MODULE.getLenient().getNid(), 
262 				loincPathNid), StandardViewCoordinates.getWbAuxiliary()).construct(cab);
263 		ts.addUncommitted(source);
264 		
265 		mergedConcepts.incrementAndGet();
266 	}

277 	/**
278 	 * @see gov.va.isaac.mojos.dbTransforms.TransformI#getWorkResultSummary()
279 	 */
280 	@Override
281 	public String getWorkResultSummary()
282 	{
283 		StringBuilder sb = new StringBuilder();
284 		sb.append("Examined " + examinedConcepts.get() + " concepts and added hierarchy linkages to " + generatedRels.get() + " concepts.  "
285 				+ "Merged " + mergedConcepts.get() + " concepts");
286 		
287 		for (Entry<Integer, AtomicInteger> x : ruleHits.entrySet())
288 		{
289 			sb.append("  Rule " + x.getKey() + " hit on " + x.getValue() + " concepts");
290 		}
291 		
292 		return sb.toString();
293 	}
