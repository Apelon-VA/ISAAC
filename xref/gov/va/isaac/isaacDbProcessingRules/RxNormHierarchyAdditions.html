47  /**
48   * {@link RxNormHierarchyAdditions}
49   * 
50   * A Transformer that executes various rules to create more taxonomy in RxNorm
51   * 
52   * So far:
53   * 
54   * Let’s further integrate RxNorm and SNOMED CT. For every RxNorm IN concept with the string “penicillin” in its name, make it a child (is-a) 
55   * of SNOMED CT fdca98cf-8720-3dbe-bb72-3377d658a85c Penicillin -class of antibiotic- (product).
56   * 
57   *   a.  What should happen here is that the relationship between the RxNorm penicillin ingredient concepts and the dummy “RxNorm Ingredients” 
58   *       is made redundant by the classifier… it would still be there in stated view, but not in inferred view.
59   * 
60   *
61   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
62   */
63  @Service
64  @Named(value = "RxNorm Hierarchy Additions")
65  public class RxNormHierarchyAdditions implements TransformArbitraryI
66  {
67  	private AtomicInteger addedRels = new AtomicInteger();
68  	private AtomicInteger examinedConcepts = new AtomicInteger();
69  	
70  	private final UUID penicillinProduct = UUID.fromString("fdca98cf-8720-3dbe-bb72-3377d658a85c");//Penicillin -class of antibiotic- (product)
71  	private final UUID termTypeIN = UUID.fromString("17114d54-ed48-5f0a-a865-4ecec3e31cdc"); //Name for an Ingredient     // IN
72  	private final UUID rxNormHPath = UUID.fromString("763c21ad-55e3-5bb3-af1e-3e4fb475de44"); //RxNormH Path
73  	private final UUID rxNormDescType = UUID.fromString("3599879d-78c6-5b1e-b442-9ef08eaedd3c");// "RxNorm Description Type" - refex that carries ingredient linkage
74  
75  	/**
76  	 * @see gov.va.isaac.mojos.dbTransforms.TransformI#getName()
77  	 */
78  	@Override
79  	public String getName()
80  	{
81  		return "RxNorm Hierarchy Additions";
82  	}
83  	
84  	/**
85  	 * @see gov.va.isaac.mojos.dbTransforms.TransformI#configure(java.io.File, org.ihtsdo.otf.tcc.api.store.TerminologyStoreDI)
86  	 */
87  	@Override
88  	public void configure(File configFile, TerminologyStoreDI ts)
89  	{
90  		// noop
91  	}
92  
93  	/**
94  	 * @throws Exception
95  	 * @see gov.va.isaac.mojos.dbTransforms.TransformI#transform(org.ihtsdo.otf.tcc.api.store.TerminologyStoreDI)
96  	 */
97  	@Override
98  	public void transform(TerminologyStoreDI ts) throws Exception
99  	{
100 		int rxNormPathNid = ts.getNidForUuids(rxNormHPath);
101 		int rxNormDescTypeAssemblageNid = ts.getNidForUuids(rxNormDescType);
102 		ts.iterateConceptDataInParallel(new ProcessUnfetchedConceptDataBI()
103 		{
104 			@Override
105 			public boolean continueWork()
106 			{
107 				return true;
108 			}
109 
110 			@Override
111 			public void processUnfetchedConceptData(int cNid, ConceptFetcherBI fetcher) throws Exception
112 			{
113 				ConceptChronicleBI cc = fetcher.fetch();
114 				examinedConcepts.getAndIncrement();
115 				
116 				boolean isRxNormPath = false;
117 				for (int stampNid : cc.getConceptAttributes().getAllStamps())
118 				{
119 					if (ts.getPathNidForStamp(stampNid) == rxNormPathNid)
120 					{
121 						isRxNormPath = true;
122 						break;
123 					}
124 				}
125 				
126 				if (!isRxNormPath)
127 				{
128 					return;
129 				}
130 				
131 				boolean isRxNormIN = false;
132 				boolean hasPenicillin = false;
133 				for (DescriptionChronicleBI desc : cc.getDescriptions())
134 				{
135 					if (hasPenicillin && isRxNormIN)
136 					{
137 						break;
138 					}
139 					DescriptionVersionBI<?> currentDescription = OTFUtility.getLatestDescVersion(desc.getVersions());
140 					
141 					if (currentDescription == null)
142 					{
143 						continue;
144 					}
145 					
146 					for (RefexDynamicChronicleBI<?> refex : currentDescription.getRefexesDynamic())
147 					{
148 						RefexDynamicVersionBI<?> currentRefex = OTFUtility.getLatestDynamicRefexVersion(refex.getVersions());
149 						if (currentRefex.getAssemblageNid() == rxNormDescTypeAssemblageNid)
150 						{
151 							if (((RefexDynamicUUID)currentRefex.getData()[0]).getDataUUID().equals(termTypeIN))
152 							{
153 								isRxNormIN = true;
154 								break;
155 							}
156 						}
157 					}
158 					
159 					if (currentDescription.getText().toLowerCase().contains("penicillin"))
160 					{
161 						hasPenicillin = true;
162 					}
163 				}
164 				
165 				if (hasPenicillin && isRxNormIN)
166 				{
167 					RelationshipCAB rCab = new RelationshipCAB(cc.getPrimordialUuid(), Snomed.IS_A.getUuids()[0], penicillinProduct,
168 							0, RelationshipType.STATED_ROLE, IdDirective.GENERATE_HASH);
169 					
170 					ts.getTerminologyBuilder(new EditCoordinate(TermAux.USER.getLenient().getConceptNid(), TermAux.UNSPECIFIED_MODULE.getLenient().getNid(), 
171 							rxNormPathNid), StandardViewCoordinates.getWbAuxiliary()).construct(rCab);
172 					ts.addUncommitted(cc);
173 					
174 					int last = addedRels.getAndIncrement();
175 					if (last % 2000 == 0)
176 					{
177 						ts.commit();
178 					}
179 				}
180 			}
181 
182 			@Override
183 			public String getTitle()
184 			{
185 				return "RxNorm Hierachy Creator";
186 			}
187 
188 			@Override
189 			public NativeIdSetBI getNidSet() throws IOException
190 			{
191 				return null;
192 			}
193 
194 			@Override
195 			public boolean allowCancel()
196 			{
197 				return false;
198 			}
199 		});
200 		ts.commit();
201 	}
