4   package gov.va.isaac.classifier;
5   import gov.va.isaac.util.ProgressReporter;
6   
7   import org.ihtsdo.otf.tcc.api.nid.IntSet;
8   
9   /**
10   * Generically represents a classifier.
11   */
12  public interface Classifier extends ProgressReporter {
13  
14    /**
15     * Classify.
16     * @param rootNid the root nid
17     * @throws Exception 
18     */
19    public void classify(int rootNid) throws Exception;
20  
21    /**
22     * Incremental classify.
23     *   This can only handle new concepts or relationships.
24     *   Retirements will require a full classification.
25     *
26     * @param conceptSet the set of concepts that is new or has been added to
27     * @throws Exception the exception
28     */
29    public void incrementalClassify() throws Exception;
30  
31    /**
32     * Indicates whether or not incremental classify ready is the case.
33     *
34     * @return <code>true</code> if so, <code>false</code> otherwise
35     * @throws Exception the exception
36     */
37    public boolean isIncrementalClassifyReady() throws Exception;
38    
39    /**
40     * Adds the to incremental classification set.
41     *
42     * @param conceptSet the concept set
43     * @throws Exception the exception
44     */
45    public void addToIncrementalClassificationSet(IntSet conceptSet) throws Exception;
46  
47    /**
48     * Clear static state.  Reset after a full classification.
49     */
50    public void clearStaticState();
51    
52    /**
53     * Sets the save cycle check report.
54     *
55     * @param flag the save cycle check report
56     */
57    public void setSaveCycleCheckReport(boolean flag);
58    
59    /**
60     * Sets the save equivalent concepts report.
61     *
62     * @param flag the save equivalent concepts report
63     */
64    public void setSaveEquivalentConceptsReport(boolean flag);
65    
66    /**
67     * Cancel.
68     */
69    public void cancel();
70  }
