69    public void classify(int pathNid) throws Exception {
70      this.pathNid = pathNid;
71      getRole(ROLE_GROUP);
72      LOG.info("Building axioms ");
73      dataStore.iterateConceptDataInSequence(this);
74      LOG.info("Axioms " + axioms.size());
75      addRoleAxioms(CONCEPT_ATTRIBUTE);
76      LOG.info("Total Axioms " + axioms.size());
77      IReasoner reasoner = new SnorocketReasoner();
78      reasoner.loadAxioms(axioms);
79  	long startTime = System.currentTimeMillis();
80      LOG.info("Running Classifier");
81      reasoner.classify();
82      LOG.info("Time to classify (ms): " + (System.currentTimeMillis() - startTime));
83      Ontology ontology = reasoner.getClassifiedOntology();
84      Map<String,Set<String>> equivalentMap = new HashMap<String, Set<String>>();
85      for(String id : ontology.getNodeMap().keySet()) {
86  		Node node = ontology.getNode(id);
87      	if(node.getEquivalentConcepts().size() == 1 && node.getEquivalentConcepts().contains(id)) {
88      		continue;
89      	}
90  		equivalentMap.put(id, node.getEquivalentConcepts());
91      }
92      for(String id : equivalentMap.keySet()) {
93      	LOG.warn(id + " equivalents " + equivalentMap.get(id));
94      }
95      LOG.info("Inferred Axioms " + ontology.getInferredAxioms().size());
96      for(Axiom axiom : ontology.getInferredAxioms()) {
97      	LOG.info(axiom.toString());
98      }
99    }

101   private void addRoleAxioms(UUID parent) throws IOException, ContradictionException {
102 	ArrayList<ConceptVersionBI> children = WBUtility.getAllChildrenOfConcept(WBUtility.getConceptVersion(parent), false);
103 	if(children.isEmpty())
104 		return;
105 	Role[] rlhs = new Role[children.size()];
106 	int i=0;
107 	for(ConceptVersionBI child : children) {
108 		addRoleAxioms(child.getPrimordialUuid());
109 		rlhs[i++] = getRole(child.getPrimordialUuid().toString());
110 	}
111 	axioms.add(Factory.createRoleInclusion(rlhs, getRole(parent.toString())));
112 }

134   private void convertToOntologyObjects(ConceptVersionBI concept)
135     throws IOException, ContradictionException {
136     ConceptAttributeVersionBI<?> attributes =
137         ConceptViewerHelper.getConceptAttributes(concept);
138     // Set<Integer> relGroups
139     Collection<? extends RelationshipVersionBI<?>> relationships =
140         concept.getRelationshipsOutgoingActive();
141     Map<Integer, Concept> mapRoleGroup = mapToRoleGroup(relationships);
142     List<Concept> parentConcepts = getParentConcepts(relationships);
143     // Root Concept ?
144     if (parentConcepts.isEmpty())
145       return;
146     if (mapRoleGroup.isEmpty() && parentConcepts.size() == 1) {
147       // Single Parent and no role relationships
148       axioms.add(Factory.createConceptInclusion(getConcept(concept
149           .getPrimordialUuid().toString()), parentConcepts.get(0)));
150     } else {
151       List<Concept> definition = new ArrayList<Concept>();
152       definition.addAll(parentConcepts);
153       definition.addAll(mapRoleGroup.values());
154       axioms.add(Factory.createConceptInclusion(getConcept(concept
155           .getPrimordialUuid().toString()), Factory
156           .createConjunction((definition.toArray(new Concept[0])))));
157       if (attributes.isDefined()) {
158         axioms.add(Factory.createConceptInclusion(
159             Factory.createConjunction(definition.toArray(new Concept[0])),
160             getConcept(concept.getPrimordialUuid().toString())));
161       }
162 
163     }

168     Collection<? extends RelationshipVersionBI<?>> relationships)

170     List<Concept> parentConcepts = new ArrayList<Concept>();
171     for (RelationshipVersionBI<?> relationship : relationships) {

181   private Map<Integer, Concept> mapToRoleGroup(
182     Collection<? extends RelationshipVersionBI<?>> relationships)
183     throws ValidationException, IOException {
184     Map<Integer, List<Concept>> map = new HashMap<>();
185     Map<Concept, Integer> conceptToRoleNid = new HashMap<>();
186     // map roleGroup -> Concepts
187     for (RelationshipVersionBI<?> relationship : relationships) {
188       if (relationship.getGroup() == Snomed.IS_A.getNid())
189         continue;
190 
191       List<Concept> concepts = new ArrayList<Concept>();
192 
193       if (map.containsKey(relationship.getGroup()))
194         concepts = map.get(relationship.getGroup());
195       else
196         map.put(relationship.getGroup(), concepts);
197 
198       Concept existentialConcept =
199           Factory.createExistential(
200               getRole(WBUtility.getConceptVersion(relationship.getTypeNid())
201                   .getPrimordialUuid().toString()), getConcept(WBUtility
202                   .getConceptVersion(relationship.getDestinationNid())
203                   .getPrimordialUuid().toString()));
204       concepts.add(existentialConcept);
205       conceptToRoleNid.put(existentialConcept, relationship.getTypeNid());
206     }
207     // map roleGroup -> Conjunction/Concept
208     Map<Integer, Concept> roles = new HashMap<Integer, Concept>();
209     for (Integer group : map.keySet()) {
210       List<Concept> concepts = map.get(group);
211       if (concepts.size() == 1) {
212         if (group != 0)
213           throw new IOException("Unexpected single grouped rel.");
214         if (isNonGroupingRole(conceptToRoleNid.get(concepts.get(0)))) {
215           roles.put(group, concepts.get(0));
216         } else {
217           roles.put(group, Factory.createExistential(getRole(ROLE_GROUP), concepts.get(0)));          
218         }
219       } else {
220         if (group == 0) {
221           for (Concept concept : concepts) {
222             if (isNonGroupingRole(conceptToRoleNid.get(concept))) {
223               roles.put(group, concept);
224             } else {
225               roles.put(group, Factory.createExistential(getRole(ROLE_GROUP),concept));              
226             }
227           }
228         } else {
229           roles.put(group,
230               Factory.createExistential(getRole(ROLE_GROUP),
231                   Factory.createConjunction(concepts.toArray(new Concept[0]))));
232         }
233       }
234     }
235     return roles;
236   }

238   private boolean isNonGroupingRole(int nid) throws ValidationException, IOException {
239     // $nevergrouped{"123005000"} = "T"; # part-of is never grouped
240     //
241     // $nevergrouped{"272741003"} = "T"; # laterality is never grouped
242     //
243     // $nevergrouped{"127489000"} = "T"; # has-active-ingredient is never
244     // grouped
245     //
246     // $nevergrouped{"411116001"} = "T"; # has-dose-form is never grouped
247 	  return (nid == Snomed.PART_OF.getNid() || nid == Snomed.LATERALITY.getNid() 
248 			  ||nid ==  Snomed.HAS_ACTIVE_INGREDIENT.getNid() || nid == Snomed.HAS_DOSE_FORM.getNid());
249   }
