45  /**
46   * Commit listener for incremental classification.
47   * @author bcarlsenca
48   */
49  @SuppressWarnings("restriction")
50  @Service
51  @Singleton
52  @Named(value = "Classifier")
53  public class ClassifierCommitListener implements PropertyChangeListener,
54      ServicesToPreloadI, CommitListenerI {
55  
56    /** The log. */
57    private final Logger LOG = LoggerFactory.getLogger(this.getClass());
58  
59    /** The enabled flag. */
60    private boolean enabled = false;
61  
62    /**
63     * Instantiates an empty {@link ClassifierCommitListener}.
64     */
65    private ClassifierCommitListener() {
66      // for HK2 to create
67    }
68  
69    /**
70     * Returns the listener name.
71     *
72     * @return the listener name
73     * @see gov.va.isaac.interfaces.utility.CommitListenerI#getListenerName()
74     */
75    @Override
76    public String getListenerName() {
77      return "Classifier";
78    }
79  
80    /**
81     * Indicates whether or not enabled is the case.
82     *
83     * @return <code>true</code> if so, <code>false</code> otherwise
84     */
85    public boolean isEnabled() {
86      return enabled;
87    }
88  
89    /**
90     * Disable.
91     *
92     * @see gov.va.isaac.interfaces.utility.CommitListenerI#disable()
93     */
94    @Override
95    public void disable() {
96      if (enabled) {
97        LOG.info("Disabling the classifier commit listener");
98        // TODO file yet another OTF bug - this doesn't work. We still get
99        // property change notifications after calling remove...
100       ExtendedAppContext.getDataStore().removePropertyChangeListener(this);
101       enabled = false;
102     }
103   }
104 
105   /**
106    * Enable.
107    *
108    * @see gov.va.isaac.interfaces.utility.CommitListenerI#enable()
109    */
110   @Override
111   public void enable() {
112     UserProfile loggedIn = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
113     if (!enabled && loggedIn != null
114         && loggedIn.isLaunchWorkflowForEachCommit()) {
115       LOG.info("Enabling the classifier commit listener");
116       ExtendedAppContext.getDataStore().addPropertyChangeListener(
117           CONCEPT_EVENT.POST_COMMIT, this);
118       enabled = true;
119     }
120   }
121 
122   /**
123    * Property change.
124    *
125    * @param evt the evt
126    */
127   @Override
128   public void propertyChange(PropertyChangeEvent evt) {
129     // TODO this shouldn't be necessary, but OTF has a bug....
130     if (!enabled) {
131       return;
132     }
133     try {
134       if (CONCEPT_EVENT.POST_COMMIT.name().equals(evt.getPropertyName())) {
135         LOG.debug("post-commit triggered in classifier commit listener");
136         final int[] allConceptNids =
137             ((NativeIdSetBI) evt.getNewValue()).getSetValues();
138 
139         //
140         Platform.runLater(() -> {
141           try {
142             //Classifier classifier = new SnomedSnorocketClassifier();
143             //classifier.incrementalClassify((IntSet)evt.getNewValue());
144           } catch (Exception e) {
145             e.printStackTrace();
146             AppContext.getCommonDialogs().showErrorDialog(e.getMessage(), e);
147           }
148 
149         });
150       }
151     } catch (Exception e) {
152       LOG.error("Unexpected error processing commit notification", e);
153     }
154   }
155 
156   /**
157    * Load requested.
158    *
159    * @see gov.va.isaac.interfaces.utility.ServicesToPreloadI#loadRequested()
160    */
161   @Override
162   public void loadRequested() {
163     AppContext.getService(UserProfileManager.class).registerLoginCallback(
164         (user) -> {
165           enable();
166         });
167   }

170    * Shutdown.
171    *
172    * @see gov.va.isaac.interfaces.utility.ServicesToPreloadI#shutdown()
173    */
174   @Override
175   public void shutdown() {
176     // noop
177   }
