50  /**
51   * Commit listener for incremental classification.
52   * @author bcarlsenca
53   */
54  @SuppressWarnings("restriction")
55  @Service
56  @Singleton
57  @Named(value = "Classifier")
58  public class ClassifierCommitListener implements PropertyChangeListener,
59      ServicesToPreloadI, CommitListenerI {
60  
61    /** The log. */
62    private final Logger LOG = LoggerFactory.getLogger(this.getClass());
63  
64    /** The enabled flag. */
65    private boolean enabled = false;
66  
67    /** The includes retirements. */
68    private boolean includesRetirements;
69  
70    /**
71     * Instantiates an empty {@link ClassifierCommitListener}.
72     */
73    private ClassifierCommitListener() {
74      // for HK2 to create
75    }
76  
77    /**
78     * Returns the listener name.
79     *
80     * @return the listener name
81     * @see gov.va.isaac.interfaces.utility.CommitListenerI#getListenerName()
82     */
83    @Override
84    public String getListenerName() {
85      return "Classifier";
86    }
87  
88    /**
89     * Indicates whether or not enabled is the case.
90     *
91     * @return <code>true</code> if so, <code>false</code> otherwise
92     */
93    public boolean isEnabled() {
94      return enabled;
95    }
96  
97    /**
98     * Disable.
99     *
100    * @see gov.va.isaac.interfaces.utility.CommitListenerI#disable()
101    */
102   @Override
103   public void disable() {
104     if (enabled) {
105       LOG.info("Disabling the classifier commit listener");
106       // TODO file yet another OTF bug - this doesn't work. We still get
107       // property change notifications after calling remove...
108       ExtendedAppContext.getDataStore().removePropertyChangeListener(this);
109       enabled = false;
110     }
111   }
112 
113   /**
114    * Enable.
115    *
116    * @see gov.va.isaac.interfaces.utility.CommitListenerI#enable()
117    */
118   @Override
119   public void enable() {
120     UserProfile loggedIn = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
121     if (!enabled && loggedIn != null
122         && loggedIn.isLaunchWorkflowForEachCommit()) {
123       LOG.info("Enabling the classifier commit listener");
124       ExtendedAppContext.getDataStore().addPropertyChangeListener(
125           CONCEPT_EVENT.PRE_COMMIT, this);
126       ExtendedAppContext.getDataStore().addPropertyChangeListener(
127           CONCEPT_EVENT.POST_COMMIT, this);
128       enabled = true;
129     }
130   }
131 
132   /**
133    * Property change.
134    *
135    * @param evt the evt
136    */
137   @Override
138   public void propertyChange(PropertyChangeEvent evt) {
139     // TODO this shouldn't be necessary, but OTF has a bug....
140     if (!enabled) {
141       return;
142     }
143     try {
144       if (CONCEPT_EVENT.PRE_COMMIT.name().equals(evt.getPropertyName())) {
145         LOG.debug("pre-commit triggered in classifier commit listener");
146         // Gather uncommitted nids.
147         includesRetirements = false;
148         for (int nid : ((NativeIdSetBI) evt.getNewValue()).getSetValues()) {
149           ConceptChronicle concept =
150               (ConceptChronicle) OTFUtility.getConceptVersion(nid)
151                   .getChronicle();
152           // Iterate and look for retired concept/relationships
153           for (int componentNid : concept.getUncommittedNids().getListArray()) {
154             LOG.debug("NID: " + nid);
155             ComponentVersionBI component =
156                 OTFUtility.getComponentVersion(componentNid);
157             LOG.debug("  " + component.getClass().getName());
158             if (!component.isActive()) {
159               // This is really awful, but there's no other way I can
160               // see to ask about the type of a component.
161               if (component instanceof ConceptAttributes.Version ||
162                   component instanceof Relationship.Version) {
163                 LOG.debug("  INCLUDES RETIREMENTS");
164                 includesRetirements = true;
165                 break;
166               }
167             }
168             // if a retired concept or relationship set includesRetirements to true
169           }
170         }
171       }
172 
173       if (CONCEPT_EVENT.POST_COMMIT.name().equals(evt.getPropertyName())) {
174         LOG.debug("post-commit triggered in classifier commit listener");
175         final int[] allConceptNids =
176             ((NativeIdSetBI) evt.getNewValue()).getSetValues();
177 
178         if (allConceptNids != null && allConceptNids.length > 0) {
179 
180           // perform action
181           Platform
182               .runLater(() -> {
183                 try {
184                   Classifier classifier = new SnomedSnorocketClassifier();
185 
186                   // Identify if any components have been retired
187                   // if so, clear the classifier state and send user a warning
188                   LOG.debug(" Check for retirements");
189                   if (includesRetirements) {
190                     LOG.debug("   retirements = true");
191                     classifier.clearStaticState();
192                     LOG.warn("Commit included retirements, you must perform full classification again.");
193                     return;
194                   }
195 
196                   LOG.debug(" Add to incremental classification set");
197                   classifier.addToIncrementalClassificationSet( 
198                       new IntSet(((NativeIdSetBI) evt.getNewValue()).getSetValues()));
199                 } catch (Exception e) {
200                   e.printStackTrace();
201                   AppContext.getCommonDialogs().showErrorDialog(e.getMessage(),
202                       e);
203                 }
204 
205               });
206         }
207       }
208     } catch (Exception e) {
209       LOG.error("Unexpected error processing commit notification", e);
210     }
211   }

214    * Load requested.

223         });
