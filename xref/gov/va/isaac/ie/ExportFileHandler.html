52  /**
53   * Handler for the ISAAC export to file functionality.
54   *
55   * @author tnaing
56   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>

59  public class ExportFileHandler {
60  
61    /** The Constant LOG. */
62    @SuppressWarnings("unused")
63    private static final Logger LOG = LoggerFactory
64        .getLogger(ExportFileHandler.class);
65  
66    /** The path nid. */
67    private int pathNid;
68  
69    /** The export type. */
70    private ExportType exportType;
71  
72    /** The folder. */
73    private File folder;
74  
75    /** The zip. */
76    private boolean zip;
77  
78    /**  The exporter. */
79    private Exporter exporter;

81    /**
82     * Instantiates an empty {@link ExportFileHandler}.
83     * @param pathNid the path nid to export
84     * @param exportType the type of export (e.g. format)
85     * @param folder the folder to export to
86     * @param zip zip flag (compress output)
87     *
88     * @throws ValidationException the validation exception
89     */
90    public ExportFileHandler(int pathNid, ExportType exportType, File folder,
91        boolean zip) throws ValidationException {
92      super();
93      this.pathNid = pathNid;
94      this.exportType = exportType;
95      this.folder = folder;
96      this.zip = zip;
97    }
98  
99    /**
100    * Method called by the ISAAC application to perform the export. Will be
101    * invoked on a background thread.
102    * @param listener the listener, leave null if not needed
103    * @throws Exception the exception
104    */
105   @SuppressWarnings("resource")
106   public void doExport(ProgressListener listener) throws Exception {
107 
108     // Make sure NOT in application thread.
109     FxUtils.checkBackgroundThread();
110 
111     // Handle eConcept
112     if (exportType == ExportType.ECONCEPT) {
113       String fileName = "eConcepts.jbin";
114       File file = new File(folder, fileName);
115       checkExists(file);
116 
117       OutputStream outputStream = null;
118       if (zip) {
119         ZipOutputStream zos =
120             new ZipOutputStream(new FileOutputStream(file + ".zip"));
121         ZipEntry entry = new ZipEntry(fileName);
122         entry.setMethod(ZipEntry.DEFLATED);
123         zos.putNextEntry(entry);
124         outputStream = zos;
125       } else {
126         outputStream = new FileOutputStream(file);
127       }
128 
129       EConceptExporter econceptExporter = new EConceptExporter(outputStream);
130       exporter = econceptExporter;
131       econceptExporter.addProgressListener(listener);
132       econceptExporter.export(pathNid);
133       outputStream.flush();
134       outputStream.close();
135 
136     }
137 
138     // Handle Owl
139     else if (exportType == ExportType.OWL) {
140       // Can only export SNOMED at this time
141       if (pathNid != Snomed.SNOMED_RELEASE_PATH.getNid()) {
142         throw new UnsupportedOperationException(
143             "OWL Exporter only supports SNOMED path at this time.");
144       }
145       String fileName = "snomed.owl";
146       File file = new File(folder, fileName);
147       checkExists(file);
148       OutputStream outputStream = null;
149       if (zip) {
150         ZipOutputStream zos =
151             new ZipOutputStream(new FileOutputStream(file + ".zip"));
152         ZipEntry entry = new ZipEntry(fileName);
153         entry.setMethod(ZipEntry.DEFLATED);
154         zos.putNextEntry(entry);
155         outputStream = zos;
156       } else {
157         outputStream = new FileOutputStream(file);
158       }
159 
160       OWLExporter owlExporter = new OWLExporter(outputStream);
161       exporter = owlExporter;
162       owlExporter.addProgressListener(listener);
163       owlExporter.export(pathNid);
164       outputStream.flush();
165       outputStream.close();
166 
167     }
168 
169     // Handle RF2
170     else if (exportType == ExportType.RF2) {
171 
172       // Settings for RF2 Export
173       // TODO: can get these from AppContext.getAppConfiguration() in the future
174       // adding new properties involves tweaking AppConfigSchema.xsd in isaac-app, and then 
175       // following the changes through the implementation... IsaacAppConfigWrapper and IsaacAppConfigI
176       String namespace = "1000000";
177       ReleaseType releaseType = ReleaseType.SNAPSHOT;
178       Collection<Integer> taxonomyParentNids = new ArrayList<>();
179       taxonomyParentNids.add(Taxonomies.SNOMED.getLenient().getNid());
180       File releaseFolder = new File(folder, releaseType.suffix);
181       checkExists(releaseFolder);
182       Rf2Export rf2Export =
183           new Rf2Export(releaseFolder, releaseType, LanguageCode.EN_US,
184               COUNTRY_CODE.US, namespace, Calendar.getInstance().getTime(),
185               pathNid, OTFUtility.getViewCoordinate(), taxonomyParentNids);
186       exporter = rf2Export;
187       rf2Export.addProgressListener(listener);
188       rf2Export.export(pathNid);
189       rf2Export.writeOneTimeFiles();
190       rf2Export.close();
191       if (zip) {
192         File file = new File(folder, releaseType.suffix + ".zip");
193         JarOutputStream output =
194             new JarOutputStream(new FileOutputStream(file));
195         recursiveAddToZip(output, releaseFolder, releaseType.suffix);
196         output.close();
197         removeDirectory(releaseFolder);
198       }
199     } else {
200       throw new UnsupportedOperationException(exportType.getDisplayName()
201           + " export not yet supported in ISAAC.");
202     }
203   }
204 
205   /**
206    * Recursive add to zip.
207    *
208    * @param output the output
209    * @param parent the parent
210    * @param prefix the prefix
211    * @throws IOException Signals that an I/O exception has occurred.
212    */
213   private static void recursiveAddToZip(ZipOutputStream output, File parent,
214     String prefix) throws IOException {
215     if (parent == null) {
216       return;
217     }
218     for (File child : parent.listFiles()) {
219       if (child.isDirectory()) {
220         ZipEntry entry =
221             new ZipEntry(prefix + File.separator + child.getName()
222                 + File.separator);
223         output.putNextEntry(entry);
224         recursiveAddToZip(output, child,
225             prefix + File.separator + child.getName());
226       } else {
227         addToZip(prefix, child, output, null);
228       }
229     }
230   }
231 
232   /**
233    * Check exists.
234    *
235    * @param file the file
236    * @throws Exception the exception if it exists.
237    */
238   private void checkExists(File file) throws Exception {
239     if (file.exists()) {
240       throw new IOException("Output already exists, clean up first. "
241           + file);
242     }
243   }

246   /**
247    * Adds the to zip.
248    *
249    * @param prefix the prefix
250    * @param f the f
251    * @param output the output
252    * @param comment the comment
253    * @throws IOException Signals that an I/O exception has occurred.
254    */
255   private static void addToZip(String prefix, File f, ZipOutputStream output,
256     String comment) throws IOException {
257     ZipEntry entry = new ZipEntry(prefix + File.separator + f.getName());
258     if (f.exists()) {
259       entry.setSize(f.length());
260       entry.setTime(f.lastModified());
261     } else {
262       entry.setSize(0);
263       entry.setTime(System.currentTimeMillis());
264     }
265     entry.setComment(comment);
266     output.putNextEntry(entry);
267     if (f.exists()) {
268       FileInputStream fis = new FileInputStream(f);
269       byte[] buffer = new byte[10240];
270       int length;
271       while ((length = fis.read(buffer)) > 0) {
272         output.write(buffer, 0, length);
273       }
274       fis.close();
275     }
276     output.closeEntry();
277   }

279   /**
280    * Removes the directory.
281    *
282    * @param directory the directory
283    * @return true, if successful
284    */
285   private static boolean removeDirectory(File directory) {
286 
287     if (directory == null)
288       return false;
289     if (!directory.exists())
290       return true;
291     if (!directory.isDirectory())
292       return false;
293 
294     String[] list = directory.list();
295 
296     if (list != null) {
297       for (int i = 0; i < list.length; i++) {
298         File entry = new File(directory, list[i]);
299         if (entry.isDirectory()) {
300           if (!removeDirectory(entry))
301             return false;
302         } else {
303           if (!entry.delete())
304             return false;
305         }
306       }
307     }
308 
309     return directory.delete();
310   }
