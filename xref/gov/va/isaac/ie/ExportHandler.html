19  package gov.va.isaac.ie;

21  import gov.va.isaac.gui.util.FxUtils;

31  
32  import javax.validation.ValidationException;
33  
34  import org.slf4j.Logger;
35  import org.slf4j.LoggerFactory;

37  /**
38   * Handles the ISAAC information model export functionality.
39   *
40   * @author ocarlsen
41   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
42   * @author bcarlsenca
43   */
44  public class ExportHandler {
45  
46    /** The Constant LOG. */
47    private static final Logger LOG = LoggerFactory
48        .getLogger(ExportHandler.class);
49  
50    /**
51     * Instantiates an empty {@link ExportHandler}.
52     *
53     * @throws ValidationException the validation exception
54     */
55    public ExportHandler() throws ValidationException {
56      super();
57    }
58  
59    /**
60     * Method called by the ISAAC application to perform the export. Will be
61     * invoked on a background thread.
62     *
63     * @param informationModel the information model
64     * @param file the file
65     * @throws Exception the exception
66     */
67    @SuppressWarnings("static-method")
68    public void doExport(InformationModel informationModel, File file)
69      throws Exception {
70      InformationModelType modelType = informationModel.getType();
71      LOG.debug("modelType=" + modelType);
72      LOG.debug("file=" + file);
73  
74      // Make sure NOT in application thread.
75      FxUtils.checkBackgroundThread();
76  
77      if (modelType == InformationModelType.CEM) {
78  
79        FileOutputStream out = new FileOutputStream(file);
80        CEMExporter exporter = new CEMExporter(out);
81        UUID modelUUID = informationModel.getUuid();
82        exporter.exportModel(modelUUID);
83        out.close();
84  
85      } else if (modelType == InformationModelType.HeD) {
86  
87        FileOutputStream out = new FileOutputStream(file);
88        HeDExporter exporter = new HeDExporter(out);
89        UUID modelUUID = informationModel.getUuid();
90        exporter.exportModel(modelUUID);
91        out.close();
92  
93      } else if (modelType == InformationModelType.FHIM) {
94  
95        FileOutputStream out = new FileOutputStream(file);
96        FHIMExporter exporter = new FHIMExporter(out);
97        UUID modelUUID = informationModel.getUuid();
98        exporter.exportModel(modelUUID);
99        out.close();
100 
101     } else {
102       throw new UnsupportedOperationException(modelType.getDisplayName()
103           + " export not yet supported in ISAAC.");
104     }
105   }
106 }
