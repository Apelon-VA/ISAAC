37    /**
38     * Indicates whether or not the concept qualifies as having a component
39     * with the indicated path.
40     *
41     * @param conceptNid the concept nid
42     * @param pathNid the path nid
43     * @return <code>true</code> if so, <code>false</code> otherwise
44     * @throws IOException
45     * @throws ContradictionException
46     */
47    public static boolean isQualifying(int conceptNid, int pathNid)
48      throws ContradictionException, IOException {
49      ConceptVersionBI cv = WBUtility.getConceptVersion(conceptNid);
50      if (cv.getPathNid() == pathNid)
51        return true;
52  
53      ViewCoordinate vc = WBUtility.getViewCoordinate();
54  
55      // atts
56      ConceptAttributeVersionBI<?> att =
57          cv.getConceptAttributes().getVersion(vc);
58      if (att == null)
59        return false;
60      if (att.getPathNid() == pathNid)
61        return true;
62      
63      // descs
64      for (DescriptionChronicleBI dc : cv.getDescriptions()) {
65        DescriptionVersionBI<?> dv = dc.getVersion(vc);
66        if (dv.getPathNid() == pathNid) 
67          return true;
68      }
69  
70      // rels
71      for (RelationshipChronicleBI rc : cv.getRelationshipsOutgoing()) {
72        RelationshipVersionBI<?> rv = rc.getVersion(vc);
73        if (rv.getPathNid() == pathNid) 
74          return true;
75      }
76      
77      // refex
78      for (RefexChronicleBI<?> rc : cv.getAnnotations()) {
79        RefexVersionBI<?> rv = rc.getVersion(vc);
80        if (rv.getPathNid() == pathNid) 
81          return true;
82      }
83      
84      return false;
85  
86    }
