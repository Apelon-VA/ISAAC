161   /**
162    * Instantiates a {@link OWLExporter} from the specified parameters.
163    *
164    * @param fileOutputStream the file output stream
165    * @throws Exception if something goes wrong in constructor
166    */
167   public OWLExporter(OutputStream fileOutputStream) throws Exception {
168     dos = new DataOutputStream(new BufferedOutputStream(fileOutputStream));
169     if (snomed == null) {
170       // Create a new ontology
171       snomed =
172           manager
173               .createOntology(new OWLOntologyID(snomedIRI, snomedVersionIRI));
174       format.setParameter("xml:base", snomedNamespace);
175 
176       // Obtain SNOMED root concept
177       // TODO: this needs to be generalized
178       UUID snomedRootUUID = Taxonomies.SNOMED.getUuids()[0];
179       ConceptVersionBI snomedRootConcept =
180           WBUtility.getConceptVersion(snomedRootUUID);
181 
182       // Add annotation based on root concept
183       for (DescriptionVersionBI<?> desc : snomedRootConcept
184           .getDescriptionsActive()) {
185         if (desc.getText().contains("Release")) {
186           manager.applyChange(new AddOntologyAnnotation(snomed, factory
187               .getOWLAnnotation(factory
188                   .getOWLAnnotationProperty(OWLRDFVocabulary.OWL_VERSION_INFO
189                       .getIRI()), factory.getOWLLiteral(desc.getText()))));
190         }
191         if (desc.getText().contains("IHTSDO")) {
192           manager.applyChange(new AddOntologyAnnotation(snomed, factory
193               .getOWLAnnotation(factory
194                   .getOWLAnnotationProperty(OWLRDFVocabulary.RDFS_COMMENT
195                       .getIRI()), factory.getOWLLiteral(desc.getText()))));
196         }
197       }
198       manager.applyChange(new AddOntologyAnnotation(snomed, factory
199           .getOWLAnnotation(factory
200               .getOWLAnnotationProperty(OWLRDFVocabulary.RDFS_LABEL.getIRI()),
201               factory.getOWLLiteral(snomedOntologyName))));
202     }
203   }

207    *
208    * @param pathNid the path nid
209    * @throws Exception the exception
210    */
211   @Override
212   public void export(int pathNid) throws Exception {
213     this.pathNid = pathNid;
214     dataStore.iterateConceptDataInSequence(this);
215     manager.addAxioms(snomed, setOfAxioms);
216     manager.saveOntology(snomed, format, dos);
217     dos.flush();
218     dos.close();
219     LOG.info("Wrote " + count + " concepts.");
220   }

252   public void processUnfetchedConceptData(int cNid, ConceptFetcherBI fetcher)
253     throws Exception {
254     if (Thread.currentThread().isInterrupted())
255       throw new InterruptedException();
256     ConceptVersionBI concept = fetcher.fetch(WBUtility.getViewCoordinate());
257     LOG.debug("Process concept " + concept.getPrimordialUuid());
258     allCount++;
259     if (Exporter.isQualifying(concept.getNid(), pathNid)) {
260       count++;
261       convertToOWLObjects(concept);
262     }
263     // Handle progress monitor
264     if ((int) ((allCount * 100) / progressMax) > progress) {
265       progress = (allCount * 100) / progressMax;
266       fireProgressEvent(progress, progress + " % finished");
267     }

292   /**
293    * Convert to owl objects.
294    *
295    * @param currentConcept the current concept
296    * @throws Exception the exception
297    */
298   public void convertToOWLObjects(ConceptVersionBI currentConcept)
299     throws Exception {
300 
301     // Bail on empty concepts
302     if (currentConcept == null) {
303       return;
304     }
305 
306     // Create an Owl class for each concept
307     OWLClass currentConceptClass =
308         createOWLAxiomsFromConceptVersionBI(setOfAxioms, factory, pm,
309             currentConcept);
310 
311     // Get labels
312     for (DescriptionVersionBI<?> desc : currentConcept.getDescriptionsActive()) {
313       // .getSynonyms was not working, thus this work around
314       if (desc.getTypeNid() == Snomed.SYNONYM_DESCRIPTION_TYPE.getNid()
315           && !desc.equals(currentConcept.getPreferredDescription())
316           && isUSLanguageRefex(desc)) {
317         OWLAnnotation synonymsTerm =
318             factory.getOWLAnnotation(factory.getOWLAnnotationProperty(IRI
319                 .create(snomedEnUsSynonymAnnotation)), factory.getOWLLiteral(
320                 desc.getText(), desc.getLang()));
321         setOfAxioms.add(factory.getOWLAnnotationAssertionAxiom(
322             currentConceptClass.getIRI(), synonymsTerm));
323       } else if (desc.getTypeNid() == Snomed.DEFINITION_DESCRIPTION_TYPE
324           .getNid() && isUSLanguageRefex(desc)) {
325         // TextDefinitions
326         OWLAnnotation textDefinition =
327             factory.getOWLAnnotation(factory.getOWLAnnotationProperty(IRI
328                 .create(snomedTextDefinitionAnnotation)), factory
329                 .getOWLLiteral(desc.getText(), desc.getLang()));
330         setOfAxioms.add(factory.getOWLAnnotationAssertionAxiom(
331             currentConceptClass.getIRI(), textDefinition));
332       }
333     }
334 
335     // Parent relationships
336     Set<OWLClass> parentClasses = new HashSet<>();
337     for (RelationshipVersionBI<?> rel : currentConcept
338         .getRelationshipsOutgoingActiveIsa()) {
339       if (rel.isStated()) {
340         parentClasses.add(factory.getOWLClass(
341             ":"
342                 + getSnomedConceptID(WBUtility.getConceptVersion(rel
343                     .getDestinationNid())), pm));
344       }
345     }
346 
347     // Handle other relationships
348     // role group
349     OWLObjectProperty groupPart =
350         factory.getOWLObjectProperty(":" + snomedRoleGroupConceptId, pm);
351     // relationships
352     SortedMap<Integer, Set<OWLClassExpression>> relationshipGroups =
353         new TreeMap<>();
354     Set<OWLClassExpression> neverGroupedRelationships = new HashSet<>();
355     for (RelationshipVersionBI<?> rel : currentConcept
356         .getRelationshipsOutgoingActive()) {
357       // Skip the is a relationship, these are taken care of in the parents
358       // relationships
359       if (rel.getTypeNid() == Snomed.IS_A.getNid() && rel.isStated()) {
360         continue;
361       }
362 
363       OWLObjectProperty relationshipTypeProperty =
364           createOWLPropertyAxiomsFromConceptVersionBI(setOfAxioms, factory, pm,
365               WBUtility.getConceptVersion(rel.getTypeNid()));
366       OWLClass destinationClass =
367           factory.getOWLClass(
368               ":"
369                   + getSnomedConceptID(WBUtility.getConceptVersion(rel
370                       .getDestinationNid())), pm);
371 
372       OWLClassExpression owlRel =
373           factory.getOWLObjectSomeValuesFrom(relationshipTypeProperty,
374               destinationClass);
375       if (rel.getTypeNid() == Snomed.LATERALITY.getNid()
376           || rel.getTypeNid() == Snomed.PART_OF.getNid()
377           || rel.getTypeNid() == Snomed.HAS_ACTIVE_INGREDIENT.getNid()
378           || rel.getTypeNid() == Snomed.HAS_DOSE_FORM.getNid()) {
379         neverGroupedRelationships.add(owlRel);
380       }
381       if (relationshipGroups.containsKey(rel.getGroup())) {
382         relationshipGroups.get(rel.getGroup()).add(owlRel);
383       } else {
384         Set<OWLClassExpression> relationships = new HashSet<>();
385         relationships.add(owlRel);
386         relationshipGroups.put(rel.getGroup(), relationships);
387       }
388     }
389     OWLClassExpression owlRelationships = null;
390     if (relationshipGroups.isEmpty()) {
391       if (parentClasses.size() == 1) {
392         owlRelationships = parentClasses.iterator().next();
393       } else if (parentClasses.size() > 1) {
394         owlRelationships = factory.getOWLObjectIntersectionOf(parentClasses);
395       }
396     } else {
397       Set<OWLClassExpression> owlInnerRelationships = new HashSet<>();
398       owlInnerRelationships.addAll(parentClasses);
399       for (int group : relationshipGroups.keySet()) {
400         OWLClassExpression innerRelationship = null;
401         if (relationshipGroups.get(group).size() > 1) {
402           // Separate into two sets, the nevergrouped and the normally grouped
403           Set<OWLClassExpression> groupedRels = new HashSet<>();
404           for (OWLClassExpression rel : relationshipGroups.get(group)) {
405             if (neverGroupedRelationships.contains(rel)) {
406               owlInnerRelationships.add(rel);
407             } else {
408               groupedRels.add(rel);
409             }
410           }
411           if (!groupedRels.isEmpty()) {
412             innerRelationship = factory.getOWLObjectIntersectionOf(groupedRels);
413           }
414         } else {
415           OWLClassExpression singleRelationship =
416               relationshipGroups.get(group).iterator().next();
417           if (neverGroupedRelationships.contains(singleRelationship)) {
418             owlInnerRelationships.add(singleRelationship);
419           } else {
420             innerRelationship = singleRelationship;
421           }
422         }
423 
424         if (innerRelationship != null) {
425           owlInnerRelationships.add(factory.getOWLObjectSomeValuesFrom(
426               groupPart, innerRelationship));
427         }
428       }
429       owlRelationships =
430           factory.getOWLObjectIntersectionOf(owlInnerRelationships);
431     }
432 
433     if (owlRelationships != null) {
434       if (currentConcept.getConceptAttributes()
435           .getVersion(WBUtility.getViewCoordinate()).isDefined()) {
436         setOfAxioms.add(factory.getOWLEquivalentClassesAxiom(
437             currentConceptClass, owlRelationships));
438       } else {
439         setOfAxioms.add(factory.getOWLSubClassOfAxiom(currentConceptClass,
440             owlRelationships));
441       }
442     }
443   }

446   /**
447    * Creates the owl axioms from concept version bi.
448    *
449    * @param setOfAxioms the set of axioms
450    * @param factory the factory
451    * @param pm the pm
452    * @param conceptVersionBI the concept version bi
453    * @return the OWL class
454    * @throws Exception the exception
455    */
456   private OWLClass createOWLAxiomsFromConceptVersionBI(
457     Set<OWLAxiom> setOfAxioms, OWLDataFactory factory, PrefixManager pm,
458     ConceptVersionBI conceptVersionBI) throws Exception {
459     // Declaration
460     OWLClass owlClass =
461         factory.getOWLClass(":" + getSnomedConceptID(conceptVersionBI), pm);
462     OWLDeclarationAxiom declarationAxiom =
463         factory.getOWLDeclarationAxiom(owlClass);
464     setOfAxioms.add(declarationAxiom);
465     // Create annotation type axiom
466     // Fully specified description
467     OWLAnnotation labelAnnotation =
468         factory.getOWLAnnotation(factory.getRDFSLabel(), factory.getOWLLiteral(
469             conceptVersionBI.getFullySpecifiedDescription().getText(),
470             conceptVersionBI.getFullySpecifiedDescription().getLang()));
471     OWLAxiom labelAnnotationAxiom =
472         factory.getOWLAnnotationAssertionAxiom(owlClass.getIRI(),
473             labelAnnotation);
474     setOfAxioms.add(labelAnnotationAxiom);
475 
476     // Preferred name
477     OWLAnnotation preferredTerm =
478         factory.getOWLAnnotation(factory.getOWLAnnotationProperty(IRI
479             .create(snomedEnUsPreferredAnnotation)), factory.getOWLLiteral(
480             conceptVersionBI.getPreferredDescription().getText(),
481             conceptVersionBI.getPreferredDescription().getLang()));
482     OWLAxiom preferredTermAx =
483         factory
484             .getOWLAnnotationAssertionAxiom(owlClass.getIRI(), preferredTerm);
485     setOfAxioms.add(preferredTermAx);
486 
487     return owlClass;
488   }

490   /**
491    * Creates the owl property axioms from concept version bi.
492    *
493    * @param setOfAxioms the set of axioms
494    * @param factory the factory
495    * @param pm the pm
496    * @param conceptVersionBI the concept version bi
497    * @return the OWL object property
498    * @throws Exception the exception
499    */
500   private OWLObjectProperty createOWLPropertyAxiomsFromConceptVersionBI(
501     Set<OWLAxiom> setOfAxioms, OWLDataFactory factory, PrefixManager pm,
502     ConceptVersionBI conceptVersionBI) throws Exception {
503     // Declaration
504     OWLObjectProperty owlPropertyClass =
505         factory.getOWLObjectProperty(
506             ":" + getSnomedConceptID(conceptVersionBI), pm);
507     OWLDeclarationAxiom declarationAxiom =
508         factory.getOWLDeclarationAxiom(owlPropertyClass);
509     setOfAxioms.add(declarationAxiom);
510     // Create annotation type axiom
511     // Fully specified description
512     OWLAnnotation labelAnnotation =
513         factory.getOWLAnnotation(factory.getRDFSLabel(), factory.getOWLLiteral(
514             conceptVersionBI.getFullySpecifiedDescription().getText(),
515             conceptVersionBI.getFullySpecifiedDescription().getLang()));
516     OWLAxiom labelAnnotationAxiom =
517         factory.getOWLAnnotationAssertionAxiom(owlPropertyClass.getIRI(),
518             labelAnnotation);
519     setOfAxioms.add(labelAnnotationAxiom);
520 
521     // Preferred name
522     OWLAnnotation preferredTerm =
523         factory.getOWLAnnotation(factory.getOWLAnnotationProperty(IRI
524             .create(snomedEnUsPreferredAnnotation)), factory.getOWLLiteral(
525             conceptVersionBI.getPreferredDescription().getText(),
526             conceptVersionBI.getPreferredDescription().getLang()));
527     OWLAxiom preferredTermAx =
528         factory.getOWLAnnotationAssertionAxiom(owlPropertyClass.getIRI(),
529             preferredTerm);
530     setOfAxioms.add(preferredTermAx);
531 
532     // SubObjectPropertyOf
533     // SubObjectPropertyOf(ObjectPropertyChain(
534     // <http://snomed.info/id/363701004> <http://snomed.info/id/127489000> )
535     // <http://snomed.info/id/363701004>)
536     for (RelationshipVersionBI<?> rel : conceptVersionBI
537         .getRelationshipsOutgoingActiveIsa()) {
538       if (rel.isStated()) {
539         String sctid =
540             getSnomedConceptID(WBUtility.getConceptVersion(rel
541                 .getDestinationNid()));
542         // Skip the root of the attributes tree
543         if (!sctid.equals(snomedConceptAttributeModelConcept)) {
544           OWLObjectProperty parent =
545               factory.getOWLObjectProperty(":" + sctid, pm);
546           setOfAxioms.add(factory.getOWLSubObjectPropertyOfAxiom(
547               owlPropertyClass, parent));
548         }
549       }
550     }
551 
552     return owlPropertyClass;
553   }

556    * Indicates whether or not US language refex is the case.
557    *
558    * @param desc the desc
559    * @return <code>true</code> if so, <code>false</code> otherwise
560    * @throws Exception the exception

562   private boolean isUSLanguageRefex(DescriptionVersionBI<?> desc)
563     throws Exception {
564     for (RefexVersionBI<?> annotation : desc.getAnnotationsActive(WBUtility
565         .getViewCoordinate())) {
566       if (annotation.getAssemblageNid() == Snomed.US_LANGUAGE_REFEX.getNid()) {
567         return true;
568       }
569     }
570     return false;
