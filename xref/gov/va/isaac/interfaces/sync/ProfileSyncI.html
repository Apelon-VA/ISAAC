21  import java.io.File;
22  import java.io.IOException;
23  import java.util.Map;
24  import java.util.Set;
25  import javax.naming.AuthenticationException;
26  import org.jvnet.hk2.annotations.Contract;

28  /**
29   * {@link ProfileSyncI}
30   * 
31   * An interface to specify the various methods available by remote profile sync services.
32   * 
33   * Implementations of this interface include GIT and SVN backed remote sync services.
34   *
35   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
36   */
37  @Contract
38  public interface ProfileSyncI
39  {
40  	public static final String DEFAULT_README_CONTENT = "ISAAC Profiles Storage \r" + "=== \r" 
41  			+ "This is a repository for storing ISAAC profiles and changesets.\r"
42  			+ "It is highly recommended that you do not make changes to this repository manually - ISAAC interfaces with this.";
43  	
44  	/**
45  	 * Set the base folder that should be used in this instance of Profile Sync.  Necessary for HK2 style 
46  	 * init where it can't be passed in via the constructor.
47  	 * 
48  	 * @param localFolder - full path the the folder that should be synchronizable
49  	 * @throws IllegalArgumentException - if the passed in file is not a folder, or does not exist.
50  	 */
51  	public abstract void setRootLocation(File localFolder) throws IllegalArgumentException;
52  	
53  	/**
54  	 * @param readmeFileContent - The content to place in the README file that the implementation will create (if it doesn't exist).
55  	 * If this method is never called, the implementation will default to the text from {@link ProfileSyncI#DEFAULT_README_CONTENT}.
56  	 * If left blank - the README file may still be created by the implementation, as some SCM systems don't allow a commit with an empty
57  	 * repository.
58  	 */
59  	public abstract void setReadmeFileContent(String readmeFileContent);
60  	
61  	/**
62  	 * Return the currently configured root location.  
63  	 * @return full path the the folder that should be synchronizable (or null, if not configured)
64  	 */
65  	public abstract File getRootLocation();
66  	
67  	/**
68  	 * Connect to remote service - pull down any existing files in the remote service.
69  	 * <br><br>
70  	 * In general, this operation should not result in any add / commit operations (though implementations may add/commit a 'README' file
71  	 * if necessary to complete the linkage in the case of empty repositories - in fact, are encouraged to ensure that a reasonable README
72  	 * file exists on both the client and the server at the end of this operation)
73  	 * <br><br>
74  	 * This is NOT intended to be called from a running application - rather - it should only be called during the initial configuration 
75  	 * (typically done during the build sequence) as it doesn't provide any feedback as to what local files changed, during its operations.
76  	 * <pre>
77  	 * Implementers should handle the following cases:
78  	 * 
79  	 *  * Remote repository exists, but is empty
80  	 *    o Local folder exists - is not a repository - may or may not contain content.
81  	 *      ==> The expected behavior for this case is that the the local folder will be turned into a repository managed folder, 
82  	 *          and the remote server should be linked to it.  Implementers may commit and push a single file, such as a 'README' file
83  	 *          if necessary - as implementations like GIT and SVN don't allow syncing when there is nothing to sync.  Any other pre-existing
84  	 *          local content should NOT be added, committed or synced as part of this operation.
85  	 *    o Local folder exists, and is already a repository - may or may not contain content.
86  	 *      ==> The expected behavior for this case is that the local repository should be linked to the remote repository.  In the case of GIT, 
87  	 *          it is acceptable to push the local repository directly to the remote server.  This will push any content that was previously committed to the 
88  	 *          repository - but any existing content in the local folder that has not been added or committed should not be pushed to the remote server.  
89  	 *          For other source control systems, this is likely an impossible state - and they may simply discard the local repository information - and link 
90  	 *          to the new location.  Any local content should be preserved, but not committed or added to the remote repository.
91  	 *
92  	 *  * Remote repository exists, and is populated
93  	 *    o Local folder exists - is not a repository - may or may not contain content
94  	 *      ==> The expected behavior for this case is that the server content should be checked out locally - and added to the local folder.  If local files 
95  	 *          already exist that would be overwritten by the checkout (because they also exist on the server) the local files should be PRESERVED - not overwritten.  
96  	 *          The end result of the operation would show these files in a MODIFIED state.
97  	 *    o Local folder exists, and is already a repository - may or may not contain content.
98  	 *      ==> The expected behavior for this case is that the server and local repositories should be linked, and the content from the server should be checked
99  	 *          out and added to the local folder.  The assumption is made that the client and server were a pair at one point - this should be the equivalent of 
100 	 *          updating the client to point to a new URL for the server if the server was relocated, for example.  
101 	 *          If the client repository is found to be incompatible with the server repository - then the client state should be discarded - and a new checkout 
102 	 *          should be done from the server.  Any local files which have naming collisions with server files should be PRESERVED during the checkout - not 
103 	 *          overwritten - leaving them in a MODIFIED state, if they happen to differ from the files that were on the server.
104 	 * </pre>
105 	 * @param remoteAddress - the URL to the remote server
106 	 * @param username - The username to use for remote operations
107 	 * @param password - The password to use for remote operations
108 	 * @throws IOException - Thrown if an error occurs accessing local or remote resources
109 	 * @throws IllegalArgumentException - if the passed parameters are invalid
110 	 * @throws AuthenticationException 
111 	 */
112 	public void linkAndFetchFromRemote(String remoteAddress, String username, String password) throws IllegalArgumentException, IOException, AuthenticationException;
113 	
114 	/**
115 	 * Fix the URL to the remote service.  This call should only be used when both the local and remote repositories exist, and are a proper pair - 
116 	 * but the URL for the remote service needs to be corrected, for whatever reason (for example, the domain name changed)
117 	 * 
118 	 * Has no impact on any local files.
119 	 *
120 	 * @param remoteAddress - the URL to the remote server
121 	 * @param username - remote credentials
122 	 * @param password - remote credentials
123 	 * @throws IOException - Thrown if an error occurs accessing local or remote resources
124 	 * @throws IllegalArgumentException - if the passed parameters are invalid
125 	 * @throws AuthenticationException - if auth fails during remote relink 
126 	 */
127 	public void relinkRemote(String remoteAddress, String username, String password) throws IllegalArgumentException, IOException, AuthenticationException;
128 	
129 	/**
130 	 * Mark the specified files as files that should be synchronized.  This is a local operation only - does not push to the server.
131 	 * @param files - The relative path of each file that should be added (relative to the localFolder)
132 	 * @throws IOException - Thrown if an error occurs accessing local or remote resources
133 	 * @throws IllegalArgumentException - if the passed parameters are invalid
134 	 */
135 	public void addFiles(String ... files) throws IllegalArgumentException, IOException;
136 	
137 	/**
138 	 * Equivalent of calling {@link #addFiles(File, Set)} for each file in the localFolder which is currently unmanaged.
139 	 * @throws IOException - Thrown if an error occurs accessing local or remote resources
140 	 * @throws IllegalArgumentException - if the passed parameters are invalid
141 	 */
142 	public void addUntrackedFiles() throws IllegalArgumentException, IOException;
143 	
144 	/**
145 	 * Mark the specified files as files that should be removed from the server.  This is a local operation only - does not push to the server.
146 	 * If the file exists locally, it will be removed by this operation.
147 	 * @param files - The relative path of each file that should be removed (deleted) (relative to the localFolder)
148 	 * @throws IOException - Thrown if an error occurs accessing local or remote resources
149 	 * @throws IllegalArgumentException - if the passed parameters are invalid
150 	 */
151 	public void removeFiles(String ... files) throws IllegalArgumentException, IOException;
152 	
153 	/**
154 	 * Update (all), commit and push the specified files to the remote server.  This can include files that have been modified, added or removed.
155 	 * Assuming that the status call reflects that state.  The implementation will also perform an update from remote as part of this operation.
156 	 * @param commitMessage - the message to attach to this commit
157 	 * @param username - the username to use to push the commit remotely
158 	 * @param password - The password to use to push the commit remotely
159 	 * @param mergeFailOption - (optional - defaults to {@link MergeFailOption#FAIL}) - the action to take if the required update results in a merge conflict.
160 	 * @param files - The list of files to commit.  May be empty, to support cases where a commit was completed, but the push failed due to a merge issue that
161 	 * required resolution.  May be null to request that all tracked files with changes be committed and pushed.
162 	 * @throws IllegalArgumentException - Thrown if an error occurs accessing local or remote resources
163 	 * @throws IOException - if the passed parameters are invalid
164 	 * @throws MergeFailure - If the update cannot be applied cleanly.  The exception will contain the list of files that were changed (cleanly, or not) during the
165 	 * update attempt.
166 	 * @return The set of files that changed during the pull from the server.
167 	 * @throws AuthenticationException 
168 	 */
169 	public Set<String> updateCommitAndPush(String commitMessage, String username, String password, MergeFailOption mergeFailOption, String ... files) 
170 			throws IllegalArgumentException, IOException, MergeFailure, AuthenticationException;
171 	
172 	/**
173 	 * Get the latest files from the server.  
174 	 * @param username - the username to use to pull the updates
175 	 * @param password - The password to use to pull the updates
176 	 * @param mergeFailOption - (optional - defaults to {@link MergeFailOption#FAIL}) - the action to take if the required update results in a merge conflit.
177 	 * @throws IllegalArgumentException - Thrown if an error occurs accessing local or remote resources
178 	 * @throws IOException - if the passed parameters are invalid
179 	 * @throws MergeFailure - If the update cannot be applied cleanly.  The exception will contain the list of files that were changed (cleanly, or not) during the
180 	 * update attempt.
181 	 * @return The set of files that changed during the pull from the server.
182 	 * @throws AuthenticationException 
183 	 */
184 	public Set<String> updateFromRemote(String username, String password, MergeFailOption mergeFailOption) 
185 			throws IllegalArgumentException, IOException, MergeFailure, AuthenticationException;
186 
187 	/**
188 	 * If {@link #updateCommitAndPush(File, String, String, String, MergeFailOption, String...) or {@link #updateFromRemote(File, String, String, MergeFailOption)}
189 	 * resulted in a MergeFailure exception, this method should be called to specify how to resolve each merge failure.
190 	 * 
191 	 * After calling this, you may call {@link #updateCommitAndPush(File, String, String, String, MergeFailOption, String...)} again.
192 	 * 
193 	 * Note - some implementations (specifically GIT) may throw another {@link MergeFailure} during this operation - this is a secondary merge failure
194 	 * which will also have to be resolved by the user (by calling this method again) before you can commit and push.
195 	 * 
196 	 * @param resolutions - A map where each key is a relative file name of a file that had a mergeFailure, and the corresponding value is
197 	 * the action that should be taken to resolve the issue.  Note that {@link MergeFailOption#FAIL} it not a valid option.
198 	 * @throws IllegalArgumentException - Thrown if an error occurs accessing local or remote resources
199 	 * @throws IOException - if the passed parameters are invalid
200 	 * @throws MergeFailure - If the update cannot be applied cleanly.  The exception will contain the list of files that were changed (cleanly, or not) during the
201 	 * update attempt.
202 	 * @return At a minimum, the set of files modified by the resolution of the merge.  Implementation may, at their choosing , return the complete set of files 
203 	 * that changed during the pull from the server that led to the merge failure, in addition to the files that were changes to resolve the conflicts.
204 	 */
205 	public Set<String> resolveMergeFailures(Map<String, MergeFailOption> resolutions) throws IllegalArgumentException, IOException, MergeFailure;
206 	
207 	/**
208 	 * Returns true if the specified location appears to be a SCM store, false otherwise.
209 	 */
210 	public boolean isRootLocationConfiguredForSCM();
211 	
212 	/**
213 	 * Check the local SCM status, and get a count of files that have changes that will be pushed.
214 	 * @throws IOException 
215 	 */
216 	public int getLocallyModifiedFileCount() throws IOException;
217 	
218 	
219 	/**
220 	 * Get the list of files currently in a merge-conflicted state.  This should normally return an empty set.  Would typically only be called 
221 	 * if a merge sequence was abnormally aborted.  This would allow you to get the files that still need to be merged, then call resolveMergeFailures...
222 	 * @throws IOException 
223 	 */
224 	public Set<String> getFilesInMergeConflict() throws IOException;
225 	
226 	/**
227 	 * An optional method that allows implementations to do an implementation specific substitution on the URL that will be used to connect.
228 	 * 
229 	 * For example, in the GIT implementation:
230 	 *  ssh://someuser@csfe.aceworkspace.net:29418/... needs to become:
231 	 *  ssh://<getUsername()>@csfe.aceworkspace.net:29418/...
232 	 * 
233 	 * The default implementation simply returns url directly back.
234 	 * 
235 	 * @param url
236 	 * @param username
237 	 * @return The substituted URL for this implementation.  Return url directly, if no substitution is required.
238 	 */
239 	public default String substituteURL(String url, String username)
240 	{
241 		return url;
242 	}
