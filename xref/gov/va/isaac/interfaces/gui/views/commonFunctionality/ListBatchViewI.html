30  import gov.va.isaac.interfaces.gui.views.EmbeddableViewI;
31  import java.util.List;
32  
33  /**
34   * {@link ListBatchViewI}
35   * 
36   * An interface that allows the creation of an ListBatchViewI implementation, which 
37   * will be a JavaFX component that extends {@link EmbeddableViewI}.  The resulting ISAAC panel
38   * is intended to allow importation into, display and manipulation of concepts to and
39   * from other ISAAC panels
40   *
41   * @author <a href="jkniaz@apelon.com">Joel Kniaz</a>
42   */
43  @Contract
44  public interface ListBatchViewI extends EmbeddableViewI {
45  	/**
46  	 * Update the view to show the specified concepts
47  	 * @param List<Integer> the list of specified concepts as int NID
48  	 */
49  	public void addConcepts(List<Integer> nids);
50  
51  	/**
52  	 * Update the view to show the specified concept
53  	 * @param int the specified concept as int NID
54  	 */
55  	public void addConcept(int nid);
56  }
