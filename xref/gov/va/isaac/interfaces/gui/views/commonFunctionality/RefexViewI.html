19  package gov.va.isaac.interfaces.gui.views.commonFunctionality;
20  
21  import gov.va.isaac.interfaces.gui.views.EmbeddableViewI;
22  import javafx.beans.property.ReadOnlyBooleanProperty;
23  import javafx.scene.layout.Region;
24  import org.jvnet.hk2.annotations.Contract;
25  
26  /**
27   * RefsetViewI
28   * 
29   * An interface that allows the creation of a RefexView implementation, which 
30   * will be a JavaFX component that extends {@link Region} that can be embedded
31   * into other views, the purpose of which is to display Refex information for 
32   * a component.
33   *
34   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
35   */
36  @Contract
37  public interface RefexViewI extends EmbeddableViewI
38  {
39  	/**
40  	 * Tell this view to display the refexes for a particular component (typically, a concept, but could also 
41  	 * be any valid thing that supportes refexes)
42  	 * 
43  	 * With this call - multiple assemblages will be shown - for a single component (concept)
44  	 * 
45  	 * @param componentNid - the component to show.
46  	 * @param showStampColumns - (optional) if provided, don't show toggle buttons for show/hiding the stamp, instead bind to this.
47  	 * @param showActiveOnly - (optional) if provided, don't show toggle buttons for activeOnly / all, instead bind to this.
48  	 * @param showFullHistory - (optional) if provided, don't show toggle buttons for show current / show all, instead bind to this.
49  	 */
50  	public void setComponent(int componentNid, ReadOnlyBooleanProperty showStampColumns, ReadOnlyBooleanProperty showActiveOnly, 
51  			ReadOnlyBooleanProperty showFullHistory);
52  	
53  	/**
54  	 * Tell this view to display the refexes for a particular assemblage concept.
55  	 * 
56  	 * With this call - only a single assemblage will be shown - but multiple components may be shown (typically concepts, 
57  	 * but could also be any valid thing that supports refexes, such as descriptions)
58  	 * 
59  	 * @param assemblageConceptNid - the assemblage to show
60  	 * @param showStampColumns - (optional) if provided, don't show toggle buttons for show/hiding the stamp, instead bind to this.
61  	 * @param showActiveOnly - (optional) if provided, don't show toggle buttons for activeOnly / all, instead bind to this.
62  	 * @param showFullHistory - (optional) if provided, don't show toggle buttons for show current / show all, instead bind to this.
63  	 */
64  	public void setAssemblage(int assemblageConceptNid, ReadOnlyBooleanProperty showStampColumns, ReadOnlyBooleanProperty showActiveOnly, 
65  			ReadOnlyBooleanProperty showFullHistory);
66  }
