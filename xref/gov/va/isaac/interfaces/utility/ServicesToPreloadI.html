19  package gov.va.isaac.interfaces.utility;
20  
21  import org.jvnet.hk2.annotations.Contract;
22  
23  
24  
25  /**
26   * {@link ServicesToPreloadI}
27   *
28   * An interface that allows implementers to easily mark their Service as something that should be 
29   * populated ASAP, when the application starts.  The main application controller will iterate over 
30   * these, requesting each to be constructed after initial startup.
31   * 
32   * Typically, this would only be used on Singleton implementations... 
33   * 
34   * All implementations should background thread any real work that happens as a consequence of being
35   * constructed.  No expensive operations should be done either in the constructor, or in the loadRequested() 
36   * method on the calling thread.
37   *
38   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
39   */
40  
41  @Contract
42  public interface ServicesToPreloadI
43  {
44  	/**
45  	 * Implementers don't actually have to do anything when called here... they can do the init in the constructor if 
46  	 * the prefer. This is just to prevent the compiler from doing something silly and dropping the for-loop over the 
47  	 * ServiceToPreloadI objects, when it looks like nothing is actually being done.  Implementors may do something here,
48  	 *  however, if they prefer to have the constructor do nothing, and instead, handle the loading when loadRequested 
49  	 *  is called.  This will be called during the startup sequence, after the database init has completed.
50  	 * 
51  	 * If any expensive works needs to be done as a result of loadRequested - it should be done in a background thread - loadRequested
52  	 * should return immediately.
53  	 */
54  	public void loadRequested();
55  	
56  	/**
57  	 * Called when the application is shutting down
58  	 */
59  	public void shutdown();
60  }
