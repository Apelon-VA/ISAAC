19  package gov.va.isaac.interfaces.utility;
20  
21  import org.jvnet.hk2.annotations.Contract;
22  
23  
24  
25  /**
26   * {@link CommitListenerI}
27   *
28   * An interface that allows implementers to easily mark their Service as something that will register 
29   * itself with {@link BdbTerminologyStore#addPropertyChangeListener} as a class that does things on 
30   * preCommit / Commit / postCommit.
31   * 
32   * The only intent of this interface is to provide a global handle to all modules which are registered to do 
33   * things - so that then can be enabled or disabled individually, or across the board prior to performing 
34   * programmatic operations where you don't want the listeners firing.
35   * 
36   * An example of this - would be disabling the Workflow Initiation listener prior to importing batch data.
37   * 
38   * =====================================
39   * Implementers of this class MUST use a @Named annotation:
40   * <code>@Named (value="the name")</code>
41   * =====================================
42   *
43   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
44   */
45  
46  @Contract
47  public interface CommitListenerI
48  {
49  	/**
50  	 * Returns the HK2 service name of this listener.  This should return the same string as the 
51  	 * <code>@Named (value="the name")</code>
52  	 * annotation on the implementation of the class.
53  	 */
54  	public String getListenerName();
55  	
56  	/**
57  	 * Tell the listener that it MAY enable itself (it does not have to).  A listener may be primarily controlled
58  	 * by a user preference, for example.  While {@link #disable()} overrides the user preferences, and orders the 
59  	 * listener disabled - this call only informs the user that it is no longer being overridden - and may return to 
60  	 * the enabled state if that was the users preference.
61  	 */
62  	public void enable();
63  	
64  	/**
65  	 * Tell the listener to disable itself - either by unregistering upstream, or ignoring all commit related events 
66  	 * until {@link #enable()} is called.
67  	 */
68  	public void disable();
69  }
