51  /**
52   * WorkflowPreferencesPluginView
53   * 
54   * @author <a href="mailto:joel.kniaz@gmail.com">Joel Kniaz</a>
55   *
56   */
57  
58  @Service
59  @Singleton
60  public class DroolsPreferencesPluginView implements PreferencesPluginViewI {
61  	private Logger logger = LoggerFactory.getLogger(DroolsPreferencesPluginView.class);
62  
63  	private GridPane gridPane = null;
64  	protected ValidBooleanBinding allValid_ = null;
65  	
66  	private final BooleanProperty runDroolsBeforeEachCommitProperty = new SimpleBooleanProperty();
67  
68  	/* (non-Javadoc)
69  	 * @see gov.va.isaac.interfaces.gui.views.commonFunctionality.PreferencesPluginViewI#getValidationFailureMessage()
70  	 */
71  	@Override
72  	public ReadOnlyStringProperty validationFailureMessageProperty() {
73  		return allValid_.getReasonWhyInvalid();
74  	}
75  
76  	/* (non-Javadoc)
77  	 * @see gov.va.isaac.interfaces.gui.views.commonFunctionality.PreferencesPluginViewI#getNode()
78  	 */
79  	@Override
80  	public Region getContent() {
81  		if (gridPane == null) {
82  			gridPane = new GridPane();
83  			
84  			allValid_ = new ValidBooleanBinding() {
85  				{
86  					setComputeOnInvalidate(true);
87  				}
88  				
89  				@Override
90  				protected boolean computeValue() {
91  					this.clearInvalidReason();
92  					return true;
93  				}
94  			};
95  			
96  			Label runDroolsBeforeEachCommitCheckBoxLabel = new Label("Run Drools Before Each Commit");
97  			runDroolsBeforeEachCommitCheckBoxLabel.setPadding(new Insets(5, 5, 5, 5));
98  			CheckBox runDroolsBeforeEachCommitCheckBox = new CheckBox();
99  			runDroolsBeforeEachCommitCheckBox.setPadding(new Insets(5, 5, 5, 5));
100 			runDroolsBeforeEachCommitCheckBox.setMaxWidth(Double.MAX_VALUE);
101 			runDroolsBeforeEachCommitCheckBox.setTooltip(new Tooltip("Default is " + UserProfileDefaults.getDefaultRunDroolsBeforeEachCommit()));
102 			runDroolsBeforeEachCommitCheckBox.selectedProperty().bindBidirectional(runDroolsBeforeEachCommitProperty);
103 			
104 			// load/set current preferences values
105 			UserProfile loggedIn = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
106 			runDroolsBeforeEachCommitCheckBox.selectedProperty().set(loggedIn.isRunDroolsBeforeEachCommit());
107 
108 			// Format GridPane
109 			int row = 0;
110 			gridPane.setMaxWidth(Double.MAX_VALUE);
111 
112 			gridPane.addRow(row++, runDroolsBeforeEachCommitCheckBoxLabel, runDroolsBeforeEachCommitCheckBox);
113 			GridPane.setHgrow(runDroolsBeforeEachCommitCheckBoxLabel, Priority.NEVER);
114 			GridPane.setFillWidth(runDroolsBeforeEachCommitCheckBox, true);
115 			GridPane.setHgrow(runDroolsBeforeEachCommitCheckBox, Priority.ALWAYS);
116 		}
117 		
118 		return gridPane;
119 	}
120 
121 	/* (non-Javadoc)
122 	 * @see gov.va.isaac.interfaces.gui.views.commonFunctionality.PreferencesPluginViewI#getName()
123 	 */
124 	@Override
125 	public String getName() {
126 		return "Drools";
127 	}
128 
129 	/* (non-Javadoc)
130 	 * @see gov.va.isaac.interfaces.gui.views.commonFunctionality.PreferencesPluginViewI#save()
131 	 */
132 	@Override
133 	public void save() throws IOException {
134 		logger.debug("Saving {} preferences", getName());
135 		
136 		UserProfile loggedIn = ExtendedAppContext.getCurrentlyLoggedInUserProfile();
137 		loggedIn.setRunDroolsBeforeEachCommit(this.runDroolsBeforeEachCommitProperty.get());
138 
139 		try {
140 			AppContext.getService(UserProfileManager.class).saveChanges(loggedIn);
141 		} catch (InvalidUserException e) {
142 			String msg = "Caught " + e.getClass().getName() + " " + e.getLocalizedMessage() + " attempting to save UserProfile";
143 			
144 			logger.error(msg, e);
145 			throw new IOException(msg, e);
146 		}
147 	}
148 	
149 	/**
150 	 * @see gov.va.isaac.interfaces.gui.views.commonFunctionality.PreferencesPluginViewI#getTabOrder()
151 	 */
152 	@Override
153 	public int getTabOrder()
154 	{
155 		return 60;
156 	}
157 }
