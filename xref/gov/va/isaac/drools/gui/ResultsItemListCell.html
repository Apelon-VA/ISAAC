29  import javafx.scene.layout.Pane;
30  import javafx.scene.layout.Priority;
31  import javafx.scene.layout.Region;

33  /**
34   * {@link ResultsItemListCell}
35   *
36   * Display code for a data column of a Dynamic Refex, when shown within a list view (one cell per refex column)
37   *
38   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
39   */
40  public class ResultsItemListCell extends ListCell<ResultsItem>
41  {
42  	//TODO display cleanup, mechnism to remove failures
43  	/**
44  	 * @see javafx.scene.control.Cell#updateItem(java.lang.Object, boolean)
45  	 */
46  	@Override
47  	protected void updateItem(ResultsItem item, boolean empty)
48  	{
49  		super.updateItem(item, empty);
50  		if (item != null)
51  		{
52  			setText("");
53  
54  			GridPane gp = new GridPane();
55  			gp.setHgap(0.0);
56  			gp.setVgap(0.0);
57  			gp.setPadding(new Insets(5, 5, 5, 5));
58  			gp.setMinWidth(250);
59  
60  			ColumnConstraints constraint1 = new ColumnConstraints();
61  			constraint1.setFillWidth(false);
62  			constraint1.setHgrow(Priority.NEVER);
63  			constraint1.setMinWidth(160);
64  			constraint1.setMaxWidth(160);
65  			gp.getColumnConstraints().add(constraint1);
66  
67  			ColumnConstraints constraint2 = new ColumnConstraints();
68  			constraint2.setFillWidth(true);
69  			constraint2.setHgrow(Priority.SOMETIMES);
70  			gp.getColumnConstraints().add(constraint2);
71  
72  			int row = 0;
73  
74  			gp.add(wrapAndStyle(makeBoldLabel("Message"), row), 0, row);
75  			Label message = new Label(item.getMessage());
76  			message.setWrapText(true);
77  			message.maxWidthProperty().bind(this.widthProperty().subtract(180));
78  			gp.add(wrapAndStyle(message, row), 1, row++);
79  
80  			gp.add(wrapAndStyle(makeBoldLabel("Severity"), row), 0, row);
81  			Label severity = new Label(item.getSeverity().getName());
82  			severity.setWrapText(true);
83  			severity.maxWidthProperty().bind(this.widthProperty().subtract(180));
84  			gp.add(wrapAndStyle(severity, row), 1, row++);
85  			
86  			gp.add(wrapAndStyle(makeBoldLabel("Rule UUID"), row), 0, row);
87  			Label ruleUUID = new Label(item.getRuleUuid());
88  			ruleUUID.setWrapText(true);
89  			ruleUUID.maxWidthProperty().bind(this.widthProperty().subtract(180));
90  			gp.add(wrapAndStyle(ruleUUID, row), 1, row++);
91  			
92  			gp.add(wrapAndStyle(makeBoldLabel("Error Code"), row), 0, row);
93  			Label errorCode = new Label(item.getErrorCode() + "");
94  			errorCode.setWrapText(true);
95  			errorCode.maxWidthProperty().bind(this.widthProperty().subtract(180));
96  			gp.add(wrapAndStyle(errorCode, row), 1, row++);
97  
98  			setGraphic(gp);
99  			this.setStyle("-fx-border-width:  0 0 2 0; -fx-border-color: grey; ");
100 
101 		}
102 		else
103 		{
104 			setText("");
105 			setGraphic(null);
106 			this.setStyle("");
107 		}
108 	}
109 
110 	private Node wrapAndStyle(Region node, int rowNumber)
111 	{
112 		Pane p = new Pane(node);
113 		node.setPadding(new Insets(5.0));
114 		GridPane.setFillWidth(p, true);
115 		p.minHeightProperty().bind(node.heightProperty());
116 		//Hack - wrapped labels don't seem to fire their height property changes at the right time - leaving the surrounding Pane node too small.
117 		//this seems to help...
118 		Platform.runLater(() -> p.autosize());
119 		p.getStyleClass().add(((rowNumber % 2 == 0) ? "evenGridRow" : "oddGridRow"));
120 		return p;
121 	}
122 
123 	private Label makeBoldLabel(String labelText)
124 	{
125 		Label l = new Label(labelText);
126 		l.getStyleClass().add("boldLabel");
127 		return l;
128 	}
