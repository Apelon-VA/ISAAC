39  public class FactFactory
40  {
41  
42  	public static Fact<?> get(Context context, Object component, ViewCoordinate vc)
43  	{
44  		if (ConceptAttributeVersionBI.class.isAssignableFrom(component.getClass()))
45  		{
46  			return new ConAttrFact(context, (ConceptAttributeVersionBI<?>) component, vc);
47  		}
48  		if (ConceptVersionBI.class.isAssignableFrom(component.getClass()))
49  		{
50  			return new ConceptFact(context, (ConceptVersionBI) component, vc);
51  		}
52  		if (DescriptionVersionBI.class.isAssignableFrom(component.getClass()))
53  		{
54  			return new DescFact(context, (DescriptionVersionBI<?>) component, vc);
55  		}
56  		if (RelationshipVersionBI.class.isAssignableFrom(component.getClass()))
57  		{
58  			return new RelFact(context, (RelationshipVersionBI<?>) component, vc);
59  		}
60  		if (RelGroupVersionBI.class.isAssignableFrom(component.getClass()))
61  		{
62  			return new RelGroupFact(context, (RelGroupVersionBI) component, vc);
63  		}
64  		if (RelSpec.class.isAssignableFrom(component.getClass()))
65  		{
66  			return new RelSpecFact(context, (RelSpec) component);
67  		}
68  		if (DescriptionSpec.class.isAssignableFrom(component.getClass()))
69  		{
70  			return new DescSpecFact(context, (DescriptionSpec) component);
71  		}
72  		if (RefexVersionBI.class.isAssignableFrom(component.getClass()))
73  		{
74  			return new RefexFact(context, (RefexVersionBI<?>) component, vc);
75  		}
76  		throw new UnsupportedOperationException("Can't handle component: " + component);
77  	}
78  
79  	public static Fact<?> get(View view)
80  	{
81  		return new ViewFact<View>(view);
82  	}
