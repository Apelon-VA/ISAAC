281 	/**
282 	 * @param variableName
283 	 * 
284 	 * Creates ComponentsForOutputVariable entry for specified variableName
285 	 * by invoking overridable methods and overridden abstract method createOutputVariableInputNode().
286 	 * 
287 	 * Note that while implementations of createOutputVariableInputNode() may include references to
288 	 * the Label and other properties, createOutputVariableInputNodeLabel() may not refer to
289 	 * any of these, as they will not exist until the Label has already been created
290 	 * and the respective ComponentsForOutputVariable has been put into the componentsForOutputVariables map
291 	 * 
292 	 * This method is final because the order of its operations is significant
293 	 * 
294 	 */
295 	protected final void addOutputVariable(String variableName) {
296 		Label newLabel = createOutputVariableInputNodeLabel(variableName);
297 		ComponentsForOutputVariable componentsForOutputVariable = new ComponentsForOutputVariable(variableName, newLabel);
298 		componentsForOutputVariables.put(variableName, componentsForOutputVariable);
299 
300 		BooleanProperty variableStatusProperty = getOutputVariableStatusProperty(variableName);
301 		variableStatusProperty.addListener(new ChangeListener<Boolean>() {
302 			@Override
303 			public void changed(
304 					ObservableValue<? extends Boolean> observable,
305 					Boolean oldValue,
306 					Boolean newValue) {
307 				if (newValue != null && newValue) {
308 					newLabel.setStyle("-fx-text-fill: -fx-text-base-color;");
309 					if (areOutputVariablesSavable()) {
310 						getOutputVariablesSavableProperty().set(true);
311 					}
312 					if (isSavable()) {
313 						getIsSavableProperty().set(true);
314 					}
315 				} else {
316 					newLabel.setStyle("-fx-text-fill: red;");
317 					getOutputVariablesSavableProperty().set(false);
318 					getIsSavableProperty().set(false);
319 				}
320 			}});
321 
322 		
323 		Node newNode = getOrCreateOutputVariableInputNode(variableName);
324 		componentsForOutputVariable.setInputNode(newNode);
325 	}

355 	public boolean areOutputVariablesSavable() {
356 		for (Map.Entry<String, ComponentsForOutputVariable> entry : componentsForOutputVariables.entrySet()) {
357 			if (! entry.getValue().getStatusProperty().get()) {
358 				LOGGER.debug("Validation failed on variable {} with value {}", entry.getKey(), entry.getValue().getValueProperty().get());
359 				return false;
360 			}
361 		}
362 		
363 		return true;
364 	}
