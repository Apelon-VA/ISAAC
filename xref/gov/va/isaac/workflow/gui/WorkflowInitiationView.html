42  /**
43   * A {@link WorkflowInitiationViewI} which can be used to initiate a new workflow instance
44   * 
45   * @author <a href="mailto:joel.kniaz@gmail.com">Joel Kniaz</a>
46   */
47  @Service
48  @PerLookup
49  public class WorkflowInitiationView extends Stage implements WorkflowInitiationViewI
50  {
51  	private final Logger logger = LoggerFactory.getLogger(WorkflowInitiationView.class);
52  
53  	private WorkflowInitiationViewController controller_;
54  
55  	private boolean shown = false;
56  	
57  	private WorkflowInitiationView() throws IOException
58  	{
59  		super();
60  
61  		URL resource = this.getClass().getResource("WorkflowInitiationView.fxml");
62  		FXMLLoader loader = new FXMLLoader(resource);
63  		Parent root = (Parent) loader.load();
64  		setScene(new Scene(root));
65  		getScene().getStylesheets().add(WorkflowInitiationView.class.getResource("/isaac-shared-styles.css").toString());
66  		getIcons().add(Images.INBOX.getImage());
67  
68  		controller_ = loader.getController();
69  		
70  		controller_.setView(this);
71  		
72  		setTitle("Workflow Initiation");
73  		setResizable(true);
74  
75  		setWidth(600);
76  		setHeight(400);
77  	}
78  
79  	/**
80  	 * Call setReferencedComponent first
81  	 * 
82  	 * @see gov.va.isaac.interfaces.gui.views.PopupViewI#showView(javafx.stage.Window)
83  	 */
84  	@Override
85  	public void showView(Window parent)
86  	{
87  		if (! shown) {
88  			shown = true;
89  
90  			initOwner(parent);
91  			initModality(Modality.NONE);
92  			initStyle(StageStyle.DECORATED);
93  		}
94  
95  		show();
96  	}
97  
98  	public void setComponent(ComponentVersionBI componentOrConcept) {
99  		// Make sure in application thread.
100 		FxUtils.checkFxUserThread();
101 		try {
102 			controller_.setComponent(componentOrConcept);
103 		} catch (Exception e) {
104 			String title = "Unexpected error loading component " + componentOrConcept;
105 			String msg = "Caught " + e.getClass().getName() + " " + e.getLocalizedMessage();
106 			logger.error(title + ". " + msg, e);
107 			AppContext.getCommonDialogs().showErrorDialog(title, msg, e.getMessage());
108 		}
109 	}
110 	
111 	@Override
112 	public void setComponent(UUID uuid) {
113 		try {
114 			setComponent(OTFUtility.getComponentVersion(uuid));
115 		} catch (Exception e) {
116 			String title = "Unexpected error loading component with UUID " + uuid;
117 			String msg = "Caught " + e.getClass().getName() + " " + e.getLocalizedMessage();
118 			logger.error(title + ". " + msg, e);
119 			AppContext.getCommonDialogs().showErrorDialog(title, msg, e.getMessage());
120 		}
121 	}
122 
123 	@Override
124 	public void setComponent(int nid) {
125 		try {
126 			//TODO (artf231900) don't do BDB lookups on the FX Thread
127 			setComponent(OTFUtility.getComponentVersion(nid));
128 		} catch (Exception e) {
129 			String title = "Unexpected error loading component with UUID " + nid;
130 			String msg = "Caught " + e.getClass().getName() + " " + e.getLocalizedMessage();
131 			logger.error(title + ". " + msg, e);
132 			AppContext.getCommonDialogs().showErrorDialog(title, msg, e.getMessage());
133 		}
134 	}
135 
136 	@Override
137 	public UUID getComponentUuid() {
138 		return controller_.getComponent().getPrimordialUuid();
139 	}
140 
141 	@Override
142 	public int getComponentNid() {
143 		return controller_.getComponent().getNid();
144 	}
145 }
