94  	private final ActionEventListener actionEventListener = new ActionEventListener() {
95  		@Override
96  		public void handle(ActionEvent actionEvent) {
97  			if (actionEvent.getActionStatus() == TaskActionStatus.Pending) {
98  				List<LocalTask> displayedItems = new ArrayList<>(taskTable.getItems());
99  				for (LocalTask taskInTable : displayedItems) {
100 					if (taskInTable.getId() == actionEvent.getTaskId()) {
101 						Platform.runLater(() -> {
102 							taskTable.getItems().remove(taskInTable);
103 							LOG.debug("ActionEventListener removed {} task id #{} due to pending {} action", taskInTable.getName(), taskInTable.getId(), taskInTable.getAction());
104 						});
105 					}
106 				}
107 			}
108 		}
109 	};

298 	private void refreshContent()
299 	{
300 		// Remove actionEventListener to prevent concurrent modification of list
301 		taskService_.removeActionEventListener(actionEventListener);
302 		taskTable.getItems().clear();
303 		try
304 		{
305 			List<LocalTask> tasksForTable = taskService_.getOpenOwnedTasks();
306 			// Remove (do not display) tasks with TaskActionStatus.Pending
307 			List<LocalTask> tasksToRemove = new ArrayList<>();
308 			for (LocalTask task : tasksForTable) {
309 				if (task.getActionStatus() == TaskActionStatus.Pending) {
310 					tasksToRemove.add(task);
311 				}
312 			}
313 
314 			tasksForTable.removeAll(tasksToRemove);
315 			taskTable.getItems().addAll(tasksForTable);
316 		}
317 		catch (DatastoreException e)
318 		{
319 			LOG.error("Unexpected error refreshing tasks", e);
320 			AppContext.getCommonDialogs().showErrorDialog("There was a problem reading the tasks", e);
321 		} finally {
322 			taskService_.addActionEventListener(actionEventListener);
323 		}
324 	}

326 	private class MyCellFactoryCallback<T> implements Callback<TableColumn<LocalTask, T>, TableCell<LocalTask, T>> {
327 		@Override
328 		public TableCell<LocalTask, T> call(TableColumn<LocalTask, T> param) {
329 			TableCell<LocalTask, T> newCell = new TableCell<LocalTask, T>() {
330 				@Override
331 				public void updateItem(T item, boolean empty) {
332 					super.updateItem(item, empty);
333 					setText(empty ? null : getString());
334 					setGraphic(null);
335 				}
336 
337 				private String getString() {
338 					return getItem() == null ? "" : getItem().toString();
339 				}
340 			};
341 			newCell.addEventFilter(MouseEvent.MOUSE_CLICKED, new EventHandler<MouseEvent>() {
342 				@Override
343 				public void handle(MouseEvent event) {
344 					if (event.getButton() == MouseButton.SECONDARY) {
345 						@SuppressWarnings("unchecked")
346 						TableCell<LocalTask, T> c = (TableCell<LocalTask, T>) event.getSource();
347 
348 						if (c != null && c.getIndex() < c.getTableView().getItems().size()) {
349 							CommonMenusDataProvider dp = new CommonMenusDataProvider() {
350 								@Override
351 								public String[] getStrings() {
352 									List<String> items = new ArrayList<>();
353 									for (Integer index : c.getTableView().getSelectionModel().getSelectedIndices()) {
354 										items.add(c.getTableColumn().getCellData(index).toString());
355 									}
356 
357 									String[] itemArray = items.toArray(new String[items.size()]);
358 
359 									// TODO: determine why we are getting here multiple (2 or 3) times for each selection
360 									//System.out.println("Selected strings: " + Arrays.toString(itemArray));
361 
362 									return itemArray;
363 								}
364 							};
365 							CommonMenusTaskIdProvider taskIdProvider = new CommonMenusTaskIdProvider() {
366 								@Override
367 								public Set<Long> getTaskIds() {
368 									Set<Long> taskIds = new HashSet<Long>();
369 									for (LocalTask r : (ObservableList<LocalTask>)c.getTableView().getSelectionModel().getSelectedItems()) {
370 										taskIds.add(r.getId());
371 									}
372 
373 									return taskIds;
374 								}
375 							};
376 
377 							ContextMenu cm = new ContextMenu();
378 							CommonMenus.addCommonMenus(cm, dp, taskIdProvider);
379 
380 							c.setContextMenu(cm);
381 						}
382 					} else {
383 						TableCell<?, ?> c = (TableCell<?,?>) event.getSource();
384 
385 						if (event.getClickCount() == 1) {
386 							//LOG.debug(event.getButton() + " single clicked. Cell text: " + c.getText());
387 						} else if (event.getClickCount() > 1) {
388 							LOG.debug(event.getButton() + " double clicked. Cell text: " + c.getText());
389 							int cellIndex = c.getIndex();
390 							LocalTask task = taskTable.getItems().get(cellIndex);
391 
392 							WorkflowTaskDetailsViewI view = AppContext.getService(WorkflowTaskDetailsViewI.class);
393 							view.setTask(task.getId());
394 							view.showView(AppContext.getMainApplicationWindow().getPrimaryStage());
395 						}
396 					}
397 				}
398 			});
399 
400 			return newCell;
401 		}
