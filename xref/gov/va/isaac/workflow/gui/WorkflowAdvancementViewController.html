88  	@FXML
89  	void initialize()
90  	{
91  		AppContext.getServiceLocator().inject(this);
92  
93  		closeButton.setOnAction((action) -> {
94  			stage.close();
95  		});
96  		
97  		// Disabling saveActionButton until dependencies met 
98  		advanceButton.setDisable(true);
99  
100 		// Activation of save depends on taskModel.isSavable()
101 		advanceButton.setOnAction((action) -> {
102 			if (taskModel.isSavable()) {
103 				Platform.runLater(() -> 
104 				{
105 					unbindSaveActionButtonFromModelIsSavableProperty();
106 					advanceButton.setDisable(true);
107 				});
108 				final BusyPopover saveActionBusyPopover = BusyPopover.createBusyPopover("Saving action...", advanceButton);
109 
110 				Utility.execute(() -> {
111 					try
112 					{
113 						taskService_.completeTask(taskModel.getTask().getId(), taskModel.getCurrentOutputVariables());
114 						
115 						addToPromotionPath();
116 						
117 						Platform.runLater(() -> 
118 						{
119 							//refreshSaveActionButtonBinding();
120 
121 							if (stage != null) {
122 								stage.close();
123 							}
124 						});
125 					}
126 					catch (Exception e)
127 					{
128 						logger.error("Error saving task: unexpected " + e.getClass().getName() + " \"" + e.getLocalizedMessage() + "\"", e);
129 					} finally {
130 						cleanupAfterSaveTaskAction(this, saveActionBusyPopover);
131 					}
132 				});
133 			} else { // ! this.taskModel.isSavable()
134 				// This should never happen, if saveActionButton.setDisable(true) used in proper places
135 				logger.error("Error completing task: fields not set: task={}", taskModel.getTask());
136 			}
137 		});
138 	}

161 	public void setTask(long taskId) {
162 		//TODO Joel - what is a user supposed to do about any of these failures?  Shouldn't all of these be putting up a dialog informing them that the 
163 		//set task failed?  You can't just silently eat them - or can you eat some of these?
164 
165 		final String errorDialogTitle = "Failed Setting Workflow Task";
166 		final String errorDialogMsg = "Failed setting initial workflow task by id #" + taskId;
167 		if (taskModel != null) {
168 			String details = "Cannot reset initialTask from #" + taskModel.getTask().getId() + " to #" + taskId;
169 			logger.error(details);
170 			AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
171 
172 			stage.close();
173 			
174 			return;
175 		}
176 
177 		if (conceptVersion != null) {
178 			String details = "Cannot set initialTask to #" + taskId + " when conceptVersion is already set to " + new SimpleDisplayConcept(conceptVersion);
179 
180 			logger.error(details);
181 			AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
182 
183 			stage.close();
184 
185 			return;
186 		}
187 		
188 		initialTask = null;
189 		try
190 		{
191 			initialTask = taskService_.getTask(taskId);
192 		}
193 		catch (DatastoreException e1)
194 		{
195 			String details = "Failed loading initial workflow task #" + taskId + ". Caught " + e1.getClass().getName() + " " + e1.getLocalizedMessage();
196 			logger.error(details, e1);
197 
198 			AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
199 
200 			stage.close();
201 
202 			return;
203 		}
204 
205 		if (initialTask == null) {
206 			String details = "Failed loading initial workflow task #" + initialTask.getId() + " (got null)";
207 			
208 			logger.error(details);
209 			
210 			AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
211 
212 			stage.close();
213 
214 			return;
215 		}
216 		if (initialTask.getComponentId() == null) {
217 			String details = "Component ID for task #" + initialTask.getId() + " is null";
218 
219 			logger.error(details);
220 
221 			AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
222 
223 			//stage.close();
224 
225 			return;
226 		}
227 		UUID componentUuid = null;
228 		try {
229 			componentUuid = UUID.fromString(initialTask.getComponentId());
230 		} catch (IllegalArgumentException e) {
231 			String details = "Component ID for task #" + initialTask.getId() + " is not a valid UUID";
232 			logger.error(details);
233 
234 			AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
235 
236 			//stage.close();
237 
238 			return;
239 		}
240 
241 		ConceptVersionBI containingConcept = null;
242 		ComponentChronicleBI<? extends ComponentVersionBI> componentChronicle = WBUtility.getComponentChronicle(componentUuid);
243 		if (componentChronicle == null) {
244 			logger.warn("Component ID for task " + initialTask.getId() + " retrieved a null componentChronicle");
245 
246 			containingConcept = WBUtility.getConceptVersion(componentUuid);
247 			if (containingConcept == null) {
248 				String details = "Component ID for task #" + initialTask.getId() + " retrieved a null concept";
249 
250 				logger.error(details);
251 				
252 				AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
253 
254 				return;
255 			}
256 		} else {
257 			try {
258 				containingConcept = componentChronicle.getEnclosingConcept().getVersion(WBUtility.getViewCoordinate());
259 			} catch (Exception e) {
260 				String details = "Failed getting version from ComponentChronicleBI task " + initialTask.getId() + ".  Caught " + e.getClass().getName() + " " + e.getLocalizedMessage();
261 
262 				logger.error(details, e);
263 				
264 				AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
265 				
266 				return;
267 			}
268 
269 			if (containingConcept == null) {
270 				String details = "ComponentChronicleBI task " + initialTask.getId() + " contained a null enclosing concept";
271 
272 				logger.error(details);
273 				
274 				AppContext.getCommonDialogs().showErrorDialog(errorDialogTitle, errorDialogMsg, details, stage);
275 
276 				return;
277 			}
278 		}
279 
280 		if (componentChronicle == null) {
281 			logger.warn("Component id " + componentUuid + " for task " + initialTask.getId() + " is a concept, not just a component.");
282 		}
283 		
284 		conceptVersion = containingConcept;
285 		
286 		taskModel = TaskModelFactory.newTaskModel(initialTask, actionComboBox);
287 
288 		taskModel.getOutputVariablesSavableProperty().addListener(new ChangeListener<Boolean>() {
289 			@Override
290 			public void changed(
291 					ObservableValue<? extends Boolean> observable,
292 					Boolean oldValue,
293 					Boolean newValue) {
294 				
295 			}
296 		});
297 		
298 		loadContent();
299 	}
