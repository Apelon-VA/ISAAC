71      public LocalTask(TaskSummary summary, boolean fetchAttachments) throws RemoteException {
72          this.id = summary.getId();
73          this.name = summary.getName();
74          this.status = summary.getStatus();
75          if (summary.getActualOwner() != null) {
76              this.owner = summary.getActualOwner().getId();
77          } else {
78              this.owner = "";
79          }
80          if (fetchAttachments) {
81              LocalWorkflowRuntimeEngineBI wfEngine = AppContext.getService(LocalWorkflowRuntimeEngineBI.class);
82              Map<String, Object> vmap = wfEngine.getVariablesMapForTaskId(summary.getId());
83              this.componentId = (String) vmap.get("in_component_id");
84              this.componentName = (String) vmap.get("in_component_name");
85              this.name = (String) vmap.get("NodeName");
86              this.setInputVariables(new HashMap<String, String>());
87              for (String key : vmap.keySet()) {
88                  this.getInputVariables().put(key, vmap.get(key).toString());
89              }
90          }
91      }
92  
93      public LocalTask(Task task, boolean fetchAttachments) throws RemoteException {
94          this.id = task.getId();
95          this.name = task.getNames().iterator().next().getText();
96          this.status = task.getTaskData().getStatus();
97          if (task.getTaskData().getActualOwner() != null) {
98              this.owner = task.getTaskData().getActualOwner().getId();
99          } else {
100             this.owner = "";
101         }
102 
103         if (fetchAttachments) {
104             LocalWorkflowRuntimeEngineBI wfEngine = AppContext.getService(LocalWorkflowRuntimeEngineBI.class);
105             Map<String, Object> vmap = wfEngine.getVariablesMapForTaskId(task.getId());
106             this.componentId = (String) vmap.get("in_component_id");
107             this.componentName = (String) vmap.get("in_component_name");
108             this.name = (String) vmap.get("NodeName");
109             this.setInputVariables(new HashMap<String, String>());
110             for (String key : vmap.keySet()) {
111                 this.getInputVariables().put(key, vmap.get(key).toString());
112             }
113         }
114     }
