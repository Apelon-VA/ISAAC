72      @Override
73      public void requestProcessInstanceCreationToServer(ProcessInstanceCreationRequestI instanceRequest) throws RemoteException, DatastoreException {
74      	ProcessInstance newInstance = null;
75      	try
76          {
77              KieSession session = AppContext.getService(RemoteWfEngine.class).getRemoteEngine().getKieSession();
78              Map<String, Object> params = new HashMap<String, Object>();
79              params.put("component_id", instanceRequest.getComponentId());
80              params.put("component_name", instanceRequest.getComponentName());
81              params.put("created_by", instanceRequest.getUserId());
82              params.putAll(instanceRequest.getVariables());
83              if (instanceRequest.getParams() != null) {
84                  params.putAll(instanceRequest.getParams());
85              }
86              newInstance = session.startProcess(instanceRequest.getProcessName(), params);
87              procApi.updateRequestStatus(instanceRequest.getId(),
88                      ProcessInstanceCreationRequestI.RequestStatus.CREATED,
89                      "Instance created on KIE Server: " + AppContext.getAppConfiguration().getCurrentWorkflowServerUrl().toString(), newInstance.getId());
90          }
91          catch (RuntimeException e)
92              {
93                  procApi.updateRequestStatus(instanceRequest.getId(),
94                          ProcessInstanceCreationRequestI.RequestStatus.REJECTED,
95                          "Instance rejected by KIE Server: " + AppContext.getAppConfiguration().getCurrentWorkflowServerUrl().toString() + " - " + e.getMessage(), newInstance != null ? newInstance.getId() : 0);
96                  throw new RemoteException("Server error", e);
97          }
98      }
99  
100     @Override
101     public Map<String, Object> getVariablesMapForTaskId(Long taskId) throws RemoteException {
102         try
103         {
104             Task task = AppContext.getService(RemoteWfEngine.class).getRemoteEngine().getTaskService().getTaskById(taskId);
105             Content contentById = AppContext.getService(RemoteWfEngine.class).getRemoteEngine().getTaskService().getContentById(task.getTaskData().getDocumentContentId());
106             JaxbContent jaxbTaskContent = (JaxbContent) contentById;
107             return jaxbTaskContent.getContentMap();
108         }
109         catch (RuntimeException e)
110         {
111             throw new RemoteException("Server error", e);
112         }
113     }
114 
115     @Override
116     public void release(Long taskId) throws DatastoreException {
117         Map<String, String> variableMap = new HashMap<>();
118         ltapi.setAction(taskId, Action.RELEASE, TaskActionStatus.Pending, variableMap);
119     }
120 
121     @Override
122     public void claim(Integer count) throws RemoteException {
123         TaskService remoteService = AppContext.getService(RemoteWfEngine.class).getRemoteTaskService();
124         String userId = ExtendedAppContext.getCurrentlyLoggedInUserProfile().getWorkflowUsername();
125         log.info("Workflow user {} requesting to claim {} tasks", userId, count);
126         List<TaskSummary> availableTasks = remoteService.getTasksAssignedAsPotentialOwner(userId, "en-UK");
127         log.debug("Available {}", availableTasks.size());
128         int claimed = 0;
129         for (TaskSummary loopTask : availableTasks) {
130             if (loopTask.getStatus().equals(Status.Ready)) {
131                 if (loopTask.getActualOwner() == null || !loopTask.getActualOwner().getId().equals(userId)) {
132                     remoteService.claim(loopTask.getId(), userId);
133                     claimed++;
134                 }
135             }
136             if (claimed >= count) break;
137         }
138     }
