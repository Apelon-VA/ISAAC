64      @Override
65      public void saveTask(LocalTask task) {
66          try {
67              PreparedStatement psInsert = conn.prepareStatement("insert into local_tasks values (?, ?, ?, ?, ?, ?)");
68              psInsert.setLong(1, task.getId());
69              psInsert.setString(2, task.getName());
70              psInsert.setString(3, task.getComponentId());
71              psInsert.setString(4, task.getComponentName());
72              psInsert.setString(5, task.getStatus());
73              psInsert.setString(6, task.getOwner());
74              psInsert.executeUpdate();
75              psInsert.closeOnCompletion();
76              System.out.println("Task " + task.getId() + " saved");
77          } catch (SQLException ex) {
78              if (ex.getSQLState().equals("23505")) {
79                  System.err.print("Task " + task.getId() + " already exists!");
80                  LocalTask taskInDb = getTask(task.getId());
81                  if (task.equals(taskInDb)) {
82                      System.err.println(" No changes.");
83                  } else {
84                      if (!task.getOwner().equals(taskInDb.getOwner())) {
85                          System.err.print(" User has changed from " + taskInDb.getOwner() + " to " + task.getOwner() + ".");
86                          try {
87                              PreparedStatement psUpdateUser = conn.prepareStatement("update local_tasks set owner = ? where id = ?");
88                              psUpdateUser.setString(1, task.getOwner());
89                              psUpdateUser.setInt(2, Integer.parseInt(task.getId().toString()));
90                              psUpdateUser.executeUpdate();
91                              psUpdateUser.closeOnCompletion();
92                          } catch (SQLException ex1) {
93                              Logger.getLogger(LocalTasksApi.class.getName()).log(Level.SEVERE, null, ex1);
94                          }
95                      }
96                      if (!task.getStatus().equals(taskInDb.getStatus())) {
97                          System.err.print(" Status has changed from " + taskInDb.getStatus() + " to " + task.getStatus() + ".");
98                          try {
99                              PreparedStatement psUpdateStatus = conn.prepareStatement("update local_tasks set status = ? where id = ?");
100                             psUpdateStatus.setString(1, task.getStatus());
101                             psUpdateStatus.setInt(2, Integer.parseInt(task.getId().toString()));
102                             psUpdateStatus.executeUpdate();
103                             psUpdateStatus.closeOnCompletion();
104                         } catch (SQLException ex1) {
105                             Logger.getLogger(LocalTasksApi.class.getName()).log(Level.SEVERE, null, ex1);
106                         }
107                     }
108                     System.err.println("-");
109                 };
110             } else {
111                 Logger.getLogger(LocalTasksApi.class.getName()).log(Level.SEVERE, null, ex);
112             }
