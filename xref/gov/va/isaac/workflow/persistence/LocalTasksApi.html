94      @Override
95      public void saveTask(LocalTask task) throws DatastoreException {
96      	final Action action = Action.NONE;
97      	final TaskActionStatus actionStatus = TaskActionStatus.None;
98  
99      	WorkflowHistoryHelper.createAndAddNewEntry(task, action, task.getInputVariables());
100 
101         try (Connection conn = ds.getConnection()) {
102             try {
103                 PreparedStatement psInsert = conn.prepareStatement("insert into local_tasks values (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)");
104                 psInsert.setLong(1, task.getId());
105                 psInsert.setString(2, task.getName());
106                 psInsert.setString(3, task.getComponentId());
107                 psInsert.setString(4, task.getComponentName());
108                 psInsert.setString(5, task.getStatus().name());
109                 psInsert.setString(6, task.getOwner());
110                 psInsert.setString(7, action.name());
111                 psInsert.setString(8, actionStatus.name());
112                 psInsert.setString(9, serializeMap(task.getInputVariables()));
113                 psInsert.setString(10, "");
114                 psInsert.executeUpdate();
115                 conn.commit();
116                 log.debug("Task {} saved", task.getId());
117 
118                 if (actionEventListeners.size() > 0) {
119                 	final ActionEvent event = new ActionEvent() {
120             			@Override
121             			public Long getTaskId() {
122             				return task.getId();
123             			}
124 
125             			@Override
126             			public Action getAction() {
127             				return action;
128             			}
129 
130             			@Override
131             			public TaskActionStatus getActionStatus() {
132             				return actionStatus;
133             			}
134 
135             			@Override
136             			public Map<String, String> getOutputVariables() {
137             				return Collections.unmodifiableMap(new HashMap<>());
138             			}
139             		};
140 
141                 	for (ActionEventListener listener : actionEventListeners) {
142                 		//Utility.execute(() -> listener.handle(event));
143                 		listener.handle(event);
144                 	}
145                 }
146             } catch (SQLException ex) {
147                 if (ex.getSQLState().equals("23505")) {
148                     log.info("Task {} already exists", task.getId());
149                     LocalTask taskInDb = getTask(task.getId());
150                     if (task.equals(taskInDb)) {
151                         log.debug(" No changes.");
152                     } else {
153                         if (!task.getOwner().equals(taskInDb.getOwner())) {
154                             log.info(" User has changed from {} to {}.", taskInDb.getOwner(), task.getOwner());
155                             PreparedStatement psUpdateUser = conn.prepareStatement("update local_tasks set owner = ? where id = ?");
156                             psUpdateUser.setString(1, task.getOwner());
157                             psUpdateUser.setInt(2, Integer.parseInt(task.getId().toString()));
158                             int updatedRowCount = psUpdateUser.executeUpdate();
159                             if (updatedRowCount != 1)
160                             {
161                                 throw new DatastoreException("update owner on existing task failed!");
162                             }
163                             conn.commit();
164                         }
165                         if (!task.getStatus().equals(taskInDb.getStatus())) {
166                             log.info(" Status has changed from {} to {}.", taskInDb.getStatus(), task.getStatus());
167                             PreparedStatement psUpdateStatus = conn.prepareStatement("update local_tasks set status = ? where id = ?");
168                             psUpdateStatus.setString(1, task.getStatus().name());
169                             psUpdateStatus.setInt(2, Integer.parseInt(task.getId().toString()));
170                             int updatedRowCount = psUpdateStatus.executeUpdate();
171                             if (updatedRowCount != 1)
172                             {
173                                 throw new DatastoreException("update status on existing task failed!");
174                             }
175                             conn.commit();
176                         }
177                         //log.error("-Unknown?-");
178                     };
179                 } else {
180                     throw new DatastoreException(ex);
181                 }
182             }
183         } catch (RuntimeException re) {
184             log.error("Caught {} \"{}\" saving task #{}: {}", re.getClass().getName(), re.getLocalizedMessage(), task.getId(), task);
185             throw re;
186         }
187         catch (SQLException e)
188         {
189             throw new DatastoreException(e);
190         }
191     }

198     @Override
199     public void setAction(Long taskId, Action action, TaskActionStatus actionStatus, Map<String,String> outputVariables) throws DatastoreException {
200         LocalTask task = this.getTask(taskId);
201         WorkflowHistoryHelper.createAndAddNewEntry(task, action, outputVariables);
202         
203         try (Connection conn = ds.getConnection()){
204             PreparedStatement psUpdateStatus = conn.prepareStatement("update local_tasks set action = ?, actionStatus = ?, outputVariables = ? where id = ?");
205             psUpdateStatus.setString(1, action.name());
206             psUpdateStatus.setString(2, actionStatus.name());
207             psUpdateStatus.setString(3, serializeMap(outputVariables));
208             psUpdateStatus.setInt(4, Integer.parseInt(taskId.toString()));
209             int updatedRowCount = psUpdateStatus.executeUpdate();
210             if (updatedRowCount != 1)
211             {
212                 throw new DatastoreException("update action on existing task failed!");
213             }
214             conn.commit();
215 
216             if (actionEventListeners.size() > 0) {
217             	final ActionEvent event = new ActionEvent() {
218         			@Override
219         			public Long getTaskId() {
220         				return taskId;
221         			}
222 
223         			@Override
224         			public Action getAction() {
225         				return action;
226         			}
227 
228         			@Override
229         			public TaskActionStatus getActionStatus() {
230         				return actionStatus;
231         			}
232 
233         			@Override
234         			public Map<String, String> getOutputVariables() {
235         				return Collections.unmodifiableMap(outputVariables);
236         			}
237         		};
238 
239             	for (ActionEventListener listener : actionEventListeners) {
240             		//Utility.execute(() -> listener.handle(event));
241             		listener.handle(event);
242             	}
243             }
244         } catch (RuntimeException re) {
245             log.error("Caught {} \"{}\" setting Action {} with TaskActionStatus {} on task {}: {}", re.getClass().getName(), re.getLocalizedMessage(), action, actionStatus, taskId, outputVariables);
246             throw re;
247         } catch (SQLException ex1) {
248             throw new DatastoreException(ex1);
249         }
250         if ((task.getName().equals("Approve content") || task.getName().equals("Adjudicate content"))
251                 && action == Action.COMPLETE
252                 && actionStatus == TaskActionStatus.Pending
253                 && outputVariables.get("out_response") != null && outputVariables.get("out_response").equals("approve")) {
254             
255             populateReleaseCandidatePath(task);
256         }
257     }

392     private LocalTask readTask(ResultSet rs) throws SQLException {
393         LocalTask task = new LocalTask();
394         
395         try {
396             task.setId(rs.getLong(1));
397             task.setName(rs.getString(2));
398             task.setComponentId(rs.getString(3));
399             task.setComponentName(rs.getString(4));
400             task.setStatus(Status.valueOf(rs.getString(5)));
401             task.setOwner(rs.getString(6));
402             task.setAction(Action.valueOf(rs.getString(7)));
403             task.setActionStatus(TaskActionStatus.valueOf(rs.getString(8)));
404             task.setInputVariables(deserializeMap(rs.getString(9)));
405             task.setOutputVariables(deserializeMap(rs.getString(10)));
406         } catch (RuntimeException e) {
407             log.error("Caught {} \"{}\" reading task #{}", e.getClass().getName(), e.getLocalizedMessage(), rs.getLong(1));
408             throw e;
409         }
410 
411         return task;
412     }

414     private static String serializeMap(Map<String, String> map) {
415         if (map == null) {
416             return "";
417         }
418         ByteArrayOutputStream bos = new ByteArrayOutputStream();
419         XMLEncoder xmlEncoder = new XMLEncoder(bos);
420         xmlEncoder.writeObject(map);
421         xmlEncoder.close();
422 
423         String serializedMap = bos.toString();
424         return serializedMap;
425     }

427     private static Map<String, String> deserializeMap(String serializedMap) {
428         if (serializedMap == null || serializedMap.isEmpty()) {
429             return new HashMap<String, String>();
430         } else {
431             XMLDecoder xmlDecoder = new XMLDecoder(new ByteArrayInputStream(serializedMap.getBytes()));
432             @SuppressWarnings("unchecked")
433             Map<String, String> parsedMap = (Map<String, String>) xmlDecoder.readObject();
434             xmlDecoder.close();
435             return parsedMap;
436         }

440     public void createSchema() throws DatastoreException {
441         try (Connection conn = ds.getConnection()){
442             log.info("Creating Workflow LOCAL_TASKS schema");
443             DatabaseMetaData dbmd = conn.getMetaData();
444             ResultSet rs = dbmd.getTables(null, "WORKFLOW", "LOCAL_TASKS", null);
445             if (!rs.next()) {
446                 Statement s = conn.createStatement();
447                 s.execute("create table LOCAL_TASKS("
448                         + "id bigint PRIMARY KEY, "
449                         + "name varchar(40), "
450                         + "componentId varchar(40), "
451                         + "componentName varchar(255), "
452                         + "status varchar(40), "
453                         + "owner varchar(40), "
454                         + "action varchar(40), "
455                         + "actionStatus varchar(40), "
456                         + "inputVariables long varchar, "
457                         + "outputVariables long varchar)");
458                 
459                 s.execute("create index LOCAL_TASKS_status_idx on LOCAL_TASKS(status)");
460                 s.execute("create index LOCAL_TASKS_actionStatus_idx on LOCAL_TASKS(actionStatus)");
461                 s.execute("create index LOCAL_TASKS_componentId_idx on LOCAL_TASKS(componentId)");
462                 
463                 conn.commit();
464                 log.info("Created table LOCAL_TASKS");
465             } else {
