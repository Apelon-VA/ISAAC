77      /**
78       * <p>
79       * Starts the actual demo activities. This includes loading the correct JDBC
80       * driver, creating a database by making a connection to Derby, creating a
81       * table in the database, and inserting, updating and retrieving some data.
82       * Some of the retrieved data is then verified (compared) against the
83       * expected results. Finally, the table is deleted and, if the embedded
84       * framework is used, the database is shut down.</p>
85       * <p>
86       * Generally, when using a client/server framework, other clients may be (or
87       * want to be) connected to the database, so you should be careful about
88       * doing shutdown unless you know that no one else needs to access the
89       * database until it is rebooted. That is why this demo will not shut down
90       * the database unless it is running Derby embedded.</p>
91       *
92       * @param args - Optional argument specifying which framework or JDBC driver
93       * to use to connect to Derby. Default is the embedded framework, see the
94       * <code>main()</code> method for details.
95       * @see #main(String[])
96       */
97      void go(String[] args) {
98          /* parse the arguments to determine which framework is desired*/
99          //args[0] = "embedded";
100         parseArguments(args);
101 
102         System.out.println("SimpleApp starting in " + framework + " mode");
103 
104         /* load the desired JDBC driver */
105         loadDriver();
106 
107         /* We will be using Statement and PreparedStatement objects for
108          * executing SQL. These objects, as well as Connections and ResultSets,
109          * are resources that should be released explicitly after use, hence
110          * the try-catch-finally pattern used below.
111          * We are storing the Statement and Prepared statement object references
112          * in an array list for convenience.
113          */
114         Connection conn = null;
115         /* This ArrayList usage may cause a warning when compiling this class
116          * with a compiler for J2SE 5.0 or newer. We are not using generics
117          * because we want the source to support J2SE 1.4.2 environments. */
118         ArrayList statements = new ArrayList(); // list of Statements, PreparedStatements
119         PreparedStatement psInsert = null;
120         PreparedStatement psUpdate = null;
121         Statement s = null;
122         ResultSet rs = null;
123         try {
124             Properties props = new Properties(); // connection properties
125             // providing a user name and password is optional in the embedded
126             // and derbyclient frameworks
127             props.put("user", "user1");
128             props.put("password", "user1");
129 
130             /* By default, the schema APP will be used when no username is
131              * provided.
132              * Otherwise, the schema name is the same as the user name (in this
133              * case "user1" or USER1.)
134              *
135              * Note that user authentication is off by default, meaning that any
136              * user can connect to your database using any password. To enable
137              * authentication, see the Derby Developer's Guide.
138              */
139             String dbName = "derbyDB"; // the name of the database
140 
141             /*
142              * This connection specifies create=true in the connection URL to
143              * cause the database to be created when connecting for the first
144              * time. To remove the database, remove the directory derbyDB (the
145              * same as the database name) and its contents.
146              *
147              * The directory derbyDB will be created under the directory that
148              * the system property derby.system.home points to, or the current
149              * directory (user.dir) if derby.system.home is not set.
150              */
151             conn = DriverManager.getConnection(protocol + dbName
152                     + ";create=true", props);
153 
154             System.out.println("Connected to and created database " + dbName);
155 
156             // We want to control transactions manually. Autocommit is on by
157             // default in JDBC.
158             conn.setAutoCommit(false);
159 
160             /* Creating a statement object that we can use for running various
161              * SQL statements commands against the database.*/
162             s = conn.createStatement();
163             statements.add(s);
164 
165             // We create a table...
166             s.execute("create table location(num int, addr varchar(40))");
167             System.out.println("Created table location");
168 
169             // and add a few rows...
170 
171             /* It is recommended to use PreparedStatements when you are
172              * repeating execution of an SQL statement. PreparedStatements also
173              * allows you to parameterize variables. By using PreparedStatements
174              * you may increase performance (because the Derby engine does not
175              * have to recompile the SQL statement each time it is executed) and
176              * improve security (because of Java type checking).
177              */
178             // parameter 1 is num (int), parameter 2 is addr (varchar)
179             psInsert = conn.prepareStatement(
180                     "insert into location values (?, ?)");
181             statements.add(psInsert);
182 
183             psInsert.setInt(1, 1956);
184             psInsert.setString(2, "Webster St.");
185             psInsert.executeUpdate();
186             System.out.println("Inserted 1956 Webster");
187 
188             psInsert.setInt(1, 1910);
189             psInsert.setString(2, "Union St.");
190             psInsert.executeUpdate();
191             System.out.println("Inserted 1910 Union");
192 
193             // Let's update some rows as well...
194             // parameter 1 and 3 are num (int), parameter 2 is addr (varchar)
195             psUpdate = conn.prepareStatement(
196                     "update location set num=?, addr=? where num=?");
197             statements.add(psUpdate);
198 
199             psUpdate.setInt(1, 180);
200             psUpdate.setString(2, "Grand Ave.");
201             psUpdate.setInt(3, 1956);
202             psUpdate.executeUpdate();
203             System.out.println("Updated 1956 Webster to 180 Grand");
204 
205             psUpdate.setInt(1, 300);
206             psUpdate.setString(2, "Lakeshore Ave.");
207             psUpdate.setInt(3, 180);
208             psUpdate.executeUpdate();
209             System.out.println("Updated 180 Grand to 300 Lakeshore");
210 
211 
212             /*
213              We select the rows and verify the results.
214              */
215             rs = s.executeQuery(
216                     "SELECT num, addr FROM location ORDER BY num");
217 
218             /* we expect the first returned column to be an integer (num),
219              * and second to be a String (addr). Rows are sorted by street
220              * number (num).
221              *
222              * Normally, it is best to use a pattern of
223              *  while(rs.next()) {
224              *    // do something with the result set
225              *  }
226              * to process all returned rows, but we are only expecting two rows
227              * this time, and want the verification code to be easy to
228              * comprehend, so we use a different pattern.
229              */
230             int number; // street number retrieved from the database
231             boolean failure = false;
232             if (!rs.next()) {
233                 failure = true;
234                 reportFailure("No rows in ResultSet");
235             }
236 
237             if ((number = rs.getInt(1)) != 300) {
238                 failure = true;
239                 reportFailure(
240                         "Wrong row returned, expected num=300, got " + number);
241             }
242 
243             if (!rs.next()) {
244                 failure = true;
245                 reportFailure("Too few rows");
246             }
247 
248             if ((number = rs.getInt(1)) != 1910) {
249                 failure = true;
250                 reportFailure(
251                         "Wrong row returned, expected num=1910, got " + number);
252             }
253 
254             if (rs.next()) {
255                 failure = true;
256                 reportFailure("Too many rows");
257             }
258 
259             if (!failure) {
260                 System.out.println("Verified the rows");
261             }
262 
263             // delete the table
264             s.execute("drop table location");
265             System.out.println("Dropped table location");
266 
267             /*
268              We commit the transaction. Any changes will be persisted to
269              the database now.
270              */
271             conn.commit();
272             System.out.println("Committed the transaction");
273 
274             /*
275              * In embedded mode, an application should shut down the database.
276              * If the application fails to shut down the database,
277              * Derby will not perform a checkpoint when the JVM shuts down.
278              * This means that it will take longer to boot (connect to) the
279              * database the next time, because Derby needs to perform a recovery
280              * operation.
281              *
282              * It is also possible to shut down the Derby system/engine, which
283              * automatically shuts down all booted databases.
284              *
285              * Explicitly shutting down the database or the Derby engine with
286              * the connection URL is preferred. This style of shutdown will
287              * always throw an SQLException.
288              *
289              * Not shutting down when in a client environment, see method
290              * Javadoc.
291              */
292             if (framework.equals("embedded")) {
293                 try {
294                     // the shutdown=true attribute shuts down Derby
295                     DriverManager.getConnection("jdbc:derby:;shutdown=true");
296 
297                     // To shut down a specific database only, but keep the
298                     // engine running (for example for connecting to other
299                     // databases), specify a database in the connection URL:
300                     //DriverManager.getConnection("jdbc:derby:" + dbName + ";shutdown=true");
301                 } catch (SQLException se) {
302                     if (((se.getErrorCode() == 50000)
303                             && ("XJ015".equals(se.getSQLState())))) {
304                         // we got the expected exception
305                         System.out.println("Derby shut down normally");
306                         // Note that for single database shutdown, the expected
307                         // SQL state is "08006", and the error code is 45000.
308                     } else {
309                         // if the error code or SQLState is different, we have
310                         // an unexpected exception (shutdown failed)
311                         System.err.println("Derby did not shut down normally");
312                         printSQLException(se);
313                     }
314                 }
315             }
316         } catch (SQLException sqle) {
317             printSQLException(sqle);
318         } finally {
319             // release all open resources to avoid unnecessary memory usage
320 
321             // ResultSet
322             try {
323                 if (rs != null) {
324                     rs.close();
325                     rs = null;
326                 }
327             } catch (SQLException sqle) {
328                 printSQLException(sqle);
329             }
330 
331             // Statements and PreparedStatements
332             int i = 0;
333             while (!statements.isEmpty()) {
334                 // PreparedStatement extend Statement
335                 Statement st = (Statement) statements.remove(i);
336                 try {
337                     if (st != null) {
338                         st.close();
339                         st = null;
340                     }
341                 } catch (SQLException sqle) {
342                     printSQLException(sqle);
343                 }
344             }
345 
346             //Connection
347             try {
348                 if (conn != null) {
349                     conn.close();
350                     conn = null;
351                 }
352             } catch (SQLException sqle) {
353                 printSQLException(sqle);
354             }
355         }
356     }

358     /**
359      * Loads the appropriate JDBC driver for this environment/framework. For
360      * example, if we are in an embedded environment, we load Derby's embedded
361      * Driver,
362      * <code>org.apache.derby.jdbc.EmbeddedDriver</code>.
363      */
364     private void loadDriver() {
365         /*
366          *  The JDBC driver is loaded by loading its class.
367          *  If you are using JDBC 4.0 (Java SE 6) or newer, JDBC drivers may
368          *  be automatically loaded, making this code optional.
369          *
370          *  In an embedded environment, this will also start up the Derby
371          *  engine (though not any databases), since it is not already
372          *  running. In a client environment, the Derby engine is being run
373          *  by the network server framework.
374          *
375          *  In an embedded environment, any static Derby system properties
376          *  must be set before loading the driver to take effect.
377          */
378         try {
379             Class.forName(driver).newInstance();
380             System.out.println("Loaded the appropriate driver");
381         } catch (ClassNotFoundException cnfe) {
382             System.err.println("\nUnable to load the JDBC driver " + driver);
383             System.err.println("Please check your CLASSPATH.");
384             cnfe.printStackTrace(System.err);
385         } catch (InstantiationException ie) {
386             System.err.println(
387                     "\nUnable to instantiate the JDBC driver " + driver);
388             ie.printStackTrace(System.err);
389         } catch (IllegalAccessException iae) {
390             System.err.println(
391                     "\nNot allowed to access the JDBC driver " + driver);
392             iae.printStackTrace(System.err);
393         }
394     }

406     /**
407      * Prints details of an SQLException chain to
408      * <code>System.err</code>. Details included are SQL State, Error code,
409      * Exception message.
410      *
411      * @param e the SQLException from which to print details.
412      */
413     public static void printSQLException(SQLException e) {
414         // Unwraps the entire exception chain to unveil the real cause of the
415         // Exception.
416         while (e != null) {
417             System.err.println("\n----- SQLException -----");
418             System.err.println("  SQL State:  " + e.getSQLState());
419             System.err.println("  Error Code: " + e.getErrorCode());
420             System.err.println("  Message:    " + e.getMessage());
421             // for stack traces, refer to derby.log or uncomment this:
422             //e.printStackTrace(System.err);
423             e = e.getNextException();
424         }
425     }

427     /**
428      * Parses the arguments given and sets the values of this class' instance
429      * variables accordingly - that is which framework to use, the name of the
430      * JDBC driver class, and which connection protocol protocol to use. The
431      * protocol should be used as part of the JDBC URL when connecting to Derby.
432      * <p>
433      * If the argument is "embedded" or invalid, this method will not change
434      * anything, meaning that the default values will be used.</p>
435      * <p>
436      * @param args JDBC connection framework, either "embedded", "derbyclient".
437      * Only the first argument will be considered, the rest will be ignored.
438      */
439     private void parseArguments(String[] args) {
440         if (args.length > 0) {
441             if (args[0].equalsIgnoreCase("derbyclient")) {
442                 framework = "derbyclient";
443                 driver = "org.apache.derby.jdbc.ClientDriver";
444                 protocol = "jdbc:derby://localhost:1527/";
445             }
446         }
447     }
