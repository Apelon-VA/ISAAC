78      @Override
79      public ProcessInstanceCreationRequestI createRequest(String processName, UUID componentId, String componentName, String author, Map<String, String> variables) 
80              throws DatastoreException {
81          try (Connection conn = ds.getConnection()) {
82              PreparedStatement psInsert = conn.prepareStatement("insert into PINST_REQUESTS(component_id, component_name, process_name, user_id, status, sync_message,"
83                  + " request_time, sync_time, wf_id, variables) values (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)",PreparedStatement.RETURN_GENERATED_KEYS);
84              if (componentId == null) {
85              	log.error("Setting null componentId for ProcessInstanceCreationRequestsAPI.createRequest(): user_id={}, process_name={}, component={}, map={}", author, processName, componentName, variables);
86              }
87              psInsert.setString(1, componentId != null ? componentId.toString() : null);
88              psInsert.setString(2, componentName);
89              psInsert.setString(3, processName);
90              psInsert.setString(4, author);
91              psInsert.setString(5, ProcessInstanceCreationRequestI.RequestStatus.REQUESTED.name());
92              psInsert.setString(6, "");
93              psInsert.setLong(7, System.currentTimeMillis());
94              psInsert.setLong(8, 0L);
95              psInsert.setLong(9, 0L);
96              psInsert.setString(10, serializeMap(variables));
97              psInsert.executeUpdate();
98              
99              ProcessInstanceCreationRequestI result = new ProcessInstanceCreationRequest();
100             result.setComponentId(componentId != null ? componentId.toString() : null);
101             result.setComponentName(componentName);
102             ResultSet generatedKeys = psInsert.getGeneratedKeys();
103             if (generatedKeys.next()) {
104                 result.setId(generatedKeys.getInt(1));
105             } else {
106                 throw new SQLException("Creating instance failed, no generated key obtained.");
107             }
108             result.setRequestTime(Long.MIN_VALUE);
109             result.setProcessName(processName);
110             result.setStatus(ProcessInstanceCreationRequestI.RequestStatus.REQUESTED);
111             result.setUserId(author);
112             result.setVariables(variables);
113             conn.commit();
114             return result;
115         } catch (SQLException ex) {
116             throw new DatastoreException(ex);
117         }
118     }

120     @Override
121     public void updateRequestStatus(int id, ProcessInstanceCreationRequestI.RequestStatus status, String syncMessage, Long wfId) throws DatastoreException {
122         try (Connection conn = ds.getConnection()){
123             PreparedStatement psUpdateRequest = conn.prepareStatement("update PINST_REQUESTS set sync_message = ?, status = ?, wf_id = ? where id = ?");
124             psUpdateRequest.setString(1, syncMessage);
125             psUpdateRequest.setString(2, status.name());
126             psUpdateRequest.setLong(3, (wfId == null ? 0 : wfId));
127             psUpdateRequest.setInt(4, id);
128             int rowCount = psUpdateRequest.executeUpdate();
129             if (rowCount != 1)
130             {
131                 throw new DatastoreException("updateRequestStatus failed to update any rows!");
132             }
133             conn.commit();
134         } catch (SQLException ex) {
135             throw new DatastoreException(ex);
136         }
137     }

254     private ProcessInstanceCreationRequestI readRequest(ResultSet rs) throws SQLException, DatastoreException {
255         try
256         {
257             ProcessInstanceCreationRequestI request = new ProcessInstanceCreationRequest();
258             request.setId(rs.getInt(1));
259             request.setWfId(rs.getLong(2));
260             request.setComponentId(rs.getString(3));
261             request.setComponentName(rs.getString(4));
262             request.setProcessName(rs.getString(5));
263             request.setUserId(rs.getString(6));
264             String status = rs.getString(7);
265             switch (status) {
266                 case "CREATED":
267                     request.setStatus(ProcessInstanceCreationRequestI.RequestStatus.CREATED);
268                     break;
269                 case "REQUESTED":
270                     request.setStatus(ProcessInstanceCreationRequestI.RequestStatus.REQUESTED);
271                     break;
272                 case "REJECTED":
273                     request.setStatus(ProcessInstanceCreationRequestI.RequestStatus.REJECTED);
274                     break;
275                 default :
276                     throw new DatastoreException("Unexpected 'status' found in DB: " + status);
277             }
278             request.setSyncMessage(rs.getString(8));
279             request.setRequestTime(rs.getLong(9));
280             request.setSyncTime(rs.getLong(10));
281             request.setVariables(rs.getString(11).isEmpty() ? new HashMap<String, String>() : deserializeMap(rs.getString(11)));
282 
283             return request;
284         }
285         catch (NumberFormatException e)
286         {
287             throw new DatastoreException("Encountered an unparseable number in a field that should have held a number: " + e.getMessage());
288         }
289     }

291     @Override
292     public void createSchema() throws DatastoreException {
293         try (Connection conn = ds.getConnection()){
294             log.info("Creating Workflow Process Instance Schema");
295             DatabaseMetaData dbmd = conn.getMetaData();
296             ResultSet rs = dbmd.getTables(null, "WORKFLOW", "PINST_REQUESTS", null);
297             if (!rs.next()) {
298                 Statement s = conn.createStatement();
299                 s.execute("create table PINST_REQUESTS "
300                         + "(id INTEGER NOT NULL GENERATED ALWAYS AS IDENTITY (START WITH 1, INCREMENT BY 1) PRIMARY KEY, "
301                         + "wf_id bigint, "
302                         + "component_id varchar(40), "
303                         + "component_name varchar(255), "
304                         + "process_name varchar(255), "
305                         + "user_id varchar(40), "
306                         + "status varchar(40), "
307                         + "sync_message varchar(255), "
308                         + "request_time bigint, "
309                         + "sync_time bigint, "
310                         + "variables long varchar)");
311                 s.execute("create index WORKFLOW_status_idx on PINST_REQUESTS(status)");
312                 s.execute("create index WORKFLOW_component_id_idx on PINST_REQUESTS(component_id)");
313                 s.execute("create index WORKFLOW_wf_id_idx on PINST_REQUESTS(wf_id)");
314                 log.debug("Created table PINST_REQUESTS");
315             } else {
316                 log.debug("PINST_REQUESTS already exists!");
317             }
318             conn.commit();
319         } catch (SQLException ex) {
320             throw new DatastoreException(ex);
321         }
322     }

340     private String serializeMap(Map<String, String> map ) {
341         if (map == null) {
342             return "";
343         }
344         ByteArrayOutputStream bos = new ByteArrayOutputStream();
345         XMLEncoder xmlEncoder = new XMLEncoder(bos);
346         xmlEncoder.writeObject(map);
347         xmlEncoder.close();
348 
349         String serializedMap = bos.toString();
350         return serializedMap;
351     }
352 
353     @SuppressWarnings("unchecked")
354     private Map<String, String> deserializeMap(String serializedMap) {
355         if (serializedMap == null || serializedMap.isEmpty()) {
356             return new HashMap<String, String>();
357         } else {
358             try (XMLDecoder xmlDecoder = new XMLDecoder(new ByteArrayInputStream(serializedMap.getBytes())))
359             {
360                 return (Map<String, String>) xmlDecoder.readObject();
361             }
362             catch (Exception e)
363             {
364                 log.error("Unexpected error while deserializing map", e);
365                 throw e;
366             }
367             
368         }
