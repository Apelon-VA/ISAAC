54  public class WorkflowHistoryHelper {
55  	private final static Logger logger = LoggerFactory.getLogger(WorkflowHistoryHelper.class);
56  
57  	private final static String WF_IN_HISTORY_MAP_VARIABLE_NAME = "in_history";
58  	private final static String WF_OUT_HISTORY_MAP_VARIABLE_NAME = "out_history";
59  
60  	private final static SimpleDateFormat sdf = new SimpleDateFormat("MMM dd yyyy @ HH:mm:ss");
61  
62  	/*
63  	 * Author
64  	 * Time
65  	 * Task name
66  	 * Complete/Release
67  	 * Output variables
68  	 *
69  	 */
70  	enum WorkflowHistoryVariable {
71  		wf_hist_action_owner,
72  		wf_hist_action_time,
73  		wf_hist_action, 
74  		wf_hist_comment,
75  		wf_hist_task_name
76  	}
77  
78  	public static List<Map<String, String>> getHistoryEntries(LocalTask task) {
79  		List<Map<String, String>> existingEntries = deserializeMaps(task.getInputVariables().get(WF_IN_HISTORY_MAP_VARIABLE_NAME));
80  
81  		logger.debug("Got {} history entries from task #{}: {}", existingEntries.size(), task.getId(), historyEntriesToString(existingEntries));
82  		return existingEntries;
83  	}
84  
85  	// For logging purposes only
86  	private static String historyEntryToString(Map<String, String> entry) {
87  		Map<String, String> tempMap = new HashMap<>(entry);
88  		
89  		tempMap.put(WorkflowHistoryVariable.wf_hist_action_time.name(), new Date(Long.valueOf(entry.get(WorkflowHistoryVariable.wf_hist_action_time.name()))).toString());
90  		
91  		return tempMap.toString();
92  	}
93  	
94  	// For logging purposes only
95  	public static String historyEntriesToString(List<Map<String, String>> entries) {
96  		StringBuilder builder = new StringBuilder();
97  		
98  		for (Map<String, String> entry : entries) {
99  			if (builder.toString().length() > 0) {
100 				builder.append("\n");
101 			}
102 			builder.append(historyEntryToString(entry));
103 		}
104 		
105 		return builder.toString();
106 	}

108 	private static Map<String, String> createNewHistoryEntry(LocalTask task, Action action, String comment) {
109 		long time = new Date().getTime();
110 
111 		Map<String, String> newEntry = new HashMap<>();
112 		newEntry.put(WorkflowHistoryVariable.wf_hist_action_owner.name(), task.getOwner());
113 		newEntry.put(WorkflowHistoryVariable.wf_hist_action_time.name(), Long.toString(time));
114 		newEntry.put(WorkflowHistoryVariable.wf_hist_action.name(), action.name());
115 		newEntry.put(WorkflowHistoryVariable.wf_hist_comment.name(), comment);
116 		newEntry.put(WorkflowHistoryVariable.wf_hist_task_name.name(), task.getName());
117 
118 		return newEntry;
119 	}

121 	public static void createAndAddNewEntry(LocalTask task, Action action, Map<String, String> outputVariables) {
122 		List<Map<String, String>> existingEntries = getHistoryEntries(task);
123 		int numExistingEntries = existingEntries.size();
124 		Map<String, String> newEntry = createNewHistoryEntry(task, action, outputVariables.get("out_comment"));
125 		//outputVariables.putAll(newEntry);
126 		//existingEntries.add(outputVariables);
127 		existingEntries.add(newEntry);
128 
129 		String serializedEntries = serializeMaps(existingEntries);
130 		outputVariables.put(WF_OUT_HISTORY_MAP_VARIABLE_NAME, serializedEntries);
131 
132 		logger.debug("Added new history entry for task #{} with {} existing entries: {}", task.getId(), numExistingEntries, existingEntries);
133 	}

135 	private static String serializeMaps(List<Map<String, String>> listOfHistoryEntryMaps) {
136 		if (listOfHistoryEntryMaps == null) {
137 			logger.warn("serialized null listOfHistoryEntryMaps into zero-length string \"\"");
138 			return "";
139 		}
140 		logger.debug("serializing listOfHistoryEntryMaps of size {}: {}", listOfHistoryEntryMaps.size(), listOfHistoryEntryMaps);
141 
142 		ByteArrayOutputStream bos = new ByteArrayOutputStream();
143 		XMLEncoder xmlEncoder = new XMLEncoder(bos);
144 		xmlEncoder.writeObject(listOfHistoryEntryMaps);
145 		xmlEncoder.close();
146 
147 		String serializedMaps = bos.toString();
148 		
149 		logger.debug("serialized listOfHistoryEntryMaps of size {} into XML: {}", listOfHistoryEntryMaps.size(), serializedMaps);
150 		return serializedMaps;
151 	}
152 
153 	private static List<Map<String, String>> deserializeMaps(String listOfSerializedMaps) {
154 		if (listOfSerializedMaps == null || listOfSerializedMaps.isEmpty()) {
155 			logger.warn("null or empty xml string passed.  Returning empty history entry map list.");
156 			return new ArrayList<Map<String, String>>();
157 		} else {
158 			XMLDecoder xmlDecoder = new XMLDecoder(new ByteArrayInputStream(listOfSerializedMaps.getBytes()));
159 			@SuppressWarnings("unchecked")
160 			List<Map<String, String>> parsedMaps = (List<Map<String, String>>) xmlDecoder.readObject();
161 			xmlDecoder.close();
162 			logger.debug("deserialized xml string \"{}\" into list of history entry maps with {} entries: {}", listOfSerializedMaps, parsedMaps.size(), parsedMaps.toString());
163 			return parsedMaps;
164 		}
165 	}

167 	public static void loadGridPane(GridPane gp, LocalTask task) {
168 		List<Map<String, String>> historyEntryMaps = getHistoryEntries(task);
169 		
170 		logger.debug("Loading GridPane with {} history entries from task #{}: {}", historyEntryMaps.size(), task.getId(), historyEntriesToString(historyEntryMaps));
171 
172 		int counter = 0;
173 		for (Map<String, String> historyEntryMap : historyEntryMaps) {
174 			String owner = "Owner unset";
175 			String timestamp = "Timestamp unset";
176 			String comment = "Comment unset";
177 			String taskName = "Task name unset";
178 			
179 			for (String key : historyEntryMap.keySet()) {
180 				if (key.equals(WorkflowHistoryVariable.wf_hist_action_owner.toString())) {
181 					owner = historyEntryMap.get(key);
182 				} else if (key.equals(WorkflowHistoryVariable.wf_hist_action_time.toString())) {
183 					Long time = Long.parseLong(historyEntryMap.get(key));
184 					
185 			        timestamp = sdf.format(time);
186 				} else if (key.equals(WorkflowHistoryVariable.wf_hist_comment.toString())) {
187 					comment = historyEntryMap.get(key);
188 				} else if (key.equals(WorkflowHistoryVariable.wf_hist_task_name.toString())) {
189 					taskName = historyEntryMap.get(key);
190 				}
191 			}
192 
193 			Label ownerLabel = new Label(owner);
194 			ownerLabel.setFont(new Font("System Bold", 14));
195 			Label taskNameLabel = new Label(taskName);
196 			taskNameLabel.setFont(new Font("System Bold", 14));
197 			Label timeLabel = new Label(timestamp);
198 			timeLabel.setFont(new Font("System Bold", 14));
199 			Label commentLabel = new Label(comment);
200 			commentLabel.setFont(new Font("System Bold", 14));
201 
202 			gp.addRow(counter++, ownerLabel, taskNameLabel, timeLabel, commentLabel);
203 		}
