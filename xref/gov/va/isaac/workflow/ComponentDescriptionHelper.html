42  /**
43   * ComponentDescriptionHelper
44   * 
45   * @author <a href="mailto:joel.kniaz@gmail.com">Joel Kniaz</a>
46   *
47   */
48  public class ComponentDescriptionHelper {
49  	private ComponentDescriptionHelper() {}
50  	
51  	public static String getComponentDescription(int nid) {
52  		return getComponentDescription(WBUtility.getComponentVersion(nid));
53  	}
54  	public static String getComponentDescription(UUID uuid) {
55  		return getComponentDescription(WBUtility.getComponentVersion(uuid));
56  	}
57  	public static String getComponentDescription(ComponentVersionBI component) {
58  		ComponentType type = ComponentTypeHelper.getComponentType(component);
59  
60  		String description = null;
61  
62  		switch(type) {
63  		case Concept: {
64  			ConceptChronicleBI concept = null;
65  			if (component instanceof ConceptAttributeVersionBI) {
66  				concept = ((ConceptAttributeVersionBI<?>) component).getEnclosingConcept();
67  			} else if (component instanceof ConceptChronicleBI) {
68  				concept = (ConceptChronicleBI)component;
69  			}
70  
71  			String fsn = WBUtility.getFullySpecifiedName(concept);
72  			UUID uuid = concept.getPrimordialUuid();
73  
74  			// Concept with FSN: <FSN> with UUID: <Concept UUID>
75  			description = ComponentType.Concept.name() + " \"" + fsn + "\" \nwith UUID: " + uuid;
76  			break;
77  
78  		}
79  
80  		case Description: {
81  			DescriptionVersionBI<?> descriptionVersion = (DescriptionVersionBI<?>)component;
82  			String typeName = WBUtility.getConPrefTerm(descriptionVersion.getTypeNid());
83  			String term = descriptionVersion.getText();
84  			ConceptVersionBI containingConcept = WBUtility.getConceptVersion(descriptionVersion.getConceptNid());
85  			String containingConceptFSN = WBUtility.getFullySpecifiedName(containingConcept);
86  			UUID containingConceptUuid = containingConcept.getPrimordialUuid();
87  
88  			// The <TYPE> term type with text: <TERM> in Concept with FSN: <FSN> with UUID: <Description UUID>
89  			description = typeName + " \"" + term + "\" \nin concept with FSN \"" + containingConceptFSN + "\" \nwith UUID: " + containingConceptUuid;
90  			break;
91  		}
92  
93  		case Refex:
94  			//The refex member with Referenced Component UUID: <REF_COMP_UUID> in Refex: <REFEX_UUID> having with UUID: <Refex MEMEBER UUID>
95  			RefexVersionBI<?> refexVersion = (RefexVersionBI<?>)component;
96  			UUID referencedComponent = WBUtility.getComponentVersion(refexVersion.getReferencedComponentNid()).getPrimordialUuid();
97  			int assemblageNid = refexVersion.getAssemblageNid();
98  			ComponentVersionBI assemblageComponentVersion = WBUtility.getComponentVersion(assemblageNid);
99  			UUID assemblageUuid = assemblageComponentVersion.getPrimordialUuid();
100 			description = ComponentType.Refex.name() + " member " + refexVersion.getPrimordialUuid() + " \nwith referenced component " + referencedComponent + " \nin refex " + assemblageUuid;
101 			break;
102 
103 		case RefexDynamic:
104 			//The refex Dynamic member with Referenced Component UUID: <REF_COMP_UUID> in Refex: <REFEX_UUID> having with UUID: <Refex MEMEBER UUID>
105 			RefexDynamicVersionBI<?> refexDynamicVersion = (RefexDynamicVersionBI<?>)component;
106 			UUID dynamicReferencedComponent = WBUtility.getComponentVersion(refexDynamicVersion.getReferencedComponentNid()).getPrimordialUuid();
107 			int assemblageDynamicNid = refexDynamicVersion.getAssemblageNid();
108 			ConceptVersionBI assemblageDynamicComponentVersion = WBUtility.getConceptVersion(assemblageDynamicNid);
109 			UUID assemblageDynamicUuid = assemblageDynamicComponentVersion.getPrimordialUuid();
110 			try {
111 				if (assemblageDynamicComponentVersion.isAnnotationStyleRefex()) {
112 					description = ComponentType.RefexDynamic.name() + " annotated member " + refexDynamicVersion.getPrimordialUuid() + " \nwith referenced component " + dynamicReferencedComponent + " \nin refex " + assemblageDynamicUuid;
113 				} else {
114 					description = ComponentType.RefexDynamic.name() + " regular member " + refexDynamicVersion.getPrimordialUuid() + " \nwith referenced component " + dynamicReferencedComponent + " \nin refex " + assemblageDynamicUuid;
115 				}
116 			} catch (Exception e) {
117 				description = "Error accessing Refex Annotation type\n" + ComponentType.RefexDynamic.name() + " member " + refexDynamicVersion.getPrimordialUuid() + " \nwith referenced component " + dynamicReferencedComponent + " \nin refex " + assemblageDynamicUuid;
118 			}
119 			break;
120 
121 		case Relationship: {
122 			RelationshipVersionBI<?> relationshipVersion = (RelationshipVersionBI<?>)component;
123 			UUID relationshipUuid = relationshipVersion.getPrimordialUuid();
124 			String typeName = WBUtility.getConPrefTerm(relationshipVersion.getTypeNid());
125 			ConceptVersionBI sourceConcept = WBUtility.getConceptVersion(relationshipVersion.getOriginNid());
126 			String sourceConceptFSN = WBUtility.getFullySpecifiedName(sourceConcept);
127 			ConceptVersionBI destinationConcept = WBUtility.getConceptVersion(relationshipVersion.getDestinationNid());
128 			String destinationConceptFSN = WBUtility.getFullySpecifiedName(destinationConcept);
129 
130 			// The <TYPE> relationship type with source FSN: <SOURCE_FSN> and destination FSN: <DESTINATION_FSN> with UUID: <Relationship UUID>
131 			description = ComponentType.Relationship.name() + " type \"" + typeName + "\" \nwith source FSN: \"" + sourceConceptFSN + "\" \nand destination FSN: \"" + destinationConceptFSN + "\" \nwith UUID: " + relationshipUuid;
132 			break;
133 		}
134 
135 		default:
136 			throw new IllegalArgumentException("Unsupported ComponentVersionBI: " + component.getClass().getName());
137 		}
138 
139 		return description;
140 	}
