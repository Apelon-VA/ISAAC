37  /**
38   * AppContext
39   *
40   * Provides convenience methods for retrieving implementations of various interfaces
41   * from the HK2 dependency management system.
42   *
43   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
44   */
45  public class AppContext
46  {
47  	public static final String READ_INHABITANT_FILES = "gov.va.isaac.AppContext.READ_INHABITANT_FILES";
48  	public static final String EXTRA_PACKAGES_TO_SEARCH = "gov.va.isaac.AppContext.EXTRA_PACKAGES_TO_SEARCH";
49  
50  	private static ServiceLocator serviceLocator_;
51  
52  	private static Logger log_ = LoggerFactory.getLogger(AppContext.class);
53  
54  	/**
55  	 * Call this once (and only once) to initialize the ISAAC HK2 service Locator.
56  	 * After this is called, you can access the service locator via convenience methods here,
57  	 * or via a call directly to HK2: 
58  	 * {@code
59  	 *     ServiceLocatorFactory.getInstance().create("ISAAC");
60  	 * }
61  	 * 
62  	 * @throws IOException
63  	 * @throws ClassNotFoundException
64  	 *
65  	 */
66  	public synchronized static void setup() throws ClassNotFoundException, IOException
67  	{
68  		if (serviceLocator_ != null)
69  		{
70  			throw new RuntimeException("Only one service locator should be set");
71  		}
72  		ArrayList<String> packagesToSearch = new ArrayList<String>(Arrays.asList("gov.va", "org.ihtsdo"));
73  
74  		boolean readInhabitantFiles = Boolean.getBoolean(System.getProperty(READ_INHABITANT_FILES, "false"));
75  		if (System.getProperty(EXTRA_PACKAGES_TO_SEARCH) != null) {
76  			String[] extraPackagesToSearch = System.getProperty(EXTRA_PACKAGES_TO_SEARCH).split(";");
77  			for (String packageToSearch: extraPackagesToSearch) {
78  				packagesToSearch.add(packageToSearch);
79  			}
80  		}
81  		serviceLocator_ = HK2RuntimeInitializer.init("ISAAC", readInhabitantFiles, packagesToSearch.toArray(new String[]{}));
82  	}

92  	public static ServiceLocator getServiceLocator()
93  	{
94  		return serviceLocator_;
95  	}
96  
97  	public static <T> T getService(Class<T> contractOrService)
98  	{
99  		List<ServiceHandle<T>> handles = serviceLocator_.getAllServiceHandles(contractOrService);
100 		
101 		for (ServiceHandle<T> handle : handles)
102 		{
103 			if (handle.getActiveDescriptor().getName() == null || handle.getActiveDescriptor().getName().length() == 0)
104 			{
105 				//prefer this un-named one
106 				return handle.getService();
107 			}
108 		}
109 		
110 		//couldn't find an un-named one - just return the default from HK2 (which still may be null)
111 		return serviceLocator_.getService(contractOrService);
112 	}
113 
114 	/**
115 	 * Find a service by name, and automatically fall back to any service which implements the contract if the named service was not available.
116 	 * 
117 	 * @param contractOrService May not be null, and is the contract or concrete implementation to get the best instance of
118 	 * @param name May be null (to indicate any name is ok), and is the name of the implementation to be returned
119 	 * @param qualifiers The set of qualifiers that must match this service definition
120 	 * @return
121 	 */
122 	public static <T> T getService(Class<T> contractOrService, String name)
123 	{
124 		T service = serviceLocator_.getService(contractOrService, name);
125 		if (service == null && name != null)
126 		{
127 			log_.info("Requested service '" + name + "' was not available, returning arbitrary service " + "which matches the contract (if any)");
128 			return serviceLocator_.getService(contractOrService);
129 		}
130 		return service;
131 	}
