36  /**
37   * {@link PasswordHasher}
38   * 
39   * A safe, modern way to 1-way hash user passwords.  
40   * Adapted and enhanced from http://stackoverflow.com/a/11038230/2163960 
41   * 
42   * Later, added the ability to encrypt and decrypt arbitrary data - using many of the same 
43   * techniques.
44   *
45   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
46   */
47  public class PasswordHasher
48  {
49  	private static final Logger log_ = LoggerFactory.getLogger(PasswordHasher.class);
50  	// The higher the number of iterations the more expensive computing the hash is for us
51  	// and also for a brute force attack.
52  	private static final int iterations = 10 * 1024;
53  	private static final int saltLen = 32;
54  	private static final int desiredKeyLen = 256;
55  	private static final String keyFactoryAlgorithm = "PBKDF2WithHmacSHA1";
56  	private static final String cipherAlgorithm = "PBEWithSHA1AndDESede";
57  	private static final Random random = new Random();  //Note, it would be more secure to use SecureRandom... but the entropy issues on Linux are a nasty issue
58  	//and it results in SecureRandom.getInstance(...).generateSeed(...) blocking for long periods of time.  A regular random is certainly good enough
59  	//for our encryption purposes.

63  	/**
64  	 * Computes a salted PBKDF2 hash of given plaintext password suitable for storing in a database.
65  	 * Empty passwords are not supported.
66  	 */
67  	public static String getSaltedHash(String password) throws Exception
68  	{
69  		long startTime = System.currentTimeMillis();
70  		byte[] salt = new byte[saltLen];
71  		random.nextBytes(salt);
72  		// store the salt with the password
73  		String result = Base64.getEncoder().encodeToString(salt) + "$$$" + hash(password, salt);
74  		log_.debug("Compute Salted Hash time {} ms", System.currentTimeMillis() - startTime);
75  		return result;
76  	}

78  	/**
79  	 * Checks whether given plaintext password corresponds to a stored salted hash of the password.
80  	 */
81  	public static boolean check(String password, String stored) throws Exception
82  	{
83  		String[] saltAndPass = stored.split("\\$\\$\\$");
84  		if (saltAndPass.length != 2)
85  		{
86  			return false;
87  		}
88  		if (password == null || password.length() == 0)
89  		{
90  			return false;
91  		}
92  		String hashOfInput = hash(password, Base64.getDecoder().decode(saltAndPass[0]));
93  		return hashOfInput.equals(saltAndPass[1]);
94  	}

96  	// using PBKDF2 from Sun
97  	private static String hash(String password, byte[] salt) throws Exception
98  	{
99  		long startTime = System.currentTimeMillis();
100 		if (password == null || password.length() == 0)
101 		{
102 			throw new IllegalArgumentException("Empty passwords are not supported.");
103 		}
104 		SecretKeyFactory f = SecretKeyFactory.getInstance(keyFactoryAlgorithm);
105 		SecretKey key = f.generateSecret(new PBEKeySpec(password.toCharArray(), salt, iterations, desiredKeyLen));
106 		String result = Base64.getEncoder().encodeToString(key.getEncoded());
107 		log_.debug("Password compute time: {} ms", System.currentTimeMillis() - startTime);
108 		return result;
109 	}

116 	public static String encrypt(String password, byte[] data) throws Exception
117 	{
118 		long startTime = System.currentTimeMillis();
119 		byte[] salt = new byte[saltLen];
120 		random.nextBytes(salt);
121 		// store the salt with the password
122 		String result = Base64.getEncoder().encodeToString(salt) + "$$$" + encrypt(password, salt, data);
123 		log_.debug("Encrypt Time {} ms", System.currentTimeMillis() - startTime);
124 		return result;
125 	}
126 	
127 	private static String encrypt(String password, byte[] salt, byte[] data) throws Exception
128 	{
129 		SecretKeyFactory keyFactory = SecretKeyFactory.getInstance(keyFactoryAlgorithm);
130 		SecretKey key = keyFactory.generateSecret(new PBEKeySpec(password.toCharArray(), salt, iterations, desiredKeyLen));
131 		Cipher pbeCipher = Cipher.getInstance(cipherAlgorithm);
132 		pbeCipher.init(Cipher.ENCRYPT_MODE, key, new PBEParameterSpec(salt, iterations));
133 		
134 		//attach a sha1 checksum to the end of the data, so we know if we decrypted it properly.
135 		byte[] dataCheckSum = computeChecksum("SHA1", data).getBytes();
136 		ByteBuffer temp = ByteBuffer.allocate(data.length + dataCheckSum.length);
137 		temp.put(data);
138 		temp.put(dataCheckSum);
139 		return Base64.getEncoder().encodeToString(pbeCipher.doFinal(temp.array()));
140 	}

147 	public static byte[] decrypt(String password, String encryptedData) throws Exception
148 	{
149 		long startTime = System.currentTimeMillis();
150 		String[] saltAndPass = encryptedData.split("\\$\\$\\$");
151 		if (saltAndPass.length != 2)
152 		{
153 			throw new Exception("Invalid encrypted data, can't find salt");
154 		}
155 		byte[] result = decrypt(password, Base64.getDecoder().decode(saltAndPass[0]), saltAndPass[1]);
156 		log_.debug("Decrypt Time {} ms", System.currentTimeMillis() - startTime);
157 		return result;
158 	}

160 	private static byte[] decrypt(String password, byte[] salt, String data) throws Exception
161 	{
162 		SecretKeyFactory keyFactory = SecretKeyFactory.getInstance(keyFactoryAlgorithm);
163 		SecretKey key = keyFactory.generateSecret(new PBEKeySpec(password.toCharArray(), salt, iterations, desiredKeyLen));
164 		Cipher pbeCipher = Cipher.getInstance(cipherAlgorithm);
165 		pbeCipher.init(Cipher.DECRYPT_MODE, key, new PBEParameterSpec(salt, iterations));
166 		byte[] decrypted;
167 		try
168 		{
169 			decrypted = pbeCipher.doFinal(Base64.getDecoder().decode(data));
170 		}
171 		catch (Exception e)
172 		{
173 			throw new Exception("Invalid decryption password");
174 		}
175 		if (decrypted.length >= 40)
176 		{
177 			//The last 40 bytes should be the SHA1 Sum
178 			String checkSum = new String(Arrays.copyOfRange(decrypted, decrypted.length - 40, decrypted.length));
179 			byte[] userData = Arrays.copyOf(decrypted, decrypted.length - 40);
180 			String computed = computeChecksum("SHA1", userData);
181 			if (!checkSum.equals(computed))
182 			{
183 				throw new Exception("Invalid decryption password, or truncated data");
184 			}
185 			else
186 			{
187 				return userData;
188 			}
189 		}
190 		else
191 		{
192 			throw new Exception("Truncated data");
193 		}
194 	}

196 	/**
197 	 * Type can be any supported type, like 'MD5' or 'SHA1'
198 	 * @throws NoSuchAlgorithmException 
199 	 */
200 	public static String computeChecksum(String type, byte[] data) throws NoSuchAlgorithmException
201 	{
202 		MessageDigest md = MessageDigest.getInstance(type);
203 		byte[] digest = md.digest(data);
204 		return new BigInteger(1, digest).toString(16);
205 	}
