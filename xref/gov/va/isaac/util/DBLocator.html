25  /**
26   * {@link DBLocator}
27   * 
28   * A utility to help ease the transition from the old paths used for the BDB, to the new paths.
29   *
30   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
31   */
32  public class DBLocator
33  {
34  	private static final Logger LOG = LoggerFactory.getLogger(DBLocator.class);
35  	
36  	
37  	/**
38  	 * Attempts to find the database folder, using the following selection criteria:
39  	 * 1) If the passed in folder is named 'berkeley-db' - it is used directly.
40  	 * 2) If the passed in folder ends with '*.bdb' then the subfolder "berkeley-db' is used, under the passed in folder.
41  	 * 3) Otherwise, we scan the children of the passed in folder, looking for a folder that ends with .bdb, and if we find one, 
42  	 * uses a subfolder of that folder named 'berkeley-db'.
43  	 * 
44  	 * 4) If still not found - it will scan the sibling folders of the passed in folder, looking for a folder that ends with .bdb
45  	 * Upon finding one, does 3.
46  	 * 5) Finally, if nothing matches, it just returns the input folder. 
47  	 * @param inputFolder
48  	 * @return
49  	 */
50  	public static File findDBFolder(File inputFolder)
51  	{
52  		inputFolder = inputFolder.getAbsoluteFile();
53  
54  		//If they pass this in - and it exists - just use it.
55  		if (inputFolder.getName().equals("berkeley-db") && inputFolder.isDirectory())
56  		{
57  			LOG.info("BDB Location set to " + inputFolder.getAbsolutePath());
58  			return inputFolder;
59  		}
60  		
61  		//If it is a folder with a '.bdb' at the end of the name, then berkeley-db will be in a sub-folder.
62  		if (inputFolder.getName().endsWith(".bdb") && inputFolder.isDirectory())
63  		{
64  			LOG.info("BDB Location set to " + inputFolder.getAbsolutePath());
65  			return new File(inputFolder, "berkeley-db");
66  		}
67  		//Otherwise see if we can find a .bdb folder as a direct child
68  		if (inputFolder.isDirectory())
69  		{
70  			for (File f : inputFolder.listFiles())
71  			{
72  				//If it is a folder with a '.bdb' at the end of the name, then berkeley-db will be in a sub-folder.
73  				if (f.getName().endsWith(".bdb") && f.isDirectory())
74  				{
75  					LOG.info("BDB Location set to " + inputFolder.getAbsolutePath());
76  					return new File(f, "berkeley-db");
77  				}
78  			}
79  		}
80  		
81  		//Or as a sibling
82  		if (inputFolder.getParentFile().isDirectory())
83  		{
84  			for (File f : inputFolder.getParentFile().listFiles())
85  			{
86  				//If it is a folder with a '.bdb' at the end of the name, then berkeley-db will be in a sub-folder.
87  				if (f.getName().endsWith(".bdb") && f.isDirectory())
88  				{
89  					File berkeley = new File(f, "berkeley-db"); 
90  					LOG.info("BDB Location set to " + berkeley.getAbsolutePath());
91  					return berkeley;
92  				}
93  			}
94  		}
95  
96  		//can't match an expected pattern... just return the input.
97  		LOG.info("BDB Location set to " + inputFolder.getAbsolutePath());
98  		return inputFolder;
99  	}

101 	public static File findLuceneIndexFolder(File dbLocation)
102 	{
103 		dbLocation = dbLocation.getAbsoluteFile();
104 		//old style - we had the db inside the berkeley-db folder - so if it already exists - use it.
105 		File lucene = new File(dbLocation, "lucene");
106 		if (lucene.isDirectory())
107 		{
108 			//But, we have to return the parent folder, because the lucene code adds on its own "lucene" subdirectory.... sigh.
109 			lucene = dbLocation;
110 			LOG.info("Lucene index location set to " + lucene.getAbsolutePath());
111 			return lucene;
112 		}
113 		//If it doesn't yet exist (we are setting up a new DB) then put it as a sibling to the dbLocation folder (and no 'lucene' 
114 		//subfolder -  it will make that on its own)
115 		lucene = dbLocation.getParentFile();
116 		LOG.info("Lucene index location set to " + lucene.getAbsolutePath());
117 		return lucene;
118 	}
