15  import java.util.Comparator;
16  
17  /**
18   * The Alphanum Algorithm is an improved sorting algorithm for strings containing numbers.
19   *
20   * Instead of sorting numbers in ASCII order like a standard sort, this algorithm sorts numbers in numeric order.
21   *
22   * The inspiration for this implementation came from http://www.DaveKoelle.com However, his implementation did not handle leading 0's properly, nor
23   * did it handle nulls or case sensitivity.
24   *
25   * I fixed all of those issues, and also added convenience methods.
26   *
27   * @author <A HREF="mailto:daniel.armbrust@gmail.com">Dan Armbrust</A>
28   *
29   * See http://armbrust.dyndns.org/programs/index.php?page=3
30   */
31  public class AlphanumComparator implements Comparator<String>
32  {
33      private boolean ignoreCase_;
34      private static AlphanumComparator caseSensitiveInstance_;
35      private static AlphanumComparator caseInsensitiveInstance_;
36  
37      /**
38       * Create a new instance of an AlphanumComparator.
39       *
40       * @param caseSensitive
41       */
42      public AlphanumComparator(boolean ignoreCase)
43      {
44          this.ignoreCase_ = ignoreCase;
45      }
46  
47      /**
48       * Get a reference to a cached, shared instance. Good for reuse, but would have multithreading issues if many threads are trying to sort at the
49       * same time.
50       *
51       * @param caseSensitive
52       */
53      public static synchronized AlphanumComparator getCachedInstance(boolean ignoreCase)
54      {
55          if (ignoreCase)
56          {
57              if (caseSensitiveInstance_ == null)
58              {
59                  caseSensitiveInstance_ = new AlphanumComparator(true);
60              }
61              return caseSensitiveInstance_;
62          }
63          else
64          {
65              if (caseInsensitiveInstance_ == null)
66              {
67                  caseInsensitiveInstance_ = new AlphanumComparator(false);
68              }
69              return caseInsensitiveInstance_;
70          }
71      }
72  
73      private boolean isDigit(char ch)
74      {
75          return ch >= 48 && ch <= 57;
76      }
77  
78      /**
79       * Length of string is passed in for improved efficiency (only need to calculate it once) *
80       */
81      private String getChunk(String s, int slength, int marker)
82      {
83          StringBuilder chunk = new StringBuilder();
84          char c = s.charAt(marker);
85          chunk.append(c);
86          marker++;
87          if (isDigit(c))
88          {
89              while (marker < slength)
90              {
91                  c = s.charAt(marker);
92                  if (!isDigit(c))
93                  {
94                      break;
95                  }
96                  chunk.append(c);
97                  marker++;
98              }
99          }
100         else
101         {
102             while (marker < slength)
103             {
104                 c = s.charAt(marker);
105                 if (isDigit(c))
106                 {
107                     break;
108                 }
109                 chunk.append(c);
110                 marker++;
111             }
112         }
113         return chunk.toString();
114     }
115 
116     /*
117      * Take in string (which we assume will pass Integer.ParseInt) and return an array of integers.
118      * An array is returned so we don't exceed the limits of int.
119      * 
120      * For example, 45600000000524566874861567 would be returned as : [456000000,005245668,74861567]
121      */
122     private int[] subChunkNumeric(String numericChunk)
123     {
124         int[] result = new int[(int) Math.ceil(numericChunk.length() / 9.0)];
125         int s = 0;
126         int e = (9 > numericChunk.length() ? numericChunk.length() : 9);
127         for (int i = 0; i < result.length; i++)
128         {
129             result[i] = Integer.parseInt(numericChunk.substring(s, e));
130             s = e;
131             e = (e + 9 > numericChunk.length() ? numericChunk.length() : e + 9);
132         }
133         return result;
134     }
135 
136     @Override
137     public int compare(String s1, String s2)
138     {
139         if (s1 == null)
140         {
141             return -1;
142         }
143         if (s2 == null)
144         {
145             return 1;
146         }
147 
148         int thisMarker = 0;
149         int thatMarker = 0;
150         int s1Length = s1.length();
151         int s2Length = s2.length();
152 
153         while (thisMarker < s1Length && thatMarker < s2Length)
154         {
155             String thisChunk = getChunk(s1, s1Length, thisMarker);
156             thisMarker += thisChunk.length();
157 
158             String thatChunk = getChunk(s2, s2Length, thatMarker);
159             thatMarker += thatChunk.length();
160 
161             // If both chunks contain numeric characters, sort them numerically
162             int result = 0;
163             if (isDigit(thisChunk.charAt(0)) && isDigit(thatChunk.charAt(0)))
164             {
165                 int[] thisChunkInt = subChunkNumeric(thisChunk);
166                 int[] thatChunkInt = subChunkNumeric(thatChunk);
167 
168                 //0 pad the shorter array, so that they have the same length.
169                 if (thisChunkInt.length > thatChunkInt.length)
170                 {
171                     int[] temp = new int[thisChunkInt.length];
172                     int insertOffset = thisChunkInt.length - thatChunkInt.length;
173                     for (int i = 0; i < thatChunkInt.length; i++)
174                     {
175                         temp[insertOffset++] = thatChunkInt[i];
176                     }
177                     thatChunkInt = temp;
178                 }
179                 else
180                 {
181                     if (thisChunkInt.length < thatChunkInt.length)
182                     {
183                         int[] temp = new int[thatChunkInt.length];
184                         int insertOffset = thatChunkInt.length - thisChunkInt.length;
185                         for (int i = 0; i < thisChunkInt.length; i++)
186                         {
187                             temp[insertOffset++] = thisChunkInt[i];
188                         }
189                         thisChunkInt = temp;
190                     }
191                 }
192 
193                 for (int i = 0; i < thisChunkInt.length; i++)
194                 {
195                     if (thisChunkInt[i] > thatChunkInt[i])
196                     {
197                         result = 1;
198                         break;
199                     }
200                     else
201                     {
202                         if (thisChunkInt[i] < thatChunkInt[i])
203                         {
204                             result = -1;
205                             break;
206                         }
207                     }
208                 }
209             }
210             else
211             {
212                 if (ignoreCase_)
213                 {
214                     result = thisChunk.compareToIgnoreCase(thatChunk);
215                 }
216                 else
217                 {
218                     result = thisChunk.compareTo(thatChunk);
219                 }
220             }
221 
222             if (result != 0)
223             {
224                 return result;
225             }
226         }
227 
228         return s1Length - s2Length;
229     }
230 
231     public static int compare(String left, String right, boolean ignoreCase)
232     {
233         return getCachedInstance(ignoreCase).compare(left, right);
234     }
