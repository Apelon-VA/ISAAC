7       public static final String punctuationRegEx = "!|\"|,|\'s|\'|:|;|\\?|`";
8       public static final String symbolsRegEx = "&|#|\\$|\\%|@|\\\\|_|\\|";
9       public static final String operatorsRegEx = "\\+|\\-|\\*|\\/|<|>|=|\\^|~";
10      public static final String parensRegEx = "\\(|\\)|\\{|\\}|\\[|\\]";
11      
12      public static final String escapedCharactersRegEx = "\\+|\\-|&|\\||!|\\(|\\)|\\{|\\}|\\[|\\]|\\^|\"|~|\\*|\\?|:|\\/";
13      
14      // Note: \xc2\xa0 is non-breaking space
15  	public static final String nonPrintableRegEx = 
16  			"\\x00|\\x01|\\x02|\\x03|\\x04|\\x05|\\x06|\\x07|\\x08|\\x09|\\x0a|\\x0b|\\x0c|\\x0d|\\x0e|\\x0f|" +
17  			"\\x10|\\x11|\\x12|\\x13|\\x14|\\x15|\\x16|\\x17|\\x18|\\x19|\\x1a|\\x1b|\\x1c|\\x1d|\\x1e|\\x1f|" +
18  			"\\xc2\\xa0";

20  	public static final ArrayList<String> stopWords = new ArrayList<String>();
21  	static {
22  		stopWords.add("a");
23  		stopWords.add("an");
24  		stopWords.add("and");
25  		stopWords.add("by");
26  		stopWords.add("for");
27  		stopWords.add("in");
28  		stopWords.add("not");
29  		stopWords.add("of");
30  		stopWords.add("on");
31  		stopWords.add("or");
32  		stopWords.add("the");
33  		stopWords.add("to");
34  		stopWords.add("with");
35  	}

37  	public static String prepareSearchString(String s) {
38  		String processedString = s;
39  		
40  		processedString = stripNonPrintable(processedString);
41  		processedString = escapeCharacters(processedString);
42  
43  		/*
44          // Remove leading and trailing periods and dashes

46

56  	}

58  	
59      public static String stripPunctuation	(String s) { return s.replaceAll(punctuationRegEx, " ").trim(); }
60      public static String stripSymbols		(String s) { return s.replaceAll(symbolsRegEx, " "); }
61      public static String stripOperators		(String s) { return s.replaceAll(operatorsRegEx, " "); }
62      public static String stripParens		(String s) { return s.replaceAll(parensRegEx, " "); }
63      public static String stripNonPrintable  (String s) { return s.replaceAll(nonPrintableRegEx, " "); }
64      
65      public static String stripAll (String s) {
66      	String allRegEx = punctuationRegEx 	+ "|" +
67      					  symbolsRegEx 		+ "|" +
68      					  operatorsRegEx	+ "|" +
69      					  parensRegEx		+ "|" +
70      					  nonPrintableRegEx;
71      	return s.replaceAll(allRegEx, " ");
72      }
73      
74      
75      public static String escapeCharacters(String s) {
76      	return s.replaceAll(escapedCharactersRegEx, "\\\\$0");
77      }
78      
79      public static String removeStopWords(String s) {
80          String [] words = s.trim().toLowerCase().split("\\s+");
81          
82          StringBuilder sb = new StringBuilder("");
83          for (String w : words) {
84              w = w.trim();
85              if (!stopWords.contains(w)){
86              	sb.append(w);
87              	sb.append(" ");
88              }
89          }
90  
91          return sb.toString().trim();
92  
93      }
