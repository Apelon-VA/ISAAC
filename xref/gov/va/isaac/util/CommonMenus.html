213 		private boolean isCommonMenuItemExcluded(CommonMenuItem item) {
214 			if (menuItemsToExclude != null) {
215 				for (CommonMenuItem itemToExclude : menuItemsToExclude) {
216 					if (item == itemToExclude) {
217 						return true;
218 					}
219 				}
220 			}
221 			
222 			return false;
223 		}

267 	public static void addCommonMenus(
268 			ContextMenu existingMenu, 
269 			CommonMenuBuilderI passedBuilder, 
270 			CommonMenusDataProvider dataProvider, 
271 			CommonMenusNIdProvider tmpNids,
272 			CommonMenusTaskIdProvider tmpTaskIds)
273 	{
274 		CommonMenuBuilder builder = null;
275 		if (passedBuilder == null) {
276 			builder = new CommonMenuBuilder();
277 		} else if (! (passedBuilder instanceof CommonMenuBuilder)) {
278 			builder = new CommonMenuBuilder(passedBuilder);
279 		} else {
280 			builder = (CommonMenuBuilder)passedBuilder;
281 		}
282 		CommonMenusDataProvider dataProviderLocal = (dataProvider == null ? new CommonMenusDataProvider() {} : dataProvider);
283 		
284 		if (tmpNids == null) {
285 			tmpNids = CommonMenusNIdProvider.getEmptyCommonMenusNIdProvider();
286 		}
287 		CommonMenusNIdProvider nids = tmpNids;
288 		
289 		if (tmpTaskIds == null) {
290 			tmpTaskIds = CommonMenusTaskIdProvider.getEmptyCommonMenusTaskIdProvider();
291 		}
292 		CommonMenusTaskIdProvider taskIds = tmpTaskIds;
293 		
294 		//Check the nid provider just before each display of the menu - and see if we have a nid or not.
295 		//If we don't have a nid, set the observable flag to false, so all of the menus that care, go invisible.
296 		//else, set to true, to menus that care about nids will be visible.
297 		existingMenu.setOnShowing((windowEvent) ->
298 		{
299 			taskIds.invalidateAll();
300 			nids.invalidateAll();
301 			dataProviderLocal.invalidateAll();
302 		});
303 		
304 		List<MenuItem> menuItems = getCommonMenus(builder, dataProviderLocal, nids, taskIds);
305 
306 		if (menuItems.size() > 0) {
307 			for (MenuItem newItem : menuItems) {
308 				MenuItem existingMatch = null;
309 				for (MenuItem existingItem : existingMenu.getItems()) {
310 					if (existingItem.getText().equals(newItem.getText())) {
311 						existingMatch = existingItem;
312 
313 						break;
314 					}
315 				}
316 
317 				switch (builder.getMergeMode()) {
318 				case ADD_TO_EXISTING:
319 					if (existingMatch != null) {
320 						Log.debug("Adding MenuItem with same name as existing MenuItem \"" + existingMatch.getText() + "\"");
321 					}
322 					existingMenu.getItems().add(newItem);
323 					break;
324 				case REPLACE_EXISTING:
325 					if (existingMatch != null) {
326 						Log.debug("Removing and replacing existing MenuItem \"" + existingMatch.getText() + "\"");
327 						existingMenu.getItems().remove(existingMatch);
328 					}
329 					existingMenu.getItems().add(newItem);
330 					break;
331 				case USE_EXISTING:
332 					if (existingMatch == null) {
333 						existingMenu.getItems().add(newItem);
334 					} else {
335 						Log.debug("Not adding MenuItem with same name as existing MenuItem \"" + existingMatch.getText() + "\"");
336 					}
337 					break;
338 				default:
339 					throw new RuntimeException("Unsupported enum value " + builder.getMergeMode() + " for " + MergeMode.class.getName());
340 				}
341 			}
342 		}
343 	}

354 	private static MenuItem createNewMenuItem(
355 			CommonMenuItem itemType, 
356 			CommonMenuBuilder builder, 
357 			BooleanSupplier canHandle, 
358 			BooleanExpression makeVisible,
359 			Runnable onHandlable,
360 			Runnable onNotHandleable) {
361 		
362 		if (builder.isCommonMenuItemExcluded(itemType))
363 		{
364 			return null;
365 		}
366 		MenuItem menuItem = new MenuItem(itemType.getText());
367 		menuItem.setGraphic(itemType.getImage().createImageView());
368 		menuItem.setOnAction(new EventHandler<ActionEvent>() {
369 			@Override
370 			public void handle(ActionEvent event)
371 			{
372 				if (canHandle.getAsBoolean())
373 				{
374 					onHandlable.run();
375 				} else {
376 					if (onNotHandleable != null) {
377 						onNotHandleable.run();
378 					} else {
379 						AppContext.getCommonDialogs().showInformationDialog("Invalid data selected for menu item \"" + itemType.getText() + "\"", "Can't invoke " + itemType.getText() + " for invalid data");
380 					}
381 				}
382 			}
383 		});
384 		
385 		menuItem.visibleProperty().bind(builder.getInvisibleWhenfalse() == null ? makeVisible : builder.getInvisibleWhenfalse().not().and(makeVisible));
386 		return menuItem;
387 	}

835 	/**
836 	 * If this nid is a component ref, rather than a concept ref, get the enclosing concept ref.
837 	 * @param nid
838 	 */
839 	private static int getComponentParentConceptNid(int nid)
840 	{
841 		//TODO Dan doesn't like this, because it is being done at menu execution time, in the FX Thread, rather
842 		//that in the background... but not sure how to restructure this class just now to properly handle other 
843 		//types of nids...
844 		ComponentChronicleBI<?> cc = OTFUtility.getComponentChronicle(nid);
845 		
846 		if (cc != null) {
847 			if (cc instanceof RefexDynamicChronicleBI) {
848 				RefexDynamicChronicleBI<?> refexChron = (RefexDynamicChronicleBI<?>)cc;
849 				
850 				try {
851 					if (OTFUtility.getConceptVersion(refexChron.getAssemblageNid()).isAnnotationStyleRefex()) {
852 						return refexChron.getAssemblageNid();
853 					} else {
854 						return refexChron.getReferencedComponentNid();
855 					}
856 				} catch (IOException e) {
857 					LOG.error("Unexpected - couldn't get RefexDynamicChronicleBI information for nid{}", nid);
858 					return nid;
859 				}
860 			} else 	{
861 				return cc.getConceptNid();
862 			}
863 		}
864 		else
865 		{
866 			LOG.error("Unexpected - couldn't find component for nid {}", nid);
867 			return nid;
868 		}
