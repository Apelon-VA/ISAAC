264 	public USCRSBatchTemplate(InputStream spreadsheetTemplate) throws IOException
265 	{
266 		wb = new HSSFWorkbook(spreadsheetTemplate);
267 		ch = wb.getCreationHelper();
268 
269 		for (int i = 0; i < wb.getNumberOfSheets(); i++)
270 		{
271 			Sheet s = wb.getSheetAt(i);
272 			SHEET sheetEnum = SHEET.valueOf(enumSafeCharExchange(s.getSheetName()));
273 			if (sheetEnum == null)
274 			{
275 				throw new RuntimeException("No enum type found for sheet " + s.getSheetName() + " - code out of sync with template");
276 			}
277 			else
278 			{
279 				sheetNamePositionMap.put(sheetEnum, i);
280 			}
281 
282 			if (s.getSheetName().equals("Help"))
283 			{
284 				continue;
285 			}
286 
287 			LinkedHashMap<COLUMN, Integer> colList = new LinkedHashMap<>();
288 			columnNamePositionMap.put(sheetEnum, colList);
289 
290 			wb.getSheetAt(i)
291 					.getRow(0)
292 					.forEach(headerCell -> {
293 						if (s.getSheetName().equals("metadata") && headerCell.getCellType() == Cell.CELL_TYPE_NUMERIC)
294 						{
295 							//SKIP - the metadata tab has a cell that is just a number - likely the release date
296 						}
297 						else
298 						{
299 							COLUMN colEnum = COLUMN.valueOf(enumSafeCharExchange(toString(headerCell)));
300 							if (colEnum == null)
301 							{
302 								throw new RuntimeException("No enum type found for colum " + toString(headerCell) + " on sheet " + s.getSheetName()
303 										+ " - code out of sync with template");
304 							}
305 							else
306 							{
307 								colList.put(colEnum, headerCell.getColumnIndex());
308 							}
309 						}
310 
311 					});
312 		}
313 	}

382 	private static String toString(Cell cell)
383 	{
384 		switch (cell.getCellType())
385 		{
386 			case Cell.CELL_TYPE_BLANK:
387 				return "";
388 
389 			case Cell.CELL_TYPE_BOOLEAN:
390 				return cell.getBooleanCellValue() + "";
391 
392 			case Cell.CELL_TYPE_NUMERIC:
393 				return cell.getNumericCellValue() + "";
394 
395 			case Cell.CELL_TYPE_STRING:
396 				return cell.getStringCellValue();
397 
398 			case Cell.CELL_TYPE_ERROR:
399 				return "_ERROR_ " + cell.getErrorCellValue();
400 
401 			case Cell.CELL_TYPE_FORMULA:
402 				return cell.getCellFormula() + "";
403 			default :
404 				throw new RuntimeException("No toString is available for the cell type!");
405 		}
406 	}

408 	private static String enumSafeCharExchange(String s)
409 	{
410 		s = s.replaceAll(" ", "_");
411 		s = s.replaceAll("\\(", "_");
412 		s = s.replaceAll("\\)", "_");
413 		s = s.replaceAll("/", "_");
414 		s = s.replaceAll("-", "_");
415 		return s;
416 	}
