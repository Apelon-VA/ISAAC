61  	@Override
62  	public UscrsContentRequestTrackingInfo submitContentRequest(int nid) throws Exception
63  	{
64  		LOG.debug("Submit content Request");
65  
66  		ConceptChronicleBI concept = OTFUtility.getConceptVersion(nid);
67  
68  		// Ideally this would connect to a request submission instance and dynamically create the request. In lieu
69  		// of that, we simply create a spreadsheet.
70  
71  		// Create workbook
72  		USCRSBatchTemplate bt = new USCRSBatchTemplate(USCRSBatchTemplate.class.getResourceAsStream("/USCRS_Batch_Template-2015-01-27.xls"));
73  
74  		currentRequestId = globalRequestCounter.getAndIncrement();
75  		
76  		// Handle new concept
77  		handleNewConcept(concept, bt);
78  
79  		// Handle non-isa relationships
80  		handleNewRels(concept, bt);
81  
82  		// Save the file
83  		LOG.info("Choose file to save");
84  		FileChooser fileChooser = new FileChooser();
85  
86  		// Set extension filter.
87  		FileChooser.ExtensionFilter xmlFilter = new FileChooser.ExtensionFilter("Excel files", "*.xls");
88  		FileChooser.ExtensionFilter allFilter = new FileChooser.ExtensionFilter("All files (*.*)", "*.*");
89  		fileChooser.getExtensionFilters().addAll(xmlFilter, allFilter);
90  		fileChooser.setInitialFileName("USCRS_Export.xls");
91  
92  		// Now determine
93  		UscrsContentRequestTrackingInfo info = new UscrsContentRequestTrackingInfo();
94  		info.setName(OTFUtility.getConPrefTerm(concept.getNid()));
95  
96  		// Show save file dialog.
97  		File file = fileChooser.showSaveDialog(null);
98  		LOG.info("  file = " + file);
99  		if (file != null)
100 		{
101 			bt.saveFile(file);
102 			info.setIsSuccessful(true);
103 			info.setFile(file.toString());
104 			info.setDetail("Batch USCRS submission spreadsheet successfully created.");
105 		}
106 		else
107 		{
108 			// Assume user cancelled
109 			info.setIsSuccessful(false);
110 			info.setDetail("Submission cancelled.");
111 		}
112 		return info;
113 	}

115 	/**
116 	 * Handle new concept spreadsheet tab.
117 	 *
118 	 * @param concept the concept
119 	 * @param bt the wb
120 	 * @throws Exception the exception
121 	 */
122 	private void handleNewConcept(ConceptChronicleBI concept, USCRSBatchTemplate bt) throws Exception
123 	{
124 		LOG.debug("  Handle new concept tab");
125 
126 		LOG.debug("    Add data for " + concept.toUserString());
127 
128 		String fsn = OTFUtility.getFullySpecifiedName(concept);
129 
130 		// PARENTS
131 		List<Double> parentIds = new ArrayList<>();
132 		for (RelationshipChronicleBI rel : concept.getRelationshipsOutgoing())
133 		{
134 			RelationshipVersionBI<?> relVersion = rel.getVersion(OTFUtility.getViewCoordinate());
135 			// check for "isa" relationship type
136 			if ((relVersion.getTypeNid() == Snomed.IS_A.getLenient().getNid()) && relVersion.isActive())
137 			{
138 				parentIds.add(Double.parseDouble(ConceptViewerHelper.getSctId(OTFUtility.getConceptVersion(relVersion.getDestinationNid())).trim()));
139 			}
140 		}
141 		LOG.debug("      parents = " + parentIds.size());
142 		if (parentIds.size() > 3)
143 		{
144 			throw new Exception("Cannot handle more than 3 parents");
145 		}
146 
147 		//Synonyms
148 		List<String> synonyms = new ArrayList<>();
149 		for (DescriptionChronicleBI desc : concept.getDescriptions())
150 		{
151 			DescriptionVersionBI<?> descVersion = desc.getVersion(OTFUtility.getViewCoordinate());
152 			// find active, non FSN descriptions not matching the preferred name
153 			if (descVersion.isActive() && (descVersion.getTypeNid() != Snomed.FULLY_SPECIFIED_DESCRIPTION_TYPE.getLenient().getNid())
154 					&& !descVersion.getText().equals(OTFUtility.getConPrefTerm(concept.getNid())))
155 			{
156 				synonyms.add(descVersion.getText());
157 			}
158 		}
159 		LOG.debug("      Synonym Count: {}", synonyms.size());
160 
161 		bt.selectSheet(SHEET.New_Concept);
162 		bt.addRow();
163 		for (COLUMN column : bt.getColumnsOfSheet(SHEET.New_Concept))
164 		{
165 			switch (column)
166 			{
167 				case Request_Id:
168 					bt.addNumericCell(column, currentRequestId);
169 					break;
170 				case Topic:
171 					// TODO: Topic - consider making the user enter this
172 					bt.addStringCell(column, "New concept");
173 					break;
174 				case Local_Code:
175 					bt.addStringCell(column, concept.getPrimordialUuid().toString());
176 					break;
177 				case Local_Term:
178 					bt.addStringCell(column, OTFUtility.getConPrefTerm(concept.getNid()));
179 					break;
180 				case Fully_Specified_Name:
181 					// Fully Specified Name (without the semantic tag)
182 					String fsnOnly = fsn;
183 					if (fsn.indexOf('(') != -1)
184 					{
185 						fsnOnly = fsn.substring(0, fsn.lastIndexOf('(') - 1);
186 					}
187 					bt.addStringCell(column, fsnOnly);
188 					break;
189 				case Semantic_Tag:
190 					PICKLIST_Semantic_Tag tag = null;
191 					if (fsn.indexOf('(') != -1)
192 					{
193 						String st = fsn.substring(fsn.lastIndexOf('(') + 1, fsn.lastIndexOf(')'));
194 						tag = PICKLIST_Semantic_Tag.find(st);
195 					}
196 					if (tag == null)
197 					{
198 						throw new Exception("Cannot submit a concept to USCRS without an FSN having a valid semantic tag.");
199 					}
200 					bt.addStringCell(column, tag.toString());
201 					break;
202 				case Preferred_Term:
203 					bt.addStringCell(column, OTFUtility.getConPrefTerm(concept.getNid()));
204 					break;
205 				case Terminology_1_:
206 				case Terminology_2_:
207 				case Terminology_3_:
208 					if (parentIds.size() == 0)
209 					{
210 						bt.addStringCell(column, "");
211 					}
212 					else
213 					{
214 						bt.addStringCell(column, PICKLIST_Source_Terminology.SNOMED_CT_International.toString());  //TODO this isn't a safe bet
215 					}
216 					break;
217 				case Parent_Concept_Id_1_:
218 				case Parent_Concept_Id_2_:
219 				case Parent_Concept__Id_3_:
220 					if (parentIds.size() == 0)
221 					{
222 						bt.addStringCell(column, "");
223 					}
224 					else
225 					{
226 						bt.addNumericCell(column, parentIds.remove(0));
227 					}
228 					break;
229 				case UMLS_CUI:
230 					bt.addStringCell(column, "");
231 					break;
232 				case Definition:
233 					// TODO: Definition - consider making the user enter this 
234 					bt.addStringCell(column, "See logical definition in relationships");
235 					break;
236 				case Proposed_Use:
237 					// TODO: Proposed Use - consider making the user enter this 
238 					bt.addStringCell(column, "");
239 					break;
240 				case Justification:
241 					// TODO: Justification - consider making the user enter this
242 					bt.addStringCell(column, "Developed as part of extension namespace " + AppContext.getAppConfiguration().getCurrentExtensionNamespace());
243 					break;
244 				case Note:
245 					StringBuilder sb = new StringBuilder();
246 					if (concept.getConceptAttributes().getVersion(OTFUtility.getViewCoordinate()).isDefined())
247 					{
248 						sb.append("NOTE: this concept is fully defined. ");
249 					}
250 					//First two synonyms have cols, any others go in Note
251 					boolean firstHasBeenSeen = false;
252 					while (synonyms.size() > 2)
253 					{
254 						sb.append("NOTE: this concept also has the following synonyms: ");
255 						if (firstHasBeenSeen)
256 						{
257 							sb.append(", ");
258 						}
259 						sb.append(synonyms.remove(0));
260 						firstHasBeenSeen = true;
261 					}
262 					bt.addStringCell(column, sb.toString());
263 					break;
264 				case Synonym:
265 					bt.addStringCell(column, (synonyms.size() > 0 ? synonyms.remove(0) : ""));
266 					break;
267 				default :
268 					throw new RuntimeException("Unexpected column type found in Sheet: " + column + " - " + SHEET.New_Concept);
269 			}
270 		}
271 	}

273 	/**
274 	 * Handle new rels spreadsheet tab
275 	 *
276 	 * @param concept the concept
277 	 * @param bt the wb
278 	 * @throws Exception the exception
279 	 */
280 	private void handleNewRels(ConceptChronicleBI concept, USCRSBatchTemplate bt) throws Exception
281 	{
282 		LOG.debug("  Handle non-ISA rels");
283 
284 		bt.selectSheet(SHEET.New_Relationship);
285 
286 		for (RelationshipChronicleBI rel : concept.getRelationshipsOutgoing())
287 		{
288 			RelationshipVersionBI<?> relVersion = rel.getVersion(OTFUtility.getViewCoordinate());
289 			// find active, non-ISA relationships
290 			// TODO should this be excluding inferred relationships?
291 			if (relVersion.isActive() && (relVersion.getTypeNid() != Snomed.IS_A.getLenient().getNid()))
292 			{
293 				bt.addRow();
294 				for (COLUMN column : bt.getColumnsOfSheet(SHEET.New_Relationship))
295 				{
296 					LOG.debug("    Add rel " + relVersion.toUserString());
297 					switch (column)
298 					{
299 						case Topic:
300 							// TODO: Topic - consider making the user enter this
301 							bt.addStringCell(column, "See new concept request");
302 							break;
303 						case Source_Terminology:
304 							// Source Concept Id - aligns with Request Id from the new concept spreadsheet
305 							bt.addStringCell(column, PICKLIST_Source_Terminology.Current_Batch_Requests.toString());
306 							break;
307 						case Source_Concept_Id:
308 							bt.addNumericCell(column, currentRequestId);
309 							break;
310 						case Relationship_Type:
311 							bt.addStringCell(column, PICKLIST_Relationship_Type.find(OTFUtility.getConPrefTerm(relVersion.getTypeNid())).toString());
312 							break;
313 						case Destination_Terminology:
314 							// Destination Termionlogy - TODO: here we're only supporting things linked to SNOMED, in the future we may need to link
315 							// to things that have been previously created, but we need tracking info integration to do that properly.
316 							bt.addStringCell(column, PICKLIST_Source_Terminology.SNOMED_CT_International.toString());
317 							break;
318 						case Destination_Concept_Id:
319 							bt.addNumericCell(column, Double.parseDouble(ConceptViewerHelper.getSctId(
320 									OTFUtility.getConceptVersion(relVersion.getDestinationNid())).trim()));
321 							break;
322 						case Characteristic_Type:
323 							bt.addStringCell(column, PICKLIST_Characteristic_Type.Defining_relationship.toString());
324 							break;
325 						case Refinability:
326 							bt.addStringCell(column, PICKLIST_Refinability.Not_refinable.toString());
327 							break;
328 						case Relationship_Group:
329 							bt.addNumericCell(column, relVersion.getGroup());
330 							break;
331 						case Justification:
332 							bt.addStringCell(column, "Developed as part of extension namespace " + AppContext.getAppConfiguration().getCurrentExtensionNamespace());
333 							break;
334 						case Note:
335 							bt.addStringCell(column, "This is a defining relationship expressed for the corresponding new concept request in the other tab");
336 							break;
337 						default :
338 							throw new RuntimeException("Unexpected column type found in Sheet: " + column + " - " + SHEET.New_Relationship);
339 					}
340 				}
341 			}
342 		}

354 	 * 
355 	 * @see
356 	 * gov.va.isaac.interfaces.gui.views.PopupViewI#showView(javafx.stage.Window)
357 	 */
358 	@Override
359 	public void showView(Window parent)
360 	{
361 		// No view, per se is needed, though we could
362 		// put a warning here if the request won't make sense
363 		ConceptVersionBI concept = OTFUtility.getConceptVersion(nid);
364 		if (concept == null)
365 		{
366 			AppContext.getCommonDialogs().showErrorDialog("USCRS Content Request", "Unable to load concept for " + nid, "This should never happen");
367 			return;
368 		}
369 
370 		try
371 		{
372 			if ((concept.getPathNid() != TermAux.SNOMED_CORE.getLenient().getNid())
373 					&& !OTFUtility.getConceptVersion(concept.getPathNid()).getPrimordialUuid().toString()
374 							.equals(AppContext.getAppConfiguration().getDefaultEditPathUuid()))
375 			{
376 				DialogResponse response = AppContext.getCommonDialogs().showYesNoDialog(
377 						"USCRS Content Request",
378 						"The concept path is neither Snomed CORE nor " + AppContext.getAppConfiguration().getDefaultEditPathName()
379 								+ ". It is recommended that you only submit " + "concepts edited on one of these paths to USCRS.\n\n" + "Do you want to continue?");
380 				if (response == DialogResponse.NO)
381 				{
382 					return;
383 				}
384 			}
385 		}
386 		catch (IOException e)
387 		{
388 			AppContext.getCommonDialogs().showErrorDialog("USCRS Content Request", "Unable to load concepts for path comparison.", "This should never happen");
389 			return;
390 		}
391 
392 		try
393 		{
394 			UscrsContentRequestTrackingInfo info = submitContentRequest(nid);
395 			if (info.isSuccessful())
396 			{
397 				AppContext.getCommonDialogs().showInformationDialog("USCRS Content Request",
398 						"Content request submission successful.\n\nUpload " + info.getFile() + " to here: " + info.getUrl());
399 
400 			}
401 		}
402 		catch (Exception e)
403 		{
404 			LOG.error("Unexpected error during submit", e);
405 			AppContext.getCommonDialogs().showErrorDialog("USCRS Content Request", "Unexpected error trying to submit request.", e.getMessage());
406 			return;
407 		}
408 	}
