64    @Override
65    public LoincContentRequestTrackingInfo submitContentRequest(int nid)
66      throws Exception {
67      LOG.debug("Submit content Request");
68  
69      ConceptChronicleBI concept = WBUtility.getConceptVersion(nid);
70  
71      // Ideally this would connect to a request submission
72      // instance and dynamically create the request. In lieu
73      // of that, we simply create a spreadsheet.
74  
75      // Here, we simply need to make a TSV to import into the LOINC spreadsheet.
76      StringBuilder sb = new StringBuilder();
77  
78      // Handle new concept
79      handleNewConcept(concept, sb);
80  
81      // Save the file
82      LOG.info("Choose file to save");
83      FileChooser fileChooser = new FileChooser();
84  
85      // Set extension filter.
86      FileChooser.ExtensionFilter xmlFilter =
87          new FileChooser.ExtensionFilter("Excel files", "*.xls", "*.xlsx");
88      FileChooser.ExtensionFilter allFilter =
89          new FileChooser.ExtensionFilter("All files (*.*)", "*.*");
90      fileChooser.getExtensionFilters().addAll(xmlFilter, allFilter);
91  
92      // Now determine
93      LoincContentRequestTrackingInfo info =
94          new LoincContentRequestTrackingInfo();
95      info.setName(WBUtility.getConPrefTerm(concept.getNid()));
96  
97      // Show save file dialog.
98      File file = fileChooser.showSaveDialog(null);
99      LOG.info("  file = " + file);
100     if (file != null) {
101       PrintWriter out =
102           new PrintWriter(new OutputStreamWriter(new FileOutputStream(file)));
103       out.print(sb.toString());
104       out.flush();
105       out.close();
106       info.setIsSuccessful(true);
107       info.setFile(file.toString());
108       info.setDetail("Batch LOINC submission tab-separated-values file successfully created.");
109     } else {
110       // Assume user cancelled
111       info.setIsSuccessful(false);
112       info.setDetail("Submission cancelled.");
113     }
114     return info;
115   }

117   /**
118    * Handle new concept spreadsheet tab.
119    *
120    * @param concept the concept
121    * @param sb the string builder
122    * @throws Exception the exception
123    */
124   private void handleNewConcept(ConceptChronicleBI concept, StringBuilder sb)
125     throws Exception {
126     LOG.debug("  Handle new concept tab");
127 
128     // Reference #
129     sb.append(concept.getPrimordialUuid()).append("\t");
130 
131     // Local observation code - TODO (artf231892): ??, do we have anything other than uuid?
132     sb.append(concept.getPrimordialUuid()).append("\t");
133 
134     // Local observation name - concept preferred name
135     sb.append(WBUtility.getConPrefTerm(concept.getNid())).append("\t");
136 
137     // Observation description - if there is a definition
138     sb.append(getDescriptionText(concept, "Definition (core metadata concept)")).append("\t");
139 
140     // Reference Info/URL
141     sb.append(concept.getPrimordialUuid()).append("\t");
142 
143     // Component
144     sb.append(getDestinationText(concept, "Has_COMPONENT")).append("\t");
145 
146     // Property
147     sb.append(getDestinationText(concept, "Has_PROPERTY")).append("\t");
148 
149     // Timing
150     sb.append(getDestinationText(concept, "Has_TIME_ASPECT")).append("\t");
151 
152     // System
153     sb.append(getDestinationText(concept, "Has_SYSTEM")).append("\t");
154 
155     // Scale
156     sb.append(getDestinationText(concept, "Has_SCALE_TYP")).append("\t");
157 
158     // Method
159     sb.append(getDestinationText(concept, "Has_METHOD_TYP")).append("\t");
160 
161     // Answers
162     sb.append(getAttributeText(concept, "EXMPL_ANSWERS")).append("\t");
163 
164     // Units
165     sb.append(getAttributeText(concept, "EXAMPLE_UNITS")).append("\t");
166 
167     // Formula
168     sb.append(getAttributeText(concept, "FORMULA")).append("\t");
169 
170     sb.append("\r\n");
171 
172     for (RefexChronicleBI<?> refex : concept.getAnnotations()) {
173       LOG.debug("  Sememe = " + refex.toUserString());
174     }
175 
176     for (IdBI id : concept.getAllIds()) {
177       LOG.debug("  id = " + id.toString());
178     }
179   }

181   /**
182    * Returns the description text.
183    *
184    * @param concept the concept
185    * @param type the type
186    * @return the description text
187    * @throws IOException Signals that an I/O exception has occurred.
188    * @throws ContradictionException the contradiction exception
189    */
190   private String getDescriptionText(ConceptChronicleBI concept, String type)
191     throws IOException, ContradictionException {
192     for (DescriptionChronicleBI desc : concept.getDescriptions()) {
193       DescriptionVersionBI<?> descVersion =
194           desc.getVersion(WBUtility.getViewCoordinate());
195       // WARNING:
196       // LOINC is created using FSN and not PT for this, the
197       // metadata concepts do not have PTs.
198       String prefName =
199           WBUtility.getConceptVersion(descVersion.getTypeNid())
200               .getFullySpecifiedDescription().getText();
201       if (descVersion.isActive() && prefName.equals(type)) {
202         return descVersion.getText();
203       }
204     }
205     return "";
206   }

208   /**
209    * Returns the attribute text.
210    *
211    * @param concept the concept
212    * @param type the type
213    * @return the attribute text
214    * @throws Exception the exception
215    */
216   @SuppressWarnings("unused")
217   private String getAttributeText(ConceptChronicleBI concept, String type)
218     throws Exception {
219     for (RefexChronicleBI<?> refex : concept.getAnnotations()) {
220       RefexVersionBI<?> refexVersion =
221           refex.getVersion(WBUtility.getViewCoordinate());
222       // WARNING:
223       // LOINC is created using FSN and not PT for this, the
224       // metadata concepts do not have PTs.
225       String prefName =
226           WBUtility.getConceptVersion(refexVersion.getAssemblageNid())
227               .getFullySpecifiedDescription().getText();
228       if (refexVersion.isActive() && refexVersion instanceof StringMember
229           && prefName.equals(type)) {
230         return ((StringMember) refexVersion).getString1();
231       }
232     }
233     return "";
234   }

236   /**
237    * Returns the destination text.
238    *
239    * @param concept the concept
240    * @param type the type
241    * @return the destination text
242    * @throws Exception the exception
243    */
244   private String getDestinationText(ConceptChronicleBI concept, String type)
245     throws Exception {
246     for (RelationshipChronicleBI rel : concept.getRelationshipsOutgoing()) {
247       RelationshipVersionBI<?> relVersion =
248           rel.getVersion(WBUtility.getViewCoordinate());
249       String prefName = WBUtility.getConPrefTerm(relVersion.getTypeNid());
250       if (relVersion.isActive() && prefName.equals(type)) {
251         return WBUtility.getConPrefTerm(relVersion.getDestinationNid());
252       }
253     }
254     return "";
255   }

271   @Override
272   public void showView(Window parent) {
273     // No view, per se is needed, though we could
274     // put a warning here if the request won't make sense
275     ConceptVersionBI concept = WBUtility.getConceptVersion(nid);
276     if (concept == null) {
277       AppContext.getCommonDialogs().showErrorDialog("LOINC Content Request",
278           "Unable to load concept for " + nid, "This should never happen");
279       return;
280     }
281 
282     // Check LOINC Path and current edit path
283     if (!WBUtility.getConceptVersion(concept.getPathNid()).getPrimordialUuid()
284         .toString().equals("b2b1cc96-9ca6-5513-aad9-aa21e61ddc29")
285         && !WBUtility.getConceptVersion(concept.getPathNid())
286             .getPrimordialUuid().toString()
287             .equals(AppContext.getAppConfiguration().getDefaultEditPathUuid())) {
288       DialogResponse response =
289           AppContext.getCommonDialogs().showYesNoDialog(
290               "LOINC Content Request",
291               "The concept path is neither LOINC Path nor "
292                   + AppContext.getAppConfiguration().getDefaultEditPathName()
293                   + ". It is recommended that you only submit "
294                   + "concepts edited on one of these paths to LOINC.\n\n"
295                   + "Do you want to continue?");
296       if (response == DialogResponse.NO) {
297         return;
298       }
299     }
300 
301     try {
302       LoincContentRequestTrackingInfo info = submitContentRequest(nid);
303       if (info.isSuccessful()) {
304         AppContext.getCommonDialogs().showInformationDialog(
305             "LOINC Content Request",
306             "Content request submission successful.\n\nDownload the "
307                 + "submission template from " + info.getUrl() + " and import "
308                 + "the data from " + info.getFile() + " and submit it.");
309       }
310     } catch (Exception e) {
311       e.printStackTrace();
312       AppContext.getCommonDialogs().showErrorDialog("LOINC Content Request",
313           "Unexpected error trying to submit request.", e.getMessage());
314       return;
315     }
316   }
