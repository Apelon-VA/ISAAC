19  package gov.va.legoEdit.importer;
20  
21  import gov.va.isaac.AppContext;
22  import gov.va.legoEdit.formats.LegoXMLUtils;
23  import gov.va.legoEdit.gui.legoListView.LegoListView;
24  import gov.va.legoEdit.model.schemaModel.Concept;
25  import gov.va.legoEdit.model.schemaModel.LegoList;
26  import gov.va.legoEdit.storage.BDBDataStoreImpl;
27  import gov.va.legoEdit.storage.wb.LegoWBUtility;
28  import java.io.File;
29  import java.util.ArrayList;
30  import java.util.HashMap;
31  import java.util.List;
32  import org.slf4j.Logger;
33  import org.slf4j.LoggerFactory;
34  
35  /**
36   * LegoImporter - The actual import logic, intended to be run in a background thread.
37   * Callbacks are used to signal the GUI that things can be updated during the import, 
38   * and again upon completion.
39   *
40   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
41   */
42  public class LegoImporter implements Runnable
43  {
44  	private List<File> files;
45  	int count = 0;
46  	StringBuilder status = new StringBuilder();
47  	HashMap<String, Concept> missingConcepts = new HashMap<>();
48  	ImportStatusCallback callback;
49  	ArrayList<String> importedLegoListIds = new ArrayList<String>();
50  	
51  	Logger logger = LoggerFactory.getLogger(LegoImporter.class);
52  
53  	public LegoImporter(List<File> files, ImportStatusCallback callback)
54  	{
55  		this.files = files;
56  		this.callback = callback;
57  	}
58  
59  	@Override
60  	public void run()
61  	{
62  		try
63  		{
64  			for (final File f : files)
65  			{
66  				final String temp = status.toString();
67  				status.setLength(0);
68  				callback.setCurrentItemName("Importing " + f.getName() + "...");
69  				callback.setProgress((double) count / (double) files.size());
70  				callback.appendDetails(temp);
71  
72  				if (f.exists() && f.isFile())
73  				{
74  					try
75  					{
76  						String fileName = f.getName().toLowerCase();
77  						if (fileName.endsWith(".xml"))
78  						{
79  							try
80  							{
81  								LegoXMLUtils.validate((f));
82  							}
83  							catch (Exception e)
84  							{
85  								status.append("Warning - The file '" + f.getName() + "' is not schema valid.  Will attempt to import, but may fail.");
86  								status.append("  The schema error was: " + e.getMessage());
87  								status.append(System.getProperty("line.separator"));
88  								status.append(System.getProperty("line.separator"));
89  							}
90  							LegoList ll = LegoXMLUtils.readLegoList(f);
91  							List<Concept> failures = LegoWBUtility.lookupAllConcepts(ll);
92  							BDBDataStoreImpl.getInstance().importLegoList(ll);
93  							importedLegoListIds.add(ll.getLegoListUUID());
94  							for (Concept c : failures)
95  							{
96  								if (c.getSctid() != null)
97  								{
98  									missingConcepts.put(c.getSctid() + "", c);
99  								}
100 								else if (c.getUuid() != null && c.getUuid().length() > 0)
101 								{
102 									missingConcepts.put(c.getUuid(), c);
103 								}
104 								else
105 								{
106 									missingConcepts.put(c.getDesc(), c);
107 								}
108 							}
109 						}
110 						else
111 						{
112 							status.append("Warning - The file '" + f.getName() + "' is does not have a supported file extension.  Must be .xml  "
113 									+ "File has been skipped.");
114 							status.append(System.getProperty("line.separator"));
115 							status.append(System.getProperty("line.separator"));
116 						}
117 						status.append("Completed " + f.getName());
118 					}
119 					catch (Exception ex)
120 					{
121 						logger.info("Error loading file " + f.getName(), ex);
122 						status.append("Error loading file " + f.getName() + ": ");
123 						status.append((ex.getLocalizedMessage() == null ? ex.toString() : ex.getLocalizedMessage()));
124 					}
125 				}
126 				else
127 				{
128 					status.append("Skipped " + f.getName());
129 				}
130 
131 				status.append(System.getProperty("line.separator"));
132 				status.append(System.getProperty("line.separator"));
133 				count++;
134 			}
135 			
136 			//TODO cleanup
137 			//LegoGUI.getInstance().getLegoGUIController().getCommonlyUsedConcept().rebuildDBStats();
138 			
139 			callback.appendDetails(status.toString());
140 
141 			callback.setCurrentItemName("Updating Editor");
142 			callback.setProgress(99.0);
143 			AppContext.getService(LegoListView.class).updateListView();
144 			callback.setProgress(100.0);
145 			callback.setCurrentItemName("Import Complete");
146 			if (missingConcepts.size() > 0)
147 			{
148 				StringBuilder temp = new StringBuilder();
149 				temp.append("Some concepts specified in the imported Legos do not exist in the SCT DB or the pending concepts file:");
150 				temp.append(System.getProperty("line.separator"));
151 				for (Concept c : missingConcepts.values())
152 				{
153 					temp.append(c.getSctid() + "\t" + c.getDesc() + (c.getUuid() != null ? "\t" + c.getUuid() : ""));
154 					temp.append(System.getProperty("line.separator"));
155 				}
156 				callback.appendDetails(temp.toString());
157 			}
158 		}
159 		finally
160 		{
161 			callback.importComplete(importedLegoListIds);
162 		}
163 	}
164 }
