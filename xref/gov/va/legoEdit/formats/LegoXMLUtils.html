89  	public static void validate(File path) throws IOException, SAXException
90  	{
91  		logger.debug("Validating the XML file {}", path);
92  		try
93  		{
94  			Source xmlFile = new StreamSource(path);
95  			Validator validator = schema.newValidator();
96  			validator.validate(xmlFile);
97  			logger.debug("The XML file {} is valid", path);
98  		}
99  		catch (SAXException | IOException e)
100 		{
101 			logger.debug("The XML file {} is invalid: {}", path, e.getLocalizedMessage());
102 			throw e;
103 		}
104 	}

157 	private static String marshall(Object o, boolean fragment) throws PropertyException, JAXBException
158 	{
159 		ByteArrayOutputStream baos = new ByteArrayOutputStream();
160 		writeXML(o, baos, fragment);
161 		return new String(baos.toByteArray());
162 	}
163 
164 	public static void writeXML(Object o, OutputStream os, boolean fragment) throws PropertyException, JAXBException
165 	{
166 		Marshaller m = jc.createMarshaller();
167 		m.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
168 		if (fragment)
169 		{
170 			m.setProperty(Marshaller.JAXB_FRAGMENT, true);
171 		}
172 		else
173 		{
174 			m.setProperty(Marshaller.JAXB_FRAGMENT, false);
175 			m.setProperty(Marshaller.JAXB_NO_NAMESPACE_SCHEMA_LOCATION, "LEGO.xsd");
176 		}
177 		m.marshal(o, os);
178 	}

248 	public static void schemaValidateLego(final Lego lego, final LegoValidateCallback callback)
249 	{
250 		Runnable r = new Runnable()
251 		{
252 			@Override
253 			public void run()
254 			{
255 				try
256 				{
257 					logger.debug("Schema validating a Lego");
258 					//In order to schema validate, we need to check a full lego list.
259 					LegoList ll = new LegoList();
260 					ll.setGroupName("foo");
261 					ll.setLegoListUUID(UUID.randomUUID().toString());
262 					ll.getLego().add(lego);
263 					
264 					Source xmlFile = new StreamSource(new ByteArrayInputStream(toXML(ll).getBytes()));
265 					Validator validator = schema.newValidator();
266 					validator.validate(xmlFile);
267 					callback.validateComplete(true, null);
268 				}
269 				catch (Exception e)
270 				{
271 					callback.validateComplete(false, e.getMessage());
272 				}
273 			}
274 		};
275 		Utility.execute(r);
276 	}

278 	public static Tidy getTidy()
279 	{
280 		if (htmlTidy == null)
281 		{
282 			htmlTidy = new Tidy();
283 			htmlTidy.setXHTML(true);
284 			htmlTidy.setIndentContent(true);
285 			htmlTidy.setSmartIndent(true);
286 			htmlTidy.setXmlTags(true);
287 			htmlTidy.setWraplen(150);
288 			htmlTidy.setQuiet(true);
289 		}
290 		return htmlTidy;
291 	}

293 	public static void transform(LegoList ll, OutputStream target, Transformer transformer, boolean tidyHtml ) throws IOException, TransformerException, PropertyException, JAXBException
294 	{
295 		String legoListAsXML = toXML(ll);
296 		if (transformer != null)
297 		{
298 			ByteArrayInputStream bais = new ByteArrayInputStream(legoListAsXML.getBytes());
299 			ByteArrayOutputStream baos = new ByteArrayOutputStream();
300 			transformer.transform(new StreamSource(bais), new StreamResult(baos));
301 			if (tidyHtml)
302 			{
303 				getTidy().parse(new ByteArrayInputStream(baos.toByteArray()), target);
304 			}
305 			else
306 			{
307 				target.write(baos.toByteArray());
308 			}
309 		}
310 		else
311 		{
312 			target.write(legoListAsXML.getBytes());
313 		}
314 	}

323 			transformer.transform(new StreamSource(bais), new StreamResult(baos));
324 			if (tidyHtml)
325 			{
326 				getTidy().parse(new ByteArrayInputStream(baos.toByteArray()), target);
327 			}
328 			else
329 			{
330 				target.write(baos.toByteArray());

332 		}
333 		else
334 		{
335 			target.write(legoAsXML.getBytes());
