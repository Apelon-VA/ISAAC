40  public class LegoListBDB

44  	@PrimaryKey protected String legoListUUID;
45  	protected String groupDescription;

47  	protected List<String> legoUniqueIds;
48  	@SecondaryKey(relate = Relationship.ONE_TO_MANY) protected Set<String> legoUUIDs;

60  	{
61  		this.groupName = groupName;
62  		this.groupDescription = groupDescription;
63  		this.comment = comment;
64  		this.legoListUUID = uuid;

66  		this.legoUUIDs = new HashSet<>();

70  	public LegoListBDB(LegoList ll) throws WriteException
71  	{
72  		groupDescription = ll.getGroupDescription();
73  		groupName = ll.getGroupName();
74  		legoListUUID = ll.getLegoListUUID();
75  		comment = ll.getComment();
76  		legoUniqueIds = new ArrayList<>();
77  		legoBDBRefs = new ArrayList<>();
78  		this.legoUUIDs = new HashSet<>();
79  		this.legoUUIDsUsage = new HashMap<>();
80  		for (Lego l : ll.getLego())
81  		{
82  			LegoBDB lBDB = new LegoBDB(l);
83  
84  			verifyLegoUUID(l.getLegoUUID());
85  			legoUniqueIds.add(lBDB.getUniqueId());
86  			legoBDBRefs.add(lBDB);
87  			legoUUIDs.add(l.getLegoUUID());
88  			Integer temp = legoUUIDsUsage.get(l.getLegoUUID());
89  			if (temp == null)
90  			{
91  				temp = new Integer(1);
92  			}
93  			else
94  			{
95  				temp = new Integer(1 + temp.intValue());
96  			}
97  			legoUUIDsUsage.put(l.getLegoUUID(), temp);
98  		}
99  	}

106 	public void addLego(LegoBDB lego) throws WriteException
107 	{
108 		verifyLegoUUID(lego.getLegoUUID());
109 		legoUniqueIds.add(lego.getUniqueId());
110 		legoUUIDs.add(lego.getLegoUUID());
111 		Integer temp = legoUUIDsUsage.get(lego.getLegoUUID());
112 		if (temp == null)
113 		{
114 			temp = new Integer(1);
115 		}
116 		else
117 		{
118 			temp = new Integer(1 + temp.intValue());
119 		}
120 		legoUUIDsUsage.put(lego.getLegoUUID(), temp);
121 	}
122 
123 	public void removeLego(String legoUUID, String legoUniqueId)
124 	{
125 		legoUniqueIds.remove(legoUniqueId);
126 		Integer temp = legoUUIDsUsage.get(legoUUID);
127 		if (temp != null)
128 		{
129 			temp = new Integer(temp.intValue() - 1);
130 			legoUUIDsUsage.put(legoUUID, temp);
131 		}
132 		if (temp == null || temp.intValue() == 0)
133 		{
134 			legoUUIDs.remove(legoUUID);
135 			legoUUIDsUsage.remove(legoUUID);
136 		}
137 	}

139 	private void verifyLegoUUID(String legoUUID) throws WriteException
140 	{
141 		// legoUUID should only be used by this legoList (no other)
142 		List<String> legoListUUIDs = BDBDataStoreImpl.getInstance().getLegoListByLego(legoUUID);
143 		for (String s : legoListUUIDs)
144 		{
145 			if (!this.legoListUUID.equals(s))
146 			{
147 				throw new WriteException("The LEGO UUID '" + legoUUID + "' is already in use by the legoList '" + s + "'.  Lego UUIDs should not cross legoLists.");
148 			}
149 		}
150 	}

200 	public LegoList toSchemaLegoList()
201 	{
202 		LegoList ll = new LegoList();
203 		ll.setGroupDescription(groupDescription);
204 		ll.setGroupName(groupName);
205 		ll.setLegoListUUID(legoListUUID);
206 		ll.setComment(comment);
207 		List<Lego> legos = ll.getLego();
208 
209 		for (String lui : legoUniqueIds)
210 		{
211 			Lego l = ((BDBDataStoreImpl) BDBDataStoreImpl.getInstance()).getLegoByUniqueId(lui);
212 			if (l == null)
213 			{
214 				throw new DataStoreException("This shouldn't have been null!");
215 			}
216 			legos.add(l);
217 		}
