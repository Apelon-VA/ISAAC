42  /**
43   * This class handles the storage of the LEGO object into the BerkeleyDB.
44   * 
45   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
46   */
47  @Entity
48  public class LegoBDB
49  {
50  	@PrimaryKey private String uniqueId;
51  	@SecondaryKey(relate = Relationship.MANY_TO_ONE) protected String legoUUID;
52  	protected String stampId;
53  	protected String comment;
54  	@SecondaryKey(relate = Relationship.MANY_TO_ONE) protected String pncsId;
55  	protected List<Assertion> assertions;
56  	@SecondaryKey(relate = Relationship.MANY_TO_MANY) protected Set<String> usedSCTIdentifiers;
57  	// This is the list of assertions defined within this lego.
58  	@SecondaryKey(relate = Relationship.MANY_TO_MANY) protected Set<String> usedAssertionUUIDs;
59  	// This is the list of assertions linked to by this lego in the form of composite assertions.
60  	@SecondaryKey(relate = Relationship.MANY_TO_MANY) protected Set<String> compositeAssertionUUIDs;
61  
62  	// not stored
63  	private transient PncsBDB pncsBDBRef;
64  	private transient StampBDB stampBDBRef;
65  
66  	@SuppressWarnings("unused")
67  	private LegoBDB()
68  	{
69  		// required by BDB
70  	}
71  
72  	public LegoBDB(Lego lego) throws WriteException
73  	{
74  		legoUUID = lego.getLegoUUID();
75  		pncsBDBRef = new PncsBDB(lego.getPncs());
76  		pncsId = pncsBDBRef.getUniqueId();
77  		comment = lego.getComment();
78  		stampBDBRef = new StampBDB(lego.getStamp());
79  		stampId = stampBDBRef.getStampId();
80  		assertions = new ArrayList<>();
81  		usedAssertionUUIDs = new HashSet<>();
82  		compositeAssertionUUIDs = new HashSet<>();
83  		usedSCTIdentifiers = new HashSet<>();
84  		for (Assertion a : lego.getAssertion())
85  		{
86  			assertions.add(a);
87  			checkAndUpdateAssertionList(a);
88  			indexExpression(a.getDiscernible().getExpression());
89  			indexExpression(a.getQualifier().getExpression());
90  			indexMeasurement(a.getTiming());
91  			indexExpression(a.getValue().getExpression());
92  			indexMeasurement(a.getValue().getMeasurement());
93  
94  			for (AssertionComponent ac : a.getAssertionComponent())
95  			{
96  				compositeAssertionUUIDs.add(ac.getAssertionUUID());
97  			}
98  		}
99  
100 		this.uniqueId = ModelUtil.makeUniqueLegoID(legoUUID, stampId);
101 	}
102 
103 	private void indexConcept(Concept c)
104 	{
105 		if (c != null)
106 		{
107 			if (c.getSctid() != null)
108 			{
109 				usedSCTIdentifiers.add(c.getSctid() + "");
110 			}
111 			if (c.getUuid() != null && c.getUuid().length() > 0)
112 			{
113 				usedSCTIdentifiers.add(c.getUuid());
114 			}
115 		}
116 	}
117 
118 	private void indexExpression(Expression expression)
119 	{
120 		if (expression != null)
121 		{
122 			indexConcept(expression.getConcept());
123 
124 			if (expression.getExpression() != null)
125 			{
126 				for (Expression e : expression.getExpression())
127 				{
128 					indexExpression(e);
129 				}
130 			}
131 
132 			if (expression.getRelation() != null)
133 			{
134 				for (Relation r : expression.getRelation())
135 				{
136 					indexRelation(r);
137 				}
138 			}
139 			if (expression.getRelationGroup() != null)
140 			{
141 				for (RelationGroup rg : expression.getRelationGroup())
142 				{
143 					for (Relation r : rg.getRelation())
144 					{
145 						indexRelation(r);
146 					}
147 				}
148 			}
149 		}
150 	}
151 
152 	private void indexRelation(Relation r)
153 	{
154 		if (r.getType() != null)
155 		{
156 			indexConcept(r.getType().getConcept());
157 		}
158 		if (r.getDestination() != null)
159 		{
160 			indexExpression(r.getDestination().getExpression());
161 			indexMeasurement(r.getDestination().getMeasurement());
162 		}
163 	}
164 
165 	private void indexMeasurement(Measurement m)
166 	{
167 		if (m != null && m.getUnits() != null)
168 		{
169 			indexConcept(m.getUnits().getConcept());
170 		}
171 	}

188 	public StampBDB getStampBDB()
189 	{
190 		if (stampBDBRef == null)
191 		{
192 			stampBDBRef = ((BDBDataStoreImpl) BDBDataStoreImpl.getInstance()).getStampByUniqueId(stampId);
193 		}
194 		return stampBDBRef;
195 	}

203 	{
204 		return pncsId;
205 	}

207 	public void addAssertion(Assertion assertion) throws WriteException
208 	{
209 		if (assertions == null)
210 		{
211 			assertions = new ArrayList<>();
212 		}
213 		assertions.add(assertion);
214 		checkAndUpdateAssertionList(assertion);
215 
216 		for (AssertionComponent ac : assertion.getAssertionComponent())
217 		{
218 			compositeAssertionUUIDs.add(ac.getAssertionUUID());
219 		}
220 
221 		indexExpression(assertion.getDiscernible().getExpression());
222 		indexExpression(assertion.getQualifier().getExpression());
223 		indexMeasurement(assertion.getTiming());
224 		indexExpression(assertion.getValue().getExpression());
225 		indexMeasurement(assertion.getValue().getMeasurement());
226 	}
227 
228 	/**
229 	 * Note, this returns a copy (you can't add to this list)
230 	 */
231 	public List<Assertion> getAssertions()
232 	{
233 		ArrayList<Assertion> result = new ArrayList<>();
234 		if (assertions == null)
235 		{
236 			return result;
237 		}
238 		for (Assertion a : assertions)
239 		{
240 			result.add(a);
241 		}
242 		return result;
243 	}

246 	{
247 		return uniqueId;
248 	}

250 	public Lego toSchemaLego()
251 	{
252 		Lego l = new Lego();
253 		l.setLegoUUID(legoUUID);
254 		l.setComment(comment);
255 		l.setStamp(getStampBDB().toSchemaStamp());
256 		l.setPncs(((BDBDataStoreImpl) BDBDataStoreImpl.getInstance()).getPncsByUniqueId(pncsId));
257 		if (l.getPncs() == null)
258 		{
259 			throw new DataStoreException("Pncs should be present!");
260 		}
261 		l.getAssertion().addAll(assertions);
262 		return l;
263 	}
264 
265 	private void checkAndUpdateAssertionList(Assertion a) throws WriteException
266 	{
267 		if (!usedAssertionUUIDs.add(a.getAssertionUUID()))
268 		{
269 			throw new WriteException("Each assertion within a Lego must have a unique UUID");
270 		}
271 		Set<String> legosUsingAssertionUUID = ((BDBDataStoreImpl) BDBDataStoreImpl.getInstance()).getLegoUUIDsContainingAssertion(a.getAssertionUUID());
272 		legosUsingAssertionUUID.remove(legoUUID);
273 		if (legosUsingAssertionUUID.size() > 0)
274 		{
275 			throw new WriteException("The assertion UUID '" + a.getAssertionUUID() + "' is already in use by the lego '" + legosUsingAssertionUUID.iterator().next()
276 					+ "'.  Assertion UUIDs should be unique");
277 		}
278 	}
