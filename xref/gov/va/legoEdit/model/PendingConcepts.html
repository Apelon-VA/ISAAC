103 	private boolean areIdentifiersUnique(Concept potentialPendingConcept)
104 	{
105 		//no load check here - will cause a startup deadlock.  
106 		if (pendingConcepts.containsKey(potentialPendingConcept.getUuid()))
107 		{
108 			return false;
109 		}
110 
111 		if (null != WBUtility.lookupSnomedIdentifierAsCV(potentialPendingConcept.getSctid().toString()))
112 		{
113 			return false;
114 		}
115 		if (potentialPendingConcept.getUuid() != null && null != WBUtility.lookupSnomedIdentifierAsCV(potentialPendingConcept.getUuid().toString()))
116 		{
117 			return false;
118 		}
119 		return true;
120 	}

122 	public void addConcept(long id, String description, Long parent) throws IllegalArgumentException
123 	{
124 		loadCheck();
125 		PendingConcept c = new PendingConcept();
126 		c.setSctid(id);
127 		c.setDesc(description);
128 		c.setUuid(UUID.nameUUIDFromBytes((c.getSctid() + "").getBytes()).toString());
129 		if (areIdentifiersUnique(c))
130 		{
131 			pendingConcepts.put(c.getUuid(), c);
132 			if (c.getSctid() > highestInUseId)
133 			{
134 				highestInUseId = c.getSctid();
135 			}
136 			if (parent != null)
137 			{
138 				//allow ref to pending, but not self....
139 				Concept parentConcept =  WBUtility.lookupSnomedIdentifier(parent + "");
140 				if (parentConcept != null && parentConcept.getSctid().longValue() != id)
141 				{
142 					parentConcepts.put(id, parentConcept);
143 				}
144 				else
145 				{
146 					throw new IllegalArgumentException("The specified parent SCTID isn't a valid snomed concept");
147 				}
148 			}
149 			notifyListeners();
150 		}
151 		else
152 		{
153 			throw new IllegalArgumentException("The provided concept is not unique");
154 		}
155 		try
156 		{
157 			rewritePendingConceptsFile();
158 		}
159 		catch (IOException e)
160 		{
161 			logger.error("Pending concepts Store failed", e);
162 			pendingConcepts.remove(c.getUuid());
163 			parentConcepts.remove(id);
164 			throw new IllegalArgumentException("Sorry, store failed");
165 		}
166 	}

168 	public void deleteConcept(long id) throws IllegalArgumentException
169 	{
170 		loadCheck();
171 		PendingConcept pending = pendingConcepts.remove(UUID.nameUUIDFromBytes((id + "").getBytes()).toString());
172 		Concept parent = parentConcepts.remove(id);
173 		try
174 		{
175 			rewritePendingConceptsFile();
176 			notifyListeners();
177 		}
178 		catch (IOException e)
179 		{
180 			logger.error("Pending concepts Store failed", e);
181 			pendingConcepts.put(pending.getUuid(),  pending);
182 			if(parent != null)
183 			{
184 				parentConcepts.put(id, parent);
185 			}
186 			throw new IllegalArgumentException("Sorry, store failed");
187 		}
188 	}

190 	public long getUnusedId()
191 	{
192 		loadCheck();
193 		while (true)
194 		{
195 			long temp = ++highestInUseId;
196 			Concept possible = new PendingConcept();
197 			possible.setSctid(temp);
198 			if (areIdentifiersUnique(possible))
199 			{
200 				return temp;
201 			}
202 		}

205 	private void rewritePendingConceptsFile() throws IOException
206 	{
207 		//Read through the existing file, keeping the comments, and any lines we don't understand.  
208 		//only keep the concepts if they our in our current list.  Finally, add any concepts that are missing.
209 		loadCheck();
210 		HashSet<String> unstoredConcepts = new HashSet<>();
211 		unstoredConcepts.addAll(pendingConcepts.keySet());
212 		
213 		List<String> lines = Files.readAllLines(pendingConceptsFile.toPath(), StandardCharsets.UTF_8);
214 		StringBuilder replacement = new StringBuilder();
215 		String eol = System.getProperty("line.separator");
216 		for (String line : lines)
217 		{
218 			if (line.startsWith("#") || line.length() == 0)
219 			{
220 				replacement.append(line);
221 				replacement.append(eol);
222 			}
223 			else
224 			{
225 				String[] parts = line.split("\t");
226 				try
227 				{
228 					long id = Long.parseLong(parts[0]);
229 					String uuid = UUID.nameUUIDFromBytes((id + "").getBytes()).toString();
230 					if (pendingConcepts.containsKey(uuid))
231 					{
232 						replacement.append(buildLine(pendingConcepts.get(uuid)));
233 						replacement.append(eol);
234 						unstoredConcepts.remove(uuid);
235 					}
236 					
237 				}
238 				catch (Exception e) 
239 				{
240 					replacement.append(line);
241 					replacement.append(eol);
242 				}
243 			}
244 		}
245 		for (String uuid : unstoredConcepts)
246 		{
247 			replacement.append(buildLine(pendingConcepts.get(uuid)));
248 			replacement.append(eol);
249 		}
250 		
251 		Files.write(pendingConceptsFile.toPath(), replacement.toString().getBytes(), 
252 				StandardOpenOption.TRUNCATE_EXISTING, StandardOpenOption.WRITE, StandardOpenOption.CREATE);
253 	}

255 	private String buildLine(Concept c)
256 	{
257 		Concept parent = parentConcepts.get(c.getSctid());
258 		return c.getSctid() + "\t" + c.getDesc() + (parent == null ? "" : "\t" + parent.getSctid() + "\t" + parent.getDesc());
259 	}

261 	public PendingConcept getConcept(String sctIdOrUUID)
262 	{
263 		if (sctIdOrUUID == null)
264 		{
265 			return null;
266 		}
267 		loadCheck();
268 		PendingConcept temp = pendingConcepts.get(sctIdOrUUID.trim());
269 		if (temp == null)
270 		{
271 			temp = pendingConcepts.get(UUID.nameUUIDFromBytes((sctIdOrUUID.trim() + "").getBytes()).toString()); 
272 		}
273 		return temp;
274 	}

315 	private class ReadData implements Runnable
316 	{
317 		@Override
318 		public void run()
319 		{
320 			logger.info("Loading pending concepts from: " + pendingConceptsFile.getAbsolutePath());
321 			try
322 			{
323 				if (pendingConceptsFile.exists())
324 				{
325 					List<String> lines = Files.readAllLines(pendingConceptsFile.toPath(), StandardCharsets.UTF_8);
326 					for (String s : lines)
327 					{
328 						if (s.startsWith("#") || s.length() == 0)
329 						{
330 							continue;
331 						}
332 						String[] parts = s.split("\t");
333 						if (parts.length > 1)
334 						{
335 							PendingConcept c = new PendingConcept();
336 							try
337 							{
338 								c.setSctid(Long.parseLong(parts[0]));
339 							}
340 							catch (NumberFormatException e)
341 							{
342 								logger.error("Invalid ID in pending concepts file - line '" + s + "'");
343 								continue;
344 							}
345 							c.setDesc(parts[1]);
346 							c.setUuid(UUID.nameUUIDFromBytes(parts[0].getBytes()).toString());
347 							
348 							Concept parent = null;
349 
350 							if (parts.length > 2)
351 							{
352 								long parentSCTID = Long.parseLong(parts[2]);
353 								//Use this lookup, since it doesn't loop back to pending.
354 								ConceptVersionBI wbParentConcept =  WBUtility.lookupSnomedIdentifierAsCV(parentSCTID + "");
355 								if (wbParentConcept == null)
356 								{
357 									//See if it is a different pending concept (must be higher in the file, for this to work)
358 									parent = pendingConcepts.get(UUID.nameUUIDFromBytes((parentSCTID + "").getBytes()).toString());
359 								}
360 								else if (wbParentConcept != null)
361 								{
362 									parent = WBUtility.convertConcept(wbParentConcept);
363 								}
364 								if (parent == null)
365 								{
366 									logger.error("The specified parent concept for " + c.getSctid() + " doesn't exist and will be ignored");
367 								}
368 							}
369 							
370 							if (!areIdentifiersUnique(c))
371 							{
372 								logger.error("Pending concepts contains a value which is a duplicate, or already exists in snomed '" + c.getSctid() + "'.  Ignoring.");
373 								continue;
374 							}
375 							
376 							pendingConcepts.put(c.getUuid(), c);
377 							if (c.getSctid() > highestInUseId)
378 							{
379 								highestInUseId = c.getSctid();
380 							}
381 							if (parent != null)
382 							{
383 								parentConcepts.put(c.getSctid(), parent);
384 							}
385 						}
386 						else
387 						{
388 							logger.error("Pending concepts need an ID and a description");
389 						}
390 					}
391 				}
392 			}
393 			catch (IOException e)
394 			{
395 				logger.error("Unexpected error loading pending concepts file", e);
396 			}
397 			synchronized (PendingConcepts.class)
398 			{
399 				loadCompleted = true;
400 				PendingConcepts.class.notifyAll();
401 			}
402 			logger.info("Loaded " + pendingConcepts.size() + " pending concepts");
403 		}
