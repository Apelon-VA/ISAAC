49  /**
50   * The toString methods more or less follow the XML schema structure, and are very verbose.  Mostly used for debugging, 
51   * and also for the copy to clipboard functions.
52   * 
53   * The summary methods are a much more compact summary, used in top of the lego tree view.
54   * SchemaToString
55   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a> 
56   * Copyright 2013
57   *
58   */
59  public class SchemaToString
60  {
61  	private static Logger logger = LoggerFactory.getLogger(SchemaToString.class);
62  	public final static String eol = System.getProperty("line.separator");
63  	public final static String infinity = "\u221e";
64  	public final static String center = " \u2716 ";
65  	public final static String downRightArrow = " \u21b3 ";
66  	public final static String rightArrow = " \u2192 ";
67  	public final static String ltEq = "\u2264";

69  	public static String toString(Object o)
70  	{
71  		if (o instanceof Lego)
72  		{
73  			return toString((Lego) o);
74  		}
75  		else if (o instanceof Assertion)
76  		{
77  			return toString((Assertion) o, "");
78  		}
79  		else if (o instanceof Discernible)
80  		{
81  			return toString((Discernible) o, "");
82  		}
83  		else if (o instanceof Qualifier)
84  		{
85  			return toString((Qualifier) o, "");
86  		}
87  		else if (o instanceof Value)
88  		{
89  			return toString((Value) o, "");
90  		}
91  		else if (o instanceof Expression)
92  		{
93  			return toString((Expression) o, "");
94  		}
95  		else
96  		{
97  			logger.warn("Unsupported use of SchemaToString for " + o);
98  			return o.toString();
99  		}
100 	}

136 	{
137 		if (ll == null)
138 		{
139 			return "<null LegoList>";
140 		}
141 		else
142 		{
143 			StringBuilder sb = new StringBuilder();
144 			sb.append("LegoList " + ll.getLegoListUUID());
145 			sb.append(eol + "  Group Name " + ll.getGroupName());
146 			if (ll.getGroupDescription() != null && ll.getGroupDescription().length() > 0)
147 			{
148 				sb.append(eol + "  Group Description " + ll.getGroupDescription());
149 			}
150 			if (ll.getComment() != null && ll.getComment().length() > 0)
151 			{
152 				sb.append(eol + "  Comment: " + ll.getComment());
153 			}
154 			for (Lego l : ll.getLego())
155 			{
156 				sb.append(eol + summary(l, "  "));
157 			}
158 			return sb.toString();
159 		}

250 		if (a.getAssertionComponent().size() > 0)
251 		{
252 			sb.append(eol + prefix + "  Assertion Components");
253 			for (AssertionComponent ac : a.getAssertionComponent())
254 			{
255 				sb.append(eol + toString(ac, prefix + "  "));
256 			}
257 		}
258 		sb.append(eol);
259 		return sb.toString();

587 	public static String toString(Expression e, String prefix)
588 	{
589 		if (e == null)
590 		{
591 			return prefix + "<null Expression>";
592 		}
593 
594 		StringBuilder sb = new StringBuilder();
595 		sb.append(prefix + "Expression");
596 
597 		if (e.getConcept() != null)
598 		{
599 			sb.append(eol + toString(e.getConcept(), prefix + "  "));
600 		}
601 		else
602 		{
603 			if (e.getExpression().size() > 0)
604 			{
605 				sb.append(eol + prefix + "  Conjunction");
606 			}
607 			for (Expression e1 : e.getExpression())
608 			{
609 				sb.append(eol + toString(e1, prefix + "  |  "));
610 			}
611 		}
612 		for (Relation r : e.getRelation())
613 		{
614 			sb.append(eol + prefix + downRightArrow);
615 			sb.append(toString(r.getType(), ""));
616 			sb.append(rightArrow);
617 			sb.append(toString(r.getDestination(), prefix + "    ", false));
618 		}
619 		for (RelationGroup rg : e.getRelationGroup())
620 		{
621 			sb.append(eol + prefix + downRightArrow +  "  Relation Group");
622 			for (Relation r : rg.getRelation())
623 			{
624 				sb.append(eol + prefix + "  " + downRightArrow);
625 				sb.append(toString(r.getType(), ""));
626 				sb.append(rightArrow);
627 				sb.append(toString(r.getDestination(), prefix + "      ", false));
628 			}
629 		}
630 		return sb.toString();
631 	}

633 	public static String summary(Expression e, String prefix, boolean prefixIndividualConcept)
634 	{
635 		if (e == null)
636 		{
637 			return prefix + "<null Expression>";
638 		}
639 		
640 		int conceptLength = 0;
641 
642 		StringBuilder sb = new StringBuilder();
643 		if (e.getConcept() != null)
644 		{
645 			sb.append(toString(e.getConcept(), (prefixIndividualConcept ? prefix : "")));
646 			conceptLength += sb.length();
647 		}
648 		else
649 		{
650 			if (e.getExpression().size() > 0)
651 			{
652 				sb.append(eol + prefix + "  Conjunction");
653 			}
654 			for (Expression e1 : e.getExpression())
655 			{
656 				sb.append(eol + summary(e1, prefix + "  |  ", true));
657 			}
658 		}
659 		for (Relation r : e.getRelation())
660 		{
661 			sb.append(eol + prefix + downRightArrow);
662 			String temp = toString(r.getType(), "");
663 			StringBuilder indent = new StringBuilder();
664 			indent.append(prefix);
665 			for (int i = 0; i < conceptLength + temp.length(); i++)
666 			{
667 				indent.append(" ");
668 			}
669 			sb.append(temp);
670 			sb.append(rightArrow);
