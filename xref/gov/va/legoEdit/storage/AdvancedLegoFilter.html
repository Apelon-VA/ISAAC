97  	static private boolean legoPassesFilter(Lego l, Concept relTypeFilter, Concept relDestFilter, String destModifier, String relAppliesToLegoSection)
98  	{
99  		for (Assertion a : l.getAssertion())
100 		{
101 			if (relAppliesToLegoSection == null && relDestFilter == null)
102 			{
103 				for (AssertionComponent ac : a.getAssertionComponent())
104 				{
105 					if (typeMatches(ac.getType(), relTypeFilter))
106 					{
107 						return true;
108 					}
109 				}
110 			}
111 			
112 			if ((relAppliesToLegoSection == null || relAppliesToLegoSection.equals(LegoFilterPaneController.DISCERNIBLE)) && a.getDiscernible() != null 
113 					&& typeMatches(a.getDiscernible().getExpression(), relTypeFilter) 
114 					&& destMatches(a.getDiscernible().getExpression(), relDestFilter, destModifier, false))
115 			{
116 					return true;
117 			}
118 			if ((relAppliesToLegoSection == null || relAppliesToLegoSection.equals(LegoFilterPaneController.QUALIFIER)) && a.getQualifier() != null 
119 					&& typeMatches(a.getQualifier().getExpression(), relTypeFilter) 
120 					&& destMatches(a.getQualifier().getExpression(), relDestFilter, destModifier, false))
121 			{
122 					return true;
123 			}
124 			
125 			if ((relAppliesToLegoSection == null || relAppliesToLegoSection.equals(LegoFilterPaneController.VALUE)) && a.getValue() != null 
126 					&& typeMatches(a.getValue().getExpression(), relTypeFilter) 
127 					&& destMatches(a.getValue().getExpression(), relDestFilter, destModifier, false))
128 			{
129 					return true;
130 			}
131 			
132 		}
133 		//If we haven't returned yet, none of the assertions matched both the type and dest filters.
134 		return false;
135 	}

137 	static private boolean destMatches(Expression e, Concept relDestFilter, String destModifier, boolean isDestConcept)
138 	{
139 		if (relDestFilter == null)
140 		{
141 			return true;
142 		}
143 		if (e == null)
144 		{
145 			return false;
146 		}
147 		
148 		if (isDestConcept)
149 		{
150 			if ((LegoFilterPaneController.IS.equals(destModifier) && conceptMatches(relDestFilter, e.getConcept()))
151 					|| (LegoFilterPaneController.CHILD_OF.equals(destModifier) && conceptHierarchyMatches(relDestFilter, e.getConcept())))
152 			{
153 				return true;
154 			}
155 		}
156 		
157 		for (Expression e1 : e.getExpression())
158 		{
159 			if (destMatches(e1, relDestFilter, destModifier, isDestConcept))
160 			{
161 				return true;
162 			}
163 		}
164 		for (Relation r : e.getRelation())
165 		{
166 			if (destMatches(r, relDestFilter, destModifier))
167 			{
168 				return true;
169 			}
170 		}
171 		
172 		for (RelationGroup rg : e.getRelationGroup())
173 		{
174 			for (Relation r : rg.getRelation())
175 			{
176 				if (destMatches(r, relDestFilter, destModifier))
177 				{
178 					return true;
179 				}
180 			}
181 		}
182 		
183 		return false;
184 	}

335 	static private boolean conceptHierarchyMatches(Concept filter, Concept concept)
336 	{
337 		if (concept == null)
338 		{
339 			return false;
340 		}
341 		
342 		//this checks filter null
343 		if (conceptMatches(filter, concept))
344 		{
345 			return true;
346 		}
347 		
348 		//The cache I'm keeping is a mapping of the concept, to the desired filter concept, to the answer (yes or no)
349 		String conceptKey = makeKey(concept);
350 		String filterKey = makeKey(filter);
351 		HashMap<String, Boolean> map = hierarchyCache.get(conceptKey);
352 		if (map != null)
353 		{
354 			Boolean result = map.get(filterKey);
355 			if (result != null)
356 			{
357 				logger.debug("Cache hit - cache said " + result);
358 				return result;
359 			}
360 		}
361 		
362 		if (map == null)
363 		{
364 			map = new HashMap<String, Boolean>();
365 			hierarchyCache.put(conceptKey, map);
366 		}
367 		
368 		ConceptVersionBI cv = WBUtility.lookupIdentifier(concept.getUuid());
369 		if (cv == null)
370 		{
371 			cv = WBUtility.lookupIdentifier(concept.getSctid() + "");
372 		}
373 		if (cv == null)
374 		{
375 			return false;
376 		}
377 		else
378 		{
379 			try
380 			{
381 				Collection<List<Integer>> pathsToRoot = cv.getNidPathsToRoot();
382 				for (List<Integer> list : pathsToRoot)
383 				{
384 					for (Integer i : list)
385 					{
386 						Concept c = LegoWBUtility.convertConcept(WBUtility.getConceptVersion(i));
387 						if (conceptMatches(filter, c))
388 						{
389 							map.put(filterKey, true);
390 							return true;
391 						}
392 					}
393 				}
394 			}
395 			catch (IOException e)
396 			{
397 				logger.error("Unexpeted error during AdvancedLegoFilter", e);
398 			}
399 		}
400 		map.put(filterKey, false);
401 		return false;
402 	}
