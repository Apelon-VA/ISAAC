19  package gov.va.legoEdit.storage;
20  
21  import gov.va.legoEdit.model.schemaModel.Lego;
22  import gov.va.legoEdit.model.schemaModel.LegoList;
23  import gov.va.legoEdit.model.schemaModel.Pncs;
24  import gov.va.legoEdit.model.schemaModel.Stamp;
25  import java.util.List;
26  
27  /**
28   * 
29   * DataStoreInterface
30   * 
31   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
32   * Copyright 2013
33   */
34  public interface DataStoreInterface
35  {
36  
37  	/**
38  	 * Get the legoList object with the specified name.
39  	 */
40  	public LegoList getLegoListByName(String legoListGroupName) throws DataStoreException;
41  
42  	/**
43  	 * Get the legoList object with the specified uuid.
44  	 */
45  	public LegoList getLegoListByID(String legoListUUID) throws DataStoreException;
46  
47  	/**
48  	 * Get an iterator that traverses all stored legoLists.
49  	 */
50  	public CloseableIterator<LegoList> getLegoLists() throws DataStoreException;
51  
52  	/**
53  	 * Get all LEGO (versions) in the DB with the provided UUID.
54  	 */
55  	public List<Lego> getLegos(String legoUUID) throws DataStoreException;
56  
57  	/**
58  	 * Get the exact lego that matches the specified ID and Stamp
59  	 */
60  	public Lego getLego(String legoUUID, String stampUUID) throws DataStoreException;
61  
62  	/**
63  	 * Get all LEGOs which contain (define) the assertion with the specified assertion UUID. (There may be more than one stamped lego with the same
64  	 * UUID - up to the caller to decide which stamp they want) Note - this does not return legos that simply link to the named assertion via an
65  	 * assertionComponent
66  	 */
67  	public List<Lego> getLegosContainingAssertion(String assertionUUID);
68  
69  	/**
70  	 * Get all LEGOs which use the assertion with the specified assertion UUID as part of an assertionComponent
71  	 */
72  	public List<Lego> getLegosUsingAssertion(String assertionUUID);
73  
74  	/**
75  	 * Get all LEGOs which contain the specified snomed concept (as any child).
76  	 */
77  	public List<Lego> getLegosContainingConceptIdentifiers(String... conceptUuidOrSCTId) throws DataStoreException;
78  
79  	/**
80  	 * Get an iterator that traverses all stored LEGO objects.
81  	 */
82  	public CloseableIterator<Lego> getLegos() throws DataStoreException;
83  
84  	/**
85  	 * Get all LEGOs which reference the specified PNCS.
86  	 */
87  	public List<Lego> getLegosForPncs(int id, String value) throws DataStoreException;
88  
89  	/**
90  	 * Get all LEGOs which reference the specified PNCS.
91  	 */
92  	public List<Lego> getLegosForPncs(int id) throws DataStoreException;
93  
94  	/**
95  	 * Get all of the PNCS objects currently in the system.
96  	 */
97  	public CloseableIterator<Pncs> getPncs() throws DataStoreException;
98  
99  	/**
100 	 * Get all of the PNCS objects which have a matching id.
101 	 */
102 	public List<Pncs> getPncs(int id) throws DataStoreException;
103 
104 	/**
105 	 * Get the PNCS for the specified id and value
106 	 */
107 	public Pncs getPncs(int id, String value);
108 
109 	/**
110 	 * Get the UUID of the legoList(s) which contains the specified LEGO.
111 	 * 
112 	 * @return null if no legoList contains the specified LEGO
113 	 */
114 	public List<String> getLegoListByLego(String legoUUID) throws DataStoreException;
115 
116 	/**
117 	 * Create and store a new (empty) LegoList.
118 	 * 
119 	 * @throws WriteException if the operation fails
120 	 */
121 	public LegoList createLegoList(String uuid, String groupName, String groupDescription, String comments) throws WriteException;
122 
123 	/**
124 	 * A method to allow updating the the name, description and comments field that is stored on the lego list.
125 	 * 
126 	 * @param groupName - the new name, or null to leave the existing name in place. Cannot be an empty string.
127 	 * @param groupDescription - the new description, or null to leave the existing description in place
128 	 * @param comments - the new comments, or null to leave the existing comments in place.
129 	 */
130 	public void updateLegoListMetadata(String legoListUUID, String groupName, String groupDescription, String comments) throws WriteException;
131 
132 	/**
133 	 * Store the specified legoList. Note - this method cannot be used to replace an existing LegoList.
134 	 * 
135 	 * This method does not make any changes to the STAMP values of any contained LEGO objects.
136 	 * 
137 	 * @throws WriteException if a legoList of the specified name already exists or if the store operation fails
138 	 */
139 	public void importLegoList(LegoList legoList) throws WriteException;
140 
141 	/**
142 	 * Delete the specified legoList (and all LEGO children contained within).
143 	 * 
144 	 * @throws WriteException if the operation fails
145 	 */
146 	public void deleteLegoList(String legoListUUID) throws WriteException;
147 
148 	/**
149 	 * Delete the exact lego that matches the specified ID and Stamp within the specified LegoList
150 	 */
151 	public void deleteLego(String legoListUUID, String legoUUID, String stampUUID) throws WriteException;
152 
153 	/**
154 	 * This method always adds a new LEGO to the specified legoList - even if a LEGO with the same UUID already exists in the legoList.
155 	 * 
156 	 * The new LEGO will be differentiated from other LEGOs with the same UUID by the STAMP value (status, time, author, module, path)
157 	 * 
158 	 * The time and UUID fields of the STAMP will be updated by this method - time set to now - UUID set to a new unique value.
159 	 * 
160 	 * If status, author, module or path are missing, they will be set to the default values.
161 	 * 
162 	 * The updated STAMP will be returned to the caller.
163 	 * 
164 	 * @return The STAMP given to the LEGO as committed.
165 	 * 
166 	 * @throws WriteException if the specified legoList does not exist or if the write operation fails for any reason.
167 	 */
168 	public Stamp commitLego(Lego lego, String legoListUUID) throws WriteException;
169 
170 	/**
171 	 * Call to notify the backend to shutdown cleanly.
172 	 */
173 	public void shutdown();
174 }
