40  /**
41   * 
42   * {@link LegoSummaryPane} Just a trivial view of the simplified toString for a LegoList or a Lego.
43   *
44   * @author <a href="mailto:daniel.armbrust.list@gmail.com">Dan Armbrust</a>
45   */
46  public class LegoSummaryPane extends VBox
47  {
48  	private TextArea summary;
49  	private String title;
50  	Logger logger = LoggerFactory.getLogger(LegoSummaryPane.class);
51  	private String legoListUUID_;
52  	
53  	public LegoSummaryPane(LegoList legoList)
54  	{
55  		legoListUUID_ = legoList.getLegoListUUID();
56  		title = "LEGO List " + legoList.getGroupName(); 
57  		setup(() ->
58  		{
59  			return SchemaToString.summary(legoList);
60  		});
61  	}
62  	
63  	public LegoSummaryPane(Lego lego)
64  	{
65  		title = "LEGO " + lego.getLegoUUID(); 
66  		setup(() ->
67  		{
68  			return SchemaToString.summary(lego);
69  		});
70  	}
71  	
72  	private void setup(Supplier<String> stringRepresentation)
73  	{
74  		getStyleClass().add("itemBorder");
75  		Label l = new Label("Summary");
76  		l.getStyleClass().add("boldLabel");
77  		getChildren().add(l);
78  		summary = new TextArea();
79  		summary.setEditable(false);
80  		summary.setWrapText(false);
81  		summary.setFocusTraversable(false);
82  		updateSummary(stringRepresentation);
83  		getChildren().add(summary);
84  		VBox.setVgrow(summary, Priority.ALWAYS);
85  		setMinWidth(400.0);
86  		setMinHeight(300.0);
87  		VBox.setVgrow(this, Priority.NEVER);
88  		if (legoListUUID_ != null)
89  		{
90  			Button export = new Button("Export");
91  			export.setOnAction((a) ->
92  			{
93  				ArrayList<LegoList> temp = new ArrayList<LegoList>();
94  				temp.add(BDBDataStoreImpl.getInstance().getLegoListByID(legoListUUID_));
95  				try
96  				{
97  					new ExportDialog(temp, getScene().getWindow());
98  				}
99  				catch (Exception e)
100 				{
101 					logger.error("Unexpected error exporting LegoList ", e);
102 					AppContext.getCommonDialogs().showErrorDialog("Error exporting Lego Lists", e);
103 				}
104 			});
105 			getChildren().add(export);
106 		}
107 	}
108 	
109 	private void updateSummary(Supplier<String> stringRepresentation)
110 	{
111 		Utility.execute(() -> 
112 		{
113 			final StringBuilder value = new StringBuilder();
114 			value.append(stringRepresentation.get());
115 			Platform.runLater(new Runnable()
116 			{
117 				@Override
118 				public void run()
119 				{
120 					double scrollTop = summary.getScrollTop();
121 					summary.setText(value.toString());
122 					summary.setScrollTop(scrollTop);
123 				}
124 			});
125 		});
126 	}
127 	
128 	public void show()
129 	{
130 		Stage stage = new Stage();
131 		stage.setScene(new Scene(this));
132 		stage.getScene().getStylesheets().add(LegoSummaryPane.class.getResource("/isaac-shared-styles.css").toString());
133 		stage.setTitle(title);
134 		stage.setWidth(700);
135 		stage.setHeight(600);
136 		stage.show();
137 	}
