59  	private void showSummaryPane(LegoList legoList)
60  	{
61  		LegoSummaryPane lsp = new LegoSummaryPane(legoList);
62  
63  		Stage stage = new Stage();
64  		stage.setScene(new Scene(lsp));
65  		stage.getScene().getStylesheets().add(LegoGUI.class.getResource("/styles.css").toString());
66  		stage.setTitle("Lego List Viewer");
67  		stage.show();
68  	}

70  	public void showFileImportChooser(Window parent) throws IOException
71  	{
72  		FileChooser fc = new FileChooser();
73  		fc.getExtensionFilters().add(new FileChooser.ExtensionFilter("LEGO xml Files (*.xml)", "*.xml"));
74  		fc.getExtensionFilters().add(new FileChooser.ExtensionFilter("All Files (*.*)", "*"));
75  		List<File> files = fc.showOpenMultipleDialog(parent);
76  		if (files != null && files.size() > 0)
77  		{
78  			for (String id : showImportDialog(files, parent))
79  			{
80  				showSummaryPane(BDBDataStoreImpl.getInstance().getLegoListByID(id));
81  			}
82  		}
83  	}
84  
85  	private List<String> showImportDialog(List<File> filesToImport, Window parent) throws IOException
86  	{
87  		Stage stage = new Stage();
88  		stage.initModality(Modality.WINDOW_MODAL);
89  		stage.initOwner(parent);
90  		stage.initStyle(StageStyle.UTILITY);
91  		FXMLLoader loader = new FXMLLoader();
92  		loader.setLocation(ImportDialogController.class.getResource("ImportDialog.fxml"));
93  		Scene scene = new Scene((Parent) loader.load(ImportDialogController.class.getResourceAsStream("ImportDialog.fxml")));
94  		ImportDialogController idc = loader.getController();
95  		scene.getStylesheets().add(LegoGUI.class.getResource("/styles.css").toString());
96  		stage.setScene(scene);
97  		idc.importFiles(filesToImport);
98  		return idc.getImportedLegoListsIDs();
99  	}

108 	 * @param legoLists

111 	protected static void export(List<LegoList> legoLists, Window parent) throws IOException
112 	{
113 			Stage exportStage = new Stage();
114 			exportStage.initModality(Modality.WINDOW_MODAL);
115 			exportStage.initOwner(parent);
116 			exportStage.initStyle(StageStyle.UTILITY);
117 			FXMLLoader loader = new FXMLLoader();
118 			loader.setLocation(ExportDialogController.class.getResource("ExportDialog.fxml"));
119 			Scene scene = new Scene((Parent) loader.load(ExportDialogController.class.getResourceAsStream("ExportDialog.fxml")));
120 			ExportDialogController exportDC = loader.getController();
121 			scene.getStylesheets().add(LegoGUI.class.getResource("/styles.css").toString());
122 			exportStage.setScene(scene);
123 			exportDC.exportFiles(legoLists);
124 	}
