405 	private boolean isValid()
406 	{
407 		if (isValid == null)
408 		{
409 			//prevent multiple threads from duplicating work
410 			synchronized (this)
411 			{
412 				if (isValid == null)
413 				{
414 					validate();
415 				}
416 			}
417 		}
418 		return isValid;
419 	}
420

422 	{
423 		if (areChildrenValid == null)
424 		{

428 				if (areChildrenValid == null)

433 		}

482 	private void validateChildren()
483 	{
484 		boolean newValue = true;
485 		for (TreeItem<String> ti : getChildren())
486 		{
487 			LegoTreeItem lti = (LegoTreeItem)ti;
488 			if (!lti.isValid() || !lti.areChildrenValid())
489 			{
490 				newValue = false;
491 				break;
492 			}
493 		}
494 		
495 		areChildrenValid = newValue;
496 	}

543 	private void updateValidityImage()
544 	{
545 		final LegoTreeNodeGraphic localTreeNodeGraphic = treeNodeGraphic;
546 		
547 		if (localTreeNodeGraphic != null && LegoTreeNodeType.concept != ltnt_ && LegoTreeNodeType.assertionUUID != ltnt_ && LegoTreeNodeType.text != ltnt_)
548 		{
549 			//run both of these in the thread that called us (probably a background thread)
550 			final boolean isValid = isValid(); 
551 			final boolean areChildrenValid = areChildrenValid();
552 	
553 			Platform.runLater(new Runnable()
554 			{
555 				@Override
556 				public void run()
557 				{
558 					localTreeNodeGraphic.showProgress(false);
559 					if (isValid && areChildrenValid)
560 		 			{
561 						localTreeNodeGraphic.showInvalid(false);
562 					}
563 					else
564 					{
565 						localTreeNodeGraphic.showInvalid(true);
566 						localTreeNodeGraphic.setInvalidReason((isValid ? "Error in child" : invalidReason_));
567 					}
568 				}
569 			});
570 		}
