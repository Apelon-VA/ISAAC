35  public class LegoListPropertiesController implements Initializable
36  {
37  	@FXML// fx:id="rootPane"
38  	private AnchorPane rootPane; // Value injected by FXMLLoader
39  	@FXML//  fx:id="cancelButton"
40  	private Button cancelButton; // Value injected by FXMLLoader
41  	@FXML//  fx:id="legoListDescription"
42  	private TextField legoListDescription; // Value injected by FXMLLoader
43  	@FXML//  fx:id="legoListName"
44  	private TextField legoListName; // Value injected by FXMLLoader
45  	@FXML//  fx:id="legoListUUID"
46  	private TextField legoListUUID; // Value injected by FXMLLoader
47  	@FXML//  fx:id="okButton"
48  	private Button okButton; // Value injected by FXMLLoader
49  	@FXML//  fx:id="legoListComments"
50  	private TextArea legoListComments; // Value injected by FXMLLoader
51  
52  	private TreeItem<String> ti_;
53  	LegoListByReference llbr_ = null;
54  
55  	BooleanProperty nameValid = new SimpleBooleanProperty(false);
56  	BooleanProperty descValid = new SimpleBooleanProperty(false);
57  	BooleanBinding formValid;
58  
59  	@Override
60  	// This method is called by the FXMLLoader when initialization is complete
61  	public void initialize(URL fxmlFileLocation, ResourceBundle resources)
62  	{
63  		assert cancelButton != null : "fx:id=\"cancelButton\" was not injected: check your FXML file 'LegoListProperties.fxml'.";
64  		assert legoListDescription != null : "fx:id=\"legoListDescription\" was not injected: check your FXML file 'LegoListProperties.fxml'.";
65  		assert legoListName != null : "fx:id=\"legoListName\" was not injected: check your FXML file 'LegoListProperties.fxml'.";
66  		assert legoListUUID != null : "fx:id=\"legoListUUID\" was not injected: check your FXML file 'LegoListProperties.fxml'.";
67  		assert okButton != null : "fx:id=\"okButton\" was not injected: check your FXML file 'LegoListProperties.fxml'.";
68  
69  		// initialize your logic here: all @FXML variables will have been injected
70  
71  		legoListName.textProperty().addListener(new ChangeListener<String>()
72  		{
73  			@Override
74  			public void changed(ObservableValue<? extends String> observable, String oldValue, String newValue)
75  			{
76  				//need to make sure it doesn't collide with something in the DB. Ok to collide with self.
77  				if (newValue.length() > 0
78  						&& ((llbr_ != null && llbr_.getGroupName().equals(newValue)) || BDBDataStoreImpl.getInstance().getLegoListByName(newValue) == null))
79  				{
80  					nameValid.set(true);
81  					legoListName.setEffect(null);
82  				}
83  				else
84  				{
85  					nameValid.set(false);
86  					legoListName.setEffect(FxUtils.redDropShadow);
87  				}
88  			}
89  		});
90  
91  		legoListDescription.textProperty().addListener(new ChangeListener<String>()
92  		{
93  			@Override
94  			public void changed(ObservableValue<? extends String> observable, String oldValue, String newValue)
95  			{
96  				if (newValue.length() > 0)
97  				{
98  					descValid.set(true);
99  					legoListDescription.setEffect(null);
100 				}
101 				else
102 				{
103 					descValid.set(false);
104 					legoListDescription.setEffect(FxUtils.redDropShadow);
105 				}
106 			}
107 		});
108 
109 		formValid = new BooleanBinding()
110 		{
111 			{
112 				bind(nameValid, descValid);
113 			}
114 
115 			@Override
116 			protected boolean computeValue()
117 			{
118 				return nameValid.get() && descValid.get();
119 			}
120 		};
121 
122 		cancelButton.setOnAction(new EventHandler<ActionEvent>()
123 		{
124 			@Override
125 			public void handle(ActionEvent event)
126 			{
127 				((Stage) rootPane.getScene().getWindow()).close();
128 			}
129 		});
130 
131 		okButton.setOnAction(new EventHandler<ActionEvent>()
132 		{
133 			@Override
134 			public void handle(ActionEvent event)
135 			{
136 				if (llbr_ == null)
137 				{
138 					LegoList ll = new LegoList();
139 					ll.setGroupDescription(legoListDescription.getText());
140 					ll.setGroupName(legoListName.getText());
141 					ll.setLegoListUUID(legoListUUID.getText());
142 					ll.setComment(legoListComments.getText());
143 					try
144 					{
145 						AppContext.getService(LegoGUIMasterModel.class).importLegoList(ll);
146 					}
147 					catch (WriteException e)
148 					{
149 						AppContext.getCommonDialogs().showErrorDialog("Unexpected Error", "Error creating Lego List", e.toString());
150 					}
151 				}
152 				else
153 				{
154 					try
155 					{
156 						AppContext.getService(LegoGUIMasterModel.class).updateLegoList(llbr_, ti_, legoListName.getText(), legoListDescription.getText(),
157 								legoListComments.getText());
158 					}
159 					catch (WriteException e)
160 					{
161 						AppContext.getCommonDialogs().showErrorDialog("Unexpected Error", "Error updating Lego List", e.toString());
162 					}
163 				}
164 				((Stage) rootPane.getScene().getWindow()).close();
165 			}
166 		});
167 
168 		okButton.disableProperty().bind(formValid.not());
169 	}
170 
171 	public void setVariables(LegoListByReference llbr, TreeItem<String> ti)
172 	{
173 		ti_ = ti;
174 		llbr_ = llbr;
175 		legoListDescription.setText(llbr == null ? "" : llbr.getGroupDescription());
176 		legoListName.setText(llbr == null ? "" : llbr.getGroupName());
177 		legoListUUID.setText(llbr == null ? UUID.randomUUID().toString() : llbr.getLegoListUUID());
178 		legoListComments.setText(llbr == null ? "" : llbr.getComments());
179 	}
