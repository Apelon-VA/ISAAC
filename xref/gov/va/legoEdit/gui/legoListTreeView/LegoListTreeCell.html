69  	public LegoListTreeCell(LegoListTreeView lltv)
70  	{
71  		// For reasons I don't understand, the getTreeView() method is unreliable, sometimes returning null. Either a bug in javafx, or really poorly
72  		// documented...
73  		this.treeView = lltv;
74  
75  		addEventFilter(MouseEvent.MOUSE_CLICKED, new EventHandler<MouseEvent>()
76  		{
77  			@Override
78  			public void handle(MouseEvent event)
79  			{
80  				if (event.getClickCount() > 1)
81  				{
82  					final LegoListTreeItem treeItem = (LegoListTreeItem) getTreeItem();
83  					if (treeItem.getNodeType() != null)
84  					{
85  						if (LegoTreeNodeType.legoReference == treeItem.getNodeType())
86  						{
87  							LegoReference lr = (LegoReference) treeItem.getExtraData();
88  							new LegoSummaryPane(BDBDataStoreImpl.getInstance().getLego(lr.getLegoUUID(), lr.getStampUUID())).show();
89  						}
90  					}
91  				}
92  			}
93  		});
94  	}
